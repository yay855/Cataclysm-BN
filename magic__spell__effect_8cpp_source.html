60struct line_iterable {
61 const std::vector<point> &delta_line;
62 point cur_origin;
63 point delta;
64 size_t index;
65
66 line_iterable( const point &origin, const point &delta, const std::vector<point> &dline )
67 : delta_line( dline ), cur_origin( origin ), delta( delta ), index( 0 ) {}
68
69 point get() const {
70 return cur_origin + delta_line[index];
71 }
72 // Move forward along point set, wrap around and move origin forward if necessary
73 void next() {
74 index = ( index + 1 ) % delta_line.size();
75 cur_origin = cur_origin + delta * ( index == 0 );
76 }
77 // Move back along point set, wrap around and move origin backward if necessary
78 void prev() {
79 cur_origin = cur_origin - delta * ( index == 0 );
80 index = ( index + delta_line.size() - 1 ) % delta_line.size();
81 }
82 void reset( const point &origin ) {
83 cur_origin = origin;
84 index = 0;
85 }
86};

88static int side_of( const point &a, const point &b, const point &c )
89{
90 int cross = ( ( b.x - a.x ) * ( c.y - a.y ) - ( b.y - a.y ) * ( c.x - a.x ) );
91 return ( cross > 0 ) - ( cross < 0 );

93// Tests if point c is between or on lines (a0, a0 + d) and (a1, a1 + d)
94static bool between_or_on( const point &a0, const point &a1, const point &d, const point &c )
95{
96 return side_of( a0, a0 + d, c ) != 1 && side_of( a1, a1 + d, c ) != -1;

239std::set<tripoint> spell_effect::spell_effect_line( const spell &, const tripoint &source,
240 const tripoint &target, const int aoe_radius, const bool ignore_walls )
241{
242 const point delta = ( target - source ).xy();
243 const int dist = square_dist( point_zero, delta );
244 // Early out to prevent unnecessary calculations
245 if( dist == 0 ) {
246 return std::set<tripoint>();
247 }
248 // Clockwise Perpendicular of Delta vector
249 const point delta_perp( -delta.y, delta.x );
250
251 const point abs_delta = delta.abs();
252 // Primary axis of delta vector
253 const point axis_delta = abs_delta.x > abs_delta.y ? point( delta.x, 0 ) : point( 0, delta.y );
254 // Clockwise Perpendicular of axis vector
255 const point cw_perp_axis( -axis_delta.y, axis_delta.x );
256 const point unit_cw_perp_axis( sgn( cw_perp_axis.x ), sgn( cw_perp_axis.y ) );
257 // bias leg length toward cw side if uneven
258 int ccw_len = aoe_radius / 2;
259 int cw_len = aoe_radius - ccw_len;
260
261 if( !trigdist ) {
262 ccw_len = ( ccw_len * ( abs_delta.x + abs_delta.y ) ) / dist;
263 cw_len = ( cw_len * ( abs_delta.x + abs_delta.y ) ) / dist;
264 }
265
266 // is delta aligned with, cw, or ccw of primary axis
267 int delta_side = spell_detail::side_of( point_zero, axis_delta, delta );
268
269 bool ( *test )( const tripoint &,
270 const tripoint & ) = ignore_walls ? test_always_true : test_passable;
271
272 // Canonical path from source to target, offset to local space
273 std::vector<point> path_to_target = line_to( point_zero, delta );
274 // Remove endpoint,
275 path_to_target.pop_back();
276 // and insert startpoint. Path is now prepared for wrapped iteration
277 path_to_target.insert( path_to_target.begin(), point_zero );
278
279 spell_detail::line_iterable base_line( point_zero, delta, path_to_target );
280
281 std::set<tripoint> result;
282
283 // Add midline points (source -> target )
284 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
285
286 // Add cw and ccw legs
287 if( delta_side == 0 ) { // delta is already axis aligned, only need straight lines
288 // cw leg
289 point prev_point;
290 for( const point &p : line_to( point_zero, unit_cw_perp_axis * cw_len ) ) {
291 base_line.reset( p );
292 if( !test( source + p, source + prev_point ) ) {
293 break;
294 }
295
296 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
297 prev_point = p;
298 }
299 // ccw leg
300 prev_point = point_zero;
301 for( const point &p : line_to( point_zero, unit_cw_perp_axis * -ccw_len ) ) {
302 base_line.reset( p );
303 if( !test( source + p, source + prev_point ) ) {
304 break;
305 }
306
307 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
308 prev_point = p;
309 }
310 } else if( delta_side == 1 ) { // delta is cw of primary axis
311 // ccw leg is behind perp axis
312 point prev_point;
313 for( const point &p : line_to( point_zero, unit_cw_perp_axis * -ccw_len ) ) {
314 base_line.reset( p );
315
316 // forward until in
317 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) == 1 ) {
318 base_line.next();
319 }
320 if( !test( source + p, source + prev_point ) ) {
321 break;
322 }
323 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
324 prev_point = p;
325 }
326 prev_point = point_zero;
327 // cw leg is before perp axis
328 for( const point &p : line_to( point_zero, unit_cw_perp_axis * cw_len ) ) {
329 base_line.reset( p );
330
331 // move back
332 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) != 1 ) {
333 base_line.prev();
334 }
335 base_line.next();
336 if( !test( source + p, source + prev_point ) ) {
337 break;
338 }
339 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
340 prev_point = p;
341 }
342 } else if( delta_side == -1 ) { // delta is ccw of primary axis
343 // ccw leg is before perp axis
344 point prev_point;
345 for( const point &p : line_to( point_zero, unit_cw_perp_axis * -ccw_len ) ) {
346 base_line.reset( p );
347
348 // move back
349 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) != 1 ) {
350 base_line.prev();
351 }
352 base_line.next();
353 if( !test( source + p, source + prev_point ) ) {
354 break;
355 }
356 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
357 prev_point = p;
358 }
359 prev_point = point_zero;
360 // cw leg is behind perp axis
361 for( const point &p : line_to( point_zero, unit_cw_perp_axis * cw_len ) ) {
362 base_line.reset( p );
363
364 // forward until in
365 while( spell_detail::side_of( point_zero, delta_perp, base_line.get() ) == 1 ) {
366 base_line.next();
367 }
368 if( !test( source + p, source + prev_point ) ) {
369 break;
370 }
371 spell_detail::build_line( base_line, source, delta, delta_perp, test, result );
372 prev_point = p;
373 }
374 }
375
376 result.erase( source );
377 return result;
378}

552// Adds node to a search tree. Returns true if new node is allocated.
553bool area_expander::enqueue( const tripoint &from, const tripoint &to, float cost )
554{
555 if( contains( to ) ) {
556 // We will modify existing node if its cost is lower.
557 int index = area_search[to];
558 node &node = area[index];
559 if( cost < node.cost ) {
560 node.from = from;
561 node.cost = cost;
562 }
563 return false;
564 }

566 area.push_back( {to, from, cost} );
567 frontier.push( index );
568 area_search[to] = index;

572// Run wave propagation
573int area_expander::run( const tripoint &center )
574{
575 enqueue( center, center, 0.0 );
576
577 static constexpr std::array<int, 8> x_offset = {{ -1, 1, 0, 0, 1, -1, -1, 1 }};
578 static constexpr std::array<int, 8> y_offset = {{ 0, 0, -1, 1, -1, 1, -1, 1 }};
579
580 // Number of nodes expanded.
581 int expanded = 0;
582
583 map &here = get_map();
584
585 while( !frontier.empty() ) {
586 int best_index = frontier.top();
587 frontier.pop();
588 node &best = area[best_index];
589
590 for( size_t i = 0; i < 8; i++ ) {
591 tripoint pt = best.position + point( x_offset[ i ], y_offset[ i ] );
592
593 if( ( here.impassable( pt ) && !here.has_flag( "THIN_OBSTACLE", pt ) ) ||
594 here.obstructed_by_vehicle_rotation( best.position, pt ) ) {
595 continue;
596 }
597
598 float center_range = static_cast<float>( rl_dist( center, pt ) );
599 if( max_range > 0 && center_range > max_range ) {
600 continue;
601 }
602
603 if( max_expand > 0 && expanded > max_expand && contains( pt ) ) {
604 continue;
605 }
606
607 float delta_range = trig_dist( best.position, pt );
608
609 if( enqueue( best.position, pt, best.cost + delta_range ) ) {
610 expanded++;
611 }
612 }
613 }
614 return expanded;
615}

639static void spell_move( const spell &sp, const Creature &caster,

641{

644 }
645

648 sp.is_valid_effect_target( target_ally ) ||
