111void input_manager::init()
112{
113 std::map<char, action_id> keymap;
114 std::string keymap_file_loaded_from;
115 std::set<action_id> unbound_keymap;
116 init_keycode_mapping();
117 reset_timeout();
118
119 try {
120 load( PATH_INFO::keybindings(), false );
121 } catch( const JsonError &err ) {
122 throw std::runtime_error( err.what() );
123 }
124 try {
125 load( PATH_INFO::keybindings_vehicle(), false );
126 } catch( const JsonError &err ) {
127 throw std::runtime_error( err.what() );
128 }
129 try {
130 load( PATH_INFO::keybindings_edit_creature(), false );
131 } catch( const JsonError &err ) {
132 throw std::runtime_error( err.what() );
133 }
134 try {
135 load( PATH_INFO::user_keybindings(), true );
136 } catch( const JsonError &err ) {
137 throw std::runtime_error( err.what() );
138 }
139
140 if( keymap_file_loaded_from.empty() || ( keymap.empty() && unbound_keymap.empty() ) ) {
141 // No keymap file was loaded, or the file has no mappings and no unmappings,
142 // we can skip the remaining part of the function, especially the save function
143 return;
144 }
145 t_actions &actions = action_contexts["DEFAULTMODE"];
146 std::set<action_id> touched;
147 for( std::map<char, action_id>::const_iterator a = keymap.begin(); a != keymap.end(); ++a ) {
148 const std::string action_id = action_ident( a->second );
149 // Put the binding from keymap either into the global context
150 // (if an action with that ident already exists there - think movement keys)
151 // or otherwise to the DEFAULTMODE context.
152 std::string context = "DEFAULTMODE";
153 if( action_contexts[default_context_id].count( action_id ) > 0 ) {
154 context = default_context_id;
155 } else if( touched.count( a->second ) == 0 ) {
156 // Note: movement keys are somehow special as the default in keymap
157 // does not contain the arrow keys, so we don't clear existing keybindings
158 // for them.
159 // If the keymap contains a binding for this action, erase all the
160 // previously (default!) existing bindings, to only keep the bindings,
161 // the user is used to
162 action_contexts[action_id].clear();
163 touched.insert( a->second );
164 }
165 add_input_for_action( action_id, context, input_event( a->first, CATA_INPUT_KEYBOARD ) );
166 }
167 // Unmap actions that are explicitly not mapped
168 for( const auto &elem : unbound_keymap ) {
169 const std::string action_id = action_ident( elem );
170 actions[action_id].input_events.clear();
171 }
172 // Imported old bindings from old keymap file, save those to the new
173 // keybindings.json file.
174 try {
175 save();
176 } catch( std::exception &err ) {
177 debugmsg( "Could not write imported keybindings: %s", err.what() );
178 return;
179 }
180 // Finally if we did import a file, and saved it to the new keybindings
181 // file, delete the old keymap file to prevent re-importing it.
182 if( !keymap_file_loaded_from.empty() ) {
183 remove_file( keymap_file_loaded_from );
184 }

187void input_manager::load( const std::string &file_name, bool is_user_preferences )
188{
189 std::ifstream data_file( file_name.c_str(), std::ifstream::in | std::ifstream::binary );
190
191 if( !data_file.good() ) {
192 // Only throw if this is the first file to load, that file _must_ exist,
193 // otherwise the keybindings can not be read at all.
194 if( action_contexts.empty() ) {
195 throw std::runtime_error( std::string( "Could not read " ) + file_name );
196 }
197 return;
198 }
199
200 JsonIn jsin( data_file, file_name );
201
202 //Crawl through once and create an entry for every definition
203 jsin.start_array();
204 while( !jsin.end_array() ) {
205 // JSON object representing the action
206 JsonObject action = jsin.get_object();
207
208 const std::string type = action.get_string( "type", "keybinding" );
209 if( type != "keybinding" ) {
210 debugmsg( "Only objects of type 'keybinding' (not %s) should appear in the "
211 "keybindings file '%s'", type, file_name );
212 continue;
213 }
214
215 const std::string action_id = action.get_string( "id" );
216 const std::string context = action.get_string( "category", default_context_id );
217 t_actions &actions = action_contexts[context];
218 if( !is_user_preferences && action.has_member( "name" ) ) {
219 // Action names are not user preferences. Some experimental builds
220 // post-0.A had written action names into the user preferences
221 // config file. Any names that exist in user preferences will be
222 // ignored.
223 action.read( "name", actions[action_id].name );
224 }
225
226 t_input_event_list events;
227 for( const JsonObject keybinding : action.get_array( "bindings" ) ) {
228 std::string input_method = keybinding.get_string( "input_method" );
229 input_event new_event;
230 if( input_method == "keyboard" ) {
231 new_event.type = CATA_INPUT_KEYBOARD;
232 } else if( input_method == "gamepad" ) {
233 new_event.type = CATA_INPUT_GAMEPAD;
234 } else if( input_method == "mouse" ) {
235 new_event.type = CATA_INPUT_MOUSE;
236 }
237
238 if( keybinding.has_array( "key" ) ) {
239 for( const std::string line : keybinding.get_array( "key" ) ) {
240 new_event.sequence.push_back( get_keycode( line ) );
241 }
242 } else { // assume string if not array, and throw if not string
243 new_event.sequence.push_back(
244 get_keycode( keybinding.get_string( "key" ) )
245 );
246 }
247
248 events.push_back( new_event );
249 }
250
251 // An invariant of this class is that user-created, local keybindings
252 // with an empty set of input_events do not exist in the
253 // action_contexts map. In prior versions of this class, this was not
254 // true, so users of experimental builds post-0.A will have empty
255 // local keybindings saved in their keybindings.json config.
256 //
257 // To be backwards compatible with keybindings.json from prior
258 // experimental builds, we will detect user-created, local keybindings
259 // with empty input_events and disregard them. When keybindings are
260 // later saved, these remnants won't be saved.
261 if( !is_user_preferences ||
262 !events.empty() ||
263 context == default_context_id ||
264 actions.count( action_id ) > 0 ) {
265 // In case this is the second file containing user preferences,
266 // this replaces the default bindings with the user's preferences.
267 action_attributes &attributes = actions[action_id];
268 attributes.input_events = events;
269 if( action.has_member( "is_user_created" ) ) {
270 attributes.is_user_created = action.get_bool( "is_user_created" );
271 }
272 }
273 }

276void input_manager::save()
277{
278 write_to_file( PATH_INFO::user_keybindings(), [&]( std::ostream & data_file ) {
279 JsonOut jsout( data_file, true );
280
281 jsout.start_array();
282 for( t_action_contexts::const_iterator a = action_contexts.begin(); a != action_contexts.end();
283 ++a ) {
284 const t_actions &actions = a->second;
285 for( const auto &action : actions ) {
286 const t_input_event_list &events = action.second.input_events;
287 jsout.start_object();
288
289 jsout.member( "id", action.first );
290 jsout.member( "category", a->first );
291 bool is_user_created = action.second.is_user_created;
292 if( is_user_created ) {
293 jsout.member( "is_user_created", is_user_created );
294 }
295
296 jsout.member( "bindings" );
297 jsout.start_array();
298 for( const auto &event : events ) {
299 jsout.start_object();
300 switch( event.type ) {
301 case CATA_INPUT_KEYBOARD:
302 jsout.member( "input_method", "keyboard" );
303 break;
304 case CATA_INPUT_GAMEPAD:
305 jsout.member( "input_method", "gamepad" );
306 break;
307 case CATA_INPUT_MOUSE:
308 jsout.member( "input_method", "mouse" );
309 break;
310 default:
311 throw std::runtime_error( "unknown input_event_t" );
312 }
313 jsout.member( "key" );
314 jsout.start_array();
315 for( size_t i = 0; i < event.sequence.size(); i++ ) {
316 jsout.write( get_keyname( event.sequence[i], event.type, true ) );
317 }
318 jsout.end_array();
319 jsout.end_object();
320 }
321 jsout.end_array();
322
323 jsout.end_object();
324 }
325 }
326 jsout.end_array();
327 }, _( "key bindings configuration" ) );

415std::string input_manager::get_keyname( int ch, input_event_t inp_type, bool portable ) const
416{
417 cata::optional<std::string> raw;
418 if( inp_type == CATA_INPUT_KEYBOARD ) {
419 const t_key_to_name_map::const_iterator a = keycode_to_keyname.find( ch );
420 if( a != keycode_to_keyname.end() ) {
421 if( IS_F_KEY( ch ) ) {
422 // special case it since F<num> key names are generated using loop
423 // and not marked individually for translation
424 if( portable ) {
425 return a->second;
426 } else {
427 return string_format( pgettext( "function key name", "F%d" ), F_KEY_NUM( ch ) );
428 }
429 } else if( ch >= char_key_beg && ch <= char_key_end && ch != ' ' ) {
430 // character keys except space need no translation
431 return a->second;
432 }
433 raw = a->second;
434 }
435 } else if( inp_type == CATA_INPUT_MOUSE ) {
436 if( ch == MOUSE_BUTTON_LEFT ) {
437 raw = translate_marker_context( "key name", "MOUSE_LEFT" );
438 } else if( ch == MOUSE_BUTTON_RIGHT ) {
439 raw = translate_marker_context( "key name", "MOUSE_RIGHT" );
440 } else if( ch == SCROLLWHEEL_UP ) {
441 raw = translate_marker_context( "key name", "SCROLL_UP" );
442 } else if( ch == SCROLLWHEEL_DOWN ) {
443 raw = translate_marker_context( "key name", "SCROLL_DOWN" );
444 } else if( ch == MOUSE_MOVE ) {
445 raw = translate_marker_context( "key name", "MOUSE_MOVE" );
446 }
447 } else if( inp_type == CATA_INPUT_GAMEPAD ) {
448 const t_key_to_name_map::const_iterator a = gamepad_keycode_to_keyname.find( ch );
449 if( a != gamepad_keycode_to_keyname.end() ) {
450 raw = a->second;
451 }
452 } else {
453 raw = translate_marker_context( "key name", "UNKNOWN" );
454 }
455 if( !raw ) {
456 if( portable ) {
457 return std::string( "UNKNOWN_" ) + int_to_str( ch );
458 }
459 return string_format( _( "unknown key %ld" ), ch );
460 }
461 return portable ? *raw : pgettext( "key name", raw->c_str() );
462}

479const action_attributes &input_manager::get_action_attributes(
480 const std::string &action_id,
481 const std::string &context,
482 bool *overwrites_default )
483{
484
485 if( context != default_context_id ) {
486 // Check if the action exists in the provided context
487 const t_action_contexts::const_iterator action_context = action_contexts.find( context );
488 if( action_context != action_contexts.end() ) {
489 const t_actions::const_iterator action = action_context->second.find( action_id );
490 if( action != action_context->second.end() ) {
491 if( overwrites_default ) {
492 *overwrites_default = true;
493 }
494
495 return action->second;
496 }
497 }
498 }
499
500 // If not, we use the default binding.
501 if( overwrites_default ) {
502 *overwrites_default = false;
503 }
504
505 t_actions &default_action_context = action_contexts[default_context_id];
506 const t_actions::const_iterator default_action = default_action_context.find( action_id );
507 if( default_action == default_action_context.end() ) {
508 // A new action is created in the event that the requested action is
509 // not in the keybindings configuration e.g. the entry is missing.
510 default_action_context[action_id].name = get_default_action_name( action_id );
511 }
512
513 return default_action_context[action_id];

551void input_manager::remove_input_for_action(
552 const std::string &action_descriptor, const std::string &context )
553{
554 const t_action_contexts::iterator action_context = action_contexts.find( context );
555 if( action_context != action_contexts.end() ) {
556 t_actions &actions = action_context->second;
557 t_actions::iterator action = actions.find( action_descriptor );
558 if( action != actions.end() ) {
559 if( action->second.is_user_created ) {
560 // Since this is a user created hotkey, remove it so that the
561 // user will fallback to the hotkey in the default context.
562 actions.erase( action );
563 } else {
564 // If a context no longer has any keybindings remaining for an action but
565 // there's an attempt to remove bindings anyway, presumably the user wants
566 // to fully remove the binding from that context.
567 if( action->second.input_events.empty() ) {
568 actions.erase( action );
569 } else {
570 action->second.input_events.clear();
571 }
572 }
573 }
574 }

604void input_context::clear_conflicting_keybindings( const input_event &event )
605{
606 // The default context is always included to cover cases where the same
607 // keybinding exists for the same action in both the global and local
608 // contexts.

622 std::vector<input_event> &events = category_action->second.input_events;

756std::string input_context::get_desc( const std::string &action_descriptor,
757 const unsigned int max_limit,
758 const input_context::input_event_filter &evt_filter ) const
759{
760 if( action_descriptor == "ANY_INPUT" ) {
761 return "(*)"; // * for wildcard
762 }
763
764 bool is_local = false;
765 const std::vector<input_event> &events = inp_mngr.get_input_for_action( action_descriptor,
766 category, &is_local );
767
768 if( events.empty() ) {
769 return is_local ? _( "Unbound locally!" ) : _( "Unbound globally!" );
770 }
771
772 std::vector<input_event> inputs_to_show;
773 for( auto &events_i : events ) {
774 const input_event &event = events_i;
775
776 if( evt_filter( event ) &&
777 // Only display gamepad buttons if a gamepad is available.
778 ( gamepad_available() || event.type != CATA_INPUT_GAMEPAD ) ) {
779
780 inputs_to_show.push_back( event );
781 }
782
783 if( max_limit > 0 && inputs_to_show.size() == max_limit ) {
784 break;
785 }
786 }
787
788 if( inputs_to_show.empty() ) {
789 return pgettext( "keybinding", "Disabled" );
790 }
791
792 std::string rval;
793 for( size_t i = 0; i < inputs_to_show.size(); ++i ) {
794 for( size_t j = 0; j < inputs_to_show[i].sequence.size(); ++j ) {
795 rval += inp_mngr.get_keyname( inputs_to_show[i].sequence[j], inputs_to_show[i].type );
796 }
797
798 // We're generating a list separated by "," and "or"
799 if( i + 2 == inputs_to_show.size() ) {
800 rval += _( " or " );
801 } else if( i + 1 < inputs_to_show.size() ) {
802 rval += ", ";
803 }
804 }
805 return rval;
806}

808std::string input_context::get_desc( const std::string &action_descriptor,
809 const std::string &text,
810 const input_context::input_event_filter &evt_filter ) const
811{
812 if( action_descriptor == "ANY_INPUT" ) {
813 // \u00A0 is the non-breaking space
814 //~ keybinding description for anykey
815 return string_format( pgettext( "keybinding", "[any]\u00A0%s" ), text );
816 }
817
818 const auto &events = inp_mngr.get_input_for_action( action_descriptor, category );
819
820 bool na = true;
821 for( const auto &evt : events ) {
822 if( evt_filter( evt ) &&
823 // Only display gamepad buttons if a gamepad is available.
824 ( gamepad_available() || evt.type != CATA_INPUT_GAMEPAD ) ) {
825
826 na = false;
827 if( evt.type == CATA_INPUT_KEYBOARD && evt.sequence.size() == 1 ) {
828 const int ch = evt.get_first_input();
829 const std::string key = utf32_to_utf8( ch );
830 const auto pos = ci_find_substr( text, key );
831 if( ch > ' ' && ch <= '~' && pos >= 0 ) {
832 return text.substr( 0, pos ) + "(" + key + ")" + text.substr( pos + key.size() );
833 }
834 }
835 }
836 }
837
838 if( na ) {
839 //~ keybinding description for unbound or disabled keys
840 return string_format( pgettext( "keybinding", "[n/a]\u00A0%s" ), text );
841 } else {
842 //~ keybinding description for bound keys
843 return string_format( pgettext( "keybinding", "[%s]\u00A0%s" ),
844 get_desc( action_descriptor, 1, evt_filter ), text );
845 }
846}

859const std::string &input_context::handle_input( const int timeout )
860{
861 const auto old_timeout = inp_mngr.get_timeout();
862 if( timeout >= 0 ) {
863 inp_mngr.set_timeout( timeout );
864 }
865 next_action.type = CATA_INPUT_ERROR;
866 const std::string *result = &CATA_ERROR;
867 while( true ) {
868 next_action = inp_mngr.get_input_event();
869 if( next_action.type == CATA_INPUT_TIMEOUT ) {
870 result = &TIMEOUT;
871 break;
872 }
873
874 const std::string &action = input_to_action( next_action );
875
876 // Special help action
877 if( action == "HELP_KEYBINDINGS" ) {
878 inp_mngr.reset_timeout();
879 display_menu();
880 inp_mngr.set_timeout( timeout );
881 result = &HELP_KEYBINDINGS;
882 break;
883 }
884
885 if( next_action.type == CATA_INPUT_MOUSE ) {
886 if( !handling_coordinate_input && action == CATA_ERROR ) {
887 continue; // Ignore this mouse input.
888 }
889
890 coordinate_input_received = true;
891 coordinate = next_action.mouse_pos;
892 } else {
893 coordinate_input_received = false;
894 }
895
896 if( action != CATA_ERROR ) {
897 result = &action;
898 break;
899 }
900
901 // If we registered to receive any input, return ANY_INPUT
902 // to signify that an unregistered key was pressed.
903 if( registered_any_input ) {
904 result = &ANY_INPUT;
905 break;
906 }
907
908 // If it's an invalid key, just keep looping until the user
909 // enters something proper.
910 }
911 inp_mngr.set_timeout( old_timeout );
912 return *result;
913}

942// dx and dy are -1, 0, or +1. Rotate the indicated direction 1/8 turn clockwise.
943void rotate_direction_cw( int &dx, int &dy )
944{
945 // convert to
946 // 0 1 2
947 // 3 4 5
948 // 6 7 8
949 int dir_num = ( dy + 1 ) * 3 + dx + 1;
950 // rotate to
951 // 1 2 5
952 // 0 4 8
953 // 3 6 7
954 static const std::array<int, 9> rotate_direction_vec = {{ 1, 2, 5, 0, 4, 8, 3, 6, 7 }};
955 dir_num = rotate_direction_vec[dir_num];
956 // convert back to -1,0,+1
957 dx = dir_num % 3 - 1;
958 dy = dir_num / 3 - 1;
959}

1272 }
1273 } else if( action == "QUIT" ) {

1275 status = s_show;
1276 } else {

1278 }

1280 // update available hotkeys in case they've changed
1281 hotkeys = ctxt.get_available_single_char_hotkeys( display_help_hotkeys );
