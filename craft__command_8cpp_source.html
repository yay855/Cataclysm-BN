223item craft_command::create_in_progress_craft()
224{
225 // Use up the components and tools
226 std::list<item> used;
227 std::vector<item_comp> comps_used;
228 if( crafter->has_trait( trait_DEBUG_HS ) ) {
229 item new_craft( rec, batch_size, used, comps_used );
230 new_craft.set_tools_to_continue( true );
231 return new_craft;
232 }
233
234 if( empty() ) {
235 debugmsg( "Warning: attempted to consume items from an empty craft_command" );
236 return item();
237 }
238
239 inventory map_inv;
240 map_inv.form_from_map( crafter->pos(), PICKUP_RANGE, crafter );
241
242 if( !check_item_components_missing( map_inv ).empty() ) {
243 debugmsg( "Aborting crafting: couldn't find cached components" );
244 return item();
245 }
246
247 const auto filter = rec->get_component_filter( flags );
248
249 for( const auto &it : item_selections ) {
250 std::list<item> tmp = crafter->consume_items( it, batch_size, filter );
251 used.splice( used.end(), tmp );
252 }
253
254 for( const comp_selection<item_comp> &selection : item_selections ) {
255 item_comp comp_used = selection.comp;
256 comp_used.count *= batch_size;
257
258 //Handle duplicate component requirement
259 auto found_it = std::find_if( comps_used.begin(),
260 comps_used.end(), [&comp_used]( const item_comp & c ) {
261 return c.type == comp_used.type;
262 } );
263 if( found_it != comps_used.end() ) {
264 item_comp &found_comp = *found_it;
265 found_comp.count += comp_used.count;
266 } else {
267 comps_used.emplace_back( comp_used );
268 }
269 }
270
271 item new_craft( rec, batch_size, used, comps_used );
272
273 new_craft.set_cached_tool_selections( tool_selections );
274 new_craft.set_tools_to_continue( true );
275 // Pass true to indicate that we are starting the craft and the remainder should be consumed as well
276 crafter->craft_consume_tools( new_craft, 1, true );
277 new_craft.set_next_failure_point( *crafter );
278
279 return new_craft;
280}

287std::vector<comp_selection<item_comp>> craft_command::check_item_components_missing(
288 const inventory &map_inv ) const
289{
290 std::vector<comp_selection<item_comp>> missing;
291
292 const auto filter = rec->get_component_filter( flags );
293
294 for( const auto &item_sel : item_selections ) {
295 itype_id type = item_sel.comp.type;
296 const item_comp component = item_sel.comp;
297 const int count = component.count > 0 ? component.count * batch_size : std::abs( component.count );
298
299 if( item::count_by_charges( type ) && count > 0 ) {
300 switch( item_sel.use_from ) {
301 case use_from_player:
302 if( !crafter->has_charges( type, count, filter ) ) {
303 missing.push_back( item_sel );
304 }
305 break;
306 case use_from_map:
307 if( !map_inv.has_charges( type, count, filter ) ) {
308 missing.push_back( item_sel );
309 }
310 break;
311 case use_from_both:
312 if( !( crafter->charges_of( type, INT_MAX, filter ) +
313 map_inv.charges_of( type, INT_MAX, filter ) >= count ) ) {
314 missing.push_back( item_sel );
315 }
316 break;
317 case use_from_none:
318 case cancel:
319 case num_usages:
320 break;
321 }
322 } else {
323 // Counting by units, not charges.
324 switch( item_sel.use_from ) {
325 case use_from_player:
326 if( !crafter->has_amount( type, count, false, filter ) ) {
327 missing.push_back( item_sel );
328 }
329 break;
330 case use_from_map:
331 if( !map_inv.has_components( type, count, filter ) ) {
332 missing.push_back( item_sel );
333 }
334 break;
335 case use_from_both:
336 if( !( crafter->amount_of( type, false, std::numeric_limits<int>::max(), filter ) +
337 map_inv.amount_of( type, false, std::numeric_limits<int>::max(), filter ) >= count ) ) {
338 missing.push_back( item_sel );
339 }
340 break;
341 case use_from_none:
342 case cancel:
343 case num_usages:
344 break;
345 }
346 }
347 }
348
349 return missing;
350}
