88/**
89 * Base class for reading generic objects from JSON.
90 * It can load members being certain containers or being a single value.
91 * The function get_next() needs to be implemented to read and convert the data from JSON.
92 * It uses the curiously recurring template pattern, you have to derive your new class
93 * `MyReader` from `generic_typed_reader<MyReader>` and implement `get_next` and
94 * optionally `erase_next`.
95 * Most function calls here are done on a `Derived`, which means it can "override" them.
96 * This even allows changing their signature and return type.
97 *
98 * - If the object is new (`was_loaded` is `false`), only the given JSON member is read
99 * and assigned, overriding any existing content of it.
100 * - If the object is not new and the member exists, it is read and assigned as well.
101 * - If the object is not new and the member does not exists, two further members are examined:
102 * entries from `"extend"` are added to the set and entries from `"delete"`
103 * are removed. This only works if the member is actually a container, not just a single value.
104 *
105 * Example:
106 * The JSON `{ "f": ["a","b","c"] }` would be loaded as the set `{"a","b","c"}`.
107 * Loading the set again from the JSON `{ "delete": { "f": ["c","x"] }, "extend": { "f": ["h"] } }`
108 * would add the "h" flag and removes the "c" and the "x" flag, resulting in `{"a","b","h"}`.
109 *
110 * @tparam Derived The class that inherits from this. It must implement the following:
111 * - `Foo get_next( JsonIn & ) const`: reads the next value from JSON, converts it into some
112 * type `Foo` and returns it. The returned value is assigned to the loaded member (see reader
113 * interface above), or is inserted into the member (if it's a container). The type `Foo` must
114 * be compatible with those uses (read: it should be the same type).
115 * - (optional) `erase_next( JsonIn &jin, C &container ) const`, the default implementation here
116 * reads a value from JSON via `get_next` and removes the matching value in the container.
117 * The value in the container must match *exactly*. You may override this function to allow
118 * a different matching algorithm, e.g. reading a simple id from JSON and remove entries with
119 * the same id from the container.
120 */
121template<typename Derived>
122class generic_typed_reader
123{
124 public:
125 template<typename C>
126 void insert_values_from( const JsonObject &jo, const std::string &member_name,
127 C &container ) const {
128 const Derived &derived = static_cast<const Derived &>( *this );
129 if( !jo.has_member( member_name ) ) {
130 return;
131 }
132 JsonIn &jin = *jo.get_raw( member_name );
133 // We allow either a single value or an array of values. Note that this will not work
134 // correctly if the thing we load from JSON is itself an array.
135 if( jin.test_array() ) {
136 jin.start_array();
137 while( !jin.end_array() ) {
138 derived.insert_next( jin, container );
139 }
140 } else {
141 derived.insert_next( jin, container );
142 }
143 }
144 template<typename C>
145 void insert_next( JsonIn &jin, C &container ) const {
146 const Derived &derived = static_cast<const Derived &>( *this );
147 reader_detail::handler<C>().insert( container, derived.get_next( jin ) );
148 }
149
150 template<typename C>
151 void erase_values_from( const JsonObject &jo, const std::string &member_name, C &container ) const {
152 const Derived &derived = static_cast<const Derived &>( *this );
153 if( !jo.has_member( member_name ) ) {
154 return;
155 }
156 JsonIn &jin = *jo.get_raw( member_name );
157 // Same as for inserting: either an array or a single value, same caveat applies.
158 if( jin.test_array() ) {
159 jin.start_array();
160 while( !jin.end_array() ) {
161 derived.erase_next( jin, container );
162 }
163 } else {
164 derived.erase_next( jin, container );
165 }
166 }
167 template<typename C>
168 void erase_next( JsonIn &jin, C &container ) const {
169 const Derived &derived = static_cast<const Derived &>( *this );
170 reader_detail::handler<C>().erase( container, derived.get_next( jin ) );
171 }

173 /**
174 * Implements the reader interface, handles members that are containers of flags.
175 * The functions forwards the actual changes to assign(), insert()
176 * and erase(), which are specialized for various container types.
177 * The `enable_if` is here to prevent the compiler from considering it
178 * when called on a simple data member, the other `operator()` will be used.
179 */
180 template<typename C, typename std::enable_if<reader_detail::handler<C>::is_container, int>::type = 0>
181 bool operator()( const JsonObject &jo, const std::string &member_name,
182 C &container, bool was_loaded ) const {
183 const Derived &derived = static_cast<const Derived &>( *this );
184 // If you get an error about "incomplete type 'struct reader_detail::handler...",
185 // you have to implement a specialization of your container type, so above for
186 // existing specializations in namespace reader_detail.
187 if( jo.has_member( member_name ) ) {
188 reader_detail::handler<C>().clear( container );
189 derived.insert_values_from( jo, member_name, container );
190 return true;
191 } else if( !was_loaded ) {
192 return false;
193 } else {
194 if( jo.has_object( "extend" ) ) {
195 JsonObject tmp = jo.get_object( "extend" );
196 tmp.allow_omitted_members();
197 derived.insert_values_from( tmp, member_name, container );
198 }
199 if( jo.has_object( "delete" ) ) {
200 JsonObject tmp = jo.get_object( "delete" );
201 tmp.allow_omitted_members();
202 derived.erase_values_from( tmp, member_name, container );
203 }
204 return true;
205 }
206 }

208 /**
209 * Implements the reader interface, handles a simple data member.
210 */
211 // was_loaded is ignored here, if the value is not found in JSON, report to
212 // the caller, which will take action on their own.
213 template < typename C, typename std::enable_if < !reader_detail::handler<C>::is_container,
214 int >::type = 0 >
215 bool operator()( const JsonObject &jo, const std::string &member_name,
216 C &member, bool /*was_loaded*/ ) const {
217 const Derived &derived = static_cast<const Derived &>( *this );
218 if( !jo.has_member( member_name ) ) {
219 return false;
220 }
221 member = derived.get_next( *jo.get_raw( member_name ) );
222 return true;
223 }

296/**
297 * Uses a map (unordered or standard) to convert strings from JSON to some other type
298 * (the mapped type of the map: `C::mapped_type`). It works for all mapped types, not just enums.
299 *
300 * One can use this if the member is `std::set<some_enum>` or `some_enum` and a
301 * map `std::map<std::string, some_enum>` with all the value enumeration values exists.
302 *
303 * The class can be conveniently instantiated for a given map `mapping` using
304 * the helper function @ref make_flag_reader (see below).
305 * The flag type (@ref flag_type) is used when the input contains invalid flags
306 * (a string that is not contained in the map). It should sound something like
307 * "my-enum-type".
308 */
309template<typename T>
310class typed_flag_reader : public generic_typed_reader<typed_flag_reader<T>>
311{
312 private:
313 using map_t = std::map<std::string, T>;
314
315 private:
316 const map_t &flag_map;
317 const std::string flag_type;
318
319 public:
320 typed_flag_reader( const map_t &flag_map, const std::string &flag_type )
321 : flag_map( flag_map )
322 , flag_type( flag_type ) {
323 }
324
325 T get_next( JsonIn &jin ) const {
326 const std::string flag = jin.get_string();
327 const auto iter = flag_map.find( flag );
328
329 if( iter == flag_map.cend() ) {
330 jin.seek( jin.tell() );
331 jin.error( string_format( "invalid %s: \"%s\"", flag_type, flag ) );
332 }
333
334 return iter->second;
335 }
336};
