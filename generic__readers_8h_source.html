11/*
12 * Template vodoo:
13 * The compiler will construct the appropriate one of these based on if the
14 * type can support the operations being done.
15 * So, it defaults to the false_type, but if it can use the *= operator
16 * against a float, it then supports proportional, and the handle_proportional
17 * template that isn't just a dummy is constructed.
18 * Similarly, if it can use a += operator against it's own type, the non-dummy
19 * handle_relative template is constructed.
20 */

114/**
115 * Base class for reading generic objects from JSON.
116 * It can load members being certain containers or being a single value.
117 * The function get_next() needs to be implemented to read and convert the data from JSON.
118 * It uses the curiously recurring template pattern, you have to derive your new class
119 * `MyReader` from `generic_typed_reader<MyReader>` and implement `get_next` and
120 * optionally `erase_next`.
121 * Most function calls here are done on a `Derived`, which means it can "override" them.
122 * This even allows changing their signature and return type.
123 *
124 * - If the object is new (`was_loaded` is `false`), only the given JSON member is read
125 * and assigned, overriding any existing content of it.
126 * - If the object is not new and the member exists, it is read and assigned as well.
127 * - If the object is not new and the member does not exists, two further members are examined:
128 * entries from `"extend"` are added to the set and entries from `"delete"`
129 * are removed. This only works if the member is actually a container, not just a single value.
130 *
131 * Example:
132 * The JSON `{ "f": ["a","b","c"] }` would be loaded as the set `{"a","b","c"}`.
133 * Loading the set again from the JSON `{ "delete": { "f": ["c","x"] }, "extend": { "f": ["h"] } }`
134 * would add the "h" flag and removes the "c" and the "x" flag, resulting in `{"a","b","h"}`.
135 *
136 * @tparam Derived The class that inherits from this. It must implement the following:
137 * - `Foo get_next( JsonIn & ) const`: reads the next value from JSON, converts it into some
138 * type `Foo` and returns it. The returned value is assigned to the loaded member (see reader
139 * interface above), or is inserted into the member (if it's a container). The type `Foo` must
140 * be compatible with those uses (read: it should be the same type).
141 * - (optional) `erase_next( JsonIn &jin, C &container ) const`, the default implementation here
142 * reads a value from JSON via `get_next` and removes the matching value in the container.
143 * The value in the container must match *exactly*. You may override this function to allow
144 * a different matching algorithm, e.g. reading a simple id from JSON and remove entries with
145 * the same id from the container.
146 */
147template<typename Derived>
148class generic_typed_reader
149{
150 public:
151 template<typename C>
152 void insert_values_from( const JsonObject &jo, const std::string &member_name,
153 C &container ) const {
154 const Derived &derived = static_cast<const Derived &>( *this );
155 if( !jo.has_member( member_name ) ) {
156 return;
157 }
158 JsonIn &jin = *jo.get_raw( member_name );
159 // We allow either a single value or an array of values. Note that this will not work
160 // correctly if the thing we load from JSON is itself an array.
161 if( jin.test_array() ) {
162 jin.start_array();
163 while( !jin.end_array() ) {
164 derived.insert_next( jin, container );
165 }
166 } else {
167 derived.insert_next( jin, container );
168 }
169 }
170 template<typename C>
171 void insert_next( JsonIn &jin, C &container ) const {
172 const Derived &derived = static_cast<const Derived &>( *this );
173 reader_detail::handler<C>().insert( container, derived.get_next( jin ) );
174 }
175
176 template<typename C>
177 void erase_values_from( const JsonObject &jo, const std::string &member_name, C &container ) const {
178 const Derived &derived = static_cast<const Derived &>( *this );
179 if( !jo.has_member( member_name ) ) {
180 return;
181 }
182 JsonIn &jin = *jo.get_raw( member_name );
183 // Same as for inserting: either an array or a single value, same caveat applies.
184 if( jin.test_array() ) {
185 jin.start_array();
186 while( !jin.end_array() ) {
187 derived.erase_next( jin, container );
188 }
189 } else {
190 derived.erase_next( jin, container );
191 }
192 }
193 template<typename C>
194 void erase_next( JsonIn &jin, C &container ) const {
195 const Derived &derived = static_cast<const Derived &>( *this );
196 reader_detail::handler<C>().erase( container, derived.get_next( jin ) );
197 }

199 /**
200 * Implements the reader interface, handles members that are containers of flags.
201 * The functions forwards the actual changes to assign(), insert()
202 * and erase(), which are specialized for various container types.
203 * The `enable_if` is here to prevent the compiler from considering it
204 * when called on a simple data member, the other `operator()` will be used.
205 */
206 template<typename C, typename std::enable_if<reader_detail::handler<C>::is_container, int>::type = 0>
207 bool operator()( const JsonObject &jo, const std::string &member_name,
208 C &container, bool was_loaded ) const {
209 const Derived &derived = static_cast<const Derived &>( *this );
210 // If you get an error about "incomplete type 'struct reader_detail::handler...",
211 // you have to implement a specialization of your container type, so above for
212 // existing specializations in namespace reader_detail.
213 if( jo.has_member( member_name ) ) {
214 reader_detail::handler<C>().clear( container );
215 derived.insert_values_from( jo, member_name, container );
216 return true;
217 } else if( !was_loaded ) {
218 return false;
219 } else {
220 if( jo.has_object( "extend" ) ) {
221 JsonObject tmp = jo.get_object( "extend" );
222 tmp.allow_omitted_members();
223 derived.insert_values_from( tmp, member_name, container );
224 }
225 if( jo.has_object( "delete" ) ) {
226 JsonObject tmp = jo.get_object( "delete" );
227 tmp.allow_omitted_members();
228 derived.erase_values_from( tmp, member_name, container );
229 }
230 return true;
231 }
232 }

234 /*
235 * These two functions are effectively handle_relative but they need to
236 * use the reader, so they must be here.
237 * proportional does not need these, because it's only reading a float
238 * whereas these are reading values of the same type.
239 */
240 // Type does not support relative
241 template < typename C, typename std::enable_if < !reader_detail::handler<C>::is_container,
242 int >::type = 0,
243 std::enable_if_t < !supports_relative<C>::value > * = nullptr
244 >
245 bool do_relative( const JsonObject &jo, const std::string &name, C & ) const {
246 if( jo.has_object( "relative" ) ) {
247 JsonObject relative = jo.get_object( "relative" );
248 relative.allow_omitted_members();
249 if( !relative.has_member( name ) ) {
250 return false;
251 }
252 debugmsg( "Member %s of type %s does not support relative", name, typeid( C ).name() );
253 }
254 return false;
255 }
256
257 // Type supports relative
258 template < typename C, typename std::enable_if < !reader_detail::handler<C>::is_container,
259 int >::type = 0, std::enable_if_t<supports_relative<C>::value> * = nullptr >
260 bool do_relative( const JsonObject &jo, const std::string &name, C &member ) const {
261 if( jo.has_object( "relative" ) ) {
262 JsonObject relative = jo.get_object( "relative" );
263 relative.allow_omitted_members();
264 const Derived &derived = static_cast<const Derived &>( *this );
265 // This needs to happen here, otherwise we get unvisited members
266 if( !relative.has_member( name ) ) {
267 return false;
268 }
269 C adder = derived.get_next( *relative.get_raw( name ) );
270 member += adder;
271 return true;
272 }
273 return false;
274 }

276 template<typename C>
277 bool read_normal( const JsonObject &jo, const std::string &name, C &member ) const {
278 if( jo.has_member( name ) ) {
279 const Derived &derived = static_cast<const Derived &>( *this );
280 member = derived.get_next( *jo.get_raw( name ) );
281 return true;
282 }
283 return false;
284 }

371/**
372 * Uses a map (unordered or standard) to convert strings from JSON to some other type
373 * (the mapped type of the map: `C::mapped_type`). It works for all mapped types, not just enums.
374 *
375 * One can use this if the member is `std::set<some_enum>` or `some_enum` and a
376 * map `std::map<std::string, some_enum>` with all the value enumeration values exists.
377 *
378 * The class can be conveniently instantiated for a given map `mapping` using
379 * the helper function @ref make_flag_reader (see below).
380 * The flag type (@ref flag_type) is used when the input contains invalid flags
381 * (a string that is not contained in the map). It should sound something like
382 * "my-enum-type".
383 */
384template<typename T>
385class typed_flag_reader : public generic_typed_reader<typed_flag_reader<T>>
386{
387 private:
388 using map_t = std::map<std::string, T>;
389
390 private:
391 const map_t &flag_map;
392 const std::string flag_type;
393
394 public:
395 typed_flag_reader( const map_t &flag_map, const std::string &flag_type )
396 : flag_map( flag_map )
397 , flag_type( flag_type ) {
398 }
399
400 T get_next( JsonIn &jin ) const {
401 const std::string flag = jin.get_string();
402 const auto iter = flag_map.find( flag );
403
404 if( iter == flag_map.cend() ) {
405 jin.seek( jin.tell() );
406 jin.error( string_format( "invalid %s: \"%s\"", flag_type, flag ) );
407 }
408
409 return iter->second;
410 }
411};
