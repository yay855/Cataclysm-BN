83{
84 // Get the user's language list (in order of preference)
85 CFArrayRef langs = CFLocaleCopyPreferredLanguages();
86 if( CFArrayGetCount( langs ) == 0 ) {
87 return "";
88 }
89

91 const char *lang_code_raw_fast = CFStringGetCStringPtr( lang, kCFStringEncodingUTF8 );
92 std::string lang_code;
93 if( lang_code_raw_fast ) { // fast way, probably it's never works
94 lang_code = lang_code_raw_fast;
95 } else { // fallback to slow way
96 CFIndex length = CFStringGetLength( lang ) + 1;
97 std::vector<char> lang_code_raw_slow( length, '\0' );

118std::string to_valid_language( const std::string &lang )
119{
120 if( lang.empty() ) {

209static std::vector<language_info> load_languages( const std::string &filepath )
210{
211 std::vector<language_info> ret;
212 try {
213 std::ifstream stream( filepath, std::ios_base::binary );
214 if( !stream.is_open() ) {
215 throw std::runtime_error( string_format( "File '%s' not found", filepath ) );
216 }
217 JsonIn json( stream );
218 JsonArray arr = json.get_array();
219 for( const JsonObject &obj : arr ) {
220 language_info info;
221 info.id = obj.get_string( "id" );
222 info.name = obj.get_string( "name" );
223 info.locale = obj.get_string( "locale" );
224 info.genders = obj.get_string_array( "genders" );
225 info.osx = obj.get_string( "osx", "" );
226 info.lcids = obj.get_int_array( "lcids" );
227 ret.push_back( info );
228 }
229 } catch( const std::exception &e ) {
230 debugmsg( "[lang] Failed to read language definitions: %s", e.what() );
231 return std::vector<language_info>();
232 }
233
234 // Sanity check genders
235 const std::vector<std::string> all_genders = {{"f", "m", "n"}};
236
237 for( language_info &info : ret ) {
238 for( const std::string &g : info.genders ) {
239 if( find( all_genders.begin(), all_genders.end(), g ) == all_genders.end() ) {
240 debugmsg( "Unexpected gender '%s' in grammatical gender list for language '%d'",
241 g, info.id );
242 }
243 }
244 if( info.genders.empty() ) {
245 info.genders.push_back( "n" );
246 }
247 }
248
249 return ret;
250}

253{

255 // (they usually default to "C") so don't bother
256 // trying to set the locale based on them.
257#if !defined(MACOSX)

260 } else {
261#endif

263 std::locale::global( std::locale( "" ) );

338void update_global_locale()
339{
340#if defined(_WIN32)
341 // Use the ANSI code page 1252 to work around some language output bugs.
342 if( setlocale( LC_ALL, ".1252" ) == nullptr ) {
343 dbg( DL::Warn ) << "Error while setlocale(LC_ALL, '.1252').";
344 }
345#else // _WIN32
346 std::string lang = ::get_option<std::string>( "USE_LANG" );
347
348 bool set_user = false;
349 if( lang.empty() ) {
350 // Restore user locale
351 set_user = true;
352 } else {
353 // Try specific locale
354 try {
355 std::locale::global( std::locale( get_lang_info( lang )->locale ) );
356 } catch( std::runtime_error &e ) {
357 // Try fairly neutral UTF-8 locale
358 try {
359 std::locale::global( std::locale( fallback_language.locale ) );
360 } catch( std::runtime_error &e ) {
361 // No choice left
362 set_user = true;
363 }
364 }
365 }
366 if( set_user ) {
367 setlocale( LC_ALL, sys_c_locale.c_str() );
368 try {
369 std::locale::global( std::locale( sys_cpp_locale ) );
370 } catch( std::runtime_error &e ) {
371 std::locale::global( std::locale::classic() );
372 }
373 }
374
375#endif // _WIN32
376
377 dbg( DL::Info ) << "C locale set to '" << setlocale( LC_ALL, nullptr ) << "'";
378 dbg( DL::Info ) << "C++ locale set to '" << std::locale().name() << "'";
379}

381std::vector<std::string> get_lang_path_substring( const std::string &lang_id )
382{
383 std::vector<std::string> ret;
384
385 const size_t p = lang_id.find( '_' );
386 if( p == std::string::npos ) {
387 // Dialect-agnostic id ('en', 'fr', 'de', etc.)
388 ret.push_back( lang_id );
389 } else {
390 // Id with dialect specified ('en_US', 'fr_FR', etc.)
391 // First try loading exact resource, then try dialect-agnostic resource.
392 std::string lang_only = lang_id.substr( 0, p );
393 ret.push_back( lang_id );
394 ret.push_back( lang_only );
395 }
396 return ret;
397}

412bool localized_comparator::operator()( const std::string &l, const std::string &r ) const
413{
414 // We need different implementations on each platform. MacOS seems to not
415 // support localized comparison of strings via the standard library at all,
416 // so resort to MacOS-specific solution. Windows cannot be expected to be
417 // using a UTF-8 locale (whereas our strings are always UTF-8) and so we
418 // must convert to wstring for comparison there. Linux seems to work as
419 // expected on regular strings; no workarounds needed.
420 // See https://github.com/CleverRaven/Cataclysm-DDA/pull/40041 for further
421 // discussion.
422#if defined(MACOSX)
423 CFStringRef lr = CFStringCreateWithCStringNoCopy( kCFAllocatorDefault, l.c_str(),
424 kCFStringEncodingUTF8, kCFAllocatorNull );
425 CFStringRef rr = CFStringCreateWithCStringNoCopy( kCFAllocatorDefault, r.c_str(),
426 kCFStringEncodingUTF8, kCFAllocatorNull );
427 bool result = CFStringCompare( lr, rr, kCFCompareLocalized ) < 0;
428 CFRelease( lr );
429 CFRelease( rr );
430 return result;
431#elif defined(_WIN32)
432 return ( *this )( utf8_to_wstr( l ), utf8_to_wstr( r ) );
433#else
434 return std::locale()( l, r );
435#endif
436}

438bool localized_comparator::operator()( const std::wstring &l, const std::wstring &r ) const

440#if defined(MACOSX)
441 return ( *this )( wstr_to_utf8( l ), wstr_to_utf8( r ) );
442#else
443 return std::locale()( l, r );

445}
