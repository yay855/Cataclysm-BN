233static mut_attack load_mutation_attack( const JsonObject &jo )
234{
235 mut_attack ret;
236 jo.read( "attack_text_u", ret.attack_text_u );
237 jo.read( "attack_text_npc", ret.attack_text_npc );
238 jo.read( "required_mutations", ret.required_mutations );
239 jo.read( "blocker_mutations", ret.blocker_mutations );
240 jo.read( "hardcoded_effect", ret.hardcoded_effect );
241
242 if( jo.has_string( "body_part" ) ) {
243 ret.bp = get_body_part_token( jo.get_string( "body_part" ) );
244 }
245
246 jo.read( "chance", ret.chance );
247
248 if( jo.has_array( "base_damage" ) ) {
249 ret.base_damage = load_damage_instance( jo.get_array( "base_damage" ) );
250 } else if( jo.has_object( "base_damage" ) ) {
251 JsonObject jo_dam = jo.get_object( "base_damage" );
252 ret.base_damage = load_damage_instance( jo_dam );
253 }
254
255 if( jo.has_array( "strength_damage" ) ) {
256 ret.strength_damage = load_damage_instance( jo.get_array( "strength_damage" ) );
257 } else if( jo.has_object( "strength_damage" ) ) {
258 JsonObject jo_dam = jo.get_object( "strength_damage" );
259 ret.strength_damage = load_damage_instance( jo_dam );
260 }
261
262 if( ret.attack_text_u.empty() || ret.attack_text_npc.empty() ) {
263 jo.throw_error( "Attack message unset" );
264 }
265
266 if( !ret.hardcoded_effect && ret.base_damage.empty() && ret.strength_damage.empty() ) {
267 jo.throw_error( "Damage unset" );
268 } else if( ret.hardcoded_effect && ( !ret.base_damage.empty() || !ret.strength_damage.empty() ) ) {
269 jo.throw_error( "Damage and hardcoded effect are both set (must be one, not both)" );
270 }
271
272 if( ret.chance <= 0 ) {
273 jo.throw_error( "Chance of procing must be set and positive" );
274 }
275
276 return ret;
277}

839void mutation_branch::add_entry( Trait_group &tg, const JsonObject &obj )
840{
841 std::unique_ptr<Trait_creation_data> ptr;
842 int probability = obj.get_int( "prob", 100 );
843 JsonArray jarr;
844
845 if( obj.has_member( "collection" ) ) {
846 ptr = std::make_unique<Trait_group_collection>( probability );
847 jarr = obj.get_array( "collection" );
848 } else if( obj.has_member( "distribution" ) ) {
849 ptr = std::make_unique<Trait_group_distribution>( probability );
850 jarr = obj.get_array( "distribution" );
851 }
852
853 if( ptr ) {
854 Trait_group &tg2 = dynamic_cast<Trait_group &>( *ptr );
855 for( const JsonObject job2 : jarr ) {
856 add_entry( tg2, job2 );
857 }
858 tg.add_entry( std::move( ptr ) );
859 return;
860 }
861
862 if( obj.has_member( "trait" ) ) {
863 trait_id id( obj.get_string( "trait" ) );
864 ptr = std::make_unique<Single_trait_creator>( id, probability );
865 } else if( obj.has_member( "group" ) ) {
866 ptr = std::make_unique<Trait_group_creator>( trait_group::Trait_group_tag(
867 obj.get_string( "group" ) ),
868 probability );
869 }
870
871 if( !ptr ) {
872 return;
873 }
874
875 tg.add_entry( std::move( ptr ) );
876}
