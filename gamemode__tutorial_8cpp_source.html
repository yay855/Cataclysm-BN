107bool tutorial_game::init()
108{
109 // TODO: clean up old tutorial
110 avatar &you = get_avatar();
111
112 // Start at noon
113 calendar::turn = calendar::turn_zero + 12_hours;
114 tutorials_seen.clear();
115 g->scent.reset();
116 get_weather().temperature = 65;
117 // We use a Z-factor of 10 so that we don't plop down tutorial rooms in the
118 // middle of the "real" game world
119 character_funcs::normalize( you );
120 you.str_cur = you.str_max;
121 you.per_cur = you.per_max;
122 you.int_cur = you.int_max;
123 you.dex_cur = you.dex_max;
124
125 you.set_all_parts_hp_to_max();
126
127 const oter_id rock( "rock" );
128 //~ default name for the tutorial
129 you.name = _( "John Smith" );
130 you.prof = profession::generic();
131 // overmap terrain coordinates
132 const tripoint_om_omt lp( 50, 50, 0 );
133 // Assume overmap zero
134 const tripoint_abs_omt lp_abs = project_combine( point_abs_om(), lp );
135 auto &starting_om = overmap_buffer.get( point_abs_om() );
136 for( int i = 0; i < OMAPX; i++ ) {
137 for( int j = 0; j < OMAPY; j++ ) {
138 tripoint_om_omt p( i, j, 0 );
139 starting_om.ter_set( p + tripoint_below, rock );
140 // Start with the overmap revealed
141 starting_om.seen( p ) = true;
142 }
143 }
144 starting_om.ter_set( lp, oter_id( "tutorial" ) );
145 starting_om.ter_set( lp + tripoint_below, oter_id( "tutorial" ) );
146 starting_om.clear_mon_groups();
147
148 you.toggle_trait( trait_QUICK );
149 item lighter( "lighter", calendar::start_of_cataclysm );
150 lighter.invlet = 'e';
151 you.inv.add_item( lighter, true, false );
152 you.set_skill_level( skill_gun, 5 );
153 you.set_skill_level( skill_melee, 5 );
154 g->load_map( project_to<coords::sm>( lp_abs ) );
155 you.setx( 2 );
156 you.sety( 4 );
157
158 // This shifts the view to center the players pos
159 g->update_map( you );
160 return true;
161}

163void tutorial_game::per_turn()
164{
165 // note that add_message does nothing if the message was already shown
166 add_message( tut_lesson::LESSON_INTRO );
167 add_message( tut_lesson::LESSON_MOVE );
168 add_message( tut_lesson::LESSON_LOOK );
169
170 if( g->light_level( g->u.posz() ) == 1 ) {
171 if( g->u.has_amount( itype_flashlight, 1 ) ) {
172 add_message( tut_lesson::LESSON_DARK );
173 } else {
174 add_message( tut_lesson::LESSON_DARK_NO_FLASH );
175 }
176 }
177
178 if( g->u.get_pain() > 0 ) {
179 add_message( tut_lesson::LESSON_PAIN );
180 }
181
182 if( g->u.recoil >= MAX_RECOIL ) {
183 add_message( tut_lesson::LESSON_RECOIL );
184 }
185
186 map &here = get_map();
187 if( !tutorials_seen[tut_lesson::LESSON_BUTCHER] ) {
188 for( const item &it : here.i_at( point( g->u.posx(), g->u.posy() ) ) ) {
189 if( it.is_corpse() ) {
190 add_message( tut_lesson::LESSON_BUTCHER );
191 break;
192 }
193 }
194 }
195
196 for( const tripoint &p : here.points_in_radius( g->u.pos(), 1 ) ) {
197 if( here.ter( p ) == t_door_o ) {
198 add_message( tut_lesson::LESSON_OPEN );
199 break;
200 } else if( here.ter( p ) == t_door_c ) {
201 add_message( tut_lesson::LESSON_CLOSE );
202 break;
203 } else if( here.ter( p ) == t_window ) {
204 add_message( tut_lesson::LESSON_SMASH );
205 break;
206 } else if( here.furn( p ) == f_rack && !here.i_at( p ).empty() ) {
207 add_message( tut_lesson::LESSON_EXAMINE );
208 break;
209 } else if( here.ter( p ) == t_stairs_down ) {
210 add_message( tut_lesson::LESSON_STAIRS );
211 break;
212 } else if( here.ter( p ) == t_water_sh ) {
213 add_message( tut_lesson::LESSON_PICKUP_WATER );
214 break;
215 }
216 }
217
218 if( !here.i_at( point( g->u.posx(), g->u.posy() ) ).empty() ) {
219 add_message( tut_lesson::LESSON_PICKUP );
220 }
221}

237void tutorial_game::post_action( action_id act )
238{
239 switch( act ) {
240 case ACTION_RELOAD_WEAPON:
241 if( g->u.weapon.is_gun() && !tutorials_seen[tut_lesson::LESSON_GUN_FIRE] ) {
242 g->place_critter_at( mon_zombie, tripoint( g->u.posx(), g->u.posy() - 6, g->u.posz() ) );
243 g->place_critter_at( mon_zombie, tripoint( g->u.posx() + 2, g->u.posy() - 5, g->u.posz() ) );
244 g->place_critter_at( mon_zombie, tripoint( g->u.posx() - 2, g->u.posy() - 5, g->u.posz() ) );
245 add_message( tut_lesson::LESSON_GUN_FIRE );
246 }
247 break;
248
249 case ACTION_OPEN:
250 add_message( tut_lesson::LESSON_CLOSE );
251 break;
252
253 case ACTION_CLOSE:
254 add_message( tut_lesson::LESSON_SMASH );
255 break;
256
257 case ACTION_USE: {
258 if( g->u.has_amount( itype_grenade_act, 1 ) ) {
259 add_message( tut_lesson::LESSON_ACT_GRENADE );
260 }
261 map &here = get_map();
262 for( const tripoint &dest : here.points_in_radius( g->u.pos(), 1 ) ) {
263 if( here.tr_at( dest ).id == trap_str_id( "tr_bubblewrap" ) ) {
264 add_message( tut_lesson::LESSON_ACT_BUBBLEWRAP );
265 }
266 }
267 }
268 break;
269
270 case ACTION_EAT:
271 if( g->u.last_item == itype_codeine ) {
272 add_message( tut_lesson::LESSON_TOOK_PAINKILLER );
273 } else if( g->u.last_item == itype_cig ) {
274 add_message( tut_lesson::LESSON_TOOK_CIG );
275 } else if( g->u.last_item == itype_water ) {
276 add_message( tut_lesson::LESSON_DRANK_WATER );
277 }
278 break;
279
280 case ACTION_WEAR: {
281 item it( g->u.last_item, calendar::start_of_cataclysm );
282 if( it.is_armor() ) {
283 if( it.get_coverage() >= 2 || it.get_thickness() >= 2 ) {
284 add_message( tut_lesson::LESSON_WORE_ARMOR );
285 }
286 if( it.get_storage() >= units::from_liter( 5 ) ) {
287 add_message( tut_lesson::LESSON_WORE_STORAGE );
288 }
289 if( it.get_env_resist() >= 2 ) {
290 add_message( tut_lesson::LESSON_WORE_MASK );
291 }
292 }
293 }
294 break;
295
296 case ACTION_WIELD:
297 if( g->u.weapon.is_gun() ) {
298 add_message( tut_lesson::LESSON_GUN_LOAD );
299 }
300 break;
301
302 case ACTION_EXAMINE:
303 add_message( tut_lesson::LESSON_INTERACT );
304 /* fallthrough */
305 case ACTION_PICKUP: {
306 item it( g->u.last_item, calendar::start_of_cataclysm );
307 if( it.is_armor() ) {
308 add_message( tut_lesson::LESSON_GOT_ARMOR );
309 } else if( it.is_gun() ) {
310 add_message( tut_lesson::LESSON_GOT_GUN );
311 } else if( it.is_ammo() ) {
312 add_message( tut_lesson::LESSON_GOT_AMMO );
313 } else if( it.is_tool() ) {
314 add_message( tut_lesson::LESSON_GOT_TOOL );
315 } else if( it.is_food() ) {
316 add_message( tut_lesson::LESSON_GOT_FOOD );
317 } else if( it.is_melee() ) {
318 add_message( tut_lesson::LESSON_GOT_WEAPON );
319 }
320
321 }
322 break;
323
324 default:
325 // TODO: add more actions here
326 break;
327
328 }
