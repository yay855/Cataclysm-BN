78void load_recipe_category( const JsonObject &jsobj )
79{
80 const std::string category = jsobj.get_string( "id" );
81 const bool is_hidden = jsobj.get_bool( "is_hidden", false );
82
83 if( category.find( "CC_" ) != 0 ) {
84 jsobj.throw_error( "Crafting category id has to be prefixed with 'CC_'" );
85 }
86
87 if( !is_hidden ) {
88 if( std::find( craft_cat_list.begin(), craft_cat_list.end(), category ) == craft_cat_list.end() ) {
89 craft_cat_list.push_back( category );
90 }
91 const std::string cat_name = get_cat_unprefixed( category );
92
93 for( const std::string subcat_id : jsobj.get_array( "recipe_subcategories" ) ) {
94 if( subcat_id.find( "CSC_" + cat_name + "_" ) != 0 && subcat_id != "CSC_ALL" ) {
95 jsobj.throw_error( "Crafting sub-category id has to be prefixed with CSC_<category_name>_" );
96 }
97 if( std::find( craft_subcat_list[category].begin(), craft_subcat_list[category].end(),
98 subcat_id ) == craft_subcat_list[category].end() ) {
99 craft_subcat_list[category].push_back( subcat_id );
100 }
101 }
102 }

134static int print_items( const recipe &r, const catacurses::window &w, point pos,
135 nc_color col, int batch )
136{
137 if( !r.has_byproducts() ) {
138 return 0;
139 }
140
141 const int oldy = pos.y;
142
143 mvwprintz( w, point( pos.x, pos.y++ ), col, _( "Byproducts:" ) );
144 for( const auto &bp : r.byproducts ) {
145 const itype *t = &*bp.first;
146 int amount = bp.second * batch;
147 std::string desc;
148 if( t->count_by_charges() ) {
149 amount *= t->charges_default();
150 desc = string_format( "> %s (%d)", t->nname( 1 ), amount );
151 } else {
152 desc = string_format( "> %d %s", amount,
153 t->nname( static_cast<unsigned int>( amount ) ) );
154 }
155 mvwprintz( w, point( pos.x, pos.y++ ), col, desc );
156 }
157
158 return pos.y - oldy;
159}

949std::string peek_related_recipe( const recipe *current, const recipe_subset &available )
950{
951 const avatar &u = get_avatar();
952
953 auto compare_second =
954 []( const std::pair<itype_id, std::string> &a,
955 const std::pair<itype_id, std::string> &b ) {
956 return localized_compare( a.second, b.second );
957 };
958

960 std::vector<std::pair<itype_id, std::string>> related_components;
961 const requirement_data &req = current->simple_requirements();
962 for( const std::vector<item_comp> &comp_list : req.get_components() ) {
963 for( const item_comp &a : comp_list ) {
964 related_components.push_back( { a.type, item::nname( a.type, 1 ) } );
965 }

968 // current recipe result
969 std::vector<std::pair<itype_id, std::string>> related_results;
970 item tmp = current->create_result();
971 itype_id tid;
972 if( tmp.contents.empty() ) { // use this item
973 tid = tmp.typeId();
974 } else { // use the contained item
975 tid = tmp.contents.front().typeId();
976 }

978 for( const auto *b : known_recipes ) {
979 if( available.contains( *b ) ) {
980 related_results.push_back( { b->result(), b->result_name() } );
981 }
982 }
983 std::stable_sort( related_results.begin(), related_results.end(), compare_second );
984
985 uilist rel_menu;
986 int np_last = -1;
987 if( !related_components.empty() ) {
988 rel_menu.addentry( ++np_last, false, -1, _( "COMPONENTS" ) );

991 if( !related_results.empty() ) {
992 rel_menu.addentry( ++np_last, false, -1, _( "RESULTS" ) );

1006int related_menu_fill( uilist &rmenu,
1007 const std::vector<std::pair<itype_id, std::string>> &related_recipes,
1008 const recipe_subset &available )
1009{
1010 const std::vector<uilist_entry> &entries = rmenu.entries;
1011 int np_last = entries.empty() ? -1 : entries.back().retval;
1012
1013 if( related_recipes.empty() ) {
1014 return np_last;
1015 }

1017 std::string recipe_name_prev;
1018 for( const std::pair<itype_id, std::string> &p : related_recipes ) {
1019
1020 // we have different recipes with the same names
1021 // list only one of them as we show and filter by name only
1022 std::string recipe_name = p.second;
1023 if( recipe_name == recipe_name_prev ) {
1024 continue;
1025 }
1026 recipe_name_prev = recipe_name;

1034 for( size_t recipe_n = 0; recipe_n < current_part.size(); recipe_n++ ) {
