86std::vector<std::vector<std::string>> query_popup::fold_query(
87 const std::string &category,
88 const std::vector<query_option> &options,
89 const int max_width, const int horz_padding )
90{
91 input_context ctxt( category );
92
93 std::vector<std::vector<std::string>> folded_query;
94 folded_query.emplace_back();
95
96 int query_cnt = 0;
97 int query_width = 0;
98 for( const auto &opt : options ) {
99 const auto &name = ctxt.get_action_name( opt.action );
100 const auto &desc = ctxt.get_desc( opt.action, name, opt.filter );
101 const int this_query_width = utf8_width( desc, true ) + horz_padding;
102 ++query_cnt;
103 query_width += this_query_width;
104 if( query_width > max_width + horz_padding ) {
105 if( query_cnt == 1 ) {
106 // Each line has at least one query, so keep this query in the current line
107 folded_query.back().emplace_back( desc );
108 folded_query.emplace_back();
109 query_cnt = 0;
110 query_width = 0;
111 } else {
112 // Wrap this query to the next line
113 folded_query.emplace_back();
114 folded_query.back().emplace_back( desc );
115 query_cnt = 1;
116 query_width = this_query_width;
117 }
118 } else {
119 folded_query.back().emplace_back( desc );
120 }
121 }
122
123 if( folded_query.back().empty() ) {
124 folded_query.pop_back();
125 }
126
127 return folded_query;
128}

145void query_popup::init() const
146{
147 constexpr int horz_padding = 2;
148 constexpr int vert_padding = 1;
149 const int max_line_width = FULL_SCREEN_WIDTH - border_width * 2;
150
151 // Fold message text
152 folded_msg = foldstring( text, max_line_width );
153
154 // Fold query buttons
155 const auto &folded_query = fold_query( category, options, max_line_width, horz_padding );
156
157 // Calculate size of message part
158 int msg_width = 0;
159 int msg_height = folded_msg.size();
160
161 for( const auto &line : folded_msg ) {
162 msg_width = std::max( msg_width, utf8_width( line, true ) );
163 }
164
165 // Calculate width with query buttons
166 for( const auto &line : folded_query ) {
167 if( !line.empty() ) {
168 int button_width = 0;
169 for( const auto &opt : line ) {
170 button_width += utf8_width( opt, true );
171 }
172 msg_width = std::max( msg_width, button_width +
173 horz_padding * static_cast<int>( line.size() - 1 ) );
174 }
175 }
176 msg_width = std::min( msg_width, max_line_width );
177
178 // Calculate height with query buttons & button positions
179 buttons.clear();
180 if( !folded_query.empty() ) {
181 msg_height += vert_padding;
182 for( const auto &line : folded_query ) {
183 if( !line.empty() ) {
184 int button_width = 0;
185 for( const auto &opt : line ) {
186 button_width += utf8_width( opt, true );
187 }
188 // Right align.
189 // TODO: multi-line buttons
190 int button_x = std::max( 0, msg_width - button_width -
191 horz_padding * static_cast<int>( line.size() - 1 ) );
192 for( const auto &opt : line ) {
193 buttons.emplace_back( opt, point( button_x, msg_height ) );
194 button_x += utf8_width( opt, true ) + horz_padding;
195 }
196 msg_height += 1 + vert_padding;
197 }
198 }
199 msg_height -= vert_padding;
200 }
201
202 // Calculate window size
203 const int win_width = std::min( TERMX,
204 fullscr ? FULL_SCREEN_WIDTH : msg_width + border_width * 2 );
205 const int win_height = std::min( TERMY,
206 fullscr ? FULL_SCREEN_HEIGHT : msg_height + border_width * 2 );
207 const int win_x = ( TERMX - win_width ) / 2;
208 const int win_y = ontop ? 0 : ( TERMY - win_height ) / 2;
209 win = catacurses::newwin( win_height, win_width, point( win_x, win_y ) );
210
211 std::shared_ptr<ui_adaptor> ui = adaptor.lock();
212 if( ui ) {
213 ui->position_from_window( win );
214 }

263query_popup::result query_popup::query_once()
264{
265 if( !anykey && !cancel && options.empty() ) {
266 return { false, "ERROR", {} };
267 }
268
269 if( test_mode ) {
270 return { false, "ERROR", {} };
271 }
272
273 std::shared_ptr<ui_adaptor> ui = create_or_get_adaptor();
274
275 ui_manager::redraw();
276
277 input_context ctxt( category );
278 if( cancel || !options.empty() ) {
279 ctxt.register_action( "HELP_KEYBINDINGS" );
280 }
281 if( !options.empty() ) {
282 ctxt.register_action( "LEFT" );
283 ctxt.register_action( "RIGHT" );
284 ctxt.register_action( "CONFIRM" );
285 for( const auto &opt : options ) {
286 ctxt.register_action( opt.action );
287 }
288 }
289 if( anykey ) {
290 ctxt.register_action( "ANY_INPUT" );
291 // Mouse movement, button, and wheel
292 ctxt.register_action( "COORDINATE" );
293 }
294 if( cancel ) {
295 ctxt.register_action( "QUIT" );
296 }
297
298 result res;
299 // Assign outside construction of `res` to ensure execution order
300 res.wait_input = !anykey;
301 do {
302 res.action = ctxt.handle_input();
303 res.evt = ctxt.get_raw_input();
304 } while(
305 // Always ignore mouse movement
306 ( res.evt.type == CATA_INPUT_MOUSE && res.evt.get_first_input() == MOUSE_MOVE ) ||
307 // Ignore window losing focus in SDL
308 ( res.evt.type == CATA_INPUT_KEYBOARD && res.evt.sequence.empty() )
309 );
310
311 if( cancel && res.action == "QUIT" ) {
312 res.wait_input = false;
313 } else if( res.action == "LEFT" ) {
314 if( cur > 0 ) {
315 --cur;
316 } else {
317 cur = options.size() - 1;
318 }
319 } else if( res.action == "RIGHT" ) {
320 if( cur + 1 < options.size() ) {
321 ++cur;
322 } else {
323 cur = 0;
324 }
325 } else if( res.action == "CONFIRM" ) {
326 if( cur < options.size() ) {
327 res.wait_input = false;
328 res.action = options[cur].action;
329 }
330 } else if( res.action == "HELP_KEYBINDINGS" ) {
331 // Keybindings may have changed, regenerate the UI
332 init();
333 } else {
334 for( size_t ind = 0; ind < options.size(); ++ind ) {
335 if( res.action == options[ind].action ) {
336 cur = ind;
337 if( options[ind].filter( res.evt ) ) {
338 res.wait_input = false;
339 break;
340 }
341 }
342 }
343 }
344
345 return res;
346}
