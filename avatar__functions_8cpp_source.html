496std::pair<int, int> gunmod_installation_odds( const avatar &you, const item &gun,
497 const item &mod )
498{
499 // Mods with INSTALL_DIFFICULT have a chance to fail, potentially damaging the gun
500 if( !mod.has_flag( "INSTALL_DIFFICULT" ) || you.has_trait( trait_DEBUG_HS ) ) {
501 return std::make_pair( 100, 0 );
502 }
503
504 int roll = 100; // chance of success (%)
505 int risk = 0; // chance of failure (%)
506 int chances = 1; // start with 1 in 6 (~17% chance)
507
508 for( const auto &e : mod.type->min_skills ) {
509 // gain an additional chance for every level above the minimum requirement
510 skill_id sk = e.first == skill_weapon ? gun.gun_skill() : e.first;
511 chances += std::max( you.get_skill_level( sk ) - e.second, 0 );
512 }
513 // cap success from skill alone to 1 in 5 (~83% chance)
514 roll = std::min( static_cast<double>( chances ), 5.0 ) / 6.0 * 100;
515 // focus is either a penalty or bonus of at most +/-10%
516 roll += ( std::max( std::min( you.focus_pool, 140 ), 60 ) - 100 ) / 4;
517 // dexterity and intelligence give +/-2% for each point above or below 12
518 roll += ( you.get_dex() - 12 ) * 2;
519 roll += ( you.get_int() - 12 ) * 2;
520 // each level of damage to the base gun reduces success by 10%
521 roll -= std::max( gun.damage_level( 4 ), 0 ) * 10;
522 roll = std::min( std::max( roll, 0 ), 100 );
523
524 // risk of causing damage on failure increases with less durable guns
525 risk = ( 100 - roll ) * ( ( 10.0 - std::min( gun.type->gun->durability, 9 ) ) / 10.0 );
526
527 return std::make_pair( roll, risk );
528}

631bool unload_item( avatar &you, item_location loc )
632{
633 item &it = *loc.get_item();
634 // Unload a container consuming moves per item successfully removed
635 if( it.is_container() || it.is_bandolier() ) {
636 if( it.contents.empty() ) {
637 add_msg( m_info, _( "The %s is already empty!" ), it.tname() );
638 return false;
639 }
640 if( !it.can_unload_liquid() ) {
641 add_msg( m_info, _( "The liquid can't be unloaded in its current state!" ) );
642 return false;
643 }
644
645 bool changed = false;
646 for( item *contained : it.contents.all_items_top() ) {
647 int old_charges = contained->charges;
648 const bool consumed = add_or_drop_with_msg( you, *contained, true );
649 changed = changed || consumed || contained->charges != old_charges;
650 if( consumed ) {
651 you.mod_moves( -you.item_handling_cost( *contained ) );
652 it.remove_item( *contained );
653 }
654 }
655
656 if( changed ) {
657 it.on_contents_changed();
658 }
659 return true;
660 }
661
662 // If item can be unloaded more than once (currently only guns) prompt user to choose
663 std::vector<std::string> msgs( 1, it.tname() );
664 std::vector<item *> opts( 1, &it );
665
666 for( auto e : it.gunmods() ) {
667 if( e->is_gun() && !e->has_flag( "NO_UNLOAD" ) &&
668 ( e->magazine_current() || e->ammo_remaining() > 0 || e->casings_count() > 0 ) ) {
669 msgs.emplace_back( e->tname() );
670 opts.emplace_back( e );
671 }
672 }
673
674 item *target = nullptr;
675 if( opts.size() > 1 ) {
676 const int ret = uilist( _( "Unload what?" ), msgs );
677 if( ret >= 0 ) {
678 target = opts[ret];
679 }
680 } else {
681 target = &it;
682 }
683
684 if( target == nullptr ) {
685 return false;
686 }
687
688 // Next check for any reasons why the item cannot be unloaded
689 if( target->ammo_types().empty() || target->ammo_capacity() <= 0 ) {
690 add_msg( m_info, _( "You can't unload a %s!" ), target->tname() );
691 return false;
692 }
693
694 if( target->has_flag( "NO_UNLOAD" ) ) {
695 if( target->has_flag( "RECHARGE" ) || target->has_flag( "USE_UPS" ) ) {
696 add_msg( m_info, _( "You can't unload a rechargeable %s!" ), target->tname() );
697 } else {
698 add_msg( m_info, _( "You can't unload a %s!" ), target->tname() );
699 }
700 return false;
701 }
702
703 if( !target->magazine_current() && target->ammo_remaining() <= 0 && target->casings_count() <= 0 ) {
704 if( target->is_tool() ) {
705 add_msg( m_info, _( "Your %s isn't charged." ), target->tname() );
706 } else {
707 add_msg( m_info, _( "Your %s isn't loaded." ), target->tname() );
708 }
709 return false;
710 }
711
712 target->casings_handle( [&]( item & e ) {
713 return you.i_add_or_drop( e );
714 } );
715
716 if( target->is_magazine() ) {
717 // Calculate the time to remove the contained ammo (consuming half as much time as required to load the magazine)
718 int mv = 0;
719 int qty = 0;
720 std::vector<item *> remove_contained;
721 for( item *contained : it.contents.all_items_top() ) {
722 mv += you.item_reload_cost( it, *contained, contained->charges ) / 2;
723 if( add_or_drop_with_msg( you, *contained, true ) ) {
724 qty += contained->charges;
725 remove_contained.push_back( contained );
726 }
727 }
728 // remove the ammo leads in the belt
729 for( item *remove : remove_contained ) {
730 it.remove_item( *remove );
731 }
732
733 // remove the belt linkage
734 if( it.is_ammo_belt() ) {
735 if( it.type->magazine->linkage ) {
736 item link( *it.type->magazine->linkage, calendar::turn, qty );
737 add_or_drop_with_msg( you, link, true );
738 }
739 add_msg( _( "You disassemble your %s." ), it.tname() );
740 } else {
741 add_msg( _( "You unload your %s." ), it.tname() );
742 }
743
744 you.mod_moves( -std::min( 200, mv ) );
745 if( loc->has_flag( "MAG_DESTROY" ) && loc->ammo_remaining() == 0 ) {
746 loc.remove_item();
747 }
748 return true;
749
750 } else if( target->magazine_current() ) {
751 if( !add_or_drop_with_msg( you, *target->magazine_current(), true ) ) {
752 return false;
753 }
754 // Eject magazine consuming half as much time as required to insert it
755 you.moves -= you.item_reload_cost( *target, *target->magazine_current(), -1 ) / 2;
756
757 target->remove_items_with( [&target]( const item & e ) {
758 return target->magazine_current() == &e;
759 } );
760
761 } else if( target->ammo_remaining() ) {
762 int qty = target->ammo_remaining();
763
764 if( target->ammo_current() == itype_plut_cell ) {
765 qty = target->ammo_remaining() / PLUTONIUM_CHARGES;
766 if( qty > 0 ) {
767 add_msg( _( "You recover %i unused plutonium." ), qty );
768 } else {
769 add_msg( m_info, _( "You can't remove partially depleted plutonium!" ) );
770 return false;
771 }
772 }
773
774 // Construct a new ammo item and try to drop it
775 item ammo( target->ammo_current(), calendar::turn, qty );
776 if( target->is_filthy() ) {
777 ammo.set_flag( "FILTHY" );
778 }
779
780 if( ammo.made_of( LIQUID ) ) {
781 if( !add_or_drop_with_msg( you, ammo, false ) ) {
782 qty -= ammo.charges; // only handled part (or none) of the liquid
783 }
784 if( qty <= 0 ) {
785 return false; // no liquid was moved
786 }
787
788 } else if( !add_or_drop_with_msg( you, ammo, qty > 1 ) ) {
789 return false;
790 }
791
792 // If successful remove appropriate qty of ammo consuming half as much time as required to load it
793 you.moves -= you.item_reload_cost( *target, ammo, qty ) / 2;
794
795 if( target->ammo_current() == itype_plut_cell ) {
796 qty *= PLUTONIUM_CHARGES;
797 }
798
799 target->ammo_set( target->ammo_current(), target->ammo_remaining() - qty );
800 }
801
802 // Turn off any active tools
803 if( target->is_tool() && target->active && target->ammo_remaining() == 0 ) {
804 target->type->invoke( you, *target, you.pos() );
805 }
806
807 add_msg( _( "You unload your %s." ), target->tname() );
808 return true;
809
810}
