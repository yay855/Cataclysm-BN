393 }

395 loc.remove_item();
396 }

398 u.moves -= moves_taken;
399 }
400
401 return picked_up || !did_prompt;

483std::vector<stacked_items> stack_for_pickup_ui( const
484 std::vector<item_stack::iterator> &unstacked )
485{
486 const std::pair<time_point, int> no_parent = std::make_pair(
487 calendar::before_time_starts, 0 );
488 std::map<std::pair<time_point, int>, parent_child_check_t> parent_child_check;
489 // First, we need to check which parent-child groups exist
490 for( item_stack::iterator it : unstacked ) {
491 const auto &token = *it->drop_token;
492 if( token.drop_number > 0 ) {
493 std::pair<time_point, int> turn_and_drop = std::make_pair( token.turn, token.drop_number );
494 parent_child_check[turn_and_drop].parent_exists = true;
495 }
496 if( token.parent_number != token.drop_number && token.parent_number > 0 ) {
497 std::pair<time_point, int> turn_and_parent = std::make_pair( token.turn, token.parent_number );
498 parent_child_check[turn_and_parent].child_exists = true;
499 }
500 }
501
502 // Second pass: we group children and parents together, but only if both sides are known to exist
503 std::map<std::pair<time_point, int>, unstacked_items> children_by_parent;
504 for( item_stack::iterator it : unstacked ) {
505 const auto &token = *it->drop_token;
506 std::pair<time_point, int> turn_and_drop = std::make_pair( token.turn, token.drop_number );
507 if( token.drop_number > 0 && parent_child_check[turn_and_drop].child_exists ) {
508 children_by_parent[turn_and_drop].parent = it;
509 continue;
510 }
511
512 std::pair<time_point, int> turn_and_parent = std::make_pair( token.turn, token.parent_number );
513 if( token.parent_number > 0 && token.parent_number != token.drop_number &&
514 parent_child_check[turn_and_parent].parent_exists ) {
515 children_by_parent[turn_and_parent].unstacked_children.push_back( it );
516 } else {
517 children_by_parent[no_parent].unstacked_children.push_back( it );
518 }
519 }
520
521 std::vector<stacked_items> restacked_with_parents;
522 for( const auto &pr : children_by_parent ) {
523 std::vector<std::list<item_stack::iterator>> restacked_children;
524 for( item_stack::iterator it : pr.second.unstacked_children ) {
525 bool found_stack = false;
526 for( std::list<item_stack::iterator> &stack : restacked_children ) {
527 const item &stack_top = *stack.front();
528 if( stack_top.display_stacked_with( *it ) ) {
529 stack.push_back( it );
530 found_stack = true;
531 break;
532 }
533 }
534 if( !found_stack ) {
535 restacked_children.emplace_back( std::list<item_stack::iterator>( { it } ) );
536 }
537 }
538
539 // Each sub-stack has to be sorted separately
540 std::sort( restacked_children.begin(), restacked_children.end(),
541 []( const std::list<item_stack::iterator> &lhs, const std::list<item_stack::iterator> &rhs ) {
542 return *lhs.front() < *rhs.front();
543 } );
544 restacked_with_parents.emplace_back( stacked_items{ pr.second.parent, restacked_children } );
545 }
546
547 // Sorting by parent is a bit arbitrary (parent-less go last) - sort by count?
548 std::sort( restacked_with_parents.begin(), restacked_with_parents.end(),
549 []( const stacked_items & lhs, stacked_items & rhs ) {
550 return lhs.parent.has_value() && ( !rhs.parent.has_value() || *lhs.parent < *rhs.parent );
551 } );
552
553
554 return restacked_with_parents;
555}
