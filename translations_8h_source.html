89 if( cached_lang_version != get_current_language_version() || cached_arg != arg ) {

92 same_as_arg = cached_translation == arg;
93 cached_arg = arg;
94 }

296// Localized comparison operator, intended for sorting strings when they should
297// be sorted according to the user's locale.
298//
299// For convenience, it also sorts pairs recursively, because a common
300// requirement is to sort some list of objects by their names, and this can be
301// achieved by sorting a list of pairs where the first element of the pair is
302// the translated name.
303struct localized_comparator {
304 template<typename T, typename U>
305 bool operator()( const std::pair<T, U> &l, const std::pair<T, U> &r ) const {
306 if( ( *this )( l.first, r.first ) ) {
307 return true;
308 }
309 if( ( *this )( r.first, l.first ) ) {
310 return false;
311 }
312 return ( *this )( l.second, r.second );
313 }
314
315 template<typename Head, typename... Tail>
316 bool operator()( const std::tuple<Head, Tail...> &l,
317 const std::tuple<Head, Tail...> &r ) const {
318 if( ( *this )( std::get<0>( l ), std::get<0>( r ) ) ) {
319 return true;
320 }
321 if( ( *this )( std::get<0>( r ), std::get<0>( l ) ) ) {
322 return false;
323 }
324 constexpr std::make_index_sequence<sizeof...( Tail )> Ints{};
325 return ( *this )( tie_tail( l, Ints ), tie_tail( r, Ints ) );
326 }
327
328 template<typename T>
329 bool operator()( const T &l, const T &r ) const {
330 return l < r;
331 }
332
333 bool operator()( const std::string &, const std::string & ) const;
334 bool operator()( const std::wstring &, const std::wstring & ) const;
335
336 template<typename Head, typename... Tail, size_t... Ints>
337 auto tie_tail( const std::tuple<Head, Tail...> &t, std::index_sequence<Ints...> ) const {
338 return std::tie( std::get < Ints + 1 > ( t )... );
339 }
340};
