229void enchantment::load( const JsonObject &jo, const std::string & )
230{
231 optional( jo, was_loaded, "id", id, enchantment_id( "" ) );
232
233 jo.read( "hit_you_effect", hit_you_effect );
234 jo.read( "hit_me_effect", hit_me_effect );
235 jo.read( "emitter", emitter );
236
237 if( jo.has_object( "intermittent_activation" ) ) {
238 JsonObject jobj = jo.get_object( "intermittent_activation" );
239 for( const JsonObject effect_obj : jobj.get_array( "effects" ) ) {
240 time_duration freq = read_from_json_string<time_duration>( *effect_obj.get_raw( "frequency" ),
241 time_duration::units );
242 if( effect_obj.has_array( "spell_effects" ) ) {
243 for( const JsonObject fake_spell_obj : effect_obj.get_array( "spell_effects" ) ) {
244 fake_spell fake;
245 fake.load( fake_spell_obj );
246 add_activation( freq, fake );
247 }
248 } else if( effect_obj.has_object( "spell_effects" ) ) {
249 fake_spell fake;
250 JsonObject fake_spell_obj = effect_obj.get_object( "spell_effects" );
251 fake.load( fake_spell_obj );
252 add_activation( freq, fake );
253 }
254 }
255 }
256
257 active_conditions.first = io::string_to_enum<has>( jo.get_string( "has", "HELD" ) );
258 active_conditions.second = io::string_to_enum<condition>( jo.get_string( "condition",
259 "ALWAYS" ) );
260
261 for( JsonObject jsobj : jo.get_array( "ench_effects" ) ) {
262 ench_effects.emplace( efftype_id( jsobj.get_string( "effect" ) ), jsobj.get_int( "intensity" ) );
263 }
264
265 optional( jo, was_loaded, "mutations", mutations );
266
267 if( jo.has_array( "values" ) ) {
268 for( const JsonObject value_obj : jo.get_array( "values" ) ) {
269 std::string value_raw = value_obj.get_string( "value" );
270 migrate_ench_vals_enums( value_raw );
271 const enchant_vals::mod value = io::string_to_enum<enchant_vals::mod>( value_raw );
272
273 const int add = value_obj.get_int( "add", 0 );
274 const double mult = value_obj.get_float( "multiply", 0.0 );
275 if( add != 0 ) {
276 values_add.emplace( value, add );
277 }
278 if( mult != 0.0 ) {
279 // Limit precision to minimize inconsistencies between platforms / compilers
280 const double mul = static_cast<int>( std::round( mult * 100'000 ) ) / 100'000.0;
281 values_multiply.emplace( value, mul );
282 }
283 }

383void enchantment::force_add( const enchantment &rhs )
384{
385 for( const std::pair<const enchant_vals::mod, int> &pair_values : rhs.values_add ) {
386 values_add[pair_values.first] += pair_values.second;
387 }
388 for( const std::pair<const enchant_vals::mod, double> &pair_values : rhs.values_multiply ) {
389 // values do not multiply against each other, they add.
390 // so +10% and -10% will add to 0%
391 values_multiply[pair_values.first] += pair_values.second;
392 }
393
394 hit_me_effect.insert( hit_me_effect.end(), rhs.hit_me_effect.begin(), rhs.hit_me_effect.end() );
395
396 hit_you_effect.insert( hit_you_effect.end(), rhs.hit_you_effect.begin(), rhs.hit_you_effect.end() );
397
398 ench_effects.insert( rhs.ench_effects.begin(), rhs.ench_effects.end() );
399
400 if( rhs.emitter ) {
401 emitter = rhs.emitter;
402 }
403
404 for( const trait_id &branch : rhs.mutations ) {
405 mutations.emplace( branch );
406 }
407
408 for( const std::pair<const time_duration, std::vector<fake_spell>> &act_pair :
409 rhs.intermittent_activation ) {
410 for( const fake_spell &fake : act_pair.second ) {
411 intermittent_activation[act_pair.first].emplace_back( fake );
412 }
413 }

434double enchantment::calc_bonus( enchant_vals::mod value, double base, bool round ) const
435{
436 bool use_add = true;
437 switch( value ) {
438 case enchant_vals::mod::METABOLISM:
439 case enchant_vals::mod::MANA_REGEN:
440 use_add = false;
441 break;
442 default:
443 break;
444 }
445 double add = use_add ? get_value_add( value ) : 0.0;
446 double mul = get_value_multiply( value );
447 double ret = add + base * mul;
448 if( round ) {
449 ret = trunc( ret );
450 }
451 return ret;
452}

503void enchantment::cast_enchantment_spell( Character &caster, const Creature *target,
504 const fake_spell &sp ) const
505{
506 // check the chances
507 if( !one_in( sp.trigger_once_in ) ) {
508 return;
509 }
510
511 if( sp.self ) {
512 caster.add_msg_player_or_npc( m_good,
513 sp.trigger_message,
514 sp.npc_trigger_message,
515 caster.name );
516 sp.get_spell( sp.level ).cast_all_effects( caster, caster.pos() );
517 } else if( target != nullptr ) {
518 const Creature &trg_crtr = *target;
519 const spell &spell_lvl = sp.get_spell( sp.level );
520 if( !spell_lvl.is_valid_target( caster, trg_crtr.pos() ) ||
521 !spell_lvl.is_target_in_range( caster, trg_crtr.pos() ) ) {
522 return;
523 }
524
525 caster.add_msg_player_or_npc( m_good,
526 sp.trigger_message,
527 sp.npc_trigger_message,
528 caster.name, trg_crtr.disp_name() );
529
530 spell_lvl.cast_all_effects( caster, trg_crtr.pos() );
531 }
