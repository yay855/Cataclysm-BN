118// Clamp p to the rectangle r.
119// This independently clamps each coordinate of p to the bounds of the
120// rectangle.
121// Useful for example to round an arbitrary point to the nearest point on the
122// screen, or the nearest point in a particular submap.
123template<typename Point>
124Point clamp( const Point &p, const half_open_rectangle<Point> &r )
125{
126 using Traits = point_traits<Point>;
127 return Point( ::clamp( Traits::x( p ), Traits::x( r.p_min ), Traits::x( r.p_max ) - 1 ),
128 ::clamp( Traits::y( p ), Traits::y( r.p_min ), Traits::y( r.p_max ) - 1 ) );
129}
