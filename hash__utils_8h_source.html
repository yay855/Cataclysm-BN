18template <class T, typename Hash = std::hash<T>>
19inline void hash_combine( std::size_t &seed, const T &v, const Hash &hash = std::hash<T>() )
20{
21 seed ^= hash( v ) + 0x9e3779b9 + ( seed << 6 ) + ( seed >> 2 );
