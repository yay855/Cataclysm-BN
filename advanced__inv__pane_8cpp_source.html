63bool advanced_inventory_pane::is_filtered( const item &it ) const
64{
65 if( it.has_flag( flag_HIDDEN_ITEM ) ) {
66 return true;
67 }
68 if( filter.empty() ) {
69 return false;
70 }
71
72 const std::string str = it.tname();
73 if( filtercache.find( str ) == filtercache.end() ) {
74 const auto filter_fn = item_filter_from_string( filter );
75 filtercache[str] = filter_fn;
76
77 return !filter_fn( it );
78 }
79
80 return !filtercache[str]( it );
81}

83void advanced_inventory_pane::add_items_from_area( advanced_inv_area &square,
84 bool vehicle_override )
85{
86 assert( square.id != AIM_ALL );
87 square.volume = 0_ml;
88 square.weight = 0_gram;
89 if( !square.canputitems() ) {
90 return;
91 }
92 map &m = get_map();
93 avatar &u = get_avatar();
94 // Existing items are *not* cleared on purpose, this might be called
95 // several times in case all surrounding squares are to be shown.
96 if( square.id == AIM_INVENTORY ) {
97 const invslice &stacks = u.inv.slice();
98 for( size_t x = 0; x < stacks.size(); ++x ) {
99 std::list<item *> item_pointers;
100 for( item &i : *stacks[x] ) {
101 item_pointers.push_back( &i );
102 }
103 advanced_inv_listitem it( item_pointers, x, square.id, false );
104 if( is_filtered( *it.items.front() ) ) {
105 continue;
106 }
107 square.volume += it.volume;
108 square.weight += it.weight;
109 items.push_back( it );
110 }
111 } else if( square.id == AIM_WORN ) {
112 auto iter = u.worn.begin();
113 for( size_t i = 0; i < u.worn.size(); ++i, ++iter ) {
114 advanced_inv_listitem it( &*iter, i, 1, square.id, false );
115 if( is_filtered( *it.items.front() ) ) {
116 continue;
117 }
118 square.volume += it.volume;
119 square.weight += it.weight;
120 items.push_back( it );
121 }
122 } else if( square.id == AIM_CONTAINER ) {
123 item *cont = square.get_container( in_vehicle() );
124 if( cont != nullptr ) {
125 if( !cont->is_container_empty() ) {
126 // filtering does not make sense for liquid in container
127 item *it = &square.get_container( in_vehicle() )->contents.front();
128 advanced_inv_listitem ait( it, 0, 1, square.id, in_vehicle() );
129 square.volume += ait.volume;
130 square.weight += ait.weight;
131 items.push_back( ait );
132 }
133 square.desc[0] = cont->tname( 1, false );
134 }
135 } else {
136 bool is_in_vehicle = square.can_store_in_vehicle() && ( in_vehicle() || vehicle_override );
137 const advanced_inv_area::itemstack &stacks = is_in_vehicle ?
138 square.i_stacked( square.veh->get_items( square.vstor ) ) :
139 square.i_stacked( m.i_at( square.pos ) );
140
141 for( size_t x = 0; x < stacks.size(); ++x ) {
142 advanced_inv_listitem it( stacks[x], x, square.id, is_in_vehicle );
143 if( is_filtered( *it.items.front() ) ) {
144 continue;
145 }
146 square.volume += it.volume;
147 square.weight += it.weight;
148 items.push_back( it );
149 }

153void advanced_inventory_pane::paginate( size_t itemsPerPage )
154{
155 // not needed as there are no category entries here.
156 if( sortby != SORTBY_CATEGORY ) {
157 return;
158 }
159 // first, we insert all the items, then we sort the result
160 for( size_t i = 0; i < items.size(); ++i ) {
161 if( i % itemsPerPage == 0 ) {
162 // first entry on the page, should be a category header
163 if( items[i].is_item_entry() ) {
164 items.insert( items.begin() + i, advanced_inv_listitem( items[i].cat ) );
165 }
166 }
167 if( ( i + 1 ) % itemsPerPage == 0 && i + 1 < items.size() ) {
168 // last entry of the page, but not the last entry at all!
169 // Must *not* be a category header!
170 if( items[i].is_category_header() ) {
171 items.insert( items.begin() + i, advanced_inv_listitem() );
172 }
173 }
174 }

230void advanced_inventory_pane::scroll_category( int offset )
231{
232 // 0 would make no sense
233 assert( offset != 0 );
234 // only those two offsets are allowed
235 assert( offset == -1 || offset == +1 );
236 if( items.empty() ) {
237 return;
238 }
239 // index must already be valid!
240 assert( get_cur_item_ptr() != nullptr );
241 auto cur_cat = items[index].cat;
242 if( offset > 0 ) {
243 while( items[index].cat == cur_cat ) {
244 index++;
245 // wrap to begin, stop there.
246 if( static_cast<size_t>( index ) >= items.size() ) {
247 index = 0;
248 break;
249 }
250 }
251 } else {
252 while( items[index].cat == cur_cat ) {
253 index--;
254 // wrap to end, stop there.
255 if( index < 0 ) {
256 index = static_cast<int>( items.size() ) - 1;
257 break;
258 }
259 }
260 }
261 // Make sure we land on an item entry.
262 skip_category_headers( offset > 0 ? +1 : -1 );
263}
