45static weather_gen_common get_common_data( const point_abs_ms &location, const time_point &t,
46 const calendar_config &calendar_config, unsigned seed )
47{
48 weather_gen_common result;
49 // Integer x position / widening factor of the Perlin function.
50 result.x = location.x() / 2000.0;
51 // Integer y position / widening factor of the Perlin function.
52 result.y = location.y() / 2000.0;
53 // Integer turn / widening factor of the Perlin function.
54 result.z = to_days<double>( t - calendar::turn_zero );
55 // Limit the random seed during noise calculation, a large value flattens the noise generator to zero
56 // Windows has a rand limit of 32768, other operating systems can have higher limits
57 result.modSEED = seed % SIMPLEX_NOISE_RANDOM_SEED_LIMIT;
58 // Eternal season = midpoint of initial season all the time
59 const double year_fraction = calendar_config.eternal_season() ?
60 ( 0.25 * static_cast<double>( calendar_config.initial_season() ) + 0.125 ) :
61 ( time_past_new_year( t ) / calendar::year_length() );
62
63 result.year_fraction = year_fraction;
64 // We add one-eighth to line it up so that +1 is at
65 // midwinter and -1 at midsummer. (Cataclysm years
66 // start when spring starts. Gregorian years start when
67 // winter starts.)
68 result.cosine_of_gregorian_year_fraction = std::cos( tau * ( year_fraction + .125 ) ); // [-1, 1]
69 result.season = season_of_year( t );
70
71 return result;
72}

74static units::temperature season_temp( const weather_generator &wg, double year_fraction )
75{
76 // Interpolate seasons temperature
77 // Scale year_fraction [0, 1) to [0.0, 4.0). So [0.0, 1.0) - spring, [1.0, 2.0) - summer, [2.0, 3.0) - autumn, [3.0, 4.0) - winter.
78 const double quadrum = year_fraction * 4;
79 const double season_midpoint_quadrum = quadrum - 0.5;
80 const size_t current_season = ( static_cast<size_t>( std::floor( season_midpoint_quadrum ) +
81 NUM_SEASONS ) ) % NUM_SEASONS;
82 const size_t next_season = ( current_season + 1 ) % NUM_SEASONS;
83 // 0 - current season just started, 1 - season just ended (shouldn't actually happen)
84 const double t = season_midpoint_quadrum - std::floor( season_midpoint_quadrum );
85 return units::multiply_any_unit( wg.season_stats[current_season].average_temperature, 1.0 - t )
86 + units::multiply_any_unit( wg.season_stats[next_season].average_temperature, t );
87}

89static units::temperature weather_temperature_from_common_data( const weather_generator &wg,
90 const weather_gen_common &common, const time_point &t )
91{
92 const double x( common.x );
93 const double y( common.y );
94 const double z( common.z );
95
96 const unsigned modSEED = common.modSEED;
97 const double dayFraction = time_past_midnight( t ) / 1_days;
98 // -1 at coldest_hour, +1 twelve hours later
99 const double dayv = std::cos( tau * ( dayFraction + .5 - coldest_hour / 24 ) );
100
101 units::temperature season_factor = season_temp( wg, common.year_fraction );
102 const double temperature_celsius =
103 units::to_celsius<double>( season_factor ) +
104 dayv * units::to_celsius<double>( wg.temperature_daily_amplitude ) +
105 raw_noise_4d( x, y, z, modSEED ) * units::to_celsius<double>( wg.temperature_noise_amplitude );
106
107 return units::from_celsius( temperature_celsius );
108}

123w_point weather_generator::get_weather( const tripoint_abs_ms &location, const time_point &t,
124 const calendar_config &calendar_config, unsigned seed ) const
125{
126 const weather_gen_common common = get_common_data( location.xy(), t, calendar_config, seed );
127
128 const double x( common.x );
129 const double y( common.y );
130 const double z( common.z );
131
132 const unsigned modSEED = common.modSEED;
133 // +1 in midwinter, -1 in midsummer
134 const double cgyf = common.cosine_of_gregorian_year_fraction;
135 const season_type season = common.season;
136
137 // Noise factors
138 const units::temperature T( weather_temperature_from_common_data( *this, common, t ) );
139 double A( raw_noise_4d( x, y, z, modSEED ) * 8.0 );
140 double W( raw_noise_4d( x / 2.5, y / 2.5, z / 200, modSEED ) * 10.0 );
141
142 // Humidity variation
143 double mod_h = season_stats[static_cast<size_t>( season )].humidity_mod;
144 // Relative humidity, a percentage.
145 double H = std::min( 100., std::max( 0.,
146 base_humidity + mod_h + 100 * (
147 .15 * -cgyf +
148 raw_noise_4d( x, y, z, modSEED + 101 ) *
149 .2 * ( cgyf + 2 ) ) ) );
150
151 // Pressure
152 double P =
153 base_pressure +
154 raw_noise_4d( x, y, z, modSEED + 211 ) *
155 10 * ( cgyf + 2 );
156
157 // Wind power
158 W = std::max( 0, static_cast<int>( base_wind * rng( 1, 2 ) / std::pow( ( P + W ) / 1014.78, rng( 9,
159 base_wind_distrib_peaks ) ) +
160 -cgyf / base_wind_season_variation * rng( 1, 2 ) ) );
161 // Initial static variable
162 if( current_winddir == 1000 ) {
163 current_winddir = get_wind_direction( season );
164 current_winddir = convert_winddir( current_winddir );
165 } else {
166 // When wind strength is low, wind direction is more variable
167 bool changedir = one_in( W * 2160 );
168 if( changedir ) {
169 current_winddir = get_wind_direction( season );
170 current_winddir = convert_winddir( current_winddir );
171 }
172 }
173 std::string wind_desc = get_wind_desc( W );
174 // Acid rains
175 const double acid_content = base_acid * A;
176 bool acid = acid_content >= 1.0;
177 return w_point{ T, H, P, W, wind_desc, current_winddir, acid };
178}

287units::temperature weather_generator::get_water_temperature(
288 const tripoint_abs_ms &location,
289 const time_point &time,
290 const calendar_config &calendar_config,
291 unsigned seed ) const
292{
293 // Instead of using a realistic model, we'll just smooth out air temperature
294 // Smooth out both in time and intensity
295 // And add caps - it must stay liquid water
296 constexpr std::array<std::pair<time_duration, double>, 7> measurement_weights = {{
297 { 7_days, 0.1 },
298 { 7_days + 12_hours, 0.1 },
299 { 3_days, 0.2 },
300 { 3_days + 12_hours, 0.2 },
301 { 1_days, 0.2 },
302 { 0_days + 12_hours, 0.2 },
303 { 0_days, 0.1 }
304 }
305 };
306 const units::temperature weighted_avg = std::accumulate( measurement_weights.begin(),
307 measurement_weights.end(),
308 0_c,
309 [this, location, time, seed, calendar_config]( units::temperature acc,
310 const std::pair<time_duration, double> &pr ) {
311 units::temperature weather_temperature =
312 get_weather_temperature( location, time - pr.first, calendar_config, seed );
313 return acc + multiply_any_unit( weather_temperature, pr.second );
314 } );
315 // Rescale the range:
316 // For avg air temp<-10C, water is 0C
317 // For avg air temp> 30C, water is 30C
318 // logarithmic_range smoothing for the in-between
319 constexpr int lower_limit = units::to_millidegree_celsius( -10_c );
320 constexpr int upper_limit = units::to_millidegree_celsius( 30_c );
321 const int weighted_average_celsius = units::to_millidegree_celsius( weighted_avg );
322 const double t = logarithmic_range( lower_limit,
323 upper_limit,
324 weighted_average_celsius );
325 return multiply_any_unit( 0_c, 1 - t ) + multiply_any_unit( 30_c, t );
326}
