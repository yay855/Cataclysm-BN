11static int test_face_size( const std::string &f, int size, int faceIndex )
12{
13 const TTF_Font_Ptr font( TTF_OpenFontIndex( f.c_str(), size, faceIndex ) );
14 if( font ) {
15 const char *font_style = TTF_FontFaceStyleName( font.get() );
16 if( font_style != nullptr ) {
17 int num_faces = TTF_FontFaces( font.get() );
18 for( int face_i = num_faces - 1; face_i >= 0; face_i-- ) {
19 const TTF_Font_Ptr face( TTF_OpenFontIndex( f.c_str(), size, face_i ) );
20 if( !face ) {
21 continue;
22 }
23 const char *face_style = TTF_FontFaceStyleName( face.get() );
24 if( !face_style ) {
25 continue;
26 }
27 if( lcequal( face_style, font_style ) && TTF_FontHeight( face.get() ) <= size ) {
28 return face_i;
29 }
30 }
31 }
32 }
33
34 return faceIndex;

274SDL_Texture_Ptr CachedTTFFont::create_glyph( const SDL_Renderer_Ptr &renderer,
275 const std::string &ch,
276 const int color )
277{
278 const auto function = fontblending ? TTF_RenderUTF8_Blended : TTF_RenderUTF8_Solid;
279 SDL_Surface_Ptr sglyph( function( font.get(), ch.c_str(), windowsPalette[color] ) );
280 if( !sglyph ) {
281 dbg( DL::Error ) << "Failed to create glyph for " << ch << ": " << TTF_GetError();
282 return nullptr;
283 }
284 /* SDL interprets each pixel as a 32-bit number, so our masks must depend
285 on the endianness (byte order) of the machine */
286#if SDL_BYTEORDER == SDL_BIG_ENDIAN
287 static const Uint32 rmask = 0xff000000;
288 static const Uint32 gmask = 0x00ff0000;
289 static const Uint32 bmask = 0x0000ff00;
290 static const Uint32 amask = 0x000000ff;
291#else
292 static const Uint32 rmask = 0x000000ff;
293 static const Uint32 gmask = 0x0000ff00;
294 static const Uint32 bmask = 0x00ff0000;
295 static const Uint32 amask = 0xff000000;
296#endif
297 const int wf = utf8_wrapper( ch ).display_width();
298 // Note: bits per pixel must be 8 to be synchronized with the surface
299 // that TTF_RenderGlyph above returns. This is important for SDL_BlitScaled
300 SDL_Surface_Ptr surface = CreateRGBSurface( 0, width * wf, height, 32, rmask, gmask, bmask,
301 amask );
302 SDL_Rect src_rect = { 0, 0, sglyph->w, sglyph->h };
303 SDL_Rect dst_rect = { 0, 0, width * wf, height };
304 if( src_rect.w < dst_rect.w ) {
305 dst_rect.x = ( dst_rect.w - src_rect.w ) / 2;
306 dst_rect.w = src_rect.w;
307 } else if( src_rect.w > dst_rect.w ) {
308 src_rect.x = ( src_rect.w - dst_rect.w ) / 2;
309 src_rect.w = dst_rect.w;
310 }
311 if( src_rect.h < dst_rect.h ) {
312 dst_rect.y = ( dst_rect.h - src_rect.h ) / 2;
313 dst_rect.h = src_rect.h;
314 } else if( src_rect.h > dst_rect.h ) {
315 src_rect.y = ( src_rect.h - dst_rect.h ) / 2;
316 src_rect.h = dst_rect.h;
317 }
318
319 if( !printErrorIf( SDL_BlitSurface( sglyph.get(), &src_rect, surface.get(), &dst_rect ) != 0,
320 "SDL_BlitSurface failed" ) ) {
321 sglyph = std::move( surface );
322 }
323
324 return CreateTextureFromSurface( renderer, sglyph );

363BitmapFont::BitmapFont(
364 SDL_Renderer_Ptr &renderer, SDL_PixelFormat_Ptr &format,
365 const int w, const int h,
366 const palette_array &palette,
367 const std::string &typeface_path )
368 : Font( w, h, palette )
369{
370 dbg( DL::Info ) << "Loading bitmap font [" + typeface_path + "].";
371 SDL_Surface_Ptr asciiload = load_image( typeface_path.c_str() );
372 assert( asciiload );
373 if( asciiload->w * asciiload->h < ( width * height * 256 ) ) {
374 throw std::runtime_error( "bitmap for font is to small" );
375 }
376 Uint32 key = SDL_MapRGB( asciiload->format, 0xFF, 0, 0xFF );
377 SDL_SetColorKey( asciiload.get(), SDL_TRUE, key );
378 SDL_Surface_Ptr ascii_surf[std::tuple_size<decltype( ascii )>::value];
379 ascii_surf[0].reset( SDL_ConvertSurface( asciiload.get(), format.get(), 0 ) );
380 SDL_SetSurfaceRLE( ascii_surf[0].get(), 1 );
381 asciiload.reset();
382
383 for( size_t a = 1; a < std::tuple_size<decltype( ascii )>::value; ++a ) {
384 ascii_surf[a].reset( SDL_ConvertSurface( ascii_surf[0].get(), format.get(), 0 ) );
385 SDL_SetSurfaceRLE( ascii_surf[a].get(), 1 );
386 }
387
388 for( size_t a = 0; a < std::tuple_size<decltype( ascii )>::value - 1; ++a ) {
389 SDL_LockSurface( ascii_surf[a].get() );
390 int size = ascii_surf[a]->h * ascii_surf[a]->w;
391 Uint32 *pixels = static_cast<Uint32 *>( ascii_surf[a]->pixels );
392 Uint32 color = ( windowsPalette[a].r << 16 ) | ( windowsPalette[a].g << 8 ) | windowsPalette[a].b;
393 for( int i = 0; i < size; i++ ) {
394 if( pixels[i] == 0xFFFFFF ) {
395 pixels[i] = color;
396 }
397 }
398 SDL_UnlockSurface( ascii_surf[a].get() );
399 }
400 tilewidth = ascii_surf[0]->w / width;
401
402 //convert ascii_surf to SDL_Texture
403 for( size_t a = 0; a < std::tuple_size<decltype( ascii )>::value; ++a ) {
404 ascii[a] = CreateTextureFromSurface( renderer, ascii_surf[a] );
405 }
406}
