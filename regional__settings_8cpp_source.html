613void apply_region_overlay( const JsonObject &jo, regional_settings &region )
614{
615 jo.read( "default_oter", region.default_oter );
616 jo.read( "river_scale", region.river_scale );
617 if( jo.has_array( "default_groundcover" ) ) {
618 region.default_groundcover_str.reset( new weighted_int_list<ter_str_id> );
619 for( JsonArray inner : jo.get_array( "default_groundcover" ) ) {
620 if( region.default_groundcover_str->add( ter_str_id( inner.get_string( 0 ) ),
621 inner.get_int( 1 ) ) == nullptr ) {
622 jo.throw_error( "'default_groundcover' must be a weighted list: an array of pairs [ \"id\", weight ]" );
623 }
624 }
625 }
626
627 JsonObject fieldjo = jo.get_object( "field_coverage" );
628 double tmpval = 0.0f;
629 if( fieldjo.read( "percent_coverage", tmpval ) ) {

631 }
632
633 fieldjo.read( "default_ter", region.field_coverage.default_ter_str );
634
635 for( const JsonMember member : fieldjo.get_object( "other" ) ) {
636 if( member.is_comment() ) {

640 }
641
642 if( fieldjo.read( "boost_chance", tmpval ) ) {
643 region.field_coverage.boost_chance = static_cast<int>( tmpval * 10000.0 );
644 }
645 if( fieldjo.read( "boosted_percent_coverage", tmpval ) ) {
646 if( region.field_coverage.boost_chance > 0.0f && tmpval == 0.0f ) {
647 fieldjo.throw_error( "boost_chance > 0 requires boosted_percent_coverage" );
648 }
649
650 region.field_coverage.boosted_mpercent_coverage = static_cast<int>( tmpval * 10000.0 );
651 }
652
653 if( fieldjo.read( "boosted_other_percent", tmpval ) ) {
654 if( region.field_coverage.boost_chance > 0.0f && tmpval == 0.0f ) {
655 fieldjo.throw_error( "boost_chance > 0 requires boosted_other_percent" );
656 }
657
658 region.field_coverage.boosted_other_mpercent = static_cast<int>( tmpval * 10000.0 );
659 }
660
661 for( const JsonMember member : fieldjo.get_object( "boosted_other" ) ) {
662 if( member.is_comment() ) {
663 continue;
664 }
665 region.field_coverage.boosted_percent_str[member.name()] = member.get_float();
666 }

669 region.field_coverage.boosted_percent_str.empty() ) {

675 load_forest_trail_settings( jo, region.forest_trail, false, true );

679 continue;
680 }

843ter_furn_id forest_biome::pick() const
844{
845 // Iterate through the biome components (which have already been put into sequence), roll for the
846 // one_in chance that component contributes a feature, and if so pick that feature and return it.
847 // If a given component does not roll as success, proceed to the next feature in sequence until
848 // a feature is picked or none are picked, in which case an empty feature is returned.
849 const ter_furn_id *result = nullptr;
850 for( auto &pr : biome_components ) {
851 if( one_in( pr.chance ) ) {
852 result = pr.types.pick();
853 break;
854 }
855 }
856
857 if( result == nullptr ) {
858 return ter_furn_id();
859 }
860
861 return *result;
862}
