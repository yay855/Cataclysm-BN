14// ===============================================================================================
15// Plural forms
16// ===============================================================================================
17//
18// Plural forms expression evaluation is done by evaluating a simple AST built from the
19// source string. The expression follows same operator precedence as C, and the parser here
20// is based on same simplified grammar as the parser used by Poedit:
21//
22// Expr: // Expression
23// OrExpr ? Expr : Expr
24// OrExpr
25// OrExpr: // Logical OR
26// AndExpr || OrExpr
27// AndExpr
28// AndExpr: // Logical AND
29// EqExpr && AndExpr
30// EqExpr
31// EqExpr: // Equality
32// CmpExpr == CmpExpr
33// CmpExpr != CmpExpr
34// CmpExpr
35// CmpExpr: // Comparison
36// ModExpr >= ModExpr
37// ModExpr > ModExpr
38// ModExpr <= ModExpr
39// ModExpr < ModExpr
40// ModExpr
41// ModExpr: // Modulo
42// Value % Value
43// Value
44// Value: // Value
45// ( Expr ) // expression in brackets
46// [0..9]+ // unsigned integer
47// n // the variable
48//
49// Note that this grammar does not cover all valid C expressions, or all valid C operators
50// (e.g. "n % 10 % 3", or "(n+1) % 2"), but that doesn't seem to cause problems.
51

61static const std::array<std::pair<std::string, PlfOp>, 14> simple_tokens = {{
62 { "?", PlfOp::TerCond },
63 { ":", PlfOp::TerDelim },
64 { "(", PlfOp::BrOpen },
65 { ")", PlfOp::BrClose },
66 { "n", PlfOp::Variable },
67 { "%", PlfOp::Mod },
68 { "==", PlfOp::Eq },

70 { ">=", PlfOp::GreaterEq },
71 { ">", PlfOp::Greater },
72 { "<=", PlfOp::LessEq },
73 { "<", PlfOp::Less },
74 { "&&", PlfOp::And },
75 { "||", PlfOp::Or },

79struct PlfTStream {
80 const std::string *s;
81 size_t pos;
82 size_t end;
83
84 PlfTStream( const std::string *raw ) {
85 s = raw;
86 pos = 0;
87 if( s ) {
88 end = s->size();
89 }
90 }
91
92 PlfTStream cloned() const {
93 return *this;
94 }
95
96 PlfToken peek_internal() const {
97 std::string tok;
98 size_t pos = this->pos;
99 while( pos < end ) {
100 char c = ( *s )[pos];
101 if( c == ' ' || c == '\t' ) {
102 // skip whitespace
103 pos++;
104 continue;
105 } else if( c >= '0' && c <= '9' ) {
106 // number
107 size_t start_pos = pos;
108 while( pos < end && c >= '0' && c <= '9' ) {
109 tok.push_back( c );
110 pos++;
111 c = ( *s )[pos];
112 }
113 try {
114 size_t ul = std::stoul( tok );
115 if( ul > std::numeric_limits<uint32_t>::max() ) {
116 throw std::out_of_range( "stoul" );
117 }
118 return PlfToken{ PlfOp::Literal, ul, start_pos, pos - start_pos };
119 } catch( const std::logic_error &err ) {
120 std::string e = string_format( "invalid number '%s' at pos %d", tok, start_pos );
121 throw std::runtime_error( e );
122 }
123 }
124 std::string ss = s->substr( pos, end - pos );
125 for( const auto &t : simple_tokens ) {
126 if( string_starts_with( ss, t.first ) ) {
127 return PlfToken{ t.second, 0, pos, t.first.size() };
128 }
129 }
130 pos++;
131 std::string e = string_format( "unexpected character '%c' at pos %d", c, pos - 1 );
132 throw std::runtime_error( e );
133 }
134 return PlfToken{ PlfOp::NumOps, 0, pos, 0 };
135 }
136
137 PlfToken get() {
138 PlfToken t = peek_internal();
139 pos = t.start + t.len;
140 return t;
141 }
142
143 PlfToken peek() const {
144 return peek_internal();
145 }
146
147 PlfTStream &skip() {
148 get();
149 return *this;
150 }
151
152 bool has_tokens() const {
153 return peek().kind != PlfOp::NumOps;
154 }

186static bool plf_try_binary_op( ParseRet &left, PlfOp op, ParserPtr parser )
187{
188 const PlfTStream &ts = left.ts;
189 if( ts.peek().kind != op ) {
190 // not found
191 return false;
192 }
193 ParseRet e = parser( ts.cloned().skip() );
194 left = ParseRet( std::make_unique<PlfNode>( PlfNode{
195 0, std::move( left.expr ), std::move( e.expr ), nullptr, op,
196 } ),
197 e.ts );
198 return true;
199}

201ParseRet plf_get_expr( const PlfTStream &ts )
202{
203 ParseRet e1 = plf_get_or( ts );
204 if( e1.ts.peek().kind != PlfOp::TerCond ) {
205 return e1;
206 }
207 ParseRet e2 = plf_get_expr( e1.ts.cloned().skip() );
208 if( e2.ts.peek().kind != PlfOp::TerDelim ) {
209 PlfToken tok = e2.ts.peek();
210 std::string e = string_format( "expected ternary delimiter at pos %d", tok.start );
211 throw std::runtime_error( e );
212 }
213 ParseRet e3 = plf_get_expr( e2.ts.cloned().skip() );
214 return ParseRet( std::make_unique<PlfNode>( PlfNode{
215 0, std::move( e1.expr ), std::move( e2.expr ), std::move( e3.expr ), PlfOp::TerCond,
216 } ),
217 e3.ts );
218}

267ParseRet plf_get_value( const PlfTStream &ts )
268{
269 PlfToken next = ts.peek();
270 if( next.kind == PlfOp::BrOpen ) {
271 // '(' expr ')'
272 ParseRet e = plf_get_expr( ts.cloned().skip() );
273 if( e.ts.peek().kind != PlfOp::BrClose ) {
274 PlfToken tok = e.ts.peek();
275 std::string e = string_format( "expected closing bracket at pos %d", tok.start );
276 throw std::runtime_error( e );
277 }
278 return ParseRet( std::move( e.expr ), e.ts.cloned().skip() );
279 } else if( next.kind == PlfOp::Literal ) {
280 // number
281 return ParseRet( std::make_unique<PlfNode>( PlfNode{
282 next.num, nullptr, nullptr, nullptr, PlfOp::Literal,
283 } ), ts.cloned().skip() );
284 } else if( next.kind == PlfOp::Variable ) {
285 // the variable
286 return ParseRet( std::make_unique<PlfNode>( PlfNode{
287 0, nullptr, nullptr, nullptr, PlfOp::Variable,
288 } ), ts.cloned().skip() );
289 } else {
290 std::string e = string_format( "expected expression at pos %d", next.start );
291 throw std::runtime_error( e );
292 }

295size_t PlfNode::eval( size_t n ) const
296{
297 switch( op ) {
298 case PlfOp::Mod: {
299 size_t right = b->eval( n );
300 if( right == 0 ) {
301 std::string e = string_format( "DBZ in PlfNode::eval( %d ), node='%s'", n, debug_dump() );
302 throw std::runtime_error( e );
303 }
304 return a->eval( n ) % right;
305 }
306 case PlfOp::Eq:
307 return a->eval( n ) == b->eval( n );
308 case PlfOp::NotEq:
309 return a->eval( n ) != b->eval( n );
310 case PlfOp::GreaterEq:
311 return a->eval( n ) >= b->eval( n );
312 case PlfOp::Greater:
313 return a->eval( n ) > b->eval( n );
314 case PlfOp::LessEq:
315 return a->eval( n ) <= b->eval( n );
316 case PlfOp::Less:
317 return a->eval( n ) < b->eval( n );
318 case PlfOp::And:
319 return a->eval( n ) && b->eval( n );
320 case PlfOp::Or:
321 return a->eval( n ) || b->eval( n );
322 case PlfOp::TerCond:
323 return a->eval( n ) ? b->eval( n ) : c->eval( n );
324 case PlfOp::Literal:
325 return literal_val;
326 case PlfOp::Variable:
327 return n;
328 default:
329 // unreachable
330 assert( false );
331 }
332 return 0;
333}

437void trans_catalogue::process_file_header()
438{
439 constexpr u32 MO_MAGIC_NUMBER_LE = 0x950412de;
440 constexpr u32 MO_MAGIC_NUMBER_BE = 0xde120495;
441 constexpr u32 MO_SUPPORTED_REVISION = 0;
442
443 constexpr u32 OFFS_MAGIC_NUMBER = 0;
444 constexpr u32 OFFS_REVISION = 4;
445 constexpr u32 OFFS_NUM_STRINGS = 8;
446 constexpr u32 OFFS_ORIG_TABLE_BEGIN = 12;
447 constexpr u32 OFFS_TRANS_TABLE_BEGIN = 16;
448
449 u32 magic = this->buffer.size() > 4 ? get_u32( OFFS_MAGIC_NUMBER ) : 0;
450 if( magic != MO_MAGIC_NUMBER_LE && magic != MO_MAGIC_NUMBER_BE ) {
451 throw std::runtime_error( "not a MO file" );
452 }
453 this->is_little_endian = magic == MO_MAGIC_NUMBER_LE;
454 if( get_u32( OFFS_REVISION ) != MO_SUPPORTED_REVISION ) {
455 std::string e = string_format(
456 "expected revision %d, got %d",
457 MO_SUPPORTED_REVISION,
458 get_u32( OFFS_REVISION )
459 );
460 throw std::runtime_error( e );
461 }
462
463 number_of_strings = get_u32( OFFS_NUM_STRINGS );
464 offs_orig_table = get_u32( OFFS_ORIG_TABLE_BEGIN );
465 offs_trans_table = get_u32( OFFS_TRANS_TABLE_BEGIN );
466}

468void trans_catalogue::check_string_terminators()
469{
470 const auto check_string = [this]( u32 offs ) {
471 // Check that string with its null terminator (not included in string length)
472 // does not extend beyond file boundaries.
473 string_descr s = get_string_descr( offs );
474 if( s.offset + s.length + 1 > buf_size() ) {
475 std::string e = string_format(
476 "string_descr at offs %#x: extends beyond EOF (len:%#x offs:%#x fsize:%#x)",
477 offs, s.length, s.offset, buf_size()
478 );
479 throw std::runtime_error( e );
480 }
481 // Also check for existence of the null byte.
482 u8 terminator = get_u8( s.offset + s.length );
483 if( terminator != 0 ) {
484 std::string e = string_format(
485 "string_descr at offs %#x: missing null terminator",
486 offs
487 );
488 throw std::runtime_error( e );
489 }
490 };
491 for( u32 i = 0; i < number_of_strings; i++ ) {
492 check_string( offs_orig_table + i * MO_STRING_DESCR_SIZE );
493 check_string( offs_trans_table + i * MO_STRING_DESCR_SIZE );
494 }
495}

497void trans_catalogue::check_string_plurals()
498{
499 // Skip metadata (the 0th entry)
500 for( u32 i = 1; i < number_of_strings; i++ ) {
501 string_descr info = get_string_descr_unsafe( offs_orig_table + i * MO_STRING_DESCR_SIZE );
502
503 // Check for null byte - msgid/msgid_plural separator
504 bool has_plurals = false;
505 for( u32 j = info.offset; j < info.offset + info.length; j++ ) {
506 if( get_u8_unsafe( j ) == 0 ) {
507 has_plurals = true;
508 break;
509 }
510 }
511
512 if( !has_plurals ) {
513 continue;
514 }
515
516 // Count null bytes - each plural form is a null-terminated string (including last one)
517 u32 offs_tr = offs_trans_table + i * MO_STRING_DESCR_SIZE;
518 string_descr info_tr = get_string_descr_unsafe( offs_tr );
519 size_t plural_forms = 0;
520 for( u32 j = info_tr.offset; j <= info_tr.offset + info_tr.length; j++ ) {
521 if( get_u8_unsafe( j ) == PLF_SEPARATOR ) {
522 plural_forms += 1;
523 }
524 }
525
526 // Number of plural forms should match the number specified in metadata
527 if( plural_forms != this->plurals.num ) {
528 std::string e = string_format(
529 "string_descr at offs %#x: expected %d plural forms, got %d",
530 offs_tr, this->plurals.num, plural_forms
531 );
532 throw std::runtime_error( e );
533 }
534 }

537void trans_catalogue::check_encoding( const meta_headers &headers )
538{
539 // HACK: The checks here are rather crude and don't account for
540 // possible extra spaces or wrong capitalization.
541 // However, we don't expect people to manually configure
542 // these headers (there're tons of software for that),
543 // and the strings here match output of our own scripts and
544 // Poedit/xgettext/Transifex, so leaving as is should be fine.
545 {
546 bool found = false;
547 for( const std::string &entry : headers ) {
548 if( !string_starts_with( entry, "Content-Type:" ) ) {

550 }
551 found = true;

556 expected );

559 break;
560 }

562 throw std::runtime_error( "failed to find Content-Type header" );
563 }
564 }
565 {

567 for( const std::string &entry : headers ) {
568 if( !string_starts_with( entry, "Content-Transfer-Encoding:" ) ) {
569 continue;
570 }
571 found = true;
572 static const std::string expected = "Content-Transfer-Encoding: 8bit";
573 if( entry != expected ) {
574 std::string e =

576 expected );

587trans_catalogue::catalogue_plurals_info trans_catalogue::parse_plf_header(
588 const meta_headers &headers )
589{
590 constexpr size_t MAX_PLURAL_FORMS = 8;
591
592 // HACK: Same as with encoding headers, we expect this header to be
593 // automatically generated. This "parser" here succeeds with
594 // MO files compiled from PO created with Transifex/Poedit/msginit,
595 // and that'll probably extend to majority of other software.
596
597 catalogue_plurals_info ret;
598
599 // Parse Plural-Forms header.
600 std::string plf_raw;
601 {
602 bool found = false;
603 for( const std::string &entry : headers ) {
604 if( !string_starts_with( entry, "Plural-Forms:" ) ) {
605 continue;
606 }
607 found = true;
608 plf_raw = entry.substr( 13 ); // length of "Plural-Forms:" string
609 break;
610 }
611 if( !found ) {
612 // Default to Germanic rules (English, German, Dutch, ...)
613 ret.num = 2;
614 ret.expr = parse_plural_rules( "n!=1" );
615 return ret;
616 }
617 }
618 std::vector<std::string> parts = string_split( plf_raw, ';' );
619 if( parts.size() != 3 ) {
620 throw std::runtime_error( "expected Plural-Forms header to have 2 ';' characters" );
621 }
622
623 // Parse & validate nplurals
624 {
625 std::string plf_n_raw = parts[0];
626 if( !string_starts_with( plf_n_raw, " nplurals=" ) ) {
627 throw std::runtime_error( "failed to parse Plural-Forms header" );
628 }
629 plf_n_raw = plf_n_raw.substr( 10 ); // 10 is length of " nplurals=" string
630 try {
631 ret.num = std::stoul( plf_n_raw );
632 } catch( const std::runtime_error &err ) {
633 std::string e = string_format( "failed to parse Plural-Forms nplurals number '%s': %s", plf_n_raw,
634 err.what() );
635 throw std::runtime_error( e );
636 }
637 if( ret.num == 0 || ret.num > MAX_PLURAL_FORMS ) {
638 std::string e = string_format( "expected at most 1-%d plural forms, got %d", MAX_PLURAL_FORMS,
639 ret.num );
640 throw std::runtime_error( e );
641 }
642 }
643
644 // Parse & validate plural formula
645 {
646 std::string plf_rules_raw = parts[1];
647 if( !string_starts_with( plf_rules_raw, " plural=" ) ) {
648 throw std::runtime_error( "failed to parse Plural-Forms header" );
649 }
650 plf_rules_raw = plf_rules_raw.substr( 8 ); // 8 is length of " plural=" string
651 try {
652 ret.expr = parse_plural_rules( plf_rules_raw );
653 } catch( const std::runtime_error &err ) {
654 std::string e = string_format( "failed to parse plural forms formula: %s", err.what() );
655 throw std::runtime_error( e );
656 }
657 }
658
659 return ret;
660}

706const char *trans_catalogue::get_nth_pl_translation( u32 n, size_t num ) const
707{
708 u32 descr_offs = offs_trans_table + n * MO_STRING_DESCR_SIZE;
709 string_descr r = get_string_descr_unsafe( descr_offs );
710
711 size_t plf = plurals.expr->eval( num );
712
713 if( plf == 0 || plf >= plurals.num ) {
714 return offs_to_cstr( r.offset );
715 }
716 size_t curr_plf = 0;
717 for( u32 offs = r.offset; offs < r.offset + r.length; offs++ ) {
718 if( get_u8_unsafe( offs ) == PLF_SEPARATOR ) {
719 curr_plf += 1;
720 if( plf == curr_plf ) {
721 return offs_to_cstr( offs + 1 );
722 }
723 }
724 }
725 return nullptr;
726}

732std::vector<trans_library::library_string_descr>::const_iterator trans_library::find_entry(
733 const char *id ) const
734{
735 auto it = std::lower_bound( strings.begin(), strings.end(),
736 id, [this]( const library_string_descr & a_descr, const char *b ) -> bool {
737 const char *a = this->catalogues[a_descr.catalogue].get_nth_orig_string( a_descr.entry );
738 return strcmp( a, b ) < 0;
739 } );
740
741 if( it != strings.end() ) {
742 const char *found = catalogues[it->catalogue].get_nth_orig_string( it->entry );
743 if( strcmp( id, found ) == 0 ) {
744 return it;
745 }
746 }
747
748 return strings.end();
749}
750
751void trans_library::build_string_table()
752{
753 assert( strings.empty() );
754
755 for( size_t i_cat = 0; i_cat < catalogues.size(); i_cat++ ) {
756 const trans_catalogue &cat = catalogues[i_cat];
757 u32 num = cat.get_num_strings();
758 // 0th entry is the metadata, we skip it
759 strings.reserve( num - 1 );
760 for( u32 i = 1; i < num; i++ ) {
761 const char *i_cstr = cat.get_nth_orig_string( i );
762
763 auto it = std::lower_bound( strings.begin(), strings.end(),
764 i_cstr, [this]( const library_string_descr & a_descr, const char *b ) -> bool {
765 const char *a = this->catalogues[a_descr.catalogue].get_nth_orig_string( a_descr.entry );
766 return strcmp( a, b ) < 0;
767 } );
768
769 library_string_descr desc = { static_cast<u32>( i_cat ), i };
770 if( it == strings.end() ) {
771 // Not found, or all elements are greater
772 strings.push_back( desc );
773 } else if( strcmp( catalogues[it->catalogue].get_nth_orig_string( it->entry ), i_cstr ) == 0 ) {
774 // Overwrite existing string only if new string has plural form(s),
775 // but existing one does not.
776 if(
777 cat.check_nth_translation_has_plf( i ) &&
778 !catalogues[it->catalogue].check_nth_translation_has_plf( it->entry )
779 ) {
780 *it = desc;
781 }
782 continue;
783 } else {
784 strings.insert( it, desc );
785 }
786 }
787 }
788}
