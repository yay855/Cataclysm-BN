224void assign_function( const JsonObject &jo, const std::string &id, Fun &target,
225 const std::map<std::string, Fun> &cont )
226{
227 if( jo.has_string( id ) ) {
228 const auto iter = cont.find( jo.get_string( id ) );

237void mission_type::load( const JsonObject &jo, const std::string &src )
238{
239 const bool strict = src == "dda";
240
241 mandatory( jo, was_loaded, "name", name );
242
243 mandatory( jo, was_loaded, "difficulty", difficulty );
244 mandatory( jo, was_loaded, "value", value );
245
246 if( jo.has_member( "origins" ) ) {
247 origins.clear();
248 for( auto &m : jo.get_tags( "origins" ) ) {
249 origins.emplace_back( io::string_to_enum<mission_origin>( m ) );
250 }
251 }
252
253 if( std::any_of( origins.begin(), origins.end(), []( mission_origin origin ) {
254 return origin == ORIGIN_ANY_NPC || origin == ORIGIN_OPENER_NPC || origin == ORIGIN_SECONDARY;
255} ) ) {
256 auto djo = jo.get_object( "dialogue" );
257 // TODO: There should be a cleaner way to do it
258 mandatory( djo, was_loaded, "describe", dialogue[ "describe" ] );
259 mandatory( djo, was_loaded, "offer", dialogue[ "offer" ] );
260 mandatory( djo, was_loaded, "accepted", dialogue[ "accepted" ] );
261 mandatory( djo, was_loaded, "rejected", dialogue[ "rejected" ] );
262 mandatory( djo, was_loaded, "advice", dialogue[ "advice" ] );
263 mandatory( djo, was_loaded, "inquire", dialogue[ "inquire" ] );
264 mandatory( djo, was_loaded, "success", dialogue[ "success" ] );
265 mandatory( djo, was_loaded, "success_lie", dialogue[ "success_lie" ] );
266 mandatory( djo, was_loaded, "failure", dialogue[ "failure" ] );
267 }
268
269 optional( jo, was_loaded, "description", description );
270 optional( jo, was_loaded, "urgent", urgent );
271 optional( jo, was_loaded, "item", item_id );
272 optional( jo, was_loaded, "item_group", group_id );
273 optional( jo, was_loaded, "count", item_count, 1 );
274 optional( jo, was_loaded, "required_container", container_id );
275 optional( jo, was_loaded, "remove_container", remove_container );
276 //intended for situations where closed and open container are different
277 optional( jo, was_loaded, "empty_container", empty_container );
278 optional( jo, was_loaded, "has_generic_rewards", has_generic_rewards, true );
279
280 goal = jo.get_enum_value<decltype( goal )>( "goal" );
281
282 assign_function( jo, "place", place, tripoint_function_map );
283 const auto parse_phase = [&]( const std::string & phase,
284 std::function<void( mission * )> &phase_func ) {
285 if( jo.has_string( phase ) ) {
286 assign_function( jo, phase, phase_func, mission_function_map );
287 } else if( jo.has_member( phase ) ) {
288 JsonObject j_start = jo.get_object( phase );
289 if( !parse_funcs( j_start, phase_func ) ) {
290 deferred.emplace_back( jo.get_source_location(), src );
291 jo.allow_omitted_members();
292 j_start.allow_omitted_members();
293 return false;
294 }
295 }
296 return true;
297 };
298 if( !parse_phase( "start", start ) ) {
299 return;
300 }
301 if( !parse_phase( "end", end ) ) {
302 return;
303 }
304 if( !parse_phase( "fail", fail ) ) {
305 return;
306 }
307
308 assign( jo, "deadline_low", deadline_low, false, 1_days );
309 assign( jo, "deadline_high", deadline_high, false, 1_days );
310
311 if( jo.has_member( "followup" ) ) {
312 follow_up = mission_type_id( jo.get_string( "followup" ) );
313 }
314
315 if( jo.has_member( "monster_species" ) ) {
316 monster_species = species_id( jo.get_string( "monster_species" ) );
317 }
318 if( jo.has_member( "monster_type" ) ) {
319 monster_type = mtype_id( jo.get_string( "monster_type" ) );
320 }
321
322 if( jo.has_member( "monster_kill_goal" ) ) {
323 monster_kill_goal = jo.get_int( "monster_kill_goal" );
324 }
325
326 assign( jo, "destination", target_id, strict );
327
328 if( jo.has_member( "goal_condition" ) ) {
329 read_condition<mission_goal_condition_context>( jo, "goal_condition", goal_condition, true );
330 }
331}
