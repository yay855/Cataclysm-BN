56folded_text::folded_text( const std::string &str, int line_width )
57{
58 // string pointer, remaining bytes, total codepoints, and total display width ...
59 // ... before current processed character
60 const char *src = str.c_str();
61 int bytes = str.length();
62 int cpts = 0;
63 int width = 0;
64 // ... before current line start
65 const char *src_start = src;
66 int cpts_start = cpts;
67 int width_start = width;
68 // ... after the last word character
69 const char *src_word = src_start;
70 // ... at the last breaking position
71 const char *src_break = src_start;
72 int cpts_break = cpts_start;
73 int width_break = width_start;
74 while( bytes > 0 ) {
75 // ... before current processed character
76 const char *const src_curr = src;
77 const int cpts_curr = cpts;
78 const int width_curr = width;
79 // ... after current processed character
80 const uint32_t uc = UTF8_getch( &src, &bytes );
81 const bool linebreak = is_linebreak( uc );
82 // assuming character has non-negative width
83 const int cw = linebreak ? 0 : std::max( 0, mk_wcwidth( uc ) );
84 cpts += 1;
85 width += cw;
86 // if the characters so far do not fit in a single line
87 if( width > width_start + line_width ) {
88 if( src_break > src_start ) {
89 // break at the last breaking position in the line if any
90 lines.emplace_back( folded_line{
91 cpts_start, cpts_break,
92 std::string( src_start, src_break )
93 } );
94 src_start = src_break;
95 cpts_start = cpts_break;
96 width_start = width_break;
97 } else if( src_curr > src_start ) {
98 // otherwise break before the current character, but ensure
99 // each line has at least one character
100 lines.emplace_back( folded_line{
101 cpts_start, cpts_curr,
102 std::string( src_start, src_curr )
103 } );
104 src_start = src_curr;
105 cpts_start = cpts_curr;
106 width_start = width_curr;
107 }
108 }
109 // always break on line breaks
110 if( linebreak ) {
111 lines.emplace_back( folded_line{
112 cpts_start, cpts,
113 std::string( src_start, src )
114 } );
115 src_start = src;
116 cpts_start = cpts;
117 width_start = width;
118 }
119 if( is_word( uc ) ) {
120 src_word = src;
121 }
122 // can we break after the current character?
123 if( break_after( uc ) && src > src_start
124 // break with at least one word character before
125 && src_word > src_start ) {
126 src_break = src;
127 cpts_break = cpts;
128 width_break = width;
129 }
130 }
131 // remaining characters (empty line if the string is empty or the last
132 // character is line break)
133 lines.emplace_back( folded_line{
134 cpts_start, cpts,
135 std::string( src_start, src )
136 } );
137}

144point folded_text::codepoint_coordinates( int cpt_idx, bool zero_x ) const
145{
146 if( lines.empty() ) {
147 return point_zero;
148 }
149 // find the line before the cursor position
150 auto it = std::lower_bound( lines.begin(), lines.end(), cpt_idx,
151 []( const folded_line & l, const int p ) {
152 return l.cpts_end < p;
153 } );
154 if( it == lines.end() ) {
155 // past the last codepoint, shouldn't happen
156 return point_zero;
157 }
158 int y = std::distance( lines.begin(), it );
159 // if zero_x is true and the line is not the last line, cursor at the end of
160 // the line is moved to the start of the next line
161 if( zero_x && static_cast<size_t>( y + 1 ) < lines.size()
162 && cpt_idx == it->cpts_end ) {
163 return point( 0, y + 1 );
164 }
165 // otherwise, calculate the width until cpt_idx
166 int x = 0;
167 const char *src = it->str.c_str();
168 int bytes = it->str.length();
169 for( int i = 0; bytes > 0 && i < cpt_idx - it->cpts_start; ++i ) {
170 const uint32_t uc = UTF8_getch( &src, &bytes );
171 if( is_linebreak( uc ) ) {
172 x = 0;
173 ++y;
174 } else {
175 x += std::max( 0, mk_wcwidth( uc ) );
176 }
177 }
178 return point( x, y );
179}
