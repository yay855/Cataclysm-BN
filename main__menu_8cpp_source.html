100static int utf8_width_notags( const char *s )
101{
102 int len = strlen( s );
103 const char *ptr = s;
104 int w = 0;
105 bool inside_tag = false;
106 while( len > 0 ) {
107 uint32_t ch = UTF8_getch( &ptr, &len );
108 if( ch == UNKNOWN_UNICODE ) {
109 continue;
110 }
111 if( ch == '<' ) {
112 inside_tag = true;
113 } else if( ch == '>' ) {
114 inside_tag = false;
115 continue;
116 }
117 if( inside_tag ) {
118 continue;
119 }
120 w += mk_wcwidth( ch );
121 }
122 return w;
123}
