155static item_location inv_internal( player &u, const inventory_selector_preset &preset,
156 const std::string &title, int radius,
157 const std::string &none_message,
158 const std::string &hint = std::string() )
159{
160 inventory_pick_selector inv_s( u, preset );
161
162 inv_s.set_title( title );
163 inv_s.set_hint( hint );
164 inv_s.set_display_stats( false );
165
166 std::pair<size_t, size_t> init_pair;
167 bool init_selection = false;
168 std::string init_filter;
169 bool has_init_filter = false;
170
171 const std::vector<activity_id> consuming {
172 ACT_EAT_MENU,
173 ACT_CONSUME_FOOD_MENU,
174 ACT_CONSUME_DRINK_MENU,
175 ACT_CONSUME_MEDS_MENU };
176
177 if( u.has_activity( consuming ) && u.activity.values.size() >= 2 ) {
178 init_pair.first = u.activity.values[0];
179 init_pair.second = u.activity.values[1];
180 init_selection = true;
181 }
182 if( u.has_activity( consuming ) && !u.activity.str_values.empty() ) {
183 init_filter = u.activity.str_values[0];
184 has_init_filter = true;
185 }
186
187 do {
188 u.inv.restack( u );
189
190 inv_s.clear_items();
191 inv_s.add_character_items( u );
192 inv_s.add_nearby_items( radius );
193
194 if( has_init_filter ) {
195 inv_s.set_filter( init_filter );
196 has_init_filter = false;
197 }
198 // Set position after filter to keep cursor at the right position
199 if( init_selection ) {
200 inv_s.select_position( init_pair );
201 init_selection = false;
202 }
203
204 if( inv_s.empty() ) {
205 const std::string msg = none_message.empty()
206 ? _( "You don't have the necessary item at hand." )
207 : none_message;
208 popup( msg, PF_GET_KEY );
209 return item_location();
210 }
211
212 item_location location = inv_s.execute();
213
214 if( inv_s.keep_open ) {
215 inv_s.keep_open = false;
216 continue;
217 }
218
219 if( u.has_activity( consuming ) ) {
220 u.activity.values.clear();
221 init_pair = inv_s.get_selection_position();
222 u.activity.values.push_back( init_pair.first );
223 u.activity.values.push_back( init_pair.second );
224 u.activity.str_values.clear();
225 u.activity.str_values.emplace_back( inv_s.get_filter() );
226 }
227
228 return location;
229
230 } while( true );

641 bool sort_compare( const inventory_entry &lhs, const inventory_entry &rhs ) const override {
642 time_duration time_a = get_time_left( lhs.any_item() );
643 time_duration time_b = get_time_left( rhs.any_item() );
644 int order_a = get_order( lhs.any_item(), time_a );
645 int order_b = get_order( rhs.any_item(), time_b );
646
647 return order_a < order_b
648 || ( order_a == order_b && time_a < time_b )
649 || ( order_a == order_b && time_a == time_b &&
650 inventory_selector_preset::sort_compare( lhs, rhs ) );
651 }
