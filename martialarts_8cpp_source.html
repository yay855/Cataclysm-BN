438bool ma_requirements::is_valid_character( const Character &u ) const
439{
440 for( const mabuff_id &buff_id : req_buffs ) {
441 if( !u.has_mabuff( buff_id ) ) {
442 return false;
443 }
444 }
445
446 //A technique is valid if it applies to unarmed strikes, if it applies generally
447 //to all weapons (such as Ninjutsu sneak attacks or innate weapon techniques like RAPID)
448 //or if the weapon is flagged as being compatible with the style. Some techniques have
449 //further restrictions on required weapon properties (is_valid_weapon).
450 bool cqb = u.has_active_bionic( bio_cqb );
451 // There are 4 different cases of "armedness":
452 // Truly unarmed, unarmed weapon, style-allowed weapon, generic weapon
453 bool melee_style = u.martial_arts_data->selected_strictly_melee();
454 bool is_armed = u.is_armed();
455 bool unarmed_weapon = is_armed && u.used_weapon().has_flag( flag_UNARMED_WEAPON );
456 bool forced_unarmed = u.martial_arts_data->selected_force_unarmed();
457 bool weapon_ok = is_valid_weapon( u.weapon );
458 bool style_weapon = u.martial_arts_data->selected_has_weapon( u.weapon.typeId() );
459 bool all_weapons = u.martial_arts_data->selected_allow_melee();
460
461 bool unarmed_ok = !is_armed || ( unarmed_weapon && unarmed_weapons_allowed );
462 bool melee_ok = melee_allowed && weapon_ok && ( style_weapon || all_weapons );
463
464 bool valid_unarmed = !melee_style && unarmed_allowed && unarmed_ok;
465 bool valid_melee = !strictly_unarmed && ( forced_unarmed || melee_ok );
466
467 if( !valid_unarmed && !valid_melee ) {
468 return false;
469 }
470
471 if( wall_adjacent && !get_map().is_wall_adjacent( u.pos() ) ) {
472 return false;
473 }
474
475 for( const auto &pr : min_skill ) {
476 if( ( cqb ? 5 : u.get_skill_level( pr.first ) ) < pr.second ) {
477 return false;
478 }
479 }
480
481 return true;
482}

1051{
1052 for( auto &elem : container ) {
1053 for( auto &eff : elem.second ) {
1054 const effect &e = eff.second;
1055 if( e.is_removed() ) {
1056 continue;
1057 }
1058 if( auto buff = ma_buff::from_effect( e ) ) {
1059 f( *buff, e );
1060 }
1061 }
1062 }

1067{
1068 for( auto &elem : container ) {

1070 const effect &e = eff.second;
1071 if( e.is_removed() ) {
1072 continue;
1073 }
1074 if( auto buff = ma_buff::from_effect( e ) ) {
1075 if( f( *buff, e ) ) {
1076 return true;
1077 }
1078 }
1079 }
1080 }
1081 return false;
