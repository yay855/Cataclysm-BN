685 def.create_name( _( info.name ) );
686 def.color = info.color;

688 def.materials.push_back( info.material );

691 // Set up the basic weapon type
692 const artifact_weapon_datum &weapon = artifact_weapon_data[info.base_weapon];
693 def.melee[DT_BASH] = rng( weapon.bash_min, weapon.bash_max );
694 def.melee[DT_CUT] = rng( weapon.cut_min, weapon.cut_max );
695 def.melee[DT_STAB] = rng( weapon.stab_min, weapon.stab_max );
696 def.m_to_hit = rng( weapon.to_hit_min, weapon.to_hit_max );
697 if( !weapon.tag.empty() ) {
698 def.item_tags.insert( weapon.tag );
699 }
700 // Add an extra weapon perhaps?
701 if( one_in( 2 ) ) {
702 const artifact_weapon_type select = random_entry_ref( info.extra_weapons );
703 if( select != NUM_ARTWEAPS ) {
704 const artifact_weapon_datum &weapon = artifact_weapon_data[select];
705 def.volume += weapon.volume;
706 def.weight += weapon.weight;
707 def.melee[DT_BASH] += rng( weapon.bash_min, weapon.bash_max );
708 def.melee[DT_CUT] += rng( weapon.cut_min, weapon.cut_max );
709 def.melee[DT_STAB] += rng( weapon.stab_min, weapon.stab_max );
710 def.m_to_hit += rng( weapon.to_hit_min, weapon.to_hit_max );
711 if( !weapon.tag.empty() ) {
712 def.item_tags.insert( weapon.tag );
713 }
714 def.create_name( std::string( _( weapon.adjective ) ) + " " + _( info.name ) );
715 }
716 }
717 def.description = no_translation(
718 string_format(
719 _( "This is the %s.\nIt is the only one of its kind.\nIt may have unknown powers; try activating them." ),
720 def.nname( 1 ) ) );
721
722 // Finally, pick some powers
723 art_effect_passive passive_tmp = AEP_NULL;
724 art_effect_active active_tmp = AEA_NULL;
725 int num_good = 0;
726 int num_bad = 0;
727 int value = 0;
728 std::vector<art_effect_passive> good_effects = fill_good_passive();
729 std::vector<art_effect_passive> bad_effects = fill_bad_passive();
730
731 // Wielded effects first
732 while( !good_effects.empty() && !bad_effects.empty() &&
733 num_good < 3 && num_bad < 3 &&
734 ( num_good < 1 || num_bad < 1 || one_in( num_good + 1 ) ||
735 one_in( num_bad + 1 ) || value > 1 ) ) {
736 if( value < 1 && one_in( 2 ) ) {
737 // Good
738 passive_tmp = random_entry_removed( good_effects );
739 num_good++;
740 } else if( !bad_effects.empty() ) {
741 // Bad effect
742 passive_tmp = random_entry_removed( bad_effects );
743 num_bad++;
744 }
745 value += passive_effect_cost[passive_tmp];
746 def.artifact->effects_wielded.push_back( passive_tmp );
747 }
748 // Next, carried effects; more likely to be just bad
749 num_good = 0;
750 num_bad = 0;
751 value = 0;
752 good_effects = fill_good_passive();
753 bad_effects = fill_bad_passive();
754 while( one_in( 2 ) && !good_effects.empty() && !bad_effects.empty() &&
755 num_good < 3 && num_bad < 3 &&
756 ( num_bad < 1 || one_in( num_bad + 1 ) || value > 1 ) ) {
757 if( value < 1 && one_in( 3 ) ) {
758 // Good
759 passive_tmp = random_entry_removed( good_effects );
760 num_good++;
761 } else {
762 // Bad effect
763 passive_tmp = random_entry_removed( bad_effects );
764 num_bad++;
765 }
766 value += passive_effect_cost[passive_tmp];
767 def.artifact->effects_carried.push_back( passive_tmp );
768 }
769 // Finally, activated effects; not necessarily good or bad
770 num_good = 0;
771 num_bad = 0;
772 value = 0;
773 std::vector<art_effect_active> good_a_effects = fill_good_active();
774 std::vector<art_effect_active> bad_a_effects = fill_bad_active();
775 while( !good_a_effects.empty() && !bad_a_effects.empty() &&
776 num_good < 3 && num_bad < 3 &&
777 ( value > 3 || ( num_bad > 0 && num_good == 0 ) ||
778 !one_in( 3 - num_good ) || !one_in( 3 - num_bad ) ) ) {
779 if( !one_in( 3 ) && value <= 1 ) {
780 // Good effect
781 active_tmp = random_entry_removed( good_a_effects );
782 num_good++;
783 value += active_effect_cost[active_tmp];
784 } else {
785 // Bad effect
786 active_tmp = random_entry_removed( bad_a_effects );
787 num_bad++;
788 value += active_effect_cost[active_tmp];
789 }
790 def.artifact->effects_activated.push_back( active_tmp );
791 def.tool->max_charges += rng( 1, 3 );
792 }
793 def.tool->def_charges = def.tool->max_charges;
794 // If we have charges, pick a recharge mechanism
795 if( def.tool->max_charges > 0 ) {
796 def.artifact->charge_type = static_cast<art_charge>( rng( ARTC_NULL + 1, NUM_ARTCS - 1 ) );
797 }
798 // 1 in 8 chance that it can't recharge!
799 if( one_in( 8 ) && num_bad + num_good >= 4 ) {
800 def.artifact->charge_type = ARTC_NULL;
801 }
802 // Maybe pick an extra recharge requirement
803 if( one_in( std::max( 1, 6 - num_good ) ) && def.artifact->charge_type != ARTC_NULL ) {
804 def.artifact->charge_req = static_cast<art_charge_req>( rng( ACR_NULL + 1, NUM_ACRS - 1 ) );
805 }
806 // Assign dream data (stored individually so they can be overridden in json)
807 def.artifact->dream_msg_unmet = artifact_dream_data[static_cast<int>
808 ( def.artifact->charge_req )].msg_unmet;
809 def.artifact->dream_msg_met = artifact_dream_data[static_cast<int>
810 ( def.artifact->charge_req )].msg_met;
811 def.artifact->dream_freq_unmet = artifact_dream_data[static_cast<int>
812 ( def.artifact->charge_req )].freq_unmet;
813 def.artifact->dream_freq_met = artifact_dream_data[static_cast<int>
814 ( def.artifact->charge_req )].freq_met;
815 // Stronger artifacts have a higher chance of picking their dream
816 def.artifact->dream_freq_unmet *= ( 1 + 0.1 * ( num_bad + num_good ) );
817 def.artifact->dream_freq_met *= ( 1 + 0.1 * ( num_bad + num_good ) );
818
819 item_controller->add_item_type( static_cast<itype &>( def ) );
820 return def.get_id();
821 } else {

823 it_artifact_armor def;

826
827 def.create_name( _( info.name ) );
828 // Armor is always [

830 def.color = info.color;

832 def.volume = info.volume;

842 def.armor->env_resist = info.env_resist;
843 def.armor->warmth = info.warmth;
844 def.armor->storage = info.storage;

846 _( "This is the %s.\nThey are the only ones of their kind." ) :
847 _( "This is the %s.\nIt is the only one of its kind." ), def.nname( 1 ) );
848
849 // Modify the armor further
850 if( !one_in( 4 ) ) {
851 const artifact_armor_mod mod = random_entry_ref( info.available_mods );
852 if( mod != ARMORMOD_NULL ) {
853 const artifact_armor_form_datum &modinfo = artifact_armor_mod_data[mod];

863 def.weight = 1_gram;

871 def.armor->coverage = 0;

873

875 def.armor->thickness += modinfo.thickness;
876 } else {
877 def.armor->thickness = 0;
878 }
879

883 def.armor->env_resist = 0;

913 !one_in( 3 - num_bad ) ) ) {
914 if( value < 1 && one_in( 2 ) ) {
915 // Good effect
916 passive_tmp = random_entry_removed( good_effects );
917 num_good++;
918 } else {

934 it_artifact_tool def;
935
936 // Pick a form
937 const artifact_shape_datum &shape_data = random_entry_ref( artifact_shape_data );
938 // Pick a property
939 const artifact_natural_property property = ( prop > ARTPROP_NULL ? prop :
940 static_cast<artifact_natural_property>( rng( ARTPROP_NULL + 1,
941 ARTPROP_MAX - 1 ) ) );
942 const artifact_property_datum &property_data = artifact_property_data[property];
943
944 def.sym = ":";
945 def.color = c_yellow;
946 def.materials.push_back( material_id( "stone" ) );
947 def.volume = rng( shape_data.volume_min, shape_data.volume_max );
948 def.weight = rng( shape_data.weight_min, shape_data.weight_max );
949 def.melee[DT_BASH] = 0;
950 def.melee[DT_CUT] = 0;
951 def.m_to_hit = 0;
952
953 def.create_name( _( property_data.name ), _( shape_data.name ) );
954 def.description = no_translation(
955 string_format( pgettext( "artifact description", "This %1$s %2$s." ),
956 _( shape_data.desc ), _( property_data.desc ) ) );
957
958 // Three possibilities: good passive + bad passive, good active + bad active,
959 // and bad passive + good active
960 bool good_passive = false, bad_passive = false,
961 good_active = false, bad_active = false;
962 switch( rng( 1, 3 ) ) {
963 case 1:
964 good_passive = true;
965 bad_passive = true;
966 break;
967 case 2:
968 good_active = true;
969 bad_active = true;
970 break;
971 case 3:
972 bad_passive = true;
973 good_active = true;
974 break;
975 }
976
977 // This is slowly incremented, allowing for better arts
978 int value_to_reach = 0;
979 int value = 0;
980 art_effect_passive aep_good = AEP_NULL, aep_bad = AEP_NULL;
981 art_effect_active aea_good = AEA_NULL, aea_bad = AEA_NULL;
982
983 do {
984 if( good_passive ) {
985 aep_good = random_entry_ref( property_data.passive_good );
986 if( aep_good == AEP_NULL || one_in( 4 ) ) {
987 aep_good = static_cast<art_effect_passive>( rng( AEP_NULL + 1, AEP_SPLIT - 1 ) );
988 }
989 }
990 if( bad_passive ) {
991 aep_bad = random_entry_ref( property_data.passive_bad );
992 if( aep_bad == AEP_NULL || one_in( 4 ) ) {
993 aep_bad = static_cast<art_effect_passive>( rng( AEP_SPLIT + 1, NUM_AEAS - 1 ) );
994 }
995 }
996 if( good_active ) {
997 aea_good = random_entry_ref( property_data.active_good );
998 if( aea_good == AEA_NULL || one_in( 4 ) ) {
999 aea_good = static_cast<art_effect_active>( rng( AEA_NULL + 1, AEA_SPLIT - 1 ) );
1000 }
1001 }
1002 if( bad_active ) {
1003 aea_bad = random_entry_ref( property_data.active_bad );
1004 if( aea_bad == AEA_NULL || one_in( 4 ) ) {
1005 aea_bad = static_cast<art_effect_active>( rng( AEA_SPLIT + 1, NUM_AEAS - 1 ) );
1006 }
1007 }
1008
1009 value = passive_effect_cost[aep_good] + passive_effect_cost[aep_bad] +
1010 active_effect_cost[aea_good] + active_effect_cost[aea_bad];
1011 // Yes, it is intentional that this is 1 the first check
1012 value_to_reach++;
1013 } while( value > value_to_reach );
1014
1015 if( aep_good != AEP_NULL ) {
1016 def.artifact->effects_carried.push_back( aep_good );
1017 }
1018 if( aep_bad != AEP_NULL ) {
1019 def.artifact->effects_carried.push_back( aep_bad );
1020 }
1021 if( aea_good != AEA_NULL ) {
1022 def.artifact->effects_activated.push_back( aea_good );
1023 }
1024 if( aea_bad != AEA_NULL ) {
1025 def.artifact->effects_activated.push_back( aea_bad );
1026 }
1027
1028 // Natural artifacts ALWAYS can recharge
1029 // (When "implanting" them in a mundane item, this ability may be lost
1030 if( !def.artifact->effects_activated.empty() ) {
1031 def.tool->def_charges = def.tool->max_charges = rng( 1, 4 );
1032 def.artifact->charge_type = static_cast<art_charge>( rng( ARTC_NULL + 1, NUM_ARTCS - 1 ) );
1033 //Maybe pick an extra recharge requirement
1034 if( one_in( 6 ) ) {
1035 def.artifact->charge_req = static_cast<art_charge_req>( rng( ACR_NULL + 1, NUM_ACRS - 1 ) );
1036 }
