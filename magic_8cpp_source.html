345static bool spell_infinite_loop_check( std::set<spell_id> spell_effects, const spell_id &sp )
346{
347 if( spell_effects.count( sp ) ) {
348 return true;
349 }
350 spell_effects.emplace( sp );
351
352 std::set<spell_id> unique_spell_list;
353 for( const fake_spell &fake_sp : sp->additional_spells ) {
354 unique_spell_list.emplace( fake_sp.id );
355 }
356
357 for( const spell_id &other_sp : unique_spell_list ) {
358 if( spell_infinite_loop_check( spell_effects, other_sp ) ) {
359 return true;
360 }
361 }
362 return false;
363}

365void spell_type::check_consistency()
366{
367 for( const spell_type &sp_t : get_all() ) {
368 if( ( sp_t.min_aoe > sp_t.max_aoe && sp_t.aoe_increment > 0 ) ||
369 ( sp_t.min_aoe < sp_t.max_aoe && sp_t.aoe_increment < 0 ) ) {
370 debugmsg( "ERROR: %s has higher min_aoe than max_aoe", sp_t.id.c_str() );
371 }
372 if( ( sp_t.min_damage > sp_t.max_damage && sp_t.damage_increment > 0 ) ||
373 ( sp_t.min_damage < sp_t.max_damage && sp_t.damage_increment < 0 ) ) {
374 debugmsg( "ERROR: %s has higher min_damage than max_damage", sp_t.id.c_str() );
375 }
376 if( ( sp_t.min_range > sp_t.max_range && sp_t.range_increment > 0 ) ||
377 ( sp_t.min_range < sp_t.max_range && sp_t.range_increment < 0 ) ) {
378 debugmsg( "ERROR: %s has higher min_range than max_range", sp_t.id.c_str() );
379 }
380 if( ( sp_t.min_dot > sp_t.max_dot && sp_t.dot_increment > 0 ) ||
381 ( sp_t.min_dot < sp_t.max_dot && sp_t.dot_increment < 0 ) ) {
382 debugmsg( "ERROR: %s has higher min_dot than max_dot", sp_t.id.c_str() );
383 }
384 if( ( sp_t.min_duration > sp_t.max_duration && sp_t.duration_increment > 0 ) ||
385 ( sp_t.min_duration < sp_t.max_duration && sp_t.duration_increment < 0 ) ) {
386 debugmsg( "ERROR: %s has higher min_dot_time than max_dot_time", sp_t.id.c_str() );
387 }
388 if( ( sp_t.min_pierce > sp_t.max_pierce && sp_t.pierce_increment > 0 ) ||
389 ( sp_t.min_pierce < sp_t.max_pierce && sp_t.pierce_increment < 0 ) ) {
390 debugmsg( "ERROR: %s has higher min_pierce than max_pierce", sp_t.id.c_str() );
391 }
392 if( sp_t.casting_time_increment < 0.0f && sp_t.base_casting_time < sp_t.final_casting_time ) {
393 debugmsg( "ERROR: %s has negative increment and base_casting_time < final_casting_time",
394 sp_t.id.c_str() );
395 }
396 if( sp_t.casting_time_increment > 0.0f && sp_t.base_casting_time > sp_t.final_casting_time ) {
397 debugmsg( "ERROR: %s has positive increment and base_casting_time > final_casting_time",
398 sp_t.id.c_str() );
399 }
400 if( sp_t.effect_name == "summon_vehicle" ) {
401 if( !sp_t.effect_str.empty() && !vproto_id( sp_t.effect_str ).is_valid() ) {
402 debugmsg( "ERROR %s specifies a vehicle to summon, but vehicle %s is not valid", sp_t.id.c_str(),
403 sp_t.effect_str );
404 }
405 }
406 std::set<spell_id> spell_effect_list;
407 if( spell_infinite_loop_check( spell_effect_list, sp_t.id ) ) {
408 debugmsg( "ERROR: %s has infinite loop in extra_effects", sp_t.id.c_str() );
409 }
410 if( sp_t.field ) {
411 if( sp_t.field_chance <= 0 ) {
412 debugmsg( "ERROR: %s must have a positive field chance.", sp_t.id.c_str() );
413 }
414 if( sp_t.field_intensity_increment > 0 && sp_t.max_field_intensity < sp_t.min_field_intensity ) {
415 debugmsg( "ERROR: max_field_intensity must be greater than min_field_intensity with positive increment: %s",
416 sp_t.id.c_str() );
417 } else if( sp_t.field_intensity_increment < 0 &&
418 sp_t.max_field_intensity > sp_t.min_field_intensity ) {
419 debugmsg( "ERROR: min_field_intensity must be greater than max_field_intensity with negative increment: %s",
420 sp_t.id.c_str() );
421 }
422 }
423 if( sp_t.spell_tags[spell_flag::WONDER] && sp_t.additional_spells.empty() ) {
424 debugmsg( "ERROR: %s has WONDER flag but no spells to choose from!", sp_t.id.c_str() );
425 }
426 }

742float spell::spell_fail( const Character &guy ) const
743{
744 if( has_flag( spell_flag::NO_FAIL ) ) {
745 return 0.0f;
746 }
747 // formula is based on the following:
748 // exponential curve
749 // effective skill of 0 or less is 100% failure
750 // effective skill of 8 (8 int, 0 spellcraft, 0 spell level, spell difficulty 0) is ~50% failure
751 // effective skill of 30 is 0% failure
752 const float effective_skill = 2 * ( get_level() - get_difficulty() ) + guy.get_int() +
753 guy.get_skill_level( skill() );
754 // add an if statement in here because sufficiently large numbers will definitely overflow because of exponents
755 if( effective_skill > 30.0f ) {
756 return 0.0f;
757 } else if( effective_skill < 0.0f ) {
758 return 1.0f;
759 }
760 float fail_chance = std::pow( ( effective_skill - 30.0f ) / 30.0f, 2 );
761 if( has_flag( spell_flag::SOMATIC ) && !guy.has_trait_flag( "SUBTLE_SPELL" ) ) {
762 // the first 20 points of encumbrance combined is ignored
763 const int arms_encumb = std::max( 0, guy.encumb( bp_arm_l ) + guy.encumb( bp_arm_r ) - 20 );
764 // each encumbrance point beyond the "gray" color counts as half an additional fail %
765 fail_chance += arms_encumb / 200.0f;
766 }
767 if( has_flag( spell_flag::VERBAL ) && !guy.has_trait_flag( "SILENT_SPELL" ) ) {
768 // a little bit of mouth encumbrance is allowed, but not much
769 const int mouth_encumb = std::max( 0, guy.encumb( bp_mouth ) - 5 );
770 fail_chance += mouth_encumb / 100.0f;
771 }
772 // concentration spells work better than you'd expect with a higher focus pool
773 if( has_flag( spell_flag::CONCENTRATE ) ) {
774 if( guy.focus_pool <= 0 ) {
775 return 0.0f;
776 }
777 fail_chance /= guy.focus_pool / 100.0f;
778 }
779 return clamp( fail_chance, 0.0f, 1.0f );
780}

785 std::string fail_str;

787 fail_fl, _( "Failure Chance" ) );

1045// helper function to calculate xp needed to be at a certain level
1046// pulled out as a helper function to make it easier to either be used in the future
1047// or easier to tweak the formula
1048static int exp_for_level( int level )
1049{
1050 // level 0 never needs xp

1052 return 0;

1062std::string spell::exp_progress() const
1063{
1064 const int level = get_level();
1065 const int this_level_xp = exp_for_level( level );
1066 const int next_level_xp = exp_for_level( level + 1 );
1067 const int denominator = next_level_xp - this_level_xp;
1068 const float progress = static_cast<float>( xp() - this_level_xp ) / std::max( 1.0f,
1069 static_cast<float>( denominator ) );
1070 return string_format( "%i%%", clamp( static_cast<int>( std::round( progress * 100 ) ), 0, 99 ) );
1071}

1201 sp.cast_all_effects( source, *new_target );

1205 sp.cast_all_effects( source, source.pos() );
1206 } else {
1207 sp.cast_all_effects( source, target );
1208 }
1209 }
1210 }
1211 } else {

1213 cast_spell_effect( source, target );
1214 for( const fake_spell &extra_spell : type->additional_spells ) {

1218 extra_spell.self ? source.pos() : target ) ) {

1220 }

1223 sp.cast_all_effects( source, source.pos() );
