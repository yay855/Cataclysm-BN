32/**
33 * @defgroup string_formatter_convert Convert functions for @ref string_formatter
34 *
35 * The `convert` functions here are used to convert the input value of
36 * @ref string_formatter::parse into the requested type, as defined by the format specifiers.
37 *
38 * @tparam T the input type, as given by the call to `string_format`.
39 * @tparam RT the requested type. The `convert` functions return such a value or they throw
40 * an exception via @ref throw_error.
41 *
42 * Each function has the same parameters:
43 * First parameter defined the requested type. The value of the pointer is ignored, callers
44 * should use a (properly casted) `nullptr`. It is required to "simulate" overloading the
45 * return value. E.g. `long convert(long*, int)` and `short convert(short*, int)` both convert
46 * a input value of type `int`, but the first converts to `long` and the second converts to
47 * `short`. Without the first parameters their signature would be identical.
48 * The second parameter is used to call @ref throw_error / @ref string_formatter_set_temp_buffer.
49 * The third parameter is the input value that is to be converted.
50 * The fourth parameter is a dummy value, it is always ignored, callers should use `0` here.
51 * It is used so the fallback with the variadic arguments is *only* chosen when no other
52 * overload matches.
53 */

180/**
181 * Type-safe and undefined-behavior free wrapper over `sprintf`.
182 * See @ref string_format for usage.
183 * Basically it extracts the format specifiers and calls `sprintf` for each one separately
184 * and with proper conversion of the input type.
185 * For example `printf("%f", 7)` would yield undefined behavior as "%f" requires a `double`
186 * as argument. This class detects the format specifier and converts the input to `double`
187 * before calling `sprintf`. Similar for `printf("%d", "foo")` (yields UB again), but this
188 * class will just throw an exception.
189 */
190// Note: argument index is always 0-based *in this code*, but `printf` has 1-based arguments.
191class string_formatter
192{
193 private:
194 /// Complete format string, including all format specifiers (the string passed
195 /// to @ref printf).
196 const std::string format;
197 /// Used during parsing to denote the *next* character in @ref format to be
198 /// parsed.
199 size_t current_index_in_format = 0;
200 /// The formatted output string, filled during parsing of @ref format,
201 /// so it's only valid after the parsing has completed.
202 std::string output;
203 /// The *currently parsed format specifiers. This is extracted from @ref format
204 /// during parsing and given to @ref sprintf (along with the actual argument).
205 /// It is filled and reset during parsing for each format specifier in @ref format.
206 std::string current_format;
207 /// The *index* (not number) of the next argument to be formatted via @ref current_format.
208 int current_argument_index = 0;
209 /// Return the next character from @ref format and increment @ref current_index_in_format.
210 /// Returns a null-character when the end of the @ref format has been reached (and does not
211 /// change @ref current_index_in_format).
212 char consume_next_input();
213 /// Returns (like @ref consume_next_input) the next character from @ref format, but
214 /// does *not* change @ref current_index_in_format.
215 char get_current_input() const;
216 /// If the next character to read from @ref format is the given character, consume it
217 /// (like @ref consume_next_input) and return `true`. Otherwise don't do anything at all
218 /// and return `false`.
219 bool consume_next_input_if( char c );
220 /// Return whether @ref get_current_input has a decimal digit ('0'...'9').
221 bool has_digit() const;
222 /// Consume decimal digits, interpret them as integer and return it.
223 /// A starting '0' is allowed. Leaves @ref format at the first non-digit
224 /// character (or the end). Returns 0 if the first character is not a digit.
225 int parse_integer();
226 /// Read and consume format flag characters and append them to @ref current_format.
227 /// Leaves @ref format at the first character that is not a flag (or the end).
228 void read_flags();
229 /// Read and forward to @ref current_format any width specifier from @ref format.
230 /// Returns nothing if the width is not specified or if it is specified as fixed number,
231 /// otherwise returns the index of the printf-argument to be used for the width.
232 cata::optional<int> read_width();
233 /// See @ref read_width. This does the same, but for the precision specifier.
234 cata::optional<int> read_precision();
235 /// Read and return the index of the printf-argument that is to be formatted. Returns
236 /// nothing if @ref format does not refer to a specific index (caller should use
237 /// @ref current_argument_index).

239 // Helper for common logic in @ref read_width and @ref read_precision.
240 cata::optional<int> read_number_or_argument_index();

256 /**
257 * Extracts a printf argument from the argument list and converts it to the requested type.
258 * @tparam RT The type that the argument should be converted to.

283 template<typename ...Args>
284 void read_conversion( const int format_arg_index, Args &&... args ) {
285 // Removes the prefix "ll", "l", "h" and "hh", "z", and "t".
286 // We later add "ll" again and that
287 // would interfere with the existing prefix. We convert *all* input to (un)signed
288 // long long int and use the "ll" modifier all the time. This will print the
289 // expected value all the time, even when the original modifier did not match.
290 if( consume_next_input_if( 'l' ) ) {
291 consume_next_input_if( 'l' );
292 } else if( consume_next_input_if( 'h' ) ) {
293 consume_next_input_if( 'h' );
294 } else if( consume_next_input_if( 'z' ) ) {
295 // done with it
296 } else if( consume_next_input_if( 't' ) ) {
297 // done with it
298 }
299 const char c = consume_next_input();
300 current_format.push_back( c );
301 switch( c ) {
302 case 'c':
303 return do_formating( get_nth_arg_as<int, 0>( format_arg_index, std::forward<Args>( args )... ) );
304 case 'd':
305 case 'i':
306 add_long_long_length_modifier();
307 return do_formating( get_nth_arg_as<signed long long int, 0>( format_arg_index,
308 std::forward<Args>( args )... ) );
309 case 'o':
310 case 'x':
311 case 'X':
312 // Workaround for fmtlib prepending number with ' '/'+'
313 // when formatting with ' '/'+' flags and 'o'/'x'/'X' specifiers
314 discard_oct_hex_sign_flag();
315 // intentional fall-through
316 case 'u':
317 add_long_long_length_modifier();
318 return do_formating( get_nth_arg_as<unsigned long long int, 0>( format_arg_index,
319 std::forward<Args>( args )... ) );
320 case 'a':
321 case 'A':
322 case 'g':
323 case 'G':
324 case 'f':
325 case 'F':
326 case 'e':
327 case 'E':
328 return do_formating( get_nth_arg_as<double, 0>( format_arg_index, std::forward<Args>( args )... ) );
329 case 'p':
330 return do_formating( get_nth_arg_as<void *, 0>( format_arg_index,
331 std::forward<Args>( args )... ) );
332 case 's':
333 return do_formating( get_nth_arg_as<const char *, 0>( format_arg_index,
334 std::forward<Args>( args )... ) );
335 default:
336 throw_error( "Unsupported format conversion: " + std::string( 1, c ) );
337 }
338 }

341 void do_formating( signed long long int value );
342 void do_formating( unsigned long long int value );
343 void do_formating( double value );
344 void do_formating( void *value );

347 public:
348 /// @param format The format string as required by `sprintf`.
349 string_formatter( std::string format ) : format( std::move( format ) ) { }
350 /// Does the actual `sprintf`. It uses @ref format and puts the formatted
351 /// string into @ref output.
352 /// Note: use @ref get_output to get the formatted string after a successful
353 /// call to this function.
354 /// @throws Exceptions when the arguments do not match the format specifiers,
355 /// see @ref get_nth_arg_as, or when the format is invalid for whatever reason.
356 /// Note: @ref string_format is a wrapper that handles those exceptions.
357 template<typename ...Args>
358 void parse( Args &&... args ) {
359 output.reserve( format.size() );
360 output.resize( 0 );
361 current_index_in_format = 0;
362 current_argument_index = 0;
363 while( const char c = consume_next_input() ) {
364 if( c != '%' ) {
365 output.push_back( c );
366 continue;
367 }
368 if( consume_next_input_if( '%' ) ) {
369 output.push_back( '%' );
370 continue;
371 }
372 current_format = "%";
373 const cata::optional<int> format_arg_index = read_argument_index();
374 read_flags();
375 if( const cata::optional<int> width_argument_index = read_width() ) {
376 const int w = get_nth_arg_as<int, 0>( *width_argument_index, std::forward<Args>( args )... );
377 current_format += std::to_string( w );
378 }
379 if( const cata::optional<int> precision_argument_index = read_precision() ) {
380 const int p = get_nth_arg_as<int, 0>( *precision_argument_index, std::forward<Args>( args )... );
381 current_format += std::to_string( p );
382 }
383 const int arg = format_arg_index ? *format_arg_index : current_argument_index++;
384 read_conversion( arg, std::forward<Args>( args )... );
385 }
386 }
387 std::string get_output() const {
388 return output;
389 }

394/**
395 * Simple wrapper over @ref string_formatter::parse. It catches any exceptions and returns
396 * some error string. Otherwise it just returns the formatted string.
397 *
398 * These functions perform string formatting according to the rules of the `printf` function,
399 * see `man 3 printf` or any other documentation.
400 *
401 * In short: the \p format parameter is a string with optional placeholders, which will be
402 * replaced with formatted data from the further arguments. The further arguments must have
403 * a type that matches the type expected by the placeholder.
404 * The placeholders look like this:
405 * - `%s` expects an argument of type `const char*` or `std::string` or numeric (which is
406 * converted to a string via `std::to_string`), which is inserted as is.
407 * - `%d` expects an argument of an integer type (int, short, ...), which is formatted as
408 * decimal number.
409 * - `%f` expects a numeric argument (integer / floating point), which is formatted as
410 * decimal number.
411 *
412 * There are more placeholders and options to them (see documentation of `printf`).
413 * Note that this wrapper (via @ref string_formatter) automatically converts the arguments
414 * to match the given format specifier (if possible) - see @ref string_formatter_convert.
415 */
416/**@{*/
417template<typename ...Args>
418inline std::string string_format( std::string format, Args &&...args )
419{
420 try {
421 cata::string_formatter formatter( std::move( format ) );
422 formatter.parse( std::forward<Args>( args )... );
423 return formatter.get_output();
424 } catch( ... ) {
425 return cata::handle_string_format_error();
426 }
427}
428template<typename ...Args>
429inline std::string string_format( const char *const format, Args &&...args )
430{
431 return string_format( std::string( format ), std::forward<Args>( args )... );
432}
433template<typename T, typename ...Args>
434inline typename std::enable_if<cata::is_translation<T>::value, std::string>::type
435string_format( T &&format, Args &&...args )
436{
437 return string_format( format.translated(), std::forward<Args>( args )... );
438}
