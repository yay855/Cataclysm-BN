12/**
13 * @name Enumeration (de)serialization to/from string.
14 *
15 * @ref enum_to_string converts an enumeration value to a string (which can be written to JSON).
16 * The result must be an non-empty string.
17 *
18 * This function must be implemented somewhere for each enumeration type for
19 * which conversion is required.
20 *
21 * Such enums must also specialize enum_traits to specify their 'last' member.
22 *
23 * @ref string_to_enum converts the string value back into an enumeration value. The input
24 * is expected to be one of the outputs of @ref enum_to_string. If the given string does
25 * not match an enumeration, an @ref InvalidEnumString is thrown.
26 *
27 * @code string_to_enum<E>(enum_to_string<E>(X)) == X @endcode must yield true for all values
28 * of the enumeration E.
29 */

41template<typename E>
42std::unordered_map<std::string, E> build_enum_lookup_map()
43{
44 static_assert( std::is_enum<E>::value, "E should be an enum type" );
45 static_assert( has_enum_traits<E>::value, "enum E needs a specialization of enum_traits" );
46 std::unordered_map<std::string, E> result;
47
48 using Int = std::underlying_type_t<E>;
49 constexpr Int max = static_cast<Int>( enum_traits<E>::last );
50
51 for( Int i = 0; i < max; ++i ) {
52 E e = static_cast<E>( i );
53 auto inserted = result.emplace( enum_to_string( e ), e );
54 if( !inserted.second ) {
55 debugmsg( "repeated enum string %s (%d and %d)", inserted.first->first,
56 static_cast<Int>( inserted.first->second ), i );
57 abort();
58 }
59 }
60
61 return result;
62}

64// Helper function to do the lookup in an associative container
65template<typename C, typename E = typename C::mapped_type>
66inline E string_to_enum_look_up( const C &container, const std::string &data )
67{
68 const auto iter = container.find( data );
69 if( iter == container.end() ) {
70 throw InvalidEnumString( "Invalid enum string '" + data + "' for '" +
71 typeid( E ).name() + "'" );
72 }
73 return iter->second;
74}
