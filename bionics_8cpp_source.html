358 rep.warn( "cut_protec specifies unknown body part \"%s\"", it.first.str() );
359 }

362 if( !it.first.is_valid() ) {
363 rep.warn( "bullet_protec specifies unknown body part \"%s\"", it.first.str() );

368 rep.warn( "bash_protec specifies unknown body part \"%s\"", it.first.str() );
369 }

373 rep.warn( "uses undefined enchantment \"%s\"", eid.str() );

463void npc::check_or_use_weapon_cbm( const bionic_id &cbm_id )
464{
465 // if we're already using a bio_weapon, keep using it
466 if( cbm_weapon_index >= 0 ) {
467 return;
468 }
469 const float allowed_ratio = static_cast<int>( rules.cbm_reserve ) / 100.0f;
470 const units::energy free_power = get_power_level() - get_max_power_level() * allowed_ratio;
471 if( free_power <= 0_J ) {
472 return;
473 }
474
475 int index = 0;
476 bool found = false;
477 for( bionic &i : *my_bionics ) {
478 if( i.id == cbm_id && !i.powered ) {
479 found = true;
480 break;
481 }
482 index += 1;
483 }
484 if( !found ) {
485 return;
486 }
487 bionic &bio = ( *my_bionics )[index];
488
489 if( bio.info().has_flag( flag_BIONIC_GUN ) ) {
490 const item cbm_weapon = item( bio.info().fake_item );
491 bool not_allowed = !rules.has_flag( ally_rule::use_guns ) ||
492 ( rules.has_flag( ally_rule::use_silent ) && !cbm_weapon.is_silent() );
493 if( is_player_ally() && not_allowed ) {
494 return;
495 }
496
497 const int ups_charges = charges_of( itype_UPS );
498 int ammo_count = weapon.ammo_remaining();
499 const int ups_drain = weapon.get_gun_ups_drain();
500 if( ups_drain > 0 ) {
501 ammo_count = std::min( ammo_count, ups_charges / ups_drain );
502 }
503 const int cbm_ammo = free_power / bio.info().power_activate;
504
505 if( npc_ai::weapon_value( *this, weapon, ammo_count ) <
506 npc_ai::weapon_value( *this, cbm_weapon, cbm_ammo ) ) {
507 real_weapon = weapon;
508 weapon = cbm_weapon;
509 cbm_weapon_index = index;
510 }
511 } else if( bio.info().has_flag( flag_BIONIC_WEAPON ) && !weapon.has_flag( flag_NO_UNWIELD ) &&
512 free_power > bio.info().power_activate ) {
513 if( is_armed() ) {
514 stow_item( weapon );
515 }
516 if( get_player_character().sees( pos() ) ) {
517 add_msg( m_info, _( "%s activates their %s." ), disp_name(), bio.info().name );
518 }
519
520 weapon = item( bio.info().fake_item );
521 mod_power_level( -bio.info().power_activate );
522 bio.powered = true;
523 cbm_weapon_index = index;
524 }

532bool Character::activate_bionic( bionic &bio, bool eff_only )
533{
534 const bool mounted = is_mounted();
535 if( bio.incapacitated_time > 0_turns ) {
536 add_msg_if_player( m_info, _( "Your %s is shorting out and can't be activated." ),
537 bio.info().name );
538 return false;
539 }
540
541 // eff_only means only do the effect without messing with stats or displaying messages
542 if( !eff_only ) {
543 if( bio.powered ) {
544 // It's already on!
545 return false;
546 }
547 if( !enough_power_for( bio.id ) ) {
548 add_msg_if_player( m_info, _( "You don't have the power to activate your %s." ),
549 bio.info().name );
550 return false;
551 }
552
553 // HACK: burn_fuel() doesn't check for available fuel in remote source on start.
554 // If CBM is successfully activated, the check will occur when it actually tries to draw power
555 if( !bio.info().is_remote_fueled ) {
556 if( !burn_fuel( bio, true ) ) {
557 return false;
558 }
559 }
560
561 // We can actually activate now, do activation-y things
562 mod_power_level( -bio.info().power_activate );

1473{
1474 const cata::optional<float> &coverage_penalty = bio.info().coverage_power_gen_penalty;
1475 float effective_efficiency = fuel_efficiency;

1479 for( const std::pair< const bodypart_str_id, int > &elem : occupied_bodyparts ) {

1482 !i.has_flag( flag_SEMITANGIBLE ) &&
1483 !i.has_flag( flag_PERSONAL ) && !i.has_flag( flag_AURA ) ) {
1484 coverage += i.get_coverage();
1485 }
1486 }
1487 }
1488 effective_efficiency = fuel_efficiency * ( 1.0 - ( coverage / ( 100.0 *
1489 occupied_bodyparts.size() ) )
1490 * coverage_penalty.value() );
1491 }

1495/**
1496 * @param p the player
1497 * @param bio the bionic that is meant to be recharged.
1498 * @param amount the amount of power that is to be spent recharging the bionic.
1499 * @param factor multiplies the power cost per turn.
1500 * @param rate divides the number of turns we may charge (rate of 2 discharges in half the time).
1501 * @return indicates whether we successfully charged the bionic.
1502 */
1503static bool attempt_recharge( Character &p, bionic &bio, units::energy &amount, int factor = 1,
1504 int rate = 1 )
1505{
1506 const bionic_data &info = bio.info();
1507 units::energy power_cost = info.power_over_time * factor;
1508 bool recharged = false;
1509
1510 if( power_cost > 0_kJ ) {
1511 if( p.get_power_level() >= power_cost ) {
1512 // Set the recharging cost and charge the bionic.
1513 amount = power_cost;
1514 // This is our first turn of charging, so subtract a turn from the recharge delay.
1515 bio.charge_timer = info.charge_time - rate;
1516 recharged = true;
1517 }
1518 }
1519
1520 return recharged;

1882// bionic manipulation chance of success
1883int bionic_manip_cos( float adjusted_skill, int bionic_difficulty )
1884{
1885 if( g->u.has_trait( trait_DEBUG_BIONICS ) ) {
1886 return 100;
1887 }
1888
1889 int chance_of_success = 0;
1890 // we will base chance_of_success on a ratio of skill and difficulty
1891 // when skill=difficulty, this gives us 1. skill < difficulty gives a fraction.
1892 float skill_difficulty_parameter = static_cast<float>( adjusted_skill /
1893 ( 4.0 * bionic_difficulty ) );
1894
1895 // when skill == difficulty, chance_of_success is 50%. Chance of success drops quickly below that
1896 // to reserve bionics for characters with the appropriate skill. For more difficult bionics, the
1897 // curve flattens out just above 80%
1898 chance_of_success = static_cast<int>( ( 100 * skill_difficulty_parameter ) /
1899 ( skill_difficulty_parameter + std::sqrt( 1 / skill_difficulty_parameter ) ) );
1900
1901 return chance_of_success;

2387void Character::bionics_install_failure( const std::string &installer,
2388 int difficulty, int success, float adjusted_skill )
2389{
2390 // "success" should be passed in as a negative integer representing how far off we
2391 // were for a successful install. We use this to determine consequences for failing.
2392 success = std::abs( success );
2393
2394 // failure level is decided by how far off the character was from a successful install, and
2395 // this is scaled up or down by the ratio of difficulty/skill. At high skill levels (or low
2396 // difficulties), only minor consequences occur. At low skill levels, severe consequences
2397 // are more likely.
2398 int failure_level = static_cast<int>( std::sqrt( success * 4.0 * difficulty / adjusted_skill ) );
2399 int fail_type = ( failure_level > 5 ? 5 : failure_level );
2400
2401 if( installer != "NOT_MED" ) {
2402 //~"Complications" is USian medical-speak for "unintended damage from a medical procedure".
2403 add_msg( m_neutral, _( "%s training helps to minimize the complications." ),
2404 installer );
2405 // In addition to the bonus, medical residents know enough OR protocol to avoid botching.
2406 // Take MD and be immune to faulty bionics.
2407 if( fail_type > 3 ) {
2408 fail_type = rng( 1, 3 );
2409 }
2410 }
2411
2412 switch( fail_type ) {
2413 case 0:
2414 case 1:
2415 do_damage_for_bionic_failure( 2, 5 );
2416 break;
2417 case 2:
2418 case 3:
2419 do_damage_for_bionic_failure( 5, difficulty * 5 );
2420 break;
2421 case 4:
2422 case 5: {
2423 std::vector<bionic_id> valid;
2424 std::copy_if( begin( faulty_bionics ), end( faulty_bionics ), std::back_inserter( valid ),
2425 [&]( const bionic_id & id ) {
2426 return !has_bionic( id );
2427 } );
2428
2429 // We've got all the bad bionics!
2430 if( valid.empty() ) {
2431 if( has_max_power() ) {
2432 units::energy old_power = get_max_power_level();
2433 add_msg( m_bad, _( "%s lose power capacity!" ), disp_name() );
2434 set_max_power_level( units::from_kilojoule( rng( 0,
2435 units::to_kilojoule( get_max_power_level() ) - 25 ) ) );
2436 if( get_max_power_level() < units::from_kilojoule( 0 ) ) {
2437 set_max_power_level( units::from_kilojoule( 0 ) );
2438 }
2439 if( is_player() ) {
2440 g->memorial().add(
2441 pgettext( "memorial_male", "Lost %d units of power capacity." ),
2442 pgettext( "memorial_female", "Lost %d units of power capacity." ),
2443 units::to_kilojoule( old_power - get_max_power_level() ) );
2444 }
2445 // If no faults available and no power capacity, downgrade to second-worst complication.
2446 } else {
2447 do_damage_for_bionic_failure( 5, difficulty * 5 );
2448 break;
2449 }
2450 } else {
2451 const bionic_id &id = random_entry( valid );
2452 add_bionic( id );
2453 g->events().send<event_type::installs_faulty_cbm>( getID(), id );
2454 add_msg( m_bad,
2455 _( "Complication in installation caused a malfunction - %s. Uninstall it to clear the malfunction." ),
2456 id.obj().name );
2457 }
2458 }
2459 break;
2460 }
