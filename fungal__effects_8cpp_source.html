64void fungal_effects::fungalize( const tripoint &p, Creature *origin, double spore_chance )
65{
66 if( monster *const mon_ptr = g->critter_at<monster>( p ) ) {
67 monster &critter = *mon_ptr;
68 if( gm.u.sees( p ) &&
69 !critter.type->in_species( FUNGUS ) ) {
70 add_msg( _( "The %s is covered in tiny spores!" ), critter.name() );
71 }
72 if( !critter.make_fungus() ) {
73 // Don't insta-kill non-fungables. Jabberwocks, for example
74 critter.add_effect( effect_stunned, rng( 1_turns, 3_turns ) );
75 critter.apply_damage( origin, bodypart_id( "torso" ), rng( 25, 50 ) );
76 }
77 } else if( gm.u.pos() == p && gm.u.get_effect_int( effect_fungus ) < 3 ) {
78 // TODO: Make this accept NPCs when they understand fungals
79 player &pl = gm.u;
80 ///\EFFECT_DEX increases chance of knocking fungal spores away with your TAIL_CATTLE
81 ///\EFFECT_MELEE increases chance of knocking fungal sports away with your TAIL_CATTLE
82 if( pl.has_trait( trait_TAIL_CATTLE ) &&
83 one_in( 20 - pl.dex_cur - pl.get_skill_level( skill_melee ) ) ) {
84 pl.add_msg_if_player(
85 _( "The spores land on you, but you quickly swat them off with your tail!" ) );
86 return;
87 }
88 // Spores hit the player--is there any hope?
89 bool hit = false;
90 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_head, 3, 9_minutes, bp_head );
91 hit |= one_in( 2 ) && pl.add_env_effect( effect_spores, bp_torso, 3, 9_minutes, bp_torso );
92 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_arm_l, 3, 9_minutes, bp_arm_l );
93 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_arm_r, 3, 9_minutes, bp_arm_r );
94 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_leg_l, 3, 9_minutes, bp_leg_l );
95 hit |= one_in( 4 ) && pl.add_env_effect( effect_spores, bp_leg_r, 3, 9_minutes, bp_leg_r );
96 if( hit ) {
97 add_msg( m_warning, _( "You're covered in tiny spores!" ) );
98 }
99 } else if( gm.num_creatures() < 250 && x_in_y( spore_chance, 1.0 ) ) { // Spawn a spore
100 if( monster *const spore = gm.place_critter_at( mon_spore, p ) ) {
101 monster *origin_mon = dynamic_cast<monster *>( origin );
102 if( origin_mon != nullptr ) {
103 spore->make_ally( *origin_mon );
104 } else if( origin != nullptr && origin->is_player() &&
105 gm.u.has_trait( trait_THRESH_MYCUS ) ) {
106 spore->friendly = 1000;
107 }
108 }
109 } else {
110 spread_fungus( p );
111 }

138void fungal_effects::spread_fungus_one_tile( const tripoint &p, const int growth )
139{
140 bool converted = false;
141 // Terrain conversion
142 if( m.has_flag_ter( flag_DIGGABLE, p ) ) {
143 if( x_in_y( growth * 10, 100 ) ) {
144 m.ter_set( p, t_fungus );
145 converted = true;
146 }
147 } else if( m.has_flag( flag_SHRUB, p ) ) {
148 if( x_in_y( growth * 10, 200 ) ) {
149 m.ter_set( p, t_shrub_fungal );
150 converted = true;
151 } else if( x_in_y( growth, 1000 ) ) {
152 m.ter_set( p, t_marloss );
153 converted = true;
154 }
155 } else if( m.has_flag( flag_WALL, p ) && m.has_flag( flag_FLAMMABLE, p ) ) {
156 if( x_in_y( growth * 10, 5000 ) ) {
157 m.ter_set( p, t_fungus_wall );
158 converted = true;
159 }
160 } else if( m.has_flag( flag_THIN_OBSTACLE, p ) ) {
161 if( x_in_y( growth * 10, 150 ) ) {
162 m.ter_set( p, t_fungus_mound );
163 converted = true;
164 }
165 } else if( m.has_flag( flag_YOUNG, p ) ) {
166 if( x_in_y( growth * 10, 500 ) ) {
167 if( m.get_field_intensity( p, fd_fungal_haze ) != 0 ) {
168 if( x_in_y( growth * 10, 800 ) ) { // young trees are vulnerable
169 m.ter_set( p, t_fungus );
170 if( gm.place_critter_at( mon_fungal_blossom, p ) && gm.u.sees( p ) ) {
171 add_msg( m_warning, _( "The young tree blooms forth into a fungal blossom!" ) );
172 }
173 } else if( x_in_y( growth * 10, 400 ) ) {
174 m.ter_set( p, t_marloss_tree );
175 }
176 } else {
177 m.ter_set( p, t_tree_fungal_young );
178 }
179 converted = true;
180 }
181 } else if( m.has_flag( flag_TREE, p ) ) {
182 if( one_in( 10 ) ) {
183 if( m.get_field_intensity( p, fd_fungal_haze ) != 0 ) {
184 if( x_in_y( growth * 10, 100 ) ) {
185 m.ter_set( p, t_fungus );
186 if( gm.place_critter_at( mon_fungal_blossom, p ) && gm.u.sees( p ) ) {
187 add_msg( m_warning, _( "The tree blooms forth into a fungal blossom!" ) );
188 }
189 } else if( x_in_y( growth * 10, 600 ) ) {
190 m.ter_set( p, t_marloss_tree );
191 }
192 } else {
193 m.ter_set( p, t_tree_fungal );
194 }
195 converted = true;
196 }
197 }
198 // Furniture conversion
199 if( converted ) {
200 if( m.has_flag( flag_FLOWER, p ) ) {
201 m.furn_set( p, f_flower_fungal );
202 } else if( m.has_flag( flag_ORGANIC, p ) ) {
203 if( m.furn( p ).obj().movecost == -10 ) {
204 m.furn_set( p, f_fungal_mass );
205 } else {
206 m.furn_set( p, f_fungal_clump );
207 }
208 } else if( m.has_flag( flag_PLANT, p ) ) {
209 // Replace the (already existing) seed
210 // Can't use item_stack::only_item() since there might be fertilizer
211 map_stack items = m.i_at( p );
212 const map_stack::iterator seed = std::find_if( items.begin(), items.end(), []( const item & it ) {
213 return it.is_seed();

215 if( seed == items.end() || !seed->is_seed() ) {
216 dbg( DL::Warn ) << "No seed item in the PLANT terrain at position " << p;
217 } else {
218 *seed = item( "fungal_seeds", calendar::turn );
219 }

224void fungal_effects::spread_fungus( const tripoint &p )
225{
226 int growth = 1;
227 map &here = get_map();
228 for( const tripoint &tmp : here.points_in_radius( p, 1 ) ) {
229 if( tmp == p ) {
230 continue;
231 }
232 if( m.has_flag( flag_FUNGUS, tmp ) ) {
233 growth += 1;
234 }
235 }
236
237 if( !m.has_flag_ter( flag_FUNGUS, p ) ) {
238 spread_fungus_one_tile( p, growth );
239 } else {
240 // Everything is already fungus
241 if( growth == 9 ) {
242 return;
243 }
244 for( const tripoint &dest : here.points_in_radius( p, 1 ) ) {
245 // One spread on average
246 if( !m.has_flag( flag_FUNGUS, dest ) && one_in( 9 - growth ) ) {
247 //growth chance is 100 in X simplified
248 spread_fungus_one_tile( dest, 10 );
249 }
250 }
251 }
