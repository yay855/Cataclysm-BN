56bool Creature_tracker::add( shared_ptr_fast<monster> critter_ptr )
57{
58 assert( critter_ptr );
59 monster &critter = *critter_ptr;
60
61 if( critter.type->id.is_null() ) { // Don't want to spawn null monsters o.O
62 return false;
63 }
64
65 if( critter.type->has_flag( MF_VERMIN ) ) {
66 // Don't spawn vermin, they aren't implemented yet
67 return false;
68 }
69
70 if( const shared_ptr_fast<monster> existing_mon_ptr = find( critter.pos() ) ) {
71 // We can spawn stuff on hallucinations, but we need to kill them first
72 if( existing_mon_ptr->is_hallucination() ) {
73 existing_mon_ptr->die( nullptr );
74 // But don't remove - that would change the monster order and could segfault
75 } else if( critter.is_hallucination() ) {
76 return false;
77 } else {
78 debugmsg( "there's already a monster at %d,%d,%d", critter.pos().x, critter.pos().y,
79 critter.pos().z );
80 return false;
81 }
82 }
83
84 if( MonsterGroupManager::monster_is_blacklisted( critter.type->id ) ) {
85 return false;
86 }
87
88 monsters_list.emplace_back( critter_ptr );
89 monsters_by_location[critter.pos()] = critter_ptr;
90 add_to_faction_map( critter_ptr );
91 return true;
92}

113bool Creature_tracker::update_pos( const monster &critter, const tripoint &new_pos )
114{
115 if( critter.is_dead() ) {
116 // find ignores dead critters anyway, changing their position in the
117 // monsters_by_location map is useless.
118 remove_from_location_map( critter );
119 return true;
120 }
121
122 if( const shared_ptr_fast<monster> new_critter_ptr = find( new_pos ) ) {
123 auto &othermon = *new_critter_ptr;
124 if( othermon.is_hallucination() ) {
125 othermon.die( nullptr );
126 } else {
127 debugmsg( "update_zombie_pos: wanted to move %s to %d,%d,%d, but new location already has %s",
128 critter.disp_name(),
129 new_pos.x, new_pos.y, new_pos.z, othermon.disp_name() );
130 return false;
131 }
132 }
133
134 const auto iter = std::find_if( monsters_list.begin(), monsters_list.end(),
135 [&]( const shared_ptr_fast<monster> &ptr ) {
136 return ptr.get() == &critter;
137 } );
138 if( iter != monsters_list.end() ) {
139 monsters_by_location.erase( critter.pos() );
140 monsters_by_location[new_pos] = *iter;
141 return true;
142 } else {
143 const tripoint &old_pos = critter.pos();
144 // We're changing the x/y/z coordinates of a zombie that hasn't been added
145 // to the game yet. `add` will update monsters_by_location for us.
146 debugmsg( "update_zombie_pos: no %s at %d,%d,%d (moving to %d,%d,%d)",
147 critter.disp_name(),
148 old_pos.x, old_pos.y, old_pos.z, new_pos.x, new_pos.y, new_pos.z );
149 // Rebuild cache in case the monster actually IS in the game, just bugged
150 rebuild_cache();
151 return false;
152 }
153}

155void Creature_tracker::remove_from_location_map( const monster &critter )
156{
157 const auto pos_iter = monsters_by_location.find( critter.pos() );
158 if( pos_iter != monsters_by_location.end() && pos_iter->second.get() == &critter ) {
159 monsters_by_location.erase( pos_iter );
160 return;
161 }
162
163 // When it's not in the map at its current location, it might still be there under,
164 // another location, so look for it.
165 const auto iter = std::find_if( monsters_by_location.begin(), monsters_by_location.end(),
166 [&]( const decltype( monsters_by_location )::value_type & v ) {
167 return v.second.get() == &critter;
168 } );
169 if( iter != monsters_by_location.end() ) {
170 monsters_by_location.erase( iter );
171 }
172}

174void Creature_tracker::remove( const monster &critter )
175{
176 const auto iter = std::find_if( monsters_list.begin(), monsters_list.end(),
177 [&]( const shared_ptr_fast<monster> &ptr ) {
178 return ptr.get() == &critter;
179 } );
180 if( iter == monsters_list.end() ) {
181 debugmsg( "Tried to remove invalid monster %s", critter.name() );
182 return;
183 }
184
185 for( auto &pair : monster_faction_map_ ) {
186 const auto fac_iter = pair.second.find( *iter );
187 if( fac_iter != pair.second.end() ) {
188 // Need to do this manually because the shared pointer containing critter is kept valid
189 // within removed_ and so the weak pointer in monster_faction_map_ is also valid.
190 pair.second.erase( fac_iter );
191 break;
192 }
193 }
194 remove_from_location_map( critter );
195 removed_.push_back( *iter );
196 monsters_list.erase( iter );
197}

217void Creature_tracker::swap_positions( monster &first, monster &second )
218{
219 if( first.pos() == second.pos() ) {
220 return;
221 }
222
223 // Either of them may be invalid!
224 const auto first_iter = monsters_by_location.find( first.pos() );
225 const auto second_iter = monsters_by_location.find( second.pos() );
226 // implied: first_iter != second_iter
227
228 shared_ptr_fast<monster> first_ptr;
229 if( first_iter != monsters_by_location.end() ) {
230 first_ptr = first_iter->second;
231 monsters_by_location.erase( first_iter );
232 }
233
234 shared_ptr_fast<monster> second_ptr;
235 if( second_iter != monsters_by_location.end() ) {
236 second_ptr = second_iter->second;
237 monsters_by_location.erase( second_iter );
238 }
239 // implied: (first_ptr != second_ptr) or (first_ptr == nullptr && second_ptr == nullptr)
240
241 tripoint temp = second.pos();
242 second.spawn( first.pos() );
243 first.spawn( temp );
244
245 // If the pointers have been taken out of the list, put them back in.
246 if( first_ptr ) {
247 monsters_by_location[first.pos()] = first_ptr;
248 }
249 if( second_ptr ) {
250 monsters_by_location[second.pos()] = second_ptr;
251 }

254bool Creature_tracker::kill_marked_for_death()
255{
256 // Important: `Creature::die` must not be called after creature objects (NPCs, monsters) have
257 // been removed, the dying creature could still have a pointer (the killer) to another creature.
258 bool monster_is_dead = false;
259 // Copy the list so we can iterate the copy safely *and* add new monsters from within monster::die
260 // This happens for example with blob monsters (they split into two smaller monsters).
261 const auto copy = monsters_list;
262 for( const shared_ptr_fast<monster> &mon_ptr : copy ) {
263 assert( mon_ptr );
264 monster &critter = *mon_ptr;
265 if( !critter.is_dead() ) {
266 continue;
267 }
268 dbg( DL::Info ) << string_format( "cleanup_dead: critter %s hp:%d %s",
269 critter.pos().to_string(), critter.get_hp(), critter.name() );
270
271 critter.die( nullptr );
272 monster_is_dead = true;
273 }
274
275 return monster_is_dead;
276}
