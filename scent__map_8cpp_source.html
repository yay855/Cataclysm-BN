130bool scent_map::inbounds( const tripoint &p ) const
131{
132 // HACK: This weird long check here is a hack around the fact that scentmap is 2D
133 // A z-level can access scentmap if it is within SCENT_MAP_Z_REACH flying z-level move from player's z-level
134 // That is, if a flying critter could move directly up or down (or stand still) and be on same z-level as player
135 const int levz = gm.get_levz();
136 const bool scent_map_z_level_inbounds = ( p.z == levz ) ||
137 ( std::abs( p.z - levz ) == SCENT_MAP_Z_REACH &&
138 get_map().valid_move( p, tripoint( p.xy(), levz ), false, true ) );
139 if( !scent_map_z_level_inbounds ) {
140 return false;
141 }
142 static constexpr point scent_map_boundary_min{};
143 static constexpr point scent_map_boundary_max( MAPSIZE_X, MAPSIZE_Y );
144
145 static constexpr half_open_rectangle<point> scent_map_boundaries(
146 scent_map_boundary_min, scent_map_boundary_max );
147
148 return scent_map_boundaries.contains( p.xy() );
149}
150void scent_map::update( const tripoint &center, map &m )
151{
152 // Stop updating scent after X turns of the player not moving.
153 // Once wind is added, need to reset this on wind shifts as well.
154 if( !player_last_position || center != *player_last_position ) {
155 player_last_position.emplace( center );
156 player_last_moved = calendar::turn;
157 } else if( player_last_moved + 1000_turns < calendar::turn ) {
158 return;
159 }
160
161 //the block and reduce scent properties are folded into a single scent_transfer value here
162 //block=0 reduce=1 normal=5
163 scent_array<char> scent_transfer;
164
165 std::array < std::array < int, 3 + SCENT_RADIUS * 2 >, 1 + SCENT_RADIUS * 2 > new_scent;
166 std::array < std::array < int, 3 + SCENT_RADIUS * 2 >, 1 + SCENT_RADIUS * 2 > sum_3_scent_y;
167 std::array < std::array < char, 3 + SCENT_RADIUS * 2 >, 1 + SCENT_RADIUS * 2 > squares_used_y;
168
169 diagonal_blocks( &blocked_cache )[MAPSIZE_X][MAPSIZE_Y] = m.access_cache(
170 center.z ).vehicle_obstructed_cache;
171
172 // for loop constants
173 const int scentmap_minx = center.x - SCENT_RADIUS;
174 const int scentmap_maxx = center.x + SCENT_RADIUS;
175 const int scentmap_miny = center.y - SCENT_RADIUS;
176 const int scentmap_maxy = center.y + SCENT_RADIUS;
177
178 // The new scent flag searching function. Should be wayyy faster than the old one.
179 m.scent_blockers( scent_transfer, point( scentmap_minx - 1, scentmap_miny - 1 ),
180 point( scentmap_maxx + 1, scentmap_maxy + 1 ) );
181
182 for( int x = 0; x < SCENT_RADIUS * 2 + 3; ++x ) {
183 sum_3_scent_y[0][x] = 0;
184 squares_used_y[0][x] = 0;
185 sum_3_scent_y[SCENT_RADIUS * 2][x] = 0;
186 squares_used_y[SCENT_RADIUS * 2][x] = 0;
187 }
188
189 for( int x = 0; x < SCENT_RADIUS * 2 + 3; ++x ) {
190 for( int y = 0; y < SCENT_RADIUS * 2 + 1; ++y ) {
191
192 point abs( x + scentmap_minx - 1, y + scentmap_miny );
193
194 // remember the sum of the scent val for the 3 neighboring squares that can defuse into
195 sum_3_scent_y[y][x] = 0;
196 squares_used_y[y][x] = 0;
197 for( int i = abs.y - 1; i <= abs.y + 1; ++i ) {
198 sum_3_scent_y[y][x] += scent_transfer[abs.x][i] * grscent[abs.x][i];
199 squares_used_y[y][x] += scent_transfer[abs.x][i];
200 }
201 }
202 }
203
204 for( int x = 1; x < SCENT_RADIUS * 2 + 2; ++x ) {
205 for( int y = 0; y < SCENT_RADIUS * 2 + 1; ++y ) {
206 const point abs( x + scentmap_minx - 1, y + scentmap_miny );
207
208 int squares_used = squares_used_y[y][x - 1] + squares_used_y[y][x] + squares_used_y[y][x + 1];
209 int total = sum_3_scent_y[y][x - 1] + sum_3_scent_y[y][x] + sum_3_scent_y[y][x + 1];
210
211 //handle vehicle holes
212 if( blocked_cache[abs.x][abs.y].nw && scent_transfer[abs.x + 1][abs.y + 1] == 5 ) {
213 squares_used -= 4;
214 total -= 4 * grscent[abs.x + 1][abs.y + 1];
215 }
216 if( blocked_cache[abs.x][abs.y].ne && scent_transfer[abs.x - 1][abs.y + 1] == 5 ) {
217 squares_used -= 4;
218 total -= 4 * grscent[abs.x - 1][abs.y + 1];
219 }
220 if( blocked_cache[abs.x - 1][abs.y - 1].nw && scent_transfer[abs.x - 1][abs.y - 1] == 5 ) {
221 squares_used -= 4;
222 total -= 4 * grscent[abs.x - 1][abs.y - 1];
223 }
224 if( blocked_cache[abs.x + 1][abs.y - 1].ne && scent_transfer[abs.x + 1][abs.y - 1] == 5 ) {
225 squares_used -= 4;
226 total -= 4 * grscent[abs.x + 1][abs.y - 1];
227 }
228
229 //Lingering scent
230 int temp_scent = grscent[abs.x][abs.y] * ( 250 - squares_used *
231 scent_transfer[abs.x][abs.y] ) ;
232 temp_scent -= grscent[abs.x][abs.y] * scent_transfer[abs.x][abs.y] *
233 ( 45 - squares_used ) / 5;
234
235 new_scent[y][x] = ( temp_scent + total * scent_transfer[abs.x][abs.y] ) / 250;
236
237 }
238 }
239 for( int x = 1; x < SCENT_RADIUS * 2 + 2; ++x ) {
240 for( int y = 0; y < SCENT_RADIUS * 2 + 1; ++y ) {
241 grscent[x + scentmap_minx - 1 ][y + scentmap_miny] = new_scent[y][x];
242 }
243 }
