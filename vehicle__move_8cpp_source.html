88int vehicle::slowdown( int at_velocity ) const
89{
90 double mps = vmiph_to_mps( std::abs( at_velocity ) );
91
92 // slowdown due to air resistance is proportional to square of speed
93 double f_total_drag = coeff_air_drag() * mps * mps;
94 if( is_watercraft() ) {
95 // same with water resistance
96 f_total_drag += coeff_water_drag() * mps * mps;
97 } else if( !is_falling && !is_flying ) {
98 double f_rolling_drag = coeff_rolling_drag() * ( vehicles::rolling_constant_to_variable + mps );
99 if( vehicle_movement::is_on_rails( get_map(), *this ) ) {
100 // vehicles on rails don't skid
101 f_total_drag += f_rolling_drag;
102 } else {
103 // increase rolling resistance by up to 25x if the vehicle is skidding at right angle to facing
104 const double skid_factor = 1.0 + 24.0 * std::abs( units::sin( face.dir() - move.dir() ) );
105 f_total_drag += f_rolling_drag * skid_factor;
106 }
107 }
108 double accel_slowdown = f_total_drag / to_kilogram( total_mass() );
109 // converting m/s^2 to vmiph/s
110 int slowdown = mps_to_vmiph( accel_slowdown );
111 if( is_towing() ) {
112 vehicle *other_veh = tow_data.get_towed();
113 if( other_veh ) {
114 slowdown += other_veh->slowdown( at_velocity );
115 }
116 }
117 if( slowdown < 0 ) {
118 debugmsg( "vehicle %s has negative drag slowdown %d\n", name, slowdown );
119 }
120 add_msg( m_debug, "%s at %d vimph, f_drag %3.2f, drag accel %d vmiph - extra drag %d",
121 name, at_velocity, f_total_drag, slowdown, static_drag() );
122 // plows slow rolling vehicles, but not falling or floating vehicles
123 if( !( is_falling || is_floating || is_flying ) ) {
124 slowdown -= static_drag();
125 }
126
127 return std::max( 1, slowdown );

131{

141 if( ( in_water && can_float() ) || ( is_rotorcraft() && ( z != 0 || is_flying ) ) ) {

152 add_msg( _( "The %s doesn't have enough wheels to move!" ), name );

154 return;

157 bool thrusting = true;

164 float traction = k_traction( get_map().vehicle_wheel_traction( *this ) );
165 int accel = current_acceleration() * traction;

169 tow_data.get_towed()->name );

171 return;
172 }
173 if( thrusting && accel == 0 ) {
174 if( pl_ctrl ) {
175 add_msg( _( "The %s is too heavy for its engine(s)!" ), name );

177 return;

180 // maximum braking is 20 mph/s, assumes high friction tires
181 const int max_brake = 20 * 100;
182 //pos or neg if accelerator or brake
183 int vel_inc = ( accel + ( thrusting ? 0 : max_brake ) ) * thd;
184 // Reverse is only 60% acceleration, unless an electric motor is in use
185 if( thd == -1 && thrusting && !has_engine_type( fuel_type_battery, true ) ) {

190 //so 1000 = 100% and 453 = 45.3%
191 int load;
192 // Keep exact cruise control speed
193 if( cruise_on && accel != 0 ) {
194 int effective_cruise = std::min( cruise_velocity, max_vel );
195 if( thd > 0 ) {
196 vel_inc = std::min( vel_inc, effective_cruise - velocity );
197 } else {
198 vel_inc = std::max( vel_inc, effective_cruise - velocity );
199 }
200 if( thrusting ) {
201 load = 1000 * std::abs( vel_inc ) / accel;
202 } else {
203 // brakes provide 20 mph/s of slowdown and the rest is engine braking
204 // TODO: braking depends on wheels, traction, driver skill
205 load = 1000 * std::max( 0, std::abs( vel_inc ) - max_brake ) / accel;
206 }
207 } else {
208 load = ( thrusting ? 1000 : 0 );
209 }

211 if( is_rotorcraft() && ( z > 0 || is_flying_in_air() ) ) {
212 load = std::max( load, z > 0 ? 200 : 50 );
213 thrusting = true;
214 }
215
216 // only consume resources if engine accelerating

218 //abort if engines not operational
219 if( total_power_w() <= 0 || !engine_on || ( z == 0 && accel == 0 ) ) {
220 if( pl_ctrl ) {

222 add_msg( m_info, _( "The %s doesn't have an engine!" ), name );
223 } else if( has_engine_type( fuel_type_muscle, true ) ) {
224 add_msg( m_info, _( "The %s's mechanism is out of reach!" ), name );
225 } else if( !engine_on ) {
226 add_msg( _( "The %s's engine isn't on!" ), name );
227 } else if( traction < 0.01f ) {
228 add_msg( _( "The %s is stuck." ), name );
229 } else {
230 add_msg( _( "The %s's engine emits a sneezing sound." ), name );
231 }
232 }
233 cruise_velocity = 0;
234 return;
235 }
236 //make noise and consume fuel
237 noise_and_smoke( load );
238 consume_fuel( load, 1 );
239 if( z != 0 && is_rotorcraft() ) {
240 requested_z_change = z;
241 }
242 //break the engines a bit, if going too fast.
243 int strn = static_cast<int>( strain() * strain() * 100 );
244 for( size_t e = 0; e < engines.size(); e++ ) {
245 do_engine_damage( e, strn );
246 }
247 }
248
249 //wheels aren't facing the right way to change velocity properly
250 //lower down, since engines should be getting damaged anyway
251 if( skidding ) {
252 return;
253 }
254
255 //change vehicles velocity
256 if( ( velocity > 0 && velocity + vel_inc < 0 ) || ( velocity < 0 && velocity + vel_inc > 0 ) ) {
257 //velocity within braking distance of 0
258 stop();
259 } else {
260 // Increase velocity up to max_vel or min_vel, but not above.
261 const int min_vel = max_reverse_velocity();
262 if( vel_inc > 0 ) {
263 // Don't allow braking by accelerating (could happen with damaged engines)
264 velocity = std::max( velocity, std::min( velocity + vel_inc, max_vel ) );
265 } else {
266 velocity = std::min( velocity, std::max( velocity + vel_inc, min_vel ) );
267 }
268 }
269 // If you are going faster than the animal can handle, harness is damaged

271 for( size_t e = 0; e < parts.size(); e++ ) {
272 const vehicle_part &vp = parts[ e ];

274 monster *mon = get_pet( e );

278 int dmg = rng( 0, 10 );
279 damage_direct( e, dmg );
280 }
281 }
282 }

286void vehicle::cruise_thrust( int amount )
287{
288 if( amount == 0 ) {
289 return;
290 }
291 int safe_vel = safe_velocity();
292 int max_vel = autopilot_on ? safe_velocity() : max_velocity();
293 int max_rev_vel = max_reverse_velocity();
294
295 //if the safe velocity is between the cruise velocity and its next value, set to safe velocity
296 if( ( cruise_velocity < safe_vel && safe_vel < ( cruise_velocity + amount ) ) ||
297 ( cruise_velocity > safe_vel && safe_vel > ( cruise_velocity + amount ) ) ) {
298 cruise_velocity = safe_vel;
299 } else {
300 if( amount < 0 && ( cruise_velocity == safe_vel || cruise_velocity == max_vel ) ) {
301 // If coming down from safe_velocity or max_velocity decrease by one so
302 // the rounding below will drop velocity to a multiple of amount.
303 cruise_velocity += -1;
304 } else if( amount > 0 && cruise_velocity == max_rev_vel ) {
305 // If increasing from max_rev_vel, do the opposite.
306 cruise_velocity += 1;
307 } else {
308 // Otherwise just add the amount.
309 cruise_velocity += amount;
310 }
311 // Integer round to lowest multiple of amount.
312 // The result is always equal to the original or closer to zero,
313 // even if negative
314 cruise_velocity = ( cruise_velocity / std::abs( amount ) ) * std::abs( amount );
315 }
316 // Can't have a cruise speed faster than max speed
317 // or reverse speed faster than max reverse speed.
318 if( cruise_velocity > max_vel ) {
319 cruise_velocity = max_vel;
320 } else if( cruise_velocity < max_rev_vel ) {
321 cruise_velocity = max_rev_vel;
322 }

641 const int vel_sign = sgn( coll_velocity );
642 // Velocity of the object we're hitting
643 // Assuming it starts at 0, but we'll probably hit it many times
644 // in one collision, so accumulate the velocity gain from each hit.
645 float vel2 = 0.0f;
646 do {
647 smashed = false;
648 // Impulse of vehicle
649 const float vel1 = coll_velocity / 100.0f;
650 // Velocity of car after collision
651 const float vel1_a = ( mass * vel1 + mass2 * vel2 + e * mass2 * ( vel2 - vel1 ) ) /
652 ( mass + mass2 );
653 // Velocity of object after collision
654 const float vel2_a = ( mass * vel1 + mass2 * vel2 + e * mass * ( vel1 - vel2 ) ) / ( mass + mass2 );
655 // Lost energy at collision -> deformation energy -> damage
656 const float E_before = 0.5f * ( mass * vel1 * vel1 ) + 0.5f * ( mass2 * vel2 * vel2 );
657 const float E_after = 0.5f * ( mass * vel1_a * vel1_a ) + 0.5f * ( mass2 * vel2_a * vel2_a );
658 const float d_E = E_before - E_after;
659 if( d_E <= 0 ) {
660 // Deformation energy is signed
661 // If it's negative, it means something went wrong
662 // But it still does happen sometimes...
663 if( std::fabs( vel1_a ) < std::fabs( vel1 ) ) {
664 // Lower vehicle's speed to prevent infinite loops
665 coll_velocity = vel1_a * 90;
666 }
667 if( std::fabs( vel2_a ) > std::fabs( vel2 ) ) {
668 vel2 = vel2_a;
669 }
670 // this causes infinite loop
671 if( mass2 == 0 ) {
672 mass2 = 1;
673 }
674 continue;
675 }
676
677 add_msg( m_debug, "Deformation energy: %.2f", d_E );
678 // Damage calculation
679 // Damage dealt overall
680 dmg += d_E / 400;
681 // Damage for vehicle-part
682 // Always if no critters, otherwise if critter is real
683 if( critter == nullptr || !critter->is_hallucination() ) {
684 part_dmg = dmg * k / 100;
685 add_msg( m_debug, "Part collision damage: %.2f", part_dmg );
686 }
687 // Damage for object
688 const float obj_dmg = dmg * ( 100 - k ) / 100;
689
690 if( ret.type == veh_coll_bashable ) {
691 // Something bashable -- use map::bash to determine outcome
692 // NOTE: Floor bashing disabled for balance reasons
693 // Floor values are still used to set damage dealt to vehicle
694 smashed = here.is_bashable_ter_furn( p, false ) &&
695 here.bash_resistance( p, bash_floor ) <= obj_dmg &&
696 here.bash( p, obj_dmg, false, false, false, this ).success;
697 if( smashed ) {
698 if( here.is_bashable_ter_furn( p, bash_floor ) ) {
699 // There's new terrain there to smash
700 smashed = false;
701 terrain_collision_data( p, bash_floor, mass2, part_dens, e );
702 ret.target_name = here.disp_name( p );
703 } else if( here.impassable_ter_furn( p ) ) {
704 // There's new terrain there, but we can't smash it!
705 smashed = false;
706 ret.type = veh_coll_other;
707 mass2 = 1000;
708 e = 0.10;
709 part_dens = 80;
710 ret.target_name = here.disp_name( p );
711 }
712 }
713 } else if( ret.type == veh_coll_body ) {
714 int dam = obj_dmg * dmg_mod / 100;
715
716 // We know critter is set for this type. Assert to inform static
717 // analysis.
718 assert( critter );
719
720 // No blood from hallucinations
721 if( !critter->is_hallucination() ) {
722 if( part_flag( ret.part, "SHARP" ) ) {
723 parts[ret.part].blood += ( 20 + dam ) * 5;
724 } else if( dam > rng( 10, 30 ) ) {
725 parts[ret.part].blood += ( 10 + dam / 2 ) * 5;
726 }
727
728 check_environmental_effects = true;
729 }
730
731 time_stunned = time_duration::from_turns( ( rng( 0, dam ) > 10 ) + ( rng( 0, dam ) > 40 ) );
732 if( time_stunned > 0_turns ) {
733 critter->add_effect( effect_stunned, time_stunned );
734 }
735
736 if( ph != nullptr ) {
737 ph->hitall( dam, 40, driver );
738 } else {
739 const int armor = part_flag( ret.part, "SHARP" ) ?
740 critter->get_armor_cut( bodypart_id( "torso" ) ) :
741 critter->get_armor_bash( bodypart_id( "torso" ) );
742 dam = std::max( 0, dam - armor );
743 critter->apply_damage( driver, bodypart_id( "torso" ), dam );
744 add_msg( m_debug, "Critter collision damage: %d", dam );
745 }
746
747 // Don't fling if vertical - critter got smashed into the ground
748 if( !vert_coll ) {
749 if( std::fabs( vel2_a ) > 10.0f ||
750 std::fabs( e * mass * vel1_a ) > std::fabs( mass2 * ( 10.0f - vel2_a ) ) ) {
751 const units::angle angle = rng_float( -60_degrees, 60_degrees );
752 // Also handle the weird case when we don't have enough force
753 // but still have to push (in such case compare momentum)
754 const float push_force = std::max<float>( std::fabs( vel2_a ), 10.1f );
755 // move.dir is where the vehicle is facing. If velocity is negative,
756 // we're moving backwards and have to adjust the angle accordingly.
757 const units::angle angle_sum =
758 angle + move.dir() + ( vel2_a > 0 ? 0_degrees : 180_degrees );
759 g->fling_creature( critter, angle_sum, push_force );
760 } else if( std::fabs( vel2_a ) > std::fabs( vel2 ) ) {
761 vel2 = vel2_a;
762 } else {
763 // Vehicle's momentum isn't big enough to push the critter
764 velocity = 0;
765 break;
766 }
767
768 if( critter->is_dead_state() ) {
769 smashed = true;
770 } else if( critter != nullptr ) {
771 // Only count critter as pushed away if it actually changed position
772 smashed = critter->pos() != p;
773 }
774 }
775 }
776
777 if( critter == nullptr || !critter->is_hallucination() ) {
778 coll_velocity = vel1_a * ( smashed ? 100 : 90 );
779 }
780 // Stop processing when sign inverts, not when we reach 0
781 } while( !smashed && sgn( coll_velocity ) == vel_sign );
782
783 // Apply special effects from collision.
784 if( critter != nullptr ) {
785 if( !critter->is_hallucination() ) {
786 if( pl_ctrl ) {
787 if( time_stunned > 0_turns ) {
788 //~ 1$s - vehicle name, 2$s - part name, 3$s - NPC or monster
789 add_msg( m_warning, _( "Your %1$s's %2$s rams into %3$s and stuns it!" ),
790 name, parts[ ret.part ].name(), ret.target_name );
791 } else {
792 //~ 1$s - vehicle name, 2$s - part name, 3$s - NPC or monster
793 add_msg( m_warning, _( "Your %1$s's %2$s rams into %3$s!" ),
794 name, parts[ ret.part ].name(), ret.target_name );
795 }
796 }
797
798 if( part_flag( ret.part, "SHARP" ) ) {
799 critter->bleed();
800 } else {
801 sounds::sound( p, 20, sounds::sound_t::combat, snd, false, "smash_success", "hit_vehicle" );
802 }
803 }
804 } else {
805 if( pl_ctrl ) {
806 if( !snd.empty() ) {
807 //~ 1$s - vehicle name, 2$s - part name, 3$s - collision object name, 4$s - sound message
808 add_msg( m_warning, _( "Your %1$s's %2$s rams into %3$s with a %4$s" ),
809 name, parts[ ret.part ].name(), ret.target_name, snd );
810 } else {
811 //~ 1$s - vehicle name, 2$s - part name, 3$s - collision object name
812 add_msg( m_warning, _( "Your %1$s's %2$s rams into %3$s." ),
813 name, parts[ ret.part ].name(), ret.target_name );
814 }
815 }
816
817 sounds::sound( p, smashed ? 80 : 50, sounds::sound_t::combat, snd, false, "smash_success",
818 "hit_vehicle" );
819 }
820
821 if( smashed && !vert_coll ) {
822 int turn_amount = rng( 1, 3 ) * std::sqrt( static_cast<double>( part_dmg ) );
823 turn_amount /= 15;
824 if( turn_amount < 1 ) {
825 turn_amount = 1;
826 }
827 turn_amount *= 15;
828 if( turn_amount > 120 ) {
829 turn_amount = 120;
830 }
831 int turn_roll = rng( 0, 100 );
832 // Probability of skidding increases with higher delta_v
833 if( turn_roll < std::abs( ( prev_velocity - coll_velocity ) / 100.0f * 2.0f ) ) {
834 //delta_v = vel1 - vel1_a
835 //delta_v = 50 mph -> 100% probability of skidding
836 //delta_v = 25 mph -> 50% probability of skidding
837 skidding = true;
838 turn( units::from_degrees( one_in( 2 ) ? turn_amount : -turn_amount ) );
839 }
840 }
841
842 ret.imp = part_dmg;
843 return ret;
844}

847{

849 if( pwh < 0 ) {
850 return;
851 }
852 map &here = get_map();

864 if( veh_data.is_falling ) {
865 return;
866 }
867
868 const bool seen = player_character.sees( p );
869 const bool known = player_character.knows_trap( p );
870 if( seen ) {

873 add_msg( m_bad, _( "The %1$s's %2$s runs over %3$s." ), name, parts[ part ].name(), tr.name() );

876 }

879 if( veh_data.chance >= rng( 1, 100 ) ) {

881 sounds::sound( p, veh_data.sound_volume, sounds::sound_t::combat, veh_data.sound, false,
882 veh_data.sound_type, veh_data.sound_variant );

887 // Hit the wheel directly since it ran right over the trap.
888 damage_direct( pwh, veh_data.damage );
889 }

891 if( veh_data.remove_trap || veh_data.do_explosion ) {

893 still_has_trap = false;
894 }

902 here.trap_set( p, veh_data.set_trap.id() );

905 if( still_has_trap ) {
906 const trap &tr = here.tr_at( p );
907 if( seen || known ) {
908 // known status has been reset by map::trap_set()
909 player_character.add_known_trap( p, tr );
910 }
911 if( seen && !known ) {
912 // hard to miss!
913 const std::string direction = direction_name( direction_from( player_character.pos(), p ) );
914 add_msg( _( "You've spotted a %1$s to the %2$s!" ), tr.name(), direction );
915 }
916 }
917 }

1086void vehicle::pldrive( Character &driver, point p, int z )
1087{
1088 if( z != 0 && is_rotorcraft() ) {
1089 driver.moves = std::min( driver.moves, 0 );
1090 thrust( 0, z );
1091 }
1092 units::angle turn_delta = 15_degrees * p.x;
1093 const float handling_diff = handling_difficulty();
1094 if( turn_delta != 0_degrees ) {
1095 float eff = steering_effectiveness();
1096 if( eff == -2 ) {
1097 driver.add_msg_if_player( m_info,
1098 _( "You cannot steer an animal-drawn vehicle with no animal harnessed." ) );
1099 return;
1100 }
1101
1102 if( eff < 0 ) {
1103 driver.add_msg_if_player( m_info,
1104 _( "This vehicle has no steering system installed, you can't turn it." ) );
1105 return;
1106 }
1107
1108 if( eff == 0 ) {
1109 driver.add_msg_if_player( m_bad, _( "The steering is completely broken!" ) );
1110 return;
1111 }
1112
1113 // If you've got more moves than speed, it's most likely time stop
1114 // Let's get rid of that
1115 driver.moves = std::min( driver.moves, driver.get_speed() );
1116
1117 ///\EFFECT_DEX reduces chance of losing control of vehicle when turning
1118
1119 ///\EFFECT_PER reduces chance of losing control of vehicle when turning
1120
1121 ///\EFFECT_DRIVING reduces chance of losing control of vehicle when turning
1122 float skill = std::min( 10.0f, driver.get_skill_level( skill_driving ) +
1123 ( driver.get_dex() + driver.get_per() ) / 10.0f );
1124 float penalty = rng_float( 0.0f, handling_diff ) - skill;
1125 int cost;
1126 if( penalty > 0.0f ) {
1127 // At 10 penalty (rather hard to get), we're taking 4 turns per turn
1128 cost = 100 * ( 1.0f + penalty / 2.5f );
1129 } else {
1130 // At 10 skill, with a perfect vehicle, we could turn up to 3 times per turn
1131 cost = std::max( driver.get_speed(), 100 ) * ( 1.0f - ( -penalty / 10.0f ) * 2 / 3 );
1132 }
1133
1134 if( penalty > skill || cost > 400 ) {
1135 driver.add_msg_if_player( m_warning, _( "You fumble with the %s's controls." ), name );
1136 // Anything from a wasted attempt to 2 turns in the intended direction
1137 turn_delta *= rng( 0, 2 );
1138 // Also wastes next turn
1139 cost = std::max( cost, driver.moves + 100 );
1140 } else if( one_in( 10 ) ) {
1141 // Don't warn all the time or it gets spammy
1142 if( cost >= driver.get_speed() * 2 ) {
1143 driver.add_msg_if_player( m_warning, _( "It takes you a very long time to steer that vehicle!" ) );
1144 } else if( cost >= driver.get_speed() * 1.5f ) {
1145 driver.add_msg_if_player( m_warning, _( "It takes you a long time to steer that vehicle!" ) );
1146 }
1147 }
1148
1149 turn( turn_delta );
1150
1151 // At most 3 turns per turn, because otherwise it looks really weird and jumpy
1152 driver.moves -= std::max( cost, driver.get_speed() / 3 + 1 );
1153 }
1154
1155 if( p.y != 0 ) {
1156 int thr_amount = 100 * ( std::abs( velocity ) < 2000 ? 4 : 5 );
1157 if( cruise_on ) {
1158 cruise_thrust( -p.y * thr_amount );
1159 } else {
1160 thrust( -p.y );
1161 driver.moves = std::min( driver.moves, 0 );
1162 }
1163 }
1164
1165 // TODO: Actually check if we're on land on water (or disable water-skidding)
1166 // Only check for recovering from a skid if we did active steering (not cruise control).
1167 if( skidding && ( p.x != 0 || ( p.y != 0 && !cruise_on ) ) && valid_wheel_config() ) {
1168 ///\EFFECT_DEX increases chance of regaining control of a vehicle
1169
1170 ///\EFFECT_DRIVING increases chance of regaining control of a vehicle
1171 if( handling_diff * rng( 1, 10 ) <
1172 driver.dex_cur + driver.get_skill_level( skill_driving ) * 2 ) {
1173 driver.add_msg_if_player( _( "You regain control of the %s." ), name );
1174 driver.as_player()->practice( skill_driving, velocity / 5 );
1175 velocity = static_cast<int>( forward_velocity() );
1176 skidding = false;
1177 move.init( turn_dir );
1178 }
1179 }
1180}

1182// A chance to stop skidding if moving in roughly the faced direction
1183void vehicle::possibly_recover_from_skid()
1184{
1185 if( last_turn > 13_degrees ) {
1186 // Turning on the initial skid is delayed, so move==face, initially. This filters out that case.
1187 return;
1188 }
1189
1190 rl_vec2d mv = move_vec();
1191 rl_vec2d fv = face_vec();
1192 float dot = mv.dot_product( fv );
1193 // Threshold of recovery is Gaussianesque.
1194
1195 if( std::fabs( dot ) * 100 > dice( 9, 20 ) ) {
1196 add_msg( _( "The %s recovers from its skid." ), name );
1197 // face_vec takes over.
1198 skidding = false;
1199 // Wheels absorb horizontal velocity.
1200 velocity *= dot;
1201 if( dot < -.8 ) {
1202 // Pointed backwards, velo-wise.
1203 // Move backwards.
1204 velocity *= -1;
1205 }
1206
1207 move = face;
1208 }
1209}

1264vehicle *vehicle::act_on_map()
1265{
1266 const tripoint pt = global_pos3();
1267 map &here = get_map();
1268 if( !here.inbounds( pt ) ) {
1269 dbg( DL::Info ) << "stopping out-of-map vehicle at global pos " << pt;
1270 stop( false );
1271 of_turn = 0;
1272 is_falling = false;
1273 return this;
1274 }
1275 if( decrement_summon_timer() ) {
1276 return nullptr;
1277 }
1278 Character &player_character = get_player_character();
1279 const bool pl_ctrl = player_in_control( player_character );
1280 // TODO: Remove this hack, have vehicle sink a z-level
1281 if( is_floating && !can_float() ) {
1282 add_msg( m_bad, _( "Your %s sank." ), name );
1283 if( pl_ctrl ) {
1284 unboard_all();
1285 }
1286 if( g->remoteveh() == this ) {
1287 g->setremoteveh( nullptr );
1288 }
1289
1290 here.on_vehicle_moved( sm_pos.z );
1291 // Destroy vehicle (sank to nowhere)
1292 here.destroy_vehicle( this );
1293 return nullptr;
1294 }
1295
1296 // It needs to fall when it has no support OR was falling before
1297 // so that vertical collisions happen.
1298 const bool should_fall = is_falling || vertical_velocity != 0;
1299
1300 // TODO: Saner diagonal movement, so that you can jump off cliffs properly
1301 // The ratio of vertical to horizontal movement should be vertical_velocity/velocity
1302 // for as long as of_turn doesn't run out.
1303 if( should_fall ) {
1304 // Convert from 100*mph to m/s
1305 const float old_vel = vmiph_to_mps( vertical_velocity );
1306 // Formula is v_2 = sqrt( 2*d*g + v_1^2 )
1307 // Note: That drops the sign
1308 const float new_vel = -std::sqrt( 2 * tile_height * GRAVITY_OF_EARTH + old_vel * old_vel );
1309 vertical_velocity = mps_to_vmiph( new_vel );
1310 is_falling = true;
1311 } else {
1312 // Not actually falling, was just marked for fall test
1313 is_falling = false;
1314 }
1315
1316 // Low enough for bicycles to go in reverse.
1317 // If the movement is due to a change in z-level, i.e a helicopter then the lateral movement will often be zero.
1318 if( !should_fall && std::abs( velocity ) < 20 && requested_z_change == 0 ) {
1319 stop();
1320 of_turn -= .321f;
1321 return this;
1322 }
1323
1324 const float wheel_traction_area = here.vehicle_wheel_traction( *this );
1325 const float traction = k_traction( wheel_traction_area );
1326 if( traction < 0.001f ) {
1327 of_turn = 0;
1328 if( !should_fall ) {
1329 stop();
1330 if( floating.empty() ) {
1331 add_msg( m_info, _( "Your %s can't move on this terrain." ), name );
1332 } else {
1333 add_msg( m_info, _( "Your %s is beached." ), name );
1334 }
1335 return this;
1336 }
1337 }
1338 const float turn_cost = vehicles::vmiph_per_tile / std::max<float>( 0.0001f, std::abs( velocity ) );
1339
1340 // Can't afford it this turn?
1341 // Low speed shouldn't prevent vehicle from falling, though
1342 bool falling_only = false;
1343 if( turn_cost >= of_turn && ( ( !is_flying && requested_z_change == 0 ) || !is_rotorcraft() ) ) {
1344 if( !should_fall ) {
1345 of_turn_carry = of_turn;
1346 of_turn = 0;
1347 return this;
1348 }
1349 falling_only = true;
1350 }
1351
1352 // Decrease of_turn if falling+moving, but not when it's lower than move cost
1353 if( !falling_only ) {
1354 of_turn -= turn_cost;
1355 }
1356
1357 const bool can_use_rails = this->can_use_rails();
1358 const bool is_on_rails = vehicle_movement::is_on_rails( here, *this );
1359 if( one_in( 10 ) ) {
1360 bool controlled = false;
1361 // It can even be a NPC, but must be at the controls
1362 for( int boarded : boarded_parts() ) {
1363 if( part_with_feature( boarded, VPFLAG_CONTROLS, true ) >= 0 ) {
1364 controlled = true;
1365 player *passenger = get_passenger( boarded );
1366 if( passenger != nullptr ) {
1367 passenger->practice( skill_driving, 1 );
1368 }
1369 }
1370 }
1371
1372 // Eventually send it skidding if no control
1373 // But not if it's remotely controlled, is in water or is on rails
1374 if( !controlled && !pl_ctrl && !is_floating && !is_on_rails && !is_flying &&
1375 requested_z_change == 0 ) {
1376 skidding = true;
1377 }
1378 }
1379
1380 if( skidding && one_in( 4 ) ) {
1381 // Might turn uncontrollably while skidding
1382 turn( one_in( 2 ) ? -15_degrees : 15_degrees );
1383 }
1384
1385 if( should_fall ) {
1386 // TODO: Insert a (hard) driving test to stop this from happening
1387 skidding = true;
1388 }
1389
1390 vehicle_movement::rail_processing_result rpres;
1391 if( can_use_rails && !falling_only ) {
1392 rpres = vehicle_movement::process_movement_on_rails( here, *this );
1393 }
1394 if( rpres.do_turn ) {
1395 turn_dir = rpres.turn_dir;
1396 }
1397
1398 // The direction we're moving
1399 tileray mdir;
1400 if( skidding || should_fall ) {
1401 // If skidding, it's the move vector
1402 // Same for falling - no air control
1403 mdir = move;
1404 } else if( turn_dir != face.dir() && ( !is_on_rails || rpres.do_turn ) ) {
1405 // Driver turned vehicle, get turn_dir
1406 mdir.init( turn_dir );
1407 } else {
1408 // Not turning, keep face.dir
1409 mdir = face;
1410 }
1411
1412 tripoint dp;
1413 if( std::abs( velocity ) >= 20 && !falling_only ) {
1414 mdir.advance( velocity < 0 ? -1 : 1 );
1415 if( rpres.do_shift ) {
1416 dp.x = rpres.shift_amount.x;
1417 dp.y = rpres.shift_amount.y;
1418 } else {
1419 dp.x = mdir.dx();
1420 dp.y = mdir.dy();
1421 }
1422 }
1423
1424 if( should_fall ) {
1425 dp.z = -1;
1426 is_flying = false;
1427 } else {
1428 dp.z = requested_z_change;
1429 requested_z_change = 0;
1430 if( dp.z > 0 && is_rotorcraft() ) {
1431 is_flying = true;
1432 }
1433 }
1434
1435 return here.move_vehicle( *this, dp, mdir );
1436}

1473void vehicle::adjust_zlevel( int idir, const tripoint &offset )
1474{
1475 //We don't need to do anything if we're not on a ramp
1476 //unless position 0 is outside the vehicle then there may be a ramp in between
1477 if( part_at( point_zero ) != -1 && !check_on_ramp( idir, offset ) ) {
1478 return;
1479 }
1480
1481 // when a vehicle part enters the low end of a down ramp, or the high end of an up ramp,
1482 // it immediately translates down or up a z-level, respectively, ending up on the low
1483 // end of an up ramp or high end of a down ramp, respectively. The two ends are set
1484 // past each other, like so:
1485 // (side view) z+1 Rdh RDl
1486 // z+0 RUh Rul
1487 // A vehicle moving left to right on z+1 drives down to z+0 by entering the ramp down low end.
1488 // A vehicle moving right to left on z+0 drives up to z+1 by entering the ramp up high end.
1489 // A vehicle moving left to right on z+0 should ideally collide into a wall before entering
1490 // the ramp up high end, but even if it does, it briefly transitions to z+1 before returning
1491 // to z0 by entering the ramp down low end.
1492 // A vehicle moving right to left on z+1 drives down to z+0 by entering the ramp down low end,
1493 // then immediately returns to z+1 by entering the ramp up high end.
1494 // When a vehicle's central point transitions a z-level via a ramp, all other pre-calc points
1495 // make the opposite transition, so that points that were above an ascending pivot point are
1496 // now level with it, and parts that were level with an ascending pivot point are now below
1497 // it.
1498
1499 auto &m = get_map();
1500 tripoint global_pos = global_pos3();
1501
1502 tripoint new_center = global_pos + offset;
1503
1504 if( m.has_flag( TFLAG_RAMP_DOWN, new_center ) ) {
1505 new_center.z--;
1506 } else if( m.has_flag( TFLAG_RAMP_UP, new_center ) ) {
1507 new_center.z++;
1508 }
1509
1510 std::map<point, int> z_cache;
1511
1512 //Draw a line from the center to each part, going up and down ramps as we do
1513 for( auto &prt : get_all_parts() ) {
1514 tripoint part_point = global_pos + offset + prt.part().precalc[idir];
1515
1516 auto cache_entry = z_cache.find( part_point.xy() );
1517 if( cache_entry != z_cache.end() ) {
1518 prt.part().precalc[idir].z = cache_entry->second;
1519 continue;
1520 }
1521
1522 tripoint line = new_center;
1523 while( line.xy() != part_point.xy() ) {
1524 if( line.x < part_point.x ) {
1525 line.x++;
1526 } else if( line.x > part_point.x ) {
1527 line.x--;
1528 }
1529 if( line.y < part_point.y ) {
1530 line.y++;
1531 } else if( line.y > part_point.y ) {
1532 line.y--;
1533 }
1534 if( m.has_flag( TFLAG_RAMP_UP, line ) ) {
1535 line.z += 1;
1536 }
1537 if( m.has_flag( TFLAG_RAMP_DOWN, line ) ) {
1538 line.z -= 1;
1539 }
1540 }
1541 prt.part().precalc[idir].z = line.z - global_pos.z;
1542 z_cache[part_point.xy()] = line.z - global_pos.z;
1543 }
1544}

1546void vehicle::check_falling_or_floating()
1547{
1548 // TODO: Make the vehicle "slide" towards its center of weight
1549 // when it's not properly supported
1550 const auto &pts = get_points( true );
1551 if( pts.empty() ) {
1552 // Dirty vehicle with no parts
1553 is_falling = false;
1554 is_floating = false;
1555 in_water = false;
1556 is_flying = false;
1557 return;
1558 }
1559
1560 map &here = get_map();
1561 is_falling = here.has_zlevels();
1562
1563 if( is_flying && is_rotorcraft() ) {
1564 is_falling = false;
1565 } else {
1566 is_flying = false;
1567 }
1568
1569 size_t deep_water_tiles = 0;
1570 size_t water_tiles = 0;
1571 for( const tripoint &p : pts ) {
1572 if( is_falling ) {
1573 tripoint below( p.xy(), p.z - 1 );
1574 is_falling &= here.has_flag_ter_or_furn( TFLAG_NO_FLOOR, p ) &&
1575 ( p.z > -OVERMAP_DEPTH ) && !here.supports_above( below );
1576 }
1577 deep_water_tiles += here.has_flag( TFLAG_DEEP_WATER, p ) ? 1 : 0;
1578 water_tiles += here.has_flag( TFLAG_SWIMMABLE, p ) ? 1 : 0;
1579 }
1580 // floating if 2/3rds of the vehicle is in deep water
1581 is_floating = 3 * deep_water_tiles >= 2 * pts.size();
1582 // in_water if 1/2 of the vehicle is in water at all
1583 in_water = 2 * water_tiles >= pts.size();
1584}

1586float map::vehicle_wheel_traction( const vehicle &veh,
1587 const bool ignore_movement_modifiers /*=false*/ ) const
1588{
1589 if( veh.is_in_water( true ) ) {
1590 return veh.can_float() ? 1.0f : -1.0f;
1591 }
1592 if( veh.is_in_water() && veh.is_watercraft() && veh.can_float() ) {
1593 return 1.0f;
1594 }
1595
1596 const auto &wheel_indices = veh.wheelcache;
1597 int num_wheels = wheel_indices.size();
1598 if( num_wheels == 0 ) {
1599 // TODO: Assume it is digging in dirt
1600 // TODO: Return something that could be reused for dragging
1601 return 0.0f;
1602 }
1603
1604 float traction_wheel_area = 0.0f;
1605
1606 if( vehicle_movement::is_on_rails( *this, veh ) ) {
1607 // Vehicles on rails are considered to have all of their wheels on rails
1608 for( int p : veh.rail_wheelcache ) {
1609 traction_wheel_area += veh.cpart( p ).wheel_area();
1610 }
1611 return traction_wheel_area;
1612 }
1613
1614 for( int p : wheel_indices ) {
1615 const tripoint &pp = veh.global_part_pos3( p );
1616 const int wheel_area = veh.cpart( p ).wheel_area();
1617
1618 const auto &tr = ter( pp ).obj();
1619 // Deep water and air
1620 if( tr.has_flag( TFLAG_DEEP_WATER ) || tr.has_flag( TFLAG_NO_FLOOR ) ) {
1621 // No traction from wheel in water or air
1622 continue;
1623 }
1624
1625 int move_mod = move_cost_ter_furn( pp );
1626 if( move_mod == 0 ) {
1627 // Vehicle locked in wall
1628 // Shouldn't happen, but does
1629 return 0.0f;
1630 }
1631
1632 for( const auto &terrain_mod : veh.part_info( p ).wheel_terrain_mod() ) {
1633 if( !tr.has_flag( terrain_mod.first ) ) {
1634 move_mod += terrain_mod.second;
1635 break;
1636 }
1637 }
1638
1639 // Ignore the movement modifier if needed.
1640 if( ignore_movement_modifiers ) {
1641 move_mod = 2;
1642 }
1643
1644 traction_wheel_area += 2.0 * wheel_area / move_mod;
1645 }
1646
1647 return traction_wheel_area;
1648}

1650units::angle map::shake_vehicle( vehicle &veh, const int velocity_before,
1651 const units::angle direction )
1652{
1653 const int d_vel = std::abs( veh.velocity - velocity_before ) / 100;
1654
1655 std::vector<rider_data> riders = veh.get_riders();
1656
1657 units::angle coll_turn = 0_degrees;
1658 for( const rider_data &r : riders ) {
1659 const int ps = r.prt;
1660 Creature *rider = r.psg;
1661 if( rider == nullptr ) {
1662 debugmsg( "throw passenger: empty passenger at part %d", ps );
1663 continue;
1664 }
1665
1666 const tripoint part_pos = veh.global_part_pos3( ps );
1667 if( rider->pos() != part_pos ) {
1668 debugmsg( "throw passenger: passenger at %d,%d,%d, part at %d,%d,%d",
1669 rider->posx(), rider->posy(), rider->posz(),
1670 part_pos.x, part_pos.y, part_pos.z );
1671 veh.part( ps ).remove_flag( vehicle_part::passenger_flag );
1672 continue;
1673 }
1674
1675 player *psg = dynamic_cast<player *>( rider );
1676 monster *pet = dynamic_cast<monster *>( rider );
1677
1678 bool throw_from_seat = false;
1679 int move_resist = 1;
1680 if( psg ) {
1681 ///\EFFECT_STR reduces chance of being thrown from your seat when not wearing a seatbelt
1682 move_resist = psg->str_cur * 150 + 500;
1683 } else {
1684 int pet_resist = 0;
1685 if( pet != nullptr ) {
1686 pet_resist = static_cast<int>( to_kilogram( pet->get_weight() ) * 200 );
1687 }
1688 move_resist = std::max( 100, pet_resist );
1689 }
1690 if( veh.part_with_feature( ps, VPFLAG_SEATBELT, true ) == -1 ) {
1691 ///\EFFECT_STR reduces chance of being thrown from your seat when not wearing a seatbelt
1692 throw_from_seat = d_vel * rng( 80, 120 ) > move_resist;
1693 }
1694
1695 // Damage passengers if d_vel is too high
1696 if( !throw_from_seat && ( 10 * d_vel ) > 6 * rng( 50, 100 ) ) {
1697 const int dmg = d_vel * rng( 70, 100 ) / 400;
1698 if( psg ) {
1699 psg->hurtall( dmg, nullptr );
1700 psg->add_msg_player_or_npc( m_bad,
1701 _( "You take %d damage by the power of the impact!" ),
1702 _( "<npcname> takes %d damage by the power of the "
1703 "impact!" ), dmg );
1704 } else {
1705 pet->apply_damage( nullptr, bodypart_id( "torso" ), dmg );
1706 }
1707 }
1708
1709 if( psg && veh.player_in_control( *psg ) ) {
1710 const int lose_ctrl_roll = rng( 0, d_vel );
1711 ///\EFFECT_DEX reduces chance of losing control of vehicle when shaken
1712
1713 ///\EFFECT_DRIVING reduces chance of losing control of vehicle when shaken
1714 if( lose_ctrl_roll > psg->dex_cur * 2 + psg->get_skill_level( skill_driving ) * 3 ) {
1715 psg->add_msg_player_or_npc( m_warning,
1716 _( "You lose control of the %s." ),
1717 _( "<npcname> loses control of the %s." ), veh.name );
1718 int turn_amount = rng( 1, 3 ) * std::sqrt( std::abs( veh.velocity ) ) / 30;
1719 if( turn_amount < 1 ) {
1720 turn_amount = 1;
1721 }
1722 units::angle turn_angle = std::min( turn_amount * 15_degrees, 120_degrees );
1723 coll_turn = one_in( 2 ) ? turn_angle : -turn_angle;
1724 }
1725 }
1726
1727 if( throw_from_seat ) {
1728 if( psg ) {
1729 psg->add_msg_player_or_npc( m_bad,
1730 _( "You are hurled from the %s's seat by "
1731 "the power of the impact!" ),
1732 _( "<npcname> is hurled from the %s's seat by "
1733 "the power of the impact!" ), veh.name );
1734 unboard_vehicle( part_pos );
1735 } else if( get_player_character().sees( part_pos ) ) {
1736 add_msg( m_bad, _( "The %s is hurled from %s's by the power of the impact!" ),
1737 pet->disp_name(), veh.name );
1738 }
1739 ///\EFFECT_STR reduces distance thrown from seat in a vehicle impact
1740 g->fling_creature( rider, direction + rng_float( -30_degrees, 30_degrees ),
1741 std::max( 10, d_vel - move_resist / 100 ) );
1742 }
1743 }
1744
1745 return coll_turn;
1746}

1786static bool scan_rails_at_shift(
1787 const map &m,
1788 const vehicle &veh,
1789 int velocity_sign,
1790 units::angle dir,
1791 int shift_sign,
1792 tripoint *shift_amt = nullptr )
1793{
1794 point ray_delta;
1795 {
1796 tileray ray( dir );
1797 ray.advance( 1 );
1798 ray_delta.x = ray.dx();
1799 ray_delta.y = ray.dy();
1800 }
1801 if( ray_delta.x != 0 && ray_delta.y != 0 ) {
1802 // We can't cleanly map diagonally oriented vehicles to rail turns.
1803 // As such, treat the vehicle as if it can have either skew at the same time.
1804 int num_cw_rots = get_num_cw_rots_of_ray_delta( ray_delta );
1805 point rd_l = point_north.rotate( num_cw_rots );
1806 point rd_r = point_east.rotate( num_cw_rots );
1807
1808 point vyp_l = point_east.rotate( num_cw_rots );
1809 point vyp_r = point_south.rotate( num_cw_rots );
1810
1811 tripoint scan_start = veh.global_pos3();
1812 if( shift_sign > 0 ) {
1813 scan_start += rd_r * velocity_sign;
1814 } else if( shift_sign < 0 ) {
1815 scan_start += rd_l * velocity_sign;
1816 }
1817
1818 point scan_vec = ray_delta * velocity_sign;
1819
1820 bool scan_res_l = scan_rails_from_veh_internal( m, veh, scan_start, vyp_l, scan_vec );
1821 bool scan_res_r = scan_rails_from_veh_internal( m, veh, scan_start, vyp_r, scan_vec );
1822
1823 if( scan_res_l || scan_res_r ) {
1824 if( shift_amt ) {
1825 *shift_amt = scan_start - veh.global_pos3();
1826 }
1827 return true;
1828 }
1829 } else {
1830 point veh_plus_y_vec = ray_delta.rotate( 1 );
1831 point scan_vec = ray_delta * velocity_sign;
1832 tripoint scan_start = veh.global_pos3();
1833 if( shift_sign != 0 ) {
1834 scan_start += scan_vec + veh_plus_y_vec * shift_sign;
1835 }
1836 if( scan_rails_from_veh_internal( m, veh, scan_start, veh_plus_y_vec, scan_vec ) ) {
1837 if( shift_amt ) {
1838 *shift_amt = scan_start - veh.global_pos3();
1839 }
1840 return true;
1841 }
1842 }
1843 return false;
1844}

1873 units::angle dir_left = normalize( dir_straight - 45_degrees );

1889 bool can_shift_left = scan_rails_at_shift( m, veh, vel_sign, dir_straight, -vel_sign,
1890 &shift_amount_left );
1891
1892 bool is_on_rails = face_dir_degrees == face_dir_snapped && ( can_go_straight || can_go_backwards );

1894 // Appraise possible vehicle orientations
1895 if( !is_on_rails ) {
1896 // The vehicle is derailed, attempt to get back on rails
1897 if( can_go_straight ) {
1898 return make_turn( dir_straight );
1899 }
1900 } else {
1901 if( veh.face.dir() == veh.turn_dir ) {
1902 // Automatic movement - prefer going straight.
1903 if( can_go_straight ) {
1904 return make_none();
1905 } else if( can_turn_left ) {
1906 return make_turn( dir_left );
1907 } else if( can_shift_left ) {
1908 return make_shift( shift_amount_left );
1909 } else if( can_turn_right ) {
1910 return make_turn( dir_right );
1911 } else if( can_shift_right ) {
1912 return make_shift( shift_amount_right );
1913 }
1914 } else {
1915 // Manual movement - prefer going in turn direction.

1917 if( dir_delta < 180_degrees ) {
1918 // Trying to turn right
1919 if( can_turn_right ) {
1920 return make_turn( dir_right );
1921 } else if( can_shift_right ) {
1922 return make_shift( shift_amount_right );
1923 } else if( can_go_straight ) {
1924 return make_none();
1925 } else if( can_turn_left ) {
1926 return make_turn( dir_left );
1927 } else if( can_shift_left ) {
1928 return make_shift( shift_amount_left );
1929 }
1930 } else {
1931 // Trying to turn left
1932 if( can_turn_left ) {
1933 return make_turn( dir_left );
1934 } else if( can_shift_left ) {
1935 return make_shift( shift_amount_left );
1936 } else if( can_go_straight ) {
1937 return make_none();

1939 return make_turn( dir_right );
1940 } else if( can_shift_right ) {
1941 return make_shift( shift_amount_right );
1942 }
1943 }
1944 }
1945 }
1946 return make_none();
