117void overmapbuffer::fix_mongroups( overmap &new_overmap )
118{
119 for( auto it = new_overmap.zg.begin(); it != new_overmap.zg.end(); ) {
120 auto &mg = it->second;
121 // spawn related code simply sets population to 0 when they have been
122 // transformed into spawn points on a submap, the group can then be removed
123 if( mg.empty() ) {
124 new_overmap.zg.erase( it++ );
125 continue;
126 }
127 // Inside the bounds of the overmap?
128 if( mg.pos.x() >= 0 && mg.pos.y() >= 0 && mg.pos.x() < OMAPX * 2 &&
129 mg.pos.y() < OMAPY * 2 ) {
130 ++it;
131 continue;
132 }
133 point_abs_sm smabs = project_combine( new_overmap.pos(), mg.pos.xy() );
134 point_abs_om omp;
135 point_om_sm sm_rem;
136 std::tie( omp, sm_rem ) = project_remain<coords::om>( smabs );
137 if( !has( omp ) ) {
138 // Don't generate new overmaps, as this can be called from the
139 // overmap-generating code.
140 ++it;
141 continue;
142 }
143 overmap &om = get( omp );
144 mg.pos = tripoint_om_sm( sm_rem, mg.pos.z() );
145 om.add_mon_group( mg );
146 new_overmap.zg.erase( it++ );
147 }

150void overmapbuffer::fix_npcs( overmap &new_overmap )
151{
152 // First step: move all npcs that are located outside of the given overmap
153 // into a separate container. After that loop, new_overmap.npcs is no
154 // accessed anymore!
155 decltype( overmap::npcs ) to_relocate;
156 for( auto it = new_overmap.npcs.begin(); it != new_overmap.npcs.end(); ) {
157 npc &np = **it;
158 const tripoint_abs_omt npc_omt_pos = np.global_omt_location();
159 const point_abs_om npc_om_pos = project_to<coords::om>( npc_omt_pos.xy() );
160 const point_abs_om loc = new_overmap.pos();
161 if( npc_om_pos == loc ) {
162 // Nothing to do
163 ++it;
164 continue;
165 }
166 to_relocate.push_back( *it );
167 it = new_overmap.npcs.erase( it );
168 }
169 // Second step: put them back where they belong. This step involves loading
170 // new overmaps (via `get`), which does in turn call this function for the
171 // newly loaded overmaps. This in turn may move NPCs from the second overmap
172 // back into the first overmap. This messes up the iteration of it. The
173 // iteration is therefore done in a separate step above (which does *not*
174 // involve loading new overmaps).
175 for( auto &ptr : to_relocate ) {
176 npc &np = *ptr;
177 const tripoint_abs_omt npc_omt_pos = np.global_omt_location();
178 const point_abs_om npc_om_pos = project_to<coords::om>( npc_omt_pos.xy() );
179 const point_abs_om loc = new_overmap.pos();
180 if( !has( npc_om_pos ) ) {
181 // This can't really happen without save editing
182 // We have no sane option here, just place the NPC on the edge
183 debugmsg( "NPC %s is out of bounds, on non-generated overmap %s",
184 np.name, loc.to_string() );
185 point_abs_sm npc_sm = project_to<coords::sm>( npc_om_pos );
186 point_abs_sm min = project_to<coords::sm>( loc );
187 point_abs_sm max =
188 project_to<coords::sm>( loc + point_south_east ) - point_south_east;
189 npc_sm.x() = clamp( npc_sm.x(), min.x(), max.x() );
190 npc_sm.y() = clamp( npc_sm.y(), min.y(), max.y() );
191 // TODO: fix point types
192 np.spawn_at_sm( tripoint_abs_sm( npc_sm, np.posz() ).raw() );
193 new_overmap.npcs.push_back( ptr );
194 continue;
195 }
196
197 // Simplest case: just move the pointer
198 get( npc_om_pos ).insert_npc( ptr );
199 }

259overmap *overmapbuffer::get_existing( const point_abs_om &p )
260{
261 if( last_requested_overmap && last_requested_overmap->pos() == p ) {
262 return last_requested_overmap;
263 }
264 const auto it = overmaps.find( p );
265 if( it != overmaps.end() ) {
266 return last_requested_overmap = it->second.get();
267 }
268 if( known_non_existing.count( p ) > 0 ) {
269 // This overmap does not exist on disk (this has already been
270 // checked in a previous call of this function).
271 return nullptr;
272 }
273 if( file_exist( terrain_filename( p ) ) ) {
274 // File exists, load it normally (the get function
275 // indirectly call overmap::open to do so).
276 return &get( p );
277 }
278 // File does not exist (or not readable which is essentially
279 // the same for our usage). A second call of this function with
280 // the same coordinates will not check the file system, and
281 // return early.
282 // If the overmap had been created in the mean time, the previous
283 // loop would have found and returned it.
284 known_non_existing.insert( p );
285 return nullptr;
286}

507std::vector<mongroup *> overmapbuffer::monsters_at( const tripoint_abs_omt &p )
508{
509 // (x,y) are overmap terrain coordinates, they spawn 2x2 submaps,
510 // but monster groups are defined with submap coordinates.
511 tripoint_abs_sm p_sm = project_to<coords::sm>( p );
512 std::vector<mongroup *> result;
513 for( const point &offset : std::array<point, 4> { { { point_zero }, { point_south }, { point_east }, { point_south_east } } } ) {
514 std::vector<mongroup *> tmp = groups_at( p_sm + offset );
515 result.insert( result.end(), tmp.begin(), tmp.end() );
516 }
517 return result;
518}

520std::vector<mongroup *> overmapbuffer::groups_at( const tripoint_abs_sm &p )
521{
522 std::vector<mongroup *> result;
523 point_om_sm sm_within_om;
524 point_abs_om omp;
525 std::tie( omp, sm_within_om ) = project_remain<coords::om>( p.xy() );
526 if( !has( omp ) ) {
527 return result;
528 }
529 overmap &om = get( omp );
530 auto groups_range = om.zg.equal_range( tripoint_om_sm( sm_within_om, p.z() ) );
531 for( auto it = groups_range.first; it != groups_range.second; ++it ) {
532 mongroup &mg = it->second;
533 if( mg.empty() ) {
534 continue;
535 }
536 result.push_back( &mg );
537 }
538 return result;
539}

684bool overmapbuffer::reveal( const tripoint_abs_omt &center, int radius,
685 const std::function<bool( const oter_id & )> &filter )
686{
687 int radius_squared = radius * radius;
688 bool result = false;
689 for( int i = -radius; i <= radius; i++ ) {
690 for( int j = -radius; j <= radius; j++ ) {
691 const tripoint_abs_omt p = center + point( i, j );
692 if( seen( p ) ) {
693 continue;
694 }
695 if( trigdist && i * i + j * j > radius_squared ) {
696 continue;
697 }
698 if( !filter( ter( p ) ) ) {
699 continue;
700 }
701 result = true;
702 set_seen( p, true );
703 }
704 }
705 return result;

766static int get_terrain_cost( const tripoint_abs_omt &omt_pos, const overmap_path_params &params )
767{
768 if( params.only_known_by_player && !overmap_buffer.seen( omt_pos ) ) {
769 return -1;

771 if( params.avoid_danger && overmap_buffer.is_marked_dangerous( omt_pos ) ) {
772 return -1;

774 const oter_id &oter = overmap_buffer.ter_existing( omt_pos );
775 if( is_ot_match( "road", oter, ot_match_type::type ) ||
776 is_ot_match( "bridge", oter, ot_match_type::type ) ||

778 is_ot_match( "bridgehead_ground", oter, ot_match_type::type ) ||
779 is_ot_match( "bridgehead_ramp", oter, ot_match_type::type ) ||
780 is_ot_match( "road_nesw_manhole", oter, ot_match_type::type ) ) {
781 return params.road_cost;
782 } else if( is_ot_match( "field", oter, ot_match_type::type ) ) {
783 return params.field_cost;
784 } else if( is_ot_match( "rural_road", oter, ot_match_type::prefix ) ||
785 is_ot_match( "dirt_road", oter, ot_match_type::prefix ) ||
786 is_ot_match( "subway", oter, ot_match_type::type ) ||
787 is_ot_match( "lab_subway", oter, ot_match_type::type ) ) {
788 return params.dirt_road_cost;
789 } else if( is_ot_match( "forest_trail", oter, ot_match_type::type ) ) {
790 return params.trail_cost;
791 } else if( is_ot_match( "forest_water", oter, ot_match_type::type ) ) {
792 return params.swamp_cost;
793 } else if( is_ot_match( "river", oter, ot_match_type::prefix ) ||
794 is_ot_match( "lake", oter, ot_match_type::prefix ) ) {
795 if( is_ot_match( "river_center", oter, ot_match_type::type ) ||
796 is_ot_match( "lake_surface", oter, ot_match_type::type ) ) {
797 return params.water_cost;
798 } else {
799 return params.shore_cost;
800 }
801 } else if( is_ot_match( "bridge_under", oter, ot_match_type::type ) ) {
802 return params.water_cost;
803 } else if( is_ot_match( "open_air", oter, ot_match_type::type ) ) {

824std::vector<tripoint_abs_omt> overmapbuffer::get_travel_path(
825 const tripoint_abs_omt &src, const tripoint_abs_omt &dest, overmap_path_params params )
826{
827 if( src == overmap::invalid_tripoint || dest == overmap::invalid_tripoint ) {
828 return {};
829 }
830
831 const pf::omt_scoring_fn estimate = [&]( tripoint_abs_omt pos ) {
832 const int cur_cost = pos == src ? 0 : get_terrain_cost( pos, params );
833 if( cur_cost < 0 ) {
834 return pf::omt_score::rejected;
835 }
836 return pf::omt_score( cur_cost, is_ramp( pos ) );
837 };
838
839 constexpr int radius = 4 * OMAPX; // radius of search in OMTs = 4 overmaps
840 const pf::simple_path<tripoint_abs_omt> path = pf::find_overmap_path( src, dest, radius, estimate );
841 return path.points;
842}

844bool overmapbuffer::reveal_route( const tripoint_abs_omt &source, const tripoint_abs_omt &dest,
845 const omt_route_params &params )
846{
847 // Maximal radius of search (in overmaps)
848 static const int RADIUS = 4;
849 // half-size of the area to search in
850 static const point_rel_omt O( RADIUS * OMAPX, RADIUS * OMAPY );
851
852 if( source == overmap::invalid_tripoint || dest == overmap::invalid_tripoint ) {
853 return false;
854 }
855
856 // Local source - center of the local area
857 const point_rel_omt start( O );
858 // To convert local coordinates to global ones
859 const tripoint_abs_omt base = source - start;
860 // Local destination - relative to base
861 const point_rel_omt finish = ( dest - base ).xy();
862
863 const auto get_ter_at = [&]( const point_rel_omt & p ) {
864 return ter( base + p );
865 };
866
867 const oter_id oter = get_ter_at( start );
868 const auto connection = overmap_connections::guess_for( oter );
869
870 if( !connection ) {
871 return false;
872 }
873
874 const pf::two_node_scoring_fn<point_rel_omt> estimate =
875 [&]( pf::directed_node<point_rel_omt> cur, cata::optional<pf::directed_node<point_rel_omt>> ) {
876 int cost = 0;
877 const oter_id oter = get_ter_at( cur.pos );
878 if( !connection->has( oter ) ) {
879 if( params.road_only ) {
880 return pf::node_score::rejected;
881 }
882 if( is_river( oter ) ) {
883 return pf::node_score::rejected; // Can't walk on water
884 }
885 // Allow going slightly off-road to overcome small obstacles (e.g. craters),
886 // but heavily penalize that to make roads preferable
887 cost = 250;
888 }
889 return pf::node_score( cost, manhattan_dist( finish, cur.pos ) );
890 };
891
892 // TODO: use overmapbuffer::get_travel_path() with appropriate params instead
893 // TODO: tick params.popup
894 const auto path = pf::greedy_path( start, finish, O * 2, estimate );
895
896 for( const auto &node : path.nodes ) {
897 reveal( base + node.pos, params.radius );
898 }
899 return !path.nodes.empty();
900}

956 const omt_find_params &params )
957{
958 bool type_matches = false;
959 if( params.existing_only ) {
960 for( const std::pair<std::string, ot_match_type> &elem : params.types ) {
961 type_matches = check_ot_existing( elem.first, elem.second, location );
962 if( type_matches ) {
963 break;
964 }
965 }
966 } else {
967 for( const std::pair<std::string, ot_match_type> &elem : params.types ) {
968 type_matches = check_ot( elem.first, elem.second, location );
969 if( type_matches ) {
970 break;
971 }
972 }
973 }
974 if( !type_matches ) {
975 return false;
976 }
977
978 if( params.must_see && !seen( location ) ) {
979 return false;
980 }
981 if( params.cant_see && seen( location ) ) {
982 return false;
983 }
984
985 if( params.om_special ) {
986 bool meets_om_special = false;
987 if( params.existing_only ) {
988 meets_om_special = check_overmap_special_type_existing( *params.om_special, location );
989 } else {
990 meets_om_special = check_overmap_special_type( *params.om_special, location );
991 }
992 if( !meets_om_special ) {
993 return false;
994 }
995 }
996
997 return true;

1010tripoint_abs_omt overmapbuffer::find_closest( const tripoint_abs_omt &origin,
1011 const omt_find_params &params )
1012{
1013 // Check the origin before searching adjacent tiles!
1014 if( params.min_distance == 0 && is_findable_location( origin, params ) ) {
1015 return origin;
1016 }
1017
1018 // By default search overmaps within a radius of 4,
1019 // i.e. C = current overmap, X = overmaps searched:
1020 // XXXXXXXXX
1021 // XXXXXXXXX
1022 // XXXXXXXXX
1023 // XXXXXXXXX
1024 // XXXXCXXXX
1025 // XXXXXXXXX
1026 // XXXXXXXXX
1027 // XXXXXXXXX
1028 // XXXXXXXXX
1029 //
1030 // See overmap::place_specials for how we attempt to insure specials are placed within this
1031 // range. The actual number is 5 because 1 covers the current overmap,
1032 // and each additional one expends the search to the next concentric circle of overmaps.
1033 const int min_dist = params.min_distance;
1034 const int max_dist = params.search_range ? params.search_range : OMAPX * 5;
1035
1036 std::vector<tripoint_abs_omt> result;
1037 cata::optional<int> found_dist;
1038
1039 size_t num_overmaps = overmaps.size();
1040 size_t counter = 0;
1041
1042 for( const point_abs_omt &loc_xy : closest_points_first( origin.xy(), min_dist, max_dist ) ) {
1043 const int dist_xy = square_dist( origin.xy(), loc_xy );
1044
1045 if( found_dist && *found_dist < dist_xy ) {
1046 break;
1047 }
1048
1049 for( int z = -OVERMAP_DEPTH; z <= OVERMAP_HEIGHT; z++ ) {
1050 const tripoint_abs_omt loc( loc_xy, z );
1051 const int dist = square_dist( origin, loc );
1052
1053 if( found_dist && *found_dist < dist ) {
1054 continue;
1055 }
1056
1057 if( is_findable_location( loc, params ) ) {
1058 found_dist = dist;
1059 result.push_back( loc );
1060 }
1061
1062 counter += 1;
1063 if( params.popup && ( num_overmaps != overmaps.size() || counter == 512 ) ) {
1064 params.popup->refresh();
1065 num_overmaps = overmaps.size();
1066 counter = 0;
1067 }
1068 }
1069 }
1070
1071 return random_entry( result, overmap::invalid_tripoint );
1072}

1074std::vector<tripoint_abs_omt> overmapbuffer::find_all( const tripoint_abs_omt &origin,
1075 const omt_find_params &params )

1181std::vector<overmap *> overmapbuffer::get_overmaps_near( const tripoint_abs_sm &location,
1182 const int radius )
1183{
1184 // Grab the corners of a square around the target location at distance radius.
1185 // Convert to overmap coordinates and iterate from the minimum to the maximum.
1186 const point_abs_om start =
1187 project_to<coords::om>( location.xy() + point( -radius, -radius ) );
1188 const point_abs_om end =
1189 project_to<coords::om>( location.xy() + point( radius, radius ) );
1190 const point_rel_om offset = end - start;
1191
1192 std::vector<overmap *> result;
1193 result.reserve( ( offset.x() + 1 ) * ( offset.y() + 1 ) );
1194
1195 for( int x = start.x(); x <= end.x(); ++x ) {
1196 for( int y = start.y(); y <= end.y(); ++y ) {
1197 if( overmap *existing_om = get_existing( point_abs_om( x, y ) ) ) {
1198 result.emplace_back( existing_om );
1199 }
1200 }
1201 }
1202
1203 // Sort the resulting overmaps so that the closest ones are first.
1204 const tripoint_abs_om center = project_to<coords::om>( location );
1205 std::sort( result.begin(), result.end(), [&center]( const overmap * lhs,
1206 const overmap * rhs ) {
1207 const tripoint_abs_om lhs_pos( lhs->pos(), 0 );
1208 const tripoint_abs_om rhs_pos( rhs->pos(), 0 );
1209 return trig_dist( center, lhs_pos ) < trig_dist( center, rhs_pos );
1210 } );
1211
1212 return result;
1213}

1586bool overmapbuffer::place_special( const overmap_special_id &special_id,
1587 const tripoint_abs_omt &center, int radius )
1588{
1589 // First find the requested special. If it doesn't exist, we're done here.
1590 bool found = false;
1591 overmap_special special;
1592 for( const auto &elem : overmap_specials::get_all() ) {
1593 if( elem.id == special_id ) {
1594 special = elem;
1595 found = true;
1596 break;
1597 }
1598 }
1599 if( !found ) {
1600 return false;
1601 }
1602
1603 // Force our special to occur just once when we're spawning it here.
1604 special.occurrences.min = 1;
1605 special.occurrences.max = 1;
1606
1607 // Figure out the longest side of the special for purposes of determining our sector size
1608 // when attempting placements.
1609 const auto calculate_longest_side = [&special]() {
1610 auto min_max_x = std::minmax_element( special.terrains.begin(),
1611 special.terrains.end(), []( const overmap_special_terrain & lhs,
1612 const overmap_special_terrain & rhs ) {
1613 return lhs.p.x < rhs.p.x;
1614 } );
1615
1616 auto min_max_y = std::minmax_element( special.terrains.begin(),
1617 special.terrains.end(), []( const overmap_special_terrain & lhs,
1618 const overmap_special_terrain & rhs ) {
1619 return lhs.p.y < rhs.p.y;
1620 } );
1621
1622 const int special_longest_side = std::max( min_max_x.second->p.x - min_max_x.first->p.x,
1623 min_max_y.second->p.y - min_max_y.first->p.y ) + 1;
1624
1625 // If our longest side is greater than the OMSPEC_FREQ, just use that instead.
1626 return std::min( special_longest_side, OMSPEC_FREQ );
1627 };
1628
1629 const int longest_side = calculate_longest_side();
1630
1631 // Predefine our sectors to search in.
1632 om_special_sectors sectors = get_sectors( longest_side );
1633
1634 // Get all of the overmaps within the defined radius of the center.
1635 for( const auto &om : get_overmaps_near(
1636 project_to<coords::sm>( center ), omt_to_sm_copy( radius ) ) ) {
1637
1638 // Build an overmap_special_batch for the special on this overmap.
1639 std::vector<const overmap_special *> specials;
1640 specials.push_back( &special );
1641 overmap_special_batch batch( om->pos(), specials );
1642
1643 // Filter the sectors to those which are in in range of our center point, so
1644 // that we don't end up creating specials in areas that are outside of our radius,
1645 // since the whole point is to create a special that is within the parameters.
1646 std::vector<point_om_omt> sector_points_in_range;
1647 std::copy_if( sectors.sectors.begin(), sectors.sectors.end(),
1648 std::back_inserter( sector_points_in_range ), [&]( point_om_omt & p ) {
1649 const point_abs_omt global_sector_point = project_combine( om->pos(), p );
1650 // We'll include this sector if it's within our radius. We reduce the radius by
1651 // the length of the longest side of our special so that we don't end up in a
1652 // scenario where one overmap terrain of the special is within the radius but the
1653 // rest of it is outside the radius (due to size, rotation, etc), which would
1654 // then result in us placing the special but then not finding it later if we
1655 // search using the same radius value we used in placing it.
1656 return square_dist( global_sector_point, center.xy() ) <= radius - longest_side;
1657 } );
1658 om_special_sectors sectors_in_range {sector_points_in_range, sectors.sector_width};
1659
1660 // Attempt to place the specials using our batch and sectors. We
1661 // require they be placed in unexplored terrain right now.
1662 om->place_specials_pass( batch, sectors_in_range, true, true );
1663
1664 // The place special pass will erase specials that have reached their
1665 // maximum number of instances so first check if its been erased.
1666 if( batch.empty() ) {
1667 return true;
1668 }
1669
1670 // Hasn't been erased, so lets check placement counts.
1671 for( const auto &special_placement : batch ) {
1672 if( special_placement.instances_placed > 0 ) {
1673 // It was placed, lets get outta here.
1674 return true;
1675 }
1676 }

1680 return false;
