130void monfactions::finalize()
131{
132 if( faction_list.empty() ) {
133 debugmsg( "No monster factions found." );
134 return;
135 }
136
137 // Create a tree of faction dependence
138 std::multimap< mfaction_id, mfaction_id > child_map;
139 std::set< mfaction_id > unloaded; // To check if cycles exist
140 std::queue< mfaction_id > queue;
141 for( auto &faction : faction_list ) {
142 unloaded.insert( faction.loadid );
143 if( faction.loadid == faction.base_faction ) {
144 // No parent = root of the (a?) tree
145 queue.push( faction.loadid );
146 continue;
147 }
148
149 // Point parent to children
150 if( faction.base_faction.to_i() >= 0 ) {
151 child_map.insert( std::make_pair( faction.base_faction, faction.loadid ) );
152 }
153
154 // Set faction as friendly to itself if not explicitly set to anything
155 if( faction.attitude_map.count( faction.loadid ) == 0 ) {
156 faction.attitude_map[faction.loadid] = MFA_FRIENDLY;
157 }
158 }
159
160 if( queue.empty() && !faction_list.empty() ) {
161 debugmsg( "No valid root monster faction!" );
162 return;
163 }
164
165 // Set uninitialized factions to be children of the root.
166 // If more than one root exists, use the first one.
167 const auto root = queue.front();
168 for( auto &faction : faction_list ) {
169 if( faction.base_faction.to_i() < 0 ) {
170 faction.base_faction = root;
171 // If it is the (new) root, connecting it to own parent (self) would create a cycle.
172 // So only try to connect it to the parent if it isn't own parent.
173 if( faction.base_faction != faction.loadid ) {
174 child_map.insert( std::make_pair( faction.base_faction, faction.loadid ) );
175 }
176 }
177 }
178
179 // Traverse the tree (breadth-first), starting from root
180 while( !queue.empty() ) {
181 mfaction_id cur = queue.front();
182 queue.pop();
183 if( unloaded.count( cur ) != 0 ) {
184 unloaded.erase( cur );
185 } else {
186 debugmsg( "Tried to load monster faction %s more than once", cur.obj().id.c_str() );
187 continue;
188 }
189 auto children = child_map.equal_range( cur );
190 for( auto &it = children.first; it != children.second; ++it ) {
191 // Copy attributes to child
192 apply_base_faction( cur, it->second );
193 queue.push( it->second );
194 }
195 }
196
197 // Bad json
198 if( !unloaded.empty() ) {
199 std::string names;
200 for( auto &fac : unloaded ) {
201 names.append( fac.id().str() );
202 names.append( " " );
203 auto &the_faction = faction_list[fac.to_i()];
204 the_faction.base_faction = root;
205 }
206
207 debugmsg( "Cycle encountered when processing monster factions. Bad factions:\n %s",
208 names.c_str() );
209 }
210
211 faction_list.shrink_to_fit(); // Save a couple of bytes
212}
