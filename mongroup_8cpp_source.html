60float mongroup::avg_speed() const
61{
62 float avg_speed = 0;
63 if( monsters.empty() ) {
64 const MonsterGroup &g = type.obj();
65 int remaining_frequency = 1000;
66 for( auto &elem : g.monsters ) {
67 avg_speed += elem.frequency * elem.name.obj().speed;
68 remaining_frequency -= elem.frequency;
69 }
70 if( remaining_frequency > 0 ) {
71 avg_speed += g.defaultMonster.obj().speed * remaining_frequency;
72 }
73 avg_speed /= 1000;
74 } else {
75 for( auto &it : monsters ) {
76 avg_speed += it.type->speed;
77 }
78 avg_speed /= monsters.size();
79 }
80 return avg_speed;
81}

98MonsterGroupResult MonsterGroupManager::GetResultFromGroup(
99 const mongroup_id &group_name, int *quantity )
100{
101 auto &group = GetUpgradedMonsterGroup( group_name );
102 int spawn_chance = rng( 1, group.freq_total ); //Default 1000 unless specified
103 //Our spawn details specify, by default, a single instance of the default monster
104 MonsterGroupResult spawn_details = MonsterGroupResult( group.defaultMonster, 1 );
105
106 bool monster_found = false;
107 // Loop invariant values
108 const time_point sunset = ::sunset( calendar::turn );
109 const time_point sunrise = ::sunrise( calendar::turn );
110 const season_type season = season_of_year( calendar::turn );
111 // Step through spawn definitions from the monster group until one is found or
112 for( auto it = group.monsters.begin(); it != group.monsters.end() && !monster_found; ++it ) {
113 // There's a lot of conditions to work through to see if this spawn definition is valid
114 bool valid_entry = true;
115 //Insure that the time is not before the spawn first appears or after it stops appearing
116 valid_entry = valid_entry && ( calendar::start_of_cataclysm + it->starts < calendar::turn );
117 valid_entry = valid_entry && ( it->lasts_forever() ||
118 calendar::start_of_cataclysm + it->ends > calendar::turn );
119
120 std::vector<std::pair<time_point, time_point> > valid_times_of_day;
121 bool season_limited = false;
122 bool season_matched = false;
123 //Collect the various spawn conditions, and then insure they are met appropriately
124 for( auto &elem : it->conditions ) {
125 //Collect valid time of day ranges
126 if( elem == "DAY" || elem == "NIGHT" || elem == "DUSK" || elem == "DAWN" ) {
127 if( elem == "DAY" ) {
128 valid_times_of_day.push_back( std::make_pair( sunrise, sunset ) );
129 } else if( elem == "NIGHT" ) {
130 valid_times_of_day.push_back( std::make_pair( sunset, sunrise ) );
131 } else if( elem == "DUSK" ) {
132 valid_times_of_day.push_back( std::make_pair( sunset - 1_hours, sunset + 1_hours ) );
133 } else if( elem == "DAWN" ) {
134 valid_times_of_day.push_back( std::make_pair( sunrise - 1_hours, sunrise + 1_hours ) );
135 }
136 }
137
138 //If we have any seasons listed, we know to limit by season, and if any season matches this season, we are good to spawn
139 if( elem == "SUMMER" || elem == "WINTER" || elem == "SPRING" || elem == "AUTUMN" ) {
140 season_limited = true;
141 if( ( season == SUMMER && elem == "SUMMER" ) ||
142 ( season == WINTER && elem == "WINTER" ) ||
143 ( season == SPRING && elem == "SPRING" ) ||
144 ( season == AUTUMN && elem == "AUTUMN" ) ) {
145 season_matched = true;
146 }
147 }
148 }
149
150 //Make sure the current time of day is within one of the valid time ranges for this spawn
151 bool is_valid_time_of_day = false;
152 if( valid_times_of_day.empty() ) {
153 //Then it can spawn whenever, since no times were defined
154 is_valid_time_of_day = true;
155 } else {
156 //Otherwise, it's valid if it matches any of the times of day
157 for( auto &elem : valid_times_of_day ) {
158 if( calendar::turn > elem.first && calendar::turn < elem.second ) {
159 is_valid_time_of_day = true;
160 }
161 }
162 }
163 if( !is_valid_time_of_day ) {
164 valid_entry = false;
165 }
166
167 //If we are limited by season, make sure we matched a season
168 if( season_limited && !season_matched ) {
169 valid_entry = false;
170 }
171
172 //If the entry was valid, check to see if we actually spawn it
173 if( valid_entry ) {
174 //If the monsters frequency is greater than the spawn_chance, select this spawn rule
175 if( it->frequency >= spawn_chance ) {
176 if( it->pack_maximum > 1 ) {
177 spawn_details = MonsterGroupResult( it->name, rng( it->pack_minimum, it->pack_maximum ) );
178 } else {
179 spawn_details = MonsterGroupResult( it->name, 1 );
180 }
181 //And if a quantity pointer with remaining value was passed, will modify the external value as a side effect
182 //We will reduce it by the spawn rule's cost multiplier
183 if( quantity ) {
184 *quantity -= std::max( 1, it->cost_multiplier * spawn_details.pack_size );
185 }
186 monster_found = true;
187 //Otherwise, subtract the frequency from spawn result for the next loop around
188 } else {
189 spawn_chance -= it->frequency;
190 }
191 }
192 }
193
194 // Force quantity to decrement regardless of whether we found a monster.
195 if( quantity && !monster_found ) {
196 ( *quantity )--;
197 }
198
199 return spawn_details;
200}

333{
334 float mon_upgrade_factor = get_option<float>( "MONSTER_UPGRADE_FACTOR" );
335
336 MonsterGroup g;
337

339 bool extending = false; //If already a group with that name, add to it instead of overwriting it

341 if( monsterGroupMap.count( g.name ) != 0 && !allow_override ) {
342 g = monsterGroupMap[g.name];

347 g.defaultMonster = mtype_id( jo.get_string( "default" ) );

350 if( jo.has_array( "monsters" ) ) {
351 for( JsonObject mon : jo.get_array( "monsters" ) ) {
352 const mtype_id name = mtype_id( mon.get_string( "monster" ) );
353
354 int freq = mon.get_int( "freq" );
355 int cost = mon.get_int( "cost_multiplier" );
356 int pack_min = 1;
357 int pack_max = 1;
358 if( mon.has_member( "pack_size" ) ) {

360 pack_min = packarr.next_int();
361 pack_max = packarr.next_int();
362 }
363 static const time_duration tdfactor = 1_hours;
364 time_duration starts = 0_turns;
365 time_duration ends = 0_turns;
366 if( mon.has_member( "starts" ) ) {
367 starts = tdfactor * mon.get_int( "starts" ) * ( mon_upgrade_factor > 0 ? mon_upgrade_factor : 1 );
368 }
369 if( mon.has_member( "ends" ) ) {
370 ends = tdfactor * mon.get_int( "ends" ) * ( mon_upgrade_factor > 0 ? mon_upgrade_factor : 1 );
371 }
372 MonsterGroupEntry new_mon_group = MonsterGroupEntry( name, freq, cost, pack_min, pack_max, starts,
373 ends );
374 if( mon.has_member( "conditions" ) ) {
375 for( const std::string line : mon.get_array( "conditions" ) ) {
376 new_mon_group.conditions.push_back( line );
377 }
378 }
379
380 g.monsters.push_back( new_mon_group );
381 }
382 }
383 g.replace_monster_group = jo.get_bool( "replace_monster_group", false );
384 g.new_monster_group = mongroup_id( jo.get_string( "new_monster_group_id",
385 mongroup_id::NULL_ID().str() ) );
386 assign( jo, "replacement_time", g.monster_group_time, false, 1_days );
387 g.is_safe = jo.get_bool( "is_safe", false );
388

390 if( jo.get_bool( "auto_total", false ) ) { //Fit the max size to the sum of all freqs
391 int total = 0;
392 for( MonsterGroupEntry &mon : g.monsters ) {
393 total += mon.frequency;
394 }
395 g.freq_total = total;
396 }
397
398 monsterGroupMap[g.name] = g;
