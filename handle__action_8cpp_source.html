156 int moves_elapsed() {
157 const float turn_duration = get_option<float>( "TURN_DURATION" );
158 // Magic number 0.005 chosen due to option menu's 2 digit precision and
159 // the option menu UI rounding <= 0.005 down to "0.00" in the display.
160 // This conditional will catch values (e.g. 0.003) that the options menu
161 // would round down to "0.00" in the options menu display. This prevents
162 // the user from being surprised by floating point rounding near zero.
163 if( turn_duration <= 0.005 ) {
164 return 0;
165 }
166 auto now = std::chrono::steady_clock::now();
167 std::chrono::milliseconds elapsed_ms =
168 std::chrono::duration_cast<std::chrono::milliseconds>( now - user_turn_start );
169 return elapsed_ms.count() / ( 10.0 * turn_duration );
170 }

780 if( !mech_smash ) {
781 u.handle_melee_wear( u.weapon );

817 cata::optional<tripoint> to_safety;

1415static void cast_spell()
1416{
1417 player &u = g->u;
1418
1419 std::vector<spell_id> spells = u.magic->spells();
1420
1421 if( spells.empty() ) {
1422 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1423 _( "You don't know any spells to cast." ) );
1424 return;
1425 }
1426
1427 bool can_cast_spells = false;
1428 for( spell_id sp : spells ) {
1429 spell temp_spell = u.magic->get_spell( sp );
1430 if( temp_spell.can_cast( u ) ) {
1431 can_cast_spells = true;
1432 }
1433 }
1434
1435 if( !can_cast_spells ) {
1436 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1437 _( "You can't cast any of the spells you know!" ) );
1438 }
1439
1440 const int spell_index = u.magic->select_spell( u );
1441 if( spell_index < 0 ) {
1442 return;
1443 }
1444
1445 spell &sp = *u.magic->get_spells()[spell_index];
1446
1447 if( u.is_armed() && !sp.has_flag( spell_flag::NO_HANDS ) &&
1448 !u.weapon.has_flag( flag_MAGIC_FOCUS ) ) {
1449 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1450 _( "You need your hands free to cast this spell!" ) );
1451 return;
1452 }
1453
1454 if( !u.magic->has_enough_energy( u, sp ) ) {
1455 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1456 _( "You don't have enough %s to cast the spell." ),
1457 sp.energy_string() );
1458 return;
1459 }
1460
1461 if( sp.energy_source() == hp_energy && !u.has_quality( qual_CUT ) ) {
1462 add_msg( game_message_params{ m_bad, gmf_bypass_cooldown },
1463 _( "You cannot cast Blood Magic without a cutting implement." ) );
1464 return;
1465 }
1466
1467 player_activity cast_spell( ACT_SPELLCASTING, sp.casting_time( u ) );
1468 // [0] this is used as a spell level override for items casting spells
1469 cast_spell.values.emplace_back( -1 );
1470 // [1] if this value is 1, the spell never fails
1471 cast_spell.values.emplace_back( 0 );
1472 // [2] this value overrides the mana cost if set to 0
1473 cast_spell.values.emplace_back( 1 );
1474 cast_spell.name = sp.id().c_str();
1475 if( u.magic->casting_ignore ) {
1476 const std::vector<distraction_type> ignored_distractions = {
1477 distraction_type::alert,
1478 distraction_type::noise,
1479 distraction_type::pain,
1480 distraction_type::attacked,
1481 distraction_type::hostile_spotted_near,
1482 distraction_type::hostile_spotted_far,
1483 distraction_type::talked_to,
1484 distraction_type::asthma,
1485 distraction_type::weather_change
1486 };
1487 for( const distraction_type ignored : ignored_distractions ) {
1488 cast_spell.ignore_distraction( ignored );
1489 }
1490 }
1491 u.assign_activity( cast_spell, false );
1492}

1520bool game::handle_action()
1521{
1522 std::string action;
1523 input_context ctxt;
1524 action_id act = ACTION_NULL;
1525 user_turn current_turn;
1526 // Check if we have an auto-move destination
1527 if( u.has_destination() ) {
1528 act = u.get_next_auto_move_direction();
1529 if( act == ACTION_NULL ) {
1530 add_msg( m_info, _( "Auto-move canceled" ) );
1531 u.clear_destination();
1532 return false;
1533 }
1534 } else if( u.has_destination_activity() ) {
1535 // starts destination activity after the player successfully reached his destination
1536 u.start_destination_activity();
1537 return false;
1538 } else {
1539 // No auto-move, ask player for input
1540 ctxt = get_player_input( action );
1541 }
1542
1543 const optional_vpart_position vp = m.veh_at( u.pos() );
1544 bool veh_ctrl = !u.is_dead_state() &&
1545 ( ( vp && vp->vehicle().player_in_control( u ) ) || remoteveh() != nullptr );
1546
1547 // If performing an action with right mouse button, co-ordinates
1548 // of location clicked.
1549 cata::optional<tripoint> mouse_target;
1550
1551 if( uquit == QUIT_WATCH && action == "QUIT" ) {
1552 uquit = QUIT_DIED;
1553 return false;
1554 }
1555
1556 if( act == ACTION_NULL ) {
1557 act = look_up_action( action );
1558
1559 if( act == ACTION_KEYBINDINGS ) {
1560 // already handled by input context
1561 return false;
1562 }
1563
1564 if( act == ACTION_MAIN_MENU ) {
1565 if( uquit == QUIT_WATCH ) {
1566 return false;
1567 }
1568 // No auto-move actions have or can be set at this point.
1569 u.clear_destination();
1570 destination_preview.clear();
1571 act = handle_main_menu();
1572 if( act == ACTION_NULL ) {
1573 return false;
1574 }
1575 }
1576
1577 if( act == ACTION_ACTIONMENU ) {
1578 if( uquit == QUIT_WATCH ) {
1579 return false;
1580 }
1581 // No auto-move actions have or can be set at this point.
1582 u.clear_destination();
1583 destination_preview.clear();
1584 act = handle_action_menu();
1585 if( act == ACTION_NULL ) {
1586 return false;
1587 }
1588#if defined(__ANDROID__)
1589 if( get_option<bool>( "ANDROID_ACTIONMENU_AUTOADD" ) && ctxt.get_category() == "DEFAULTMODE" ) {
1590 add_best_key_for_action_to_quick_shortcuts( act, ctxt.get_category(), false );
1591 }
1592#endif
1593 }
1594
1595 if( act == ACTION_KEYBINDINGS ) {
1596 u.clear_destination();
1597 destination_preview.clear();
1598 act = ctxt.display_menu( true );
1599 if( act == ACTION_NULL ) {
1600 return false;
1601 }
1602 }
1603
1604 if( can_action_change_worldstate( act ) ) {
1605 user_action_counter += 1;
1606 }
1607
1608 if( act == ACTION_SELECT || act == ACTION_SEC_SELECT ) {
1609 // Mouse button click
1610 if( veh_ctrl ) {
1611 // No mouse use in vehicle
1612 return false;
1613 }
1614
1615 if( u.is_dead_state() ) {
1616 // do not allow mouse actions while dead
1617 return false;
1618 }
1619
1620 const cata::optional<tripoint> mouse_pos = ctxt.get_coordinates( w_terrain );
1621 if( !mouse_pos ) {
1622 return false;
1623 } else if( !u.sees( *mouse_pos ) ) {
1624 // Not clicked in visible terrain
1625 return false;
1626 }
1627 mouse_target = mouse_pos;
1628
1629 if( act == ACTION_SELECT ) {
1630 // Note: The following has the potential side effect of
1631 // setting auto-move destination state in addition to setting
1632 // act.
1633 if( !try_get_left_click_action( act, *mouse_target ) ) {
1634 return false;
1635 }
1636 } else if( act == ACTION_SEC_SELECT ) {
1637 if( !try_get_right_click_action( act, *mouse_target ) ) {
1638 return false;
1639 }
1640 }
1641 } else if( act != ACTION_TIMEOUT ) {
1642 // act has not been set for an auto-move, so clearing possible
1643 // auto-move destinations. Since initializing an auto-move with
1644 // the mouse may span across multiple actions, we do not clear the
1645 // auto-move destination if the action is only a timeout, as this
1646 // would require the user to double click quicker than the
1647 // timeout delay.
1648 u.clear_destination();
1649 destination_preview.clear();
1650 }
1651 }
1652
1653 if( act == ACTION_NULL ) {
1654 const input_event &&evt = ctxt.get_raw_input();
1655 if( !evt.sequence.empty() ) {
1656 const int ch = evt.get_first_input();
1657 const std::string &&name = inp_mngr.get_keyname( ch, evt.type, true );
1658 if( !get_option<bool>( "NO_UNKNOWN_COMMAND_MSG" ) ) {
1659 add_msg( m_info, _( "Unknown command: \"%s\" (%ld)" ), name, ch );
1660 if( const cata::optional<std::string> hint =
1661 press_x_if_bound( ACTION_KEYBINDINGS ) ) {
1662 add_msg( m_info, _( "%s at any time to see and edit keybindings relevant to "
1663 "the current context." ),
1664 *hint );
1665 }
1666 }
1667 }
1668 return false;
1669 }
1670
1671 // This has no action unless we're in a special game mode.
1672 gamemode->pre_action( act );
1673
1674 int soffset = get_option<int>( "MOVE_VIEW_OFFSET" );
1675
1676 int before_action_moves = u.moves;
1677
1678 // These actions are allowed while deathcam is active. Registered in game::get_player_input
1679 if( uquit == QUIT_WATCH || !u.is_dead_state() ) {
1680 switch( act ) {
1681 case ACTION_TOGGLE_MAP_MEMORY:
1682 u.toggle_map_memory();
1683 break;
1684
1685 case ACTION_CENTER:
1686 u.view_offset.x = driving_view_offset.x;
1687 u.view_offset.y = driving_view_offset.y;
1688 break;
1689
1690 case ACTION_SHIFT_N:
1691 case ACTION_SHIFT_NE:
1692 case ACTION_SHIFT_E:
1693 case ACTION_SHIFT_SE:
1694 case ACTION_SHIFT_S:
1695 case ACTION_SHIFT_SW:
1696 case ACTION_SHIFT_W:
1697 case ACTION_SHIFT_NW: {
1698 static const std::map<action_id, std::pair<point, point>> shift_delta = {
1699 { ACTION_SHIFT_N, { point_north, point_north_east } },
1700 { ACTION_SHIFT_NE, { point_north_east, point_east } },
1701 { ACTION_SHIFT_E, { point_east, point_south_east } },
1702 { ACTION_SHIFT_SE, { point_south_east, point_south } },
1703 { ACTION_SHIFT_S, { point_south, point_south_west } },
1704 { ACTION_SHIFT_SW, { point_south_west, point_west } },
1705 { ACTION_SHIFT_W, { point_west, point_north_west } },
1706 { ACTION_SHIFT_NW, { point_north_west, point_north } },
1707 };
1708 u.view_offset += use_tiles && tile_iso ?
1709 shift_delta.at( act ).second * soffset : shift_delta.at( act ).first * soffset;
1710 }
1711 break;
1712
1713 case ACTION_LOOK:
1714 look_around();
1715 break;
1716
1717 case ACTION_KEYBINDINGS:
1718 // already handled by input context
1719 break;
1720
1721 default:
1722 break;
1723 }
1724 }
1725
1726 // actions allowed only while alive
1727 if( !u.is_dead_state() ) {
1728 switch( act ) {
1729 case ACTION_NULL:
1730 case NUM_ACTIONS:
1731 break; // dummy entries
1732 case ACTION_ACTIONMENU:
1733 case ACTION_MAIN_MENU:
1734 case ACTION_KEYBINDINGS:
1735 break; // handled above
1736
1737 case ACTION_TIMEOUT:
1738 if( check_safe_mode_allowed( false ) ) {
1739 character_funcs::do_pause( u );
1740 }
1741 break;
1742
1743 case ACTION_PAUSE:
1744 if( check_safe_mode_allowed() ) {
1745 character_funcs::do_pause( u );
1746 }
1747 break;
1748
1749 case ACTION_CYCLE_MOVE:
1750 u.cycle_move_mode();
1751 break;
1752
1753 case ACTION_RESET_MOVE:
1754 u.reset_move_mode();
1755 break;
1756
1757 case ACTION_TOGGLE_RUN:
1758 u.toggle_run_mode();
1759 break;
1760
1761 case ACTION_TOGGLE_CROUCH:
1762 u.toggle_crouch_mode();
1763 break;
1764
1765 case ACTION_OPEN_MOVEMENT:
1766 open_movement_mode_menu();
1767 break;
1768
1769 case ACTION_MOVE_FORTH:
1770 case ACTION_MOVE_FORTH_RIGHT:
1771 case ACTION_MOVE_RIGHT:
1772 case ACTION_MOVE_BACK_RIGHT:
1773 case ACTION_MOVE_BACK:
1774 case ACTION_MOVE_BACK_LEFT:
1775 case ACTION_MOVE_LEFT:
1776 case ACTION_MOVE_FORTH_LEFT:
1777 if( !u.get_value( "remote_controlling" ).empty() &&
1778 ( u.has_active_item( itype_radiocontrol ) ||
1779 u.has_active_bionic( bio_remote ) ) ) {
1780 rcdrive( get_delta_from_movement_action( act, iso_rotate::yes ) );
1781 } else if( veh_ctrl ) {
1782 // vehicle control uses x for steering and y for ac/deceleration,
1783 // so no rotation needed
1784 pldrive( get_delta_from_movement_action( act, iso_rotate::no ) );
1785 } else {
1786 point dest_delta = get_delta_from_movement_action( act, iso_rotate::yes );
1787 if( auto_travel_mode && !u.is_auto_moving() ) {
1788 for( int i = 0; i < SEEX; i++ ) {
1789 tripoint auto_travel_destination( u.posx() + dest_delta.x * ( SEEX - i ),
1790 u.posy() + dest_delta.y * ( SEEX - i ),
1791 u.posz() );
1792 destination_preview = m.route( u.pos(),
1793 auto_travel_destination,
1794 u.get_pathfinding_settings(),
1795 u.get_path_avoid() );
1796 if( !destination_preview.empty() ) {
1797 destination_preview.erase( destination_preview.begin() + 1, destination_preview.end() );
1798 u.set_destination( destination_preview );
1799 break;
1800 }
1801 }
1802 act = u.get_next_auto_move_direction();
1803 const point dest_next = get_delta_from_movement_action( act, iso_rotate::yes );
1804 if( dest_next == point_zero ) {
1805 u.clear_destination();
1806 }
1807 dest_delta = dest_next;
1808 }
1809 if( !avatar_action::move( u, m, dest_delta ) ) {
1810 // auto-move should be canceled due to a failed move or obstacle
1811 u.clear_destination();
1812 }
1813 }
1814 break;
1815 case ACTION_MOVE_DOWN:
1816 if( u.is_mounted() ) {
1817 auto mon = u.mounted_creature.get();
1818 if( !mon->has_flag( MF_RIDEABLE_MECH ) ) {
1819 add_msg( m_info, _( "You can't go down stairs while you're riding." ) );
1820 break;
1821 }
1822 }
1823 if( !u.in_vehicle ) {
1824 vertical_move( -1, false );
1825 } else if( veh_ctrl && vp->vehicle().is_rotorcraft() ) {
1826 pldrive( tripoint_below );
1827 }
1828 break;
1829
1830 case ACTION_MOVE_UP:
1831 if( u.is_mounted() ) {
1832 auto mon = u.mounted_creature.get();
1833 if( !mon->has_flag( MF_RIDEABLE_MECH ) ) {
1834 add_msg( m_info, _( "You can't go down stairs while you're riding." ) );
1835 break;
1836 }
1837 }
1838 if( !u.in_vehicle ) {
1839 vertical_move( 1, false );
1840 } else if( veh_ctrl && vp->vehicle().is_rotorcraft() ) {
1841 pldrive( tripoint_above );
