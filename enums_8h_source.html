66// Matching rules for comparing a string to an overmap terrain id.
67enum ot_match_type {
68 // The provided string must completely match the overmap terrain id, including
69 // linear direction suffixes for linear terrain types or rotation suffixes
70 // for rotated terrain types.
71 exact,
72 // The provided string must completely match the base type id of the overmap
73 // terrain id, which means that suffixes for rotation and linear terrain types
74 // are ignored.
75 type,
76 // The provided string must be a complete prefix (with additional parts delimited
77 // by an underscore) of the overmap terrain id. For example, "forest" will match
78 // "forest" or "forest_thick" but not "forestcabin".
79 prefix,
80 // The provided string must be contained within the overmap terrain id, but may
81 // occur at the beginning, end, or middle and does not have any rules about
82 // underscore delimiting.
83 contains,
84 num_ot_match_type
85};
