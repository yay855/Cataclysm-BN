52std::string gettext_gendered( const GenderMap &genders, const std::string &msg )
53{
54 const std::vector<std::string> &language_genders = get_language().genders;
55
56 std::vector<std::string> chosen_genders;
57 for( const auto &subject_genders : genders ) {
58 // default if no match
59 std::string chosen_gender = language_genders[0];
60 for( const std::string &gender : subject_genders.second ) {
61 if( std::find( language_genders.begin(), language_genders.end(), gender ) !=
62 language_genders.end() ) {
63 chosen_gender = gender;
64 break;
65 }
66 }
67 chosen_genders.push_back( subject_genders.first + ":" + chosen_gender );
68 }
69 std::string context = join( chosen_genders, " " );
70 return pgettext( context.c_str(), msg.c_str() );

309std::string translation::translated( const int num ) const
310{
311 if( !needs_translation || raw.empty() ) {
312 return raw;
313 }
314 // Note1: `raw`, `raw_pl` and `ctxt` are effectively immutable for caching purposes:
315 // in the places where they are changed, cache is explicitly invalidated
316 // Note2: if `raw_pl` is defined, `num` becomes part of the "cache key"
317 // otherwise `num` is ignored (for both translation and cache)
318 if( cached_language_version != current_language_version ||
319 ( raw_pl && cached_num != num ) || !cached_translation ) {
320 cached_language_version = current_language_version;
321 cached_num = num;
322
323 if( !ctxt ) {
324 if( !raw_pl ) {
325 cached_translation = cata::make_value<std::string>( detail::_translate_internal( raw ) );
326 } else {
327 cached_translation = cata::make_value<std::string>(
328 vgettext( raw.c_str(), raw_pl->c_str(), num ) );
329 }
330 } else {
331 if( !raw_pl ) {
332 cached_translation = cata::make_value<std::string>( pgettext( ctxt->c_str(), raw.c_str() ) );
333 } else {
334 cached_translation = cata::make_value<std::string>(
335 vpgettext( ctxt->c_str(), raw.c_str(), raw_pl->c_str(), num ) );
336 }
337 }
338 }
339 return *cached_translation;
340}
