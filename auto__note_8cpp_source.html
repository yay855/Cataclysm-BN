66void auto_note_settings::load()
67{
68 clear();
69
70 const auto parseJson = [&]( JsonIn & jin ) {
71 jin.start_object();
72
73 while( !jin.end_object() ) {
74 const std::string name = jin.get_member_name();
75
76 if( name == "enabled" ) {
77 jin.start_array();
78 while( !jin.end_array() ) {
79 const std::string entry = jin.get_string();
80 autoNoteEnabled.insert( string_id<map_extra> {entry} );
81 }
82 } else if( name == "discovered" ) {
83 jin.start_array();
84 while( !jin.end_array() ) {
85 const std::string entry = jin.get_string();
86 discovered.insert( string_id<map_extra> {entry} );
87 }
88 } else {
89 jin.skip_value();
90 }
91
92 }
93 };
94
95 if( !read_from_file_optional_json( build_save_path(), parseJson ) ) {
96 default_initialize();
97 save();
98 }
