94/**
95 * Value and metadata for one property of an item
96 *
97 * Contains the value of one property of an item, as well as various metadata items required to
98 * output that value. This is used primarily for user output of information about an item, for
99 * example in the various inventory menus. See @ref item::info() for the main example of how a
100 * class desiring to provide user output might obtain a class of this type.
101 *
102 * As an example, if the item being queried was a piece of clothing, then several properties might
103 * be returned. All would have sType "ARMOR". There would be one for the coverage stat with
104 * sName "Coverage: ", another for the warmth stat with sName "Warmth: ", etc.
105 */
106struct iteminfo {
107 public:
108 /** Category of item that owns this iteminfo. See @ref item_category. */
109 std::string sType;
110
111 /** Main text of this property's name */
112 std::string sName;
113
114 /** Formatting text to be placed between the name and value of this item. */
115 std::string sFmt;
116
117 /** Numerical value of this property. Set to -999 if no compare value is present */
118 std::string sValue;
119
120 /** Internal double floating point version of value, for numerical comparisons */
121 double dValue;
122
123 /** Flag indicating type of sValue. True if integer, false if single decimal */
124 bool is_int;
125
126 /** Flag indicating whether a newline should be printed after printing this item */
127 bool bNewLine;
128
129 /** Reverses behavior of red/green text coloring; smaller values are green if true */
130 bool bLowerIsBetter;
131
132 /** Whether to print sName. If false, use for comparisons but don't print for user. */
133 bool bDrawName;
134
135 /** Whether to print a sign on positive values */
136 bool bShowPlus;
137
138 /** Flag indicating decimal with three points of precision. */
139 bool three_decimal;
140
141 enum flags {
142 no_flags = 0,
143 is_decimal = 1 << 0, ///< Print as decimal rather than integer
144 is_three_decimal = 1 << 1, ///< Print as decimal with three points of precision
145 no_newline = 1 << 2, ///< Do not follow with a newline
146 lower_is_better = 1 << 3, ///< Lower values are better for this stat
147 no_name = 1 << 4, ///< Do not print the name
148 show_plus = 1 << 5, ///< Use a + sign for positive values
149 };
150
151 /**
152 * @param Type The item type of the item this iteminfo belongs to.
153 * @param Name The name of the property this iteminfo describes.
154 * @param Fmt Formatting text desired between item name and value
155 * @param Flags Additional flags to customize this entry
156 * @param Value Numerical value of this property, -999 for none.
157 */
158 iteminfo( const std::string &Type, const std::string &Name, const std::string &Fmt = "",
159 flags Flags = no_flags, double Value = -999 );
160 iteminfo( const std::string &Type, const std::string &Name, double Value );
161};
