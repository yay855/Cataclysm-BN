72// A portable thread-safe version of strerror.
73// Sets buffer to point to a string describing the error code.
74// This can be either a pointer to a string stored in buffer,
75// or a pointer to some static immutable string.
76// Returns one of the following values:
77// 0 - success
78// ERANGE - buffer is not large enough to store the error message
79// other - failure
80// Buffer should be at least of size 1.
81inline int safe_strerror( int error_code, char *&buffer,
82 size_t buffer_size ) FMT_NOEXCEPT {
83 FMT_ASSERT( buffer != nullptr &&buffer_size != 0, "invalid buffer" );
84
85 class dispatcher {
86 private:
87 int error_code_;
88 char *&buffer_;
89 size_t buffer_size_;
90
91 // A noop assignment operator to avoid bogus warnings.
92 void operator=( const dispatcher & ) {}
93
94 // Handle the result of XSI-compliant version of strerror_r.
95 int handle( int result )
96 {
97 // glibc versions before 2.13 return result in errno.
98 return result == -1 ? errno : result;
99 }
100
101 // Handle the result of GNU-specific version of strerror_r.
102 FMT_MAYBE_UNUSED
103 int handle( char *message )
104 {
105 // If the buffer is full then the message is probably truncated.
106 if( message == buffer_ && strlen( buffer_ ) == buffer_size_ - 1 ) {
107 return ERANGE;
108 }
109 buffer_ = message;
110 return 0;
111 }
112
113 // Handle the case when strerror_r is not available.
114 FMT_MAYBE_UNUSED
115 int handle( detail::null<> )
116 {
117 return fallback( strerror_s( buffer_, buffer_size_, error_code_ ) );
118 }
119
120 // Fallback to strerror_s when strerror_r is not available.
121 FMT_MAYBE_UNUSED
122 int fallback( int result )
123 {
124 // If the buffer is full then the message is probably truncated.
125 return result == 0 && strlen( buffer_ ) == buffer_size_ - 1 ? ERANGE
126 : result;
127 }
128
129#if !FMT_MSC_VER
130 // Fallback to strerror if strerror_r and strerror_s are not available.
131 int fallback( detail::null<> )
132 {
133 errno = 0;
134 buffer_ = strerror( error_code_ );
135 return errno;
136 }
137#endif
138
139 public:
140 dispatcher( int err_code, char *&buf, size_t buf_size )
141 : error_code_( err_code ), buffer_( buf ), buffer_size_( buf_size ) {}
142
143 int run()
144 {
145 return handle( strerror_r( error_code_, buffer_, buffer_size_ ) );
146 }
147 };
148 return dispatcher( error_code, buffer, buffer_size ).run();
149}

151// NOLINTNEXTLINE(misc-definitions-in-headers, bugprone-exception-escape)
152FMT_FUNC void format_error_code( detail::buffer<char> &out, int error_code,
153 string_view message ) FMT_NOEXCEPT {
154 // Report error code making sure that the output fits into
155 // inline_buffer_size to avoid dynamic memory allocation and potential
156 // bad_alloc.
157 out.try_resize( 0 );
158 static const char SEP[] = ": ";
159 static const char ERROR_STR[] = "error ";
160 // Subtract 2 to account for terminating null characters in SEP and ERROR_STR.
161 size_t error_code_size = sizeof( SEP ) + sizeof( ERROR_STR ) - 2;
162 auto abs_value = static_cast<uint32_or_64_or_128_t<int>>( error_code );
163 if( detail::is_negative( error_code ) )
164 {
165 abs_value = 0 - abs_value;
166 ++error_code_size;
167 }
168 error_code_size += detail::to_unsigned( detail::count_digits( abs_value ) );
169 auto it = buffer_appender<char>( out );
170 if( message.size() <= inline_buffer_size - error_code_size )
171 {
172 format_to( it, "{}{}", message, SEP );
173 }
174 format_to( it, "{}{}", ERROR_STR, error_code );
175 assert( out.size() <= inline_buffer_size );
176}

178// NOLINTNEXTLINE(misc-definitions-in-headers)
179FMT_FUNC void report_error( format_func func, int error_code,
180 string_view message ) FMT_NOEXCEPT {
181 memory_buffer full_message;
182 func( full_message, error_code, message );
183 // Don't use fwrite_fully because the latter may throw.
184 ( void )std::fwrite( full_message.data(), full_message.size(), 1, stderr );
185 std::fputc( '\n', stderr );
186}

188// A wrapper around fwrite that throws on error.
189inline void fwrite_fully( const void *ptr, size_t size, size_t count,
190 FILE *stream )
191{
192 size_t written = std::fwrite( ptr, size, count, stream );
193 if( written < count ) {
194 FMT_THROW( system_error( errno, "cannot write to file" ) );
195 }
196}

266template <> FMT_FUNC int count_digits<4>( detail::fallback_uintptr n )
267{
268 // fallback_uintptr is always stored in little endian.
269 int i = static_cast<int>( sizeof( void * ) ) - 1;
270 while( i > 0 && n.value[i] == 0 ) {
271 --i;
272 }
273 auto char_digits = std::numeric_limits<unsigned char>::digits / 4;
274 return i >= 0 ? i * char_digits + count_digits<4, unsigned>( n.value[i] ) : 1;
275}

1155{
1156 private:

1161 sizeof( Float ) == sizeof( uint32_t ) >;

1164 significand_type f;
1165 int e;

1167 // All sizes are in bits.
1168 // Subtract 1 to account for an implicit most significant bit in the
1169 // normalized form.
1170 static FMT_CONSTEXPR_DECL const int double_significand_size =
1171 std::numeric_limits<double>::digits - 1;
1172 static FMT_CONSTEXPR_DECL const uint64_t implicit_bit =
1173 1ULL << double_significand_size;

1175 bits<significand_type>::value;

1181 // errors on platforms where double is not IEEE754.

1186 // Assigns d to this and return true iff predecessor is closer than successor.
1187 template <typename Float, FMT_ENABLE_IF( is_supported_float<Float>::value )>
1188 bool assign( Float d ) {
1189 // Assume float is in the format [sign][exponent][significand].
1190 using limits = std::numeric_limits<Float>;
1191 const int float_significand_size = limits::digits - 1;
1192 const int exponent_size =
1193 bits<Float>::value - float_significand_size - 1; // -1 for sign
1194 const uint64_t float_implicit_bit = 1ULL << float_significand_size;
1195 const uint64_t significand_mask = float_implicit_bit - 1;
1196 const uint64_t exponent_mask = ( ~0ULL >> 1 ) & ~significand_mask;
1197 const int exponent_bias = ( 1 << exponent_size ) - limits::max_exponent - 1;
1198 constexpr bool is_double = sizeof( Float ) == sizeof( uint64_t );
1199 auto u = bit_cast<conditional_t<is_double, uint64_t, uint32_t>>( d );
1200 f = u & significand_mask;
1201 int biased_e =
1202 static_cast<int>( ( u & exponent_mask ) >> float_significand_size );
1203 // Predecessor is closer if d is a normalized power of 2 (f == 0) other than
1204 // the smallest normalized number (biased_e > 1).
1205 bool is_predecessor_closer = f == 0 && biased_e > 1;
1206 if( biased_e != 0 ) {
1207 f += float_implicit_bit;
1208 } else {
1209 biased_e = 1; // Subnormals use biased exponent 1 (min exponent).
1210 }
1211 e = biased_e - exponent_bias - float_significand_size;
1212 return is_predecessor_closer;
1213 }

1215 template < typename Float, FMT_ENABLE_IF( !is_supported_float<Float>::value ) >
1216 bool assign( Float ) {
1217 *this = fp();
1218 return false;
1219 }

1222// Normalizes the value converted from double and multiplied by (1 << SHIFT).
1223template <int SHIFT> fp normalize( fp value )
1224{
1225 // Handle subnormals.
1226 const auto shifted_implicit_bit = fp::implicit_bit << SHIFT;
1227 while( ( value.f & shifted_implicit_bit ) == 0 ) {
1228 value.f <<= 1;
1229 --value.e;
1230 }
1231 // Subtract 1 to account for hidden bit.
1232 const auto offset =
1233 fp::significand_size - fp::double_significand_size - SHIFT - 1;
1234 value.f <<= offset;
1235 value.e -= offset;
1236 return value;
1237}

1244// Computes lhs * rhs / pow(2, 64) rounded to nearest with half-up tie breaking.
1245inline uint64_t multiply( uint64_t lhs, uint64_t rhs )
1246{
1247#if FMT_USE_INT128
1248 auto product = static_cast<__uint128_t>( lhs ) * rhs;
1249 auto f = static_cast<uint64_t>( product >> 64 );
1250 return ( static_cast<uint64_t>( product ) & ( 1ULL << 63 ) ) != 0 ? f + 1 : f;
1251#else
1252 // Multiply 32-bit parts of significands.
1253 uint64_t mask = ( 1ULL << 32 ) - 1;
1254 uint64_t a = lhs >> 32, b = lhs & mask;
1255 uint64_t c = rhs >> 32, d = rhs & mask;
1256 uint64_t ac = a * c, bc = b * c, ad = a * d, bd = b * d;
1257 // Compute mid 64-bit of result and round.
1258 uint64_t mid = ( bd >> 32 ) + ( ad & mask ) + ( bc & mask ) + ( 1U << 31 );
1259 return ac + ( ad >> 32 ) + ( bc >> 32 ) + ( mid >> 32 );
1260#endif
1261}

1264{

1481 int num_rhs_bigits = rhs.num_bigits();
1482 if( max_lhs_bigits + 1 < num_rhs_bigits ) {
1483 return -1;

1485 if( max_lhs_bigits > num_rhs_bigits ) {
1486 return 1;

1489 return i >= n.exp_ && i < n.num_bigits() ? n[i - n.exp_] : 0;
1490 };
1491 double_bigit borrow = 0;
1492 int min_exp = ( std::min )( ( std::min )( lhs1.exp_, lhs2.exp_ ), rhs.exp_ );
1493 for( int i = num_rhs_bigits - 1; i >= min_exp; --i ) {

1495 static_cast<double_bigit>( get_bigit( lhs1, i ) ) + get_bigit( lhs2, i );
1496 bigit rhs_bigit = get_bigit( rhs, i );
1497 if( sum > rhs_bigit + borrow ) {
1498 return 1;
1499 }
1500 borrow = rhs_bigit + borrow - sum;
1501 if( borrow > 1 ) {
1502 return -1;
1503 }
1504 borrow <<= bigit_bits;
1505 }

1522 // repeated squaring and multiplication.

1526 square();

1528 *this *= 5;

1530 bitmask >>= 1;
1531 }
1532 *this <<= exp; // Multiply by pow(2, exp) by shifting.

1536 basic_memory_buffer<bigit, bigits_capacity> n( std::move( bigits_ ) );
1537 int num_bigits = static_cast<int>( bigits_.size() );
1538 int num_result_bigits = 2 * num_bigits;
1539 bigits_.resize( to_unsigned( num_result_bigits ) );
1540 using accumulator_t = conditional_t<FMT_USE_INT128, uint128_t, accumulator>;
1541 auto sum = accumulator_t();
1542 for( int bigit_index = 0; bigit_index < num_bigits; ++bigit_index ) {
1543 // Compute bigit at position bigit_index of the result by adding
1544 // cross-product terms n[i] * n[j] such that i + j == bigit_index.
1545 for( int i = 0, j = bigit_index; j >= 0; ++i, --j ) {
1546 // Most terms are multiplied twice which can be optimized in the future.
1547 sum += static_cast<double_bigit>( n[i] ) * n[j];
1548 }
1549 ( *this )[bigit_index] = static_cast<bigit>( sum );
1550 sum >>= bits<bigit>::value; // Compute the carry.
1551 }
1552 // Do the same for the top half.
1553 for( int bigit_index = num_bigits; bigit_index < num_result_bigits;
1554 ++bigit_index ) {
1555 for( int j = num_bigits - 1, i = bigit_index - j; i < num_bigits; ) {
1556 sum += static_cast<double_bigit>( n[i++] ) * n[j--];
1557 }
1558 ( *this )[bigit_index] = static_cast<bigit>( sum );
1559 sum >>= bits<bigit>::value;
1560 }
1561 --num_result_bigits;
1562 remove_leading_zeros();
1563 exp_ *= 2;
1564 }
1565
1566 // If this bigint has a bigger exponent than other, adds trailing zero to make
1567 // exponents equal. This simplifies some operations such as subtraction.
1568 void align( const bigint &other ) {
1569 int exp_difference = exp_ - other.exp_;
1570 if( exp_difference <= 0 ) {
1571 return;
1572 }
1573 int num_bigits = static_cast<int>( bigits_.size() );
1574 bigits_.resize( to_unsigned( num_bigits + exp_difference ) );
1575 for( int i = num_bigits - 1, j = i + exp_difference; i >= 0; --i, --j ) {
1576 bigits_[j] = bigits_[i];
1577 }
1578 std::uninitialized_fill_n( bigits_.data(), exp_difference, 0 );
1579 exp_ -= exp_difference;
1580 }
1581
1582 // Divides this bignum by divisor, assigning the remainder to this and
1583 // returning the quotient.
1584 int divmod_assign( const bigint &divisor ) {
1585 FMT_ASSERT( this != &divisor, "" );
1586 if( compare( *this, divisor ) < 0 ) {
1587 return 0;
1588 }
1589 FMT_ASSERT( divisor.bigits_[divisor.bigits_.size() - 1u] != 0, "" );
1590 align( divisor );
1591 int quotient = 0;
1592 do {
1593 subtract_aligned( divisor );
1594 ++quotient;
1595 } while( compare( *this, divisor ) >= 0 );
1596 return quotient;
1597 }
1598};

1600enum class round_direction { unknown, up, down };
1601
1602// Given the divisor (normally a power of 10), the remainder = v % divisor for
1603// some number v and the error, returns whether v should be rounded up, down, or
1604// whether the rounding direction can't be determined due to error.
1605// error should be less than divisor / 2.
1606inline round_direction get_round_direction( uint64_t divisor, uint64_t remainder,
1607 uint64_t error )
1608{
1609 FMT_ASSERT( remainder < divisor, "" ); // divisor - remainder won't overflow.
1610 FMT_ASSERT( error < divisor, "" ); // divisor - error won't overflow.
1611 FMT_ASSERT( error < divisor - error, "" ); // error * 2 won't overflow.
1612 // Round down if (remainder + error) * 2 <= divisor.
1613 if( remainder <= divisor - remainder && error * 2 <= divisor - remainder * 2 ) {
1614 return round_direction::down;
1615 }
1616 // Round up if (remainder - error) * 2 >= divisor.
1617 if( remainder >= error &&
1618 remainder - error >= divisor - ( remainder - error ) ) {
1619 return round_direction::up;
1620 }
1621 return round_direction::unknown;
1622}

1633// Generates output using the Grisu digit-gen algorithm.
1634// error: the size of the region (lower, upper) outside of which numbers
1635// definitely do not round to value (Delta in Grisu3).
1636template <typename Handler>
1637FMT_ALWAYS_INLINE digits::result grisu_gen_digits( fp value, uint64_t error,
1638 int &exp, Handler &handler )
1639{
1640 const fp one( 1ULL << -value.e, value.e );
1641 // The integral part of scaled value (p1 in Grisu) = value / one. It cannot be
1642 // zero because it contains a product of two 64-bit numbers with MSB set (due
1643 // to normalization) - 1, shifted right by at most 60 bits.
1644 auto integral = static_cast<uint32_t>( value.f >> -one.e );
1645 FMT_ASSERT( integral != 0, "" );
1646 FMT_ASSERT( integral == value.f >> -one.e, "" );
1647 // The fractional part of scaled value (p2 in Grisu) c = value % one.
1648 uint64_t fractional = value.f & ( one.f - 1 );
1649 exp = count_digits( integral ); // kappa in Grisu.
1650 // Divide by 10 to prevent overflow.
1651 auto result = handler.on_start( data::powers_of_10_64[exp - 1] << -one.e,
1652 value.f / 10, error * 10, exp );
1653 if( result != digits::more ) {
1654 return result;
1655 }
1656 // Generate digits for the integral part. This can produce up to 10 digits.
1657 do {
1658 uint32_t digit = 0;
1659 auto divmod_integral = [&]( uint32_t divisor ) {
1660 digit = integral / divisor;
1661 integral %= divisor;
1662 };
1663 // This optimization by Milo Yip reduces the number of integer divisions by
1664 // one per iteration.
1665 switch( exp ) {
1666 case 10:
1667 divmod_integral( 1000000000 );
1668 break;
1669 case 9:
1670 divmod_integral( 100000000 );
1671 break;
1672 case 8:
1673 divmod_integral( 10000000 );
1674 break;
1675 case 7:
1676 divmod_integral( 1000000 );
1677 break;
1678 case 6:
1679 divmod_integral( 100000 );
1680 break;
1681 case 5:
1682 divmod_integral( 10000 );
1683 break;
1684 case 4:
1685 divmod_integral( 1000 );
1686 break;
1687 case 3:
1688 divmod_integral( 100 );
1689 break;
1690 case 2:
1691 divmod_integral( 10 );
1692 break;
1693 case 1:
1694 digit = integral;
1695 integral = 0;
1696 break;
1697 default:
1698 FMT_ASSERT( false, "invalid number of digits" );
1699 }
1700 --exp;
1701 auto remainder = ( static_cast<uint64_t>( integral ) << -one.e ) + fractional;
1702 result = handler.on_digit( static_cast<char>( '0' + digit ),
1703 data::powers_of_10_64[exp] << -one.e, remainder,
1704 error, exp, true );
1705 if( result != digits::more ) {
1706 return result;
1707 }
1708 } while( exp > 0 );
1709 // Generate digits for the fractional part.
1710 for( ;; ) {
1711 fractional *= 10;
1712 error *= 10;
1713 char digit = static_cast<char>( '0' + ( fractional >> -one.e ) );
1714 fractional &= one.f - 1;
1715 --exp;
1716 result = handler.on_digit( digit, one.f, fractional, error, exp, false );
1717 if( result != digits::more ) {
1718 return result;
1719 }
1720 }
1721}

1723// The fixed precision digit handler.
1724struct fixed_handler {
1725 char *buf;
1726 int size;
1727 int precision;
1728 int exp10;
1729 bool fixed;
1730
1731 digits::result on_start( uint64_t divisor, uint64_t remainder, uint64_t error,
1732 int &exp ) {
1733 // Non-fixed formats require at least one digit and no precision adjustment.
1734 if( !fixed ) {
1735 return digits::more;
1736 }
1737 // Adjust fixed precision by exponent because it is relative to decimal
1738 // point.
1739 precision += exp + exp10;
1740 // Check if precision is satisfied just by leading zeros, e.g.
1741 // format("{:.2f}", 0.001) gives "0.00" without generating any digits.
1742 if( precision > 0 ) {
1743 return digits::more;
1744 }
1745 if( precision < 0 ) {
1746 return digits::done;
1747 }
1748 auto dir = get_round_direction( divisor, remainder, error );
1749 if( dir == round_direction::unknown ) {
1750 return digits::error;
1751 }
1752 buf[size++] = dir == round_direction::up ? '1' : '0';
1753 return digits::done;
1754 }
1755
1756 digits::result on_digit( char digit, uint64_t divisor, uint64_t remainder,
1757 uint64_t error, int, bool integral ) {
1758 FMT_ASSERT( remainder < divisor, "" );
1759 buf[size++] = digit;
1760 if( !integral && error >= remainder ) {
1761 return digits::error;
1762 }
1763 if( size < precision ) {
1764 return digits::more;
1765 }
1766 if( !integral ) {
1767 // Check if error * 2 < divisor with overflow prevention.
1768 // The check is not needed for the integral part because error = 1
1769 // and divisor > (1 << 32) there.
1770 if( error >= divisor || error >= divisor - error ) {
1771 return digits::error;
1772 }
1773 } else {
1774 FMT_ASSERT( error == 1 && divisor > 2, "" );
1775 }
1776 auto dir = get_round_direction( divisor, remainder, error );
1777 if( dir != round_direction::up ) {
1778 return dir == round_direction::down ? digits::done : digits::error;
1779 }
1780 ++buf[size - 1];
1781 for( int i = size - 1; i > 0 && buf[i] > '9'; --i ) {
1782 buf[i] = '0';
1783 ++buf[i - 1];
1784 }
1785 if( buf[0] > '9' ) {
1786 buf[0] = '1';
1787 if( fixed ) {
1788 buf[size++] = '0';
1789 } else {
1790 ++exp10;
1791 }
1792 }
1793 return digits::done;
1794 }
1795};

1800// Computes 128-bit result of multiplication of two 64-bit unsigned integers.
1801// NOLINTNEXTLINE(cata-xy)
1802FMT_SAFEBUFFERS inline uint128_wrapper umul128( uint64_t x, uint64_t y ) FMT_NOEXCEPT {
1803#if FMT_USE_INT128
1804 return static_cast<uint128_t>( x ) * static_cast<uint128_t>( y );
1805#elif defined(_MSC_VER) && defined(_M_X64)
1806 uint128_wrapper result;
1807 result.low_ = _umul128( x, y, &result.high_ );
1808 return result;
1809#else
1810 const uint64_t mask = ( uint64_t( 1 ) << 32 ) - uint64_t( 1 );
1811
1812 uint64_t a = x >> 32;
1813 uint64_t b = x & mask;
1814 uint64_t c = y >> 32;
1815 uint64_t d = y & mask;
1816
1817 uint64_t ac = a * c;
1818 uint64_t bc = b * c;
1819 uint64_t ad = a * d;
1820 uint64_t bd = b * d;
1821
1822 uint64_t intermediate = ( bd >> 32 ) + ( ad & mask ) + ( bc & mask );
1823
1824 return {
1825 ac + ( intermediate >> 32 ) + ( ad >> 32 ) + ( bc >> 32 ),
1826 ( intermediate << 32 ) + ( bd & mask )};
1827#endif
1828}

1836#elif defined(_MSC_VER) && defined(_M_X64)

1838#else
1839 return umul128( x, y ).high();
1840#endif

2042 FMT_ASSERT( k >= float_info<double>::min_k &&k <= float_info<double>::max_k,
2043 "k is out of range" );
2044
2045#if FMT_USE_FULL_CACHE_DRAGONBOX
2046 return data::dragonbox_pow10_significands_128[k -
2047 float_info<double>::min_k];
2048#else
2049 static const int compression_ratio = 27;
2050
2051 // Compute base index.
2052 int cache_index = ( k - float_info<double>::min_k ) / compression_ratio;
2053 int kb = cache_index * compression_ratio + float_info<double>::min_k;
2054 int offset = k - kb;
2055
2056 // Get base cache.
2057 uint128_wrapper base_cache =
2058 data::dragonbox_pow10_significands_128[cache_index];
2059 if( offset == 0 ) {
2060 return base_cache;
2061 }
2062
2063 // Compute the required amount of bit-shift.
2064 int alpha = floor_log2_pow10( kb + offset ) - floor_log2_pow10( kb ) - offset;
2065 FMT_ASSERT( alpha > 0 && alpha < 64, "shifting error detected" );
2066
2067 // Try to recover the real cache.
2068 uint64_t pow5 = data::powers_of_5_64[offset];
2069 uint128_wrapper recovered_cache = umul128( base_cache.high(), pow5 );
2070 uint128_wrapper middle_low =
2071 umul128( base_cache.low() - ( kb < 0 ? 1u : 0u ), pow5 );
2072
2073 recovered_cache += middle_low.high();
2074
2075 uint64_t high_to_middle = recovered_cache.high() << ( 64 - alpha );
2076 uint64_t middle_to_low = recovered_cache.low() << ( 64 - alpha );
2077
2078 recovered_cache =
2079 uint128_wrapper{
2080 ( recovered_cache.low() >> alpha ) | high_to_middle,
2081 ( ( middle_low.low() >> alpha ) | middle_to_low )};
2082
2083 if( kb < 0 ) {
2084 recovered_cache += 1;
2085 }
2086
2087 // Get error.
2088 int error_idx = ( k - float_info<double>::min_k ) / 16;
2089 uint32_t error = ( data::dragonbox_pow10_recovery_errors[error_idx] >>
2090 ( ( k - float_info<double>::min_k ) % 16 ) * 2 ) &
2091 0x3;
2092
2093 // Add the error back.
2094 FMT_ASSERT( recovered_cache.low() + error >= recovered_cache.low(), "" );
2095 return {recovered_cache.high(), recovered_cache.low() + error};
2096#endif
2097 }

2323template <class T>
2324FMT_ALWAYS_INLINE FMT_SAFEBUFFERS decimal_fp<T> shorter_interval_case(
2325 int exponent ) FMT_NOEXCEPT {
2326 decimal_fp<T> ret_value;
2327 // Compute k and beta
2328 const int minus_k = floor_log10_pow2_minus_log10_4_over_3( exponent );
2329 const int beta_minus_1 = exponent + floor_log2_pow10( -minus_k );
2330
2331 // Compute xi and zi
2332 using cache_entry_type = typename cache_accessor<T>::cache_entry_type;
2333 const cache_entry_type cache = cache_accessor<T>::get_cached_power( -minus_k );
2334
2335 auto xi = cache_accessor<T>::compute_left_endpoint_for_shorter_interval_case(
2336 cache, beta_minus_1 );
2337 auto zi = cache_accessor<T>::compute_right_endpoint_for_shorter_interval_case(
2338 cache, beta_minus_1 );
2339
2340 // If the left endpoint is not an integer, increase it
2341 if( !is_left_endpoint_integer_shorter_interval<T>( exponent ) )
2342 {
2343 ++xi;
2344 }
2345
2346 // Try bigger divisor
2347 ret_value.significand = zi / 10;
2348
2349 // If succeed, remove trailing zeros if necessary and return
2350 if( ret_value.significand * 10 >= xi )
2351 {
2352 ret_value.exponent = minus_k + 1;
2353 ret_value.exponent += remove_trailing_zeros( ret_value.significand );
2354 return ret_value;
2355 }
2356
2357 // Otherwise, compute the round-up of y
2358 ret_value.significand =
2359 cache_accessor<T>::compute_round_up_for_shorter_interval_case(
2360 cache, beta_minus_1 );
2361 ret_value.exponent = minus_k;
2362
2363 // When tie occurs, choose one of them according to the rule
2364 if( exponent >= float_info<T>::shorter_interval_tie_lower_threshold &&
2365 exponent <= float_info<T>::shorter_interval_tie_upper_threshold )
2366 {
2367 ret_value.significand = ret_value.significand % 2 == 0
2368 ? ret_value.significand
2369 : ret_value.significand - 1;
2370 } else if( ret_value.significand < xi )
2371 {
2372 ++ret_value.significand;
2373 }
2374 return ret_value;
2375}

2377template <typename T>
2378FMT_SAFEBUFFERS decimal_fp<T> to_decimal( T x ) FMT_NOEXCEPT {
2379 // Step 1: integer promotion & Schubfach multiplier calculation.
2380
2381 using carrier_uint = typename float_info<T>::carrier_uint;
2382 using cache_entry_type = typename cache_accessor<T>::cache_entry_type;
2383 auto br = bit_cast<carrier_uint>( x );
2384
2385 // Extract significand bits and exponent bits.
2386 const carrier_uint significand_mask =
2387 ( static_cast<carrier_uint>( 1 ) << float_info<T>::significand_bits ) - 1;
2388 carrier_uint significand = ( br & significand_mask );
2389 int exponent = static_cast<int>( ( br & exponent_mask<T>() ) >>
2390 float_info<T>::significand_bits );
2391
2392 if( exponent != 0 ) // Check if normal.
2393 {
2394 exponent += float_info<T>::exponent_bias - float_info<T>::significand_bits;
2395
2396 // Shorter interval case; proceed like Schubfach.
2397 if( significand == 0 ) {
2398 return shorter_interval_case<T>( exponent );
2399 }
2400
2401 significand |=
2402 ( static_cast<carrier_uint>( 1 ) << float_info<T>::significand_bits );
2403 } else
2404 {
2405 // Subnormal case; the interval is always regular.
2406 if( significand == 0 ) return {0, 0};
2407 exponent = float_info<T>::min_exponent - float_info<T>::significand_bits;
2408 }
2409
2410 const bool include_left_endpoint = ( significand % 2 == 0 );
2411 const bool include_right_endpoint = include_left_endpoint;
2412
2413 // Compute k and beta.
2414 const int minus_k = floor_log10_pow2( exponent ) - float_info<T>::kappa;
2415 const cache_entry_type cache = cache_accessor<T>::get_cached_power( -minus_k );
2416 const int beta_minus_1 = exponent + floor_log2_pow10( -minus_k );
2417
2418 // Compute zi and deltai
2419 // 10^kappa <= deltai < 10^(kappa + 1)
2420 const uint32_t deltai = cache_accessor<T>::compute_delta( cache, beta_minus_1 );
2421 const carrier_uint two_fc = significand << 1;
2422 const carrier_uint two_fr = two_fc | 1;
2423 const carrier_uint zi =
2424 cache_accessor<T>::compute_mul( two_fr << beta_minus_1, cache );
2425
2426 // Step 2: Try larger divisor; remove trailing zeros if necessary
2427
2428 // Using an upper bound on zi, we might be able to optimize the division
2429 // better than the compiler; we are computing zi / big_divisor here
2430 decimal_fp<T> ret_value;
2431 ret_value.significand = divide_by_10_to_kappa_plus_1( zi );
2432 uint32_t r = static_cast<uint32_t>( zi - float_info<T>::big_divisor *
2433 ret_value.significand );
2434
2435 if( r > deltai )
2436 {
2437 goto small_divisor_case_label;
2438 } else if( r < deltai )
2439 {
2440 // Exclude the right endpoint if necessary
2441 if( r == 0 && !include_right_endpoint &&
2442 is_endpoint_integer<T>( two_fr, exponent, minus_k ) ) {
2443 --ret_value.significand;
2444 r = float_info<T>::big_divisor;
2445 goto small_divisor_case_label;
2446 }
2447 } else
2448 {
2449 // r == deltai; compare fractional parts
2450 // Check conditions in the order different from the paper
2451 // to take advantage of short-circuiting
2452 const carrier_uint two_fl = two_fc - 1;
2453 if( ( !include_left_endpoint ||
2454 !is_endpoint_integer<T>( two_fl, exponent, minus_k ) ) &&
2455 !cache_accessor<T>::compute_mul_parity( two_fl, cache, beta_minus_1 ) ) {
2456 goto small_divisor_case_label;
2457 }
2458 }
2459 ret_value.exponent = minus_k + float_info<T>::kappa + 1;
2460
2461 // We may need to remove trailing zeros
2462 ret_value.exponent += remove_trailing_zeros( ret_value.significand );
2463 return ret_value;
2464
2465 // Step 3: Find the significand with the smaller divisor
2466
2467small_divisor_case_label:
2468 ret_value.significand *= 10;
2469 ret_value.exponent = minus_k + float_info<T>::kappa;
2470
2471 const uint32_t mask = ( 1u << float_info<T>::kappa ) - 1;
2472 auto dist = r - ( deltai / 2 ) + ( float_info<T>::small_divisor / 2 );
2473
2474 // Is dist divisible by 2^kappa?
2475 if( ( dist & mask ) == 0 )
2476 {
2477 const bool approx_y_parity =
2478 ( ( dist ^ ( float_info<T>::small_divisor / 2 ) ) & 1 ) != 0;
2479 dist >>= float_info<T>::kappa;
2480
2481 // Is dist divisible by 5^kappa?
2482 if( check_divisibility_and_divide_by_pow5<float_info<T>::kappa>( dist ) ) {
2483 ret_value.significand += dist;
2484
2485 // Check z^(f) >= epsilon^(f)
2486 // We have either yi == zi - epsiloni or yi == (zi - epsiloni) - 1,
2487 // where yi == zi - epsiloni if and only if z^(f) >= epsilon^(f)
2488 // Since there are only 2 possibilities, we only need to care about the
2489 // parity. Also, zi and r should have the same parity since the divisor
2490 // is an even number
2491 if( cache_accessor<T>::compute_mul_parity( two_fc, cache, beta_minus_1 ) !=
2492 approx_y_parity ) {
2493 --ret_value.significand;
2494 } else {
2495 // If z^(f) >= epsilon^(f), we might have a tie
2496 // when z^(f) == epsilon^(f), or equivalently, when y is an integer
2497 if( is_center_integer<T>( two_fc, exponent, minus_k ) ) {
2498 ret_value.significand = ret_value.significand % 2 == 0
2499 ? ret_value.significand
2500 : ret_value.significand - 1;
2501 }
2502 }
2503 }
2504 // Is dist not divisible by 5^kappa?
2505 else {
2506 ret_value.significand += dist;
2507 }
2508 }
2509 // Is dist not divisible by 2^kappa?
2510 else
2511 {
2512 // Since we know dist is small, we might be able to optimize the division
2513 // better than the compiler; we are computing dist / small_divisor here
2514 ret_value.significand +=
2515 small_division_by_pow10<float_info<T>::kappa>( dist );
2516 }
2517 return ret_value;
2518}

2521// Formats value using a variation of the Fixed-Precision Positive
2522// Floating-Point Printout ((FPP)^2) algorithm by Steele & White:
2523// https://fmt.dev/p372-steele.pdf.
2524template <typename Double>
2525void fallback_format( Double d, int num_digits, bool binary32, buffer<char> &buf,
2526 int &exp10 )
2527{
2528 bigint numerator; // 2 * R in (FPP)^2.
2529 bigint denominator; // 2 * S in (FPP)^2.
2530 // lower and upper are differences between value and corresponding boundaries.
2531 bigint lower; // (M^- in (FPP)^2).
2532 bigint upper_store; // upper's value if different from lower.
2533 bigint *upper = nullptr; // (M^+ in (FPP)^2).
2534 fp value;
2535 // Shift numerator and denominator by an extra bit or two (if lower boundary
2536 // is closer) to make lower and upper integers. This eliminates multiplication
2537 // by 2 during later computations.
2538 const bool is_predecessor_closer =
2539 binary32 ? value.assign( static_cast<float>( d ) ) : value.assign( d );
2540 int shift = is_predecessor_closer ? 2 : 1;
2541 uint64_t significand = value.f << shift;
2542 if( value.e >= 0 ) {
2543 numerator.assign( significand );
2544 numerator <<= value.e;
2545 lower.assign( 1 );
2546 lower <<= value.e;
2547 if( shift != 1 ) {
2548 upper_store.assign( 1 );
2549 upper_store <<= value.e + 1;
2550 upper = &upper_store;
2551 }
2552 denominator.assign_pow10( exp10 );
2553 denominator <<= shift;
2554 } else if( exp10 < 0 ) {
2555 numerator.assign_pow10( -exp10 );
2556 lower.assign( numerator );
2557 if( shift != 1 ) {
2558 upper_store.assign( numerator );
2559 upper_store <<= 1;
2560 upper = &upper_store;
2561 }
2562 numerator *= significand;
2563 denominator.assign( 1 );
2564 denominator <<= shift - value.e;
2565 } else {
2566 numerator.assign( significand );
2567 denominator.assign_pow10( exp10 );
2568 denominator <<= shift - value.e;
2569 lower.assign( 1 );
2570 if( shift != 1 ) {
2571 upper_store.assign( 1ULL << 1 );
2572 upper = &upper_store;
2573 }
2574 }
2575 // Invariant: value == (numerator / denominator) * pow(10, exp10).
2576 if( num_digits < 0 ) {
2577 // Generate the shortest representation.
2578 if( !upper ) {
2579 upper = &lower;
2580 }
2581 bool even = ( value.f & 1 ) == 0;
2582 num_digits = 0;
2583 char *data = buf.data();
2584 for( ;; ) {
2585 int digit = numerator.divmod_assign( denominator );
2586 bool low = compare( numerator, lower ) - even < 0; // numerator <[=] lower.
2587 // numerator + upper >[=] pow10:
2588 bool high = add_compare( numerator, *upper, denominator ) + even > 0;
2589 data[num_digits++] = static_cast<char>( '0' + digit );
2590 if( low || high ) {
2591 if( !low ) {
2592 ++data[num_digits - 1];
2593 } else if( high ) {
2594 int result = add_compare( numerator, numerator, denominator );
2595 // Round half to even.
2596 if( result > 0 || ( result == 0 && ( digit % 2 ) != 0 ) ) {
2597 ++data[num_digits - 1];
2598 }
2599 }
2600 buf.try_resize( to_unsigned( num_digits ) );
2601 exp10 -= num_digits - 1;
2602 return;
2603 }
2604 numerator *= 10;
2605 lower *= 10;
2606 if( upper != &lower ) {
2607 *upper *= 10;
2608 }
2609 }
2610 }
2611 // Generate the given number of digits.
2612 exp10 -= num_digits - 1;
2613 if( num_digits == 0 ) {
2614 buf.try_resize( 1 );
2615 denominator *= 10;
2616 buf[0] = add_compare( numerator, numerator, denominator ) > 0 ? '1' : '0';
2617 return;
2618 }
2619 buf.try_resize( to_unsigned( num_digits ) );
2620 for( int i = 0; i < num_digits - 1; ++i ) {
2621 int digit = numerator.divmod_assign( denominator );
2622 buf[i] = static_cast<char>( '0' + digit );
2623 numerator *= 10;
2624 }
2625 int digit = numerator.divmod_assign( denominator );
2626 auto result = add_compare( numerator, numerator, denominator );
2627 if( result > 0 || ( result == 0 && ( digit % 2 ) != 0 ) ) {
2628 if( digit == 9 ) {
2629 const auto overflow = '0' + 10;
2630 buf[num_digits - 1] = overflow;
2631 // Propagate the carry.
2632 for( int i = num_digits - 1; i > 0 && buf[i] == overflow; --i ) {
2633 buf[i] = '0';
2634 ++buf[i - 1];
2635 }
2636 if( buf[0] == overflow ) {
2637 buf[0] = '1';
2638 ++exp10;
2639 }
2640 return;
2641 }
2642 ++digit;
2643 }
2644 buf[num_digits - 1] = static_cast<char>( '0' + digit );
2645}

2647template <typename T>
2648int format_float( T value, int precision, float_specs specs, buffer<char> &buf )
2649{
2650 static_assert( !std::is_same<T, float>::value, "" );
2651 FMT_ASSERT( value >= 0, "value is negative" );
2652
2653 const bool fixed = specs.format == float_format::fixed;
2654 if( value <= 0 ) { // <= instead of == to silence a warning.
2655 if( precision <= 0 || !fixed ) {
2656 buf.push_back( '0' );
2657 return 0;
2658 }
2659 buf.try_resize( to_unsigned( precision ) );
2660 std::uninitialized_fill_n( buf.data(), precision, '0' );
2661 return -precision;
2662 }
2663
2664 if( !specs.use_grisu ) {
2665 return snprintf_float( value, precision, specs, buf );
2666 }
2667
2668 if( precision < 0 ) {
2669 // Use Dragonbox for the shortest format.
2670 if( specs.binary32 ) {
2671 auto dec = dragonbox::to_decimal( static_cast<float>( value ) );
2672 write<char>( buffer_appender<char>( buf ), dec.significand );
2673 return dec.exponent;
2674 }
2675 auto dec = dragonbox::to_decimal( static_cast<double>( value ) );
2676 write<char>( buffer_appender<char>( buf ), dec.significand );
2677 return dec.exponent;
2678 }
2679
2680 // Use Grisu + Dragon4 for the given precision:
2681 // https://www.cs.tufts.edu/~nr/cs257/archive/florian-loitsch/printf.pdf.
2682 int exp = 0;
2683 const int min_exp = -60; // alpha in Grisu.
2684 int cached_exp10 = 0; // K in Grisu.
2685 fp normalized = normalize( fp( value ) );
2686 const auto cached_pow = get_cached_power(
2687 min_exp - ( normalized.e + fp::significand_size ), cached_exp10 );
2688 normalized = normalized * cached_pow;
2689 // Limit precision to the maximum possible number of significant digits in an
2690 // IEEE754 double because we don't need to generate zeros.
2691 const int max_double_digits = 767;
2692 if( precision > max_double_digits ) {
2693 precision = max_double_digits;
2694 }
2695 fixed_handler handler{buf.data(), 0, precision, -cached_exp10, fixed};
2696 if( grisu_gen_digits( normalized, 1, exp, handler ) == digits::error ) {
2697 exp += handler.size - cached_exp10 - 1;
2698 fallback_format( value, handler.precision, specs.binary32, buf, exp );
2699 } else {
2700 exp += handler.exp10;
2701 buf.try_resize( to_unsigned( handler.size ) );
2702 }
2703 if( !fixed && !specs.showpoint ) {
2704 // Remove trailing zeros.
2705 auto num_digits = buf.size();
2706 while( num_digits > 0 && buf[num_digits - 1] == '0' ) {
2707 --num_digits;
2708 ++exp;
2709 }
2710 buf.try_resize( num_digits );
2711 }
2712 return exp;

2715template <typename T>
2716int snprintf_float( T value, int precision, float_specs specs,
2717 buffer<char> &buf )
2718{
2719 // Buffer capacity must be non-zero, otherwise MSVC's vsnprintf_s will fail.
2720 FMT_ASSERT( buf.capacity() > buf.size(), "empty buffer" );
2721 static_assert( !std::is_same<T, float>::value, "" );

2724 // both general and exponent format.

2731 enum { max_format_size = 7 }; // The longest format is "%#.*Le".
2732 char format[max_format_size];

2734 *format_ptr++ = '%';
2735 if( specs.showpoint && specs.format == float_format::hex ) {
2736 *format_ptr++ = '#';
2737 }
2738 if( precision >= 0 ) {
2739 *format_ptr++ = '.';
2740 *format_ptr++ = '*';
2741 }
2742 if( std::is_same<T, long double>() ) {
2743 *format_ptr++ = 'L';
2744 }
2745 *format_ptr++ = specs.format != float_format::hex
2746 ? ( specs.format == float_format::fixed ? 'f' : 'e' )
2747 : ( specs.upper ? 'A' : 'a' );
2748 *format_ptr = '\0';

2750 // Format using snprintf.
2751 auto offset = buf.size();
2752 for( ;; ) {
2753 auto begin = buf.data() + offset;
2754 auto capacity = buf.capacity() - offset;
2755#ifdef FMT_FUZZ
2756 if( precision > 100000 )
2757 throw std::runtime_error(
2758 "fuzz mode - avoid large allocation inside snprintf" );
2759#endif

2761 // Cannot use auto because of a bug in MinGW (#1532).
2762 int ( *snprintf_ptr )( char *, size_t, const char *, ... ) = FMT_SNPRINTF;
2763 int result = precision >= 0
2764 ? snprintf_ptr( begin, capacity, format, precision, value )
2765 : snprintf_ptr( begin, capacity, format, value );

2767 // The buffer will grow exponentially.
2768 buf.try_reserve( buf.capacity() + 1 );
2769 continue;
2770 }
2771 auto size = to_unsigned( result );
2772 // Size equal to capacity means that the last character was truncated.
2773 if( size >= capacity ) {
2774 buf.try_reserve( size + offset + 1 ); // Add 1 for the terminating '\0'.
2775 continue;
2776 }
2777 auto is_digit = []( char c ) {
2778 return c >= '0' && c <= '9';
2779 };
2780 if( specs.format == float_format::fixed ) {
2781 if( precision == 0 ) {
2782 buf.try_resize( size );
2783 return 0;
2784 }
2785 // Find and remove the decimal point.
2786 auto end = begin + size, p = end;
2787 do {
2788 --p;
2789 } while( is_digit( *p ) );
2790 int fraction_size = static_cast<int>( end - p - 1 );
2791 std::memmove( p, p + 1, to_unsigned( fraction_size ) );
2792 buf.try_resize( size - 1 );
2793 return -fraction_size;
2794 }
2795 if( specs.format == float_format::hex ) {
2796 buf.try_resize( size + offset );
2797 return 0;
2798 }
2799 // Find and parse the exponent.
2800 auto end = begin + size, exp_pos = end;
2801 do {
2802 --exp_pos;
2803 } while( *exp_pos != 'e' );
2804 char sign = exp_pos[1];
2805 assert( sign == '+' || sign == '-' );
2806 int exp = 0;
2807 auto p = exp_pos + 2; // Skip 'e' and sign.
2808 do {
2809 assert( is_digit( *p ) );
2810 exp = exp * 10 + ( *p++ - '0' );
2811 } while( p != end );
2812 if( sign == '-' ) {
2813 exp = -exp;
2814 }
2815 int fraction_size = 0;
2816 if( exp_pos != begin + 1 ) {
2817 // Remove trailing zeros.
2818 auto fraction_end = exp_pos - 1;
2819 while( *fraction_end == '0' ) {
2820 --fraction_end;
2821 }
2822 // Move the fractional part left to get rid of the decimal point.
2823 fraction_size = static_cast<int>( fraction_end - begin - 1 );
2824 std::memmove( begin + 1, begin + 2, to_unsigned( fraction_size ) );
2825 }
2826 buf.try_resize( to_unsigned( fraction_size ) + offset + 1 );
2827 return exp - fraction_size;
2828 }

2831// A public domain branchless UTF-8 decoder by Christopher Wellons:
2832// https://github.com/skeeto/branchless-utf8
2833/* Decode the next character, c, from buf, reporting errors in e.
2834 *
2835 * Since this is a branchless decoder, four bytes will be read from the
2836 * buffer regardless of the actual length of the next character. This
2837 * means the buffer _must_ have at least three bytes of zero padding
2838 * following the end of the data stream.
2839 *
2840 * Errors are reported in e, which will be non-zero if the parsed
2841 * character was somehow invalid: invalid byte sequence, non-canonical
2842 * encoding, or a surrogate half.
2843 *
2844 * The function returns a pointer to the next character. When an error
2845 * occurs, this pointer will be a guess that depends on the particular
2846 * error, but it will always advance at least one byte.
2847 */
2848inline const char *utf8_decode( const char *buf, uint32_t *c, int *e )
2849{
2850 static const int masks[] = {0x00, 0x7f, 0x1f, 0x0f, 0x07};
2851 static const uint32_t mins[] = {4194304, 0, 128, 2048, 65536};
2852 static const int shiftc[] = {0, 18, 12, 6, 0};
2853 static const int shifte[] = {0, 6, 4, 2, 0};
2854
2855 int len = code_point_length( buf );
2856 const char *next = buf + len;
2857
2858 // Assume a four-byte character and load four bytes. Unused bits are
2859 // shifted out.
2860 auto s = reinterpret_cast<const unsigned char *>( buf );
2861 *c = uint32_t( s[0] & masks[len] ) << 18;
2862 *c |= uint32_t( s[1] & 0x3f ) << 12;
2863 *c |= uint32_t( s[2] & 0x3f ) << 6;
2864 *c |= uint32_t( s[3] & 0x3f ) << 0;
2865 *c >>= shiftc[len];
2866
2867 // Accumulate the various error conditions.
2868 *e = ( *c < mins[len] ) << 6; // non-canonical encoding
2869 *e |= ( ( *c >> 11 ) == 0x1b ) << 7; // surrogate half?
2870 *e |= ( *c > 0x10FFFF ) << 8; // out of range?
2871 *e |= ( s[1] & 0xc0 ) >> 2;
2872 *e |= ( s[2] & 0xc0 ) >> 4;
2873 *e |= ( s[3] ) >> 6;
2874 *e ^= 0x2a; // top two bits of each tail byte correct?
2875 *e >>= shifte[len];
2876
2877 return next;
2878}
