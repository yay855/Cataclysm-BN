1003{

1005
1006 std::vector<const construction *> cons = constructions_by_group( group );
1007 std::map<tripoint, const construction *> valid;
1008 map &here = get_map();

1010 for( const construction *con : cons ) {
1011 if( p != g->u.pos() && can_construct( *con, p ) && player_can_build( g->u, total_inv, *con ) ) {
1012 valid[ p ] = con;

1020 here.drawsq( g->w_terrain, elem.first, drawsq_params().highlight( true ).show_items( true ) );
1021 }

1026 if( !pnt_ ) {
1027 return;

1029 const tripoint pnt = *pnt_;
1030
1031 if( valid.find( pnt ) == valid.end() ) {
1032 cons.front()->explain_failure( pnt );
1033 return;
1034 }
1035 // Maybe there is already a partial_con on an existing trap, that isn't caught by the usual trap-checking.
1036 // because the pre-requisite construction is already a trap anyway.
1037 // This shouldn't normally happen, unless it's a spike pit being built on a pit for example.
1038 partial_con *pre_c = here.partial_con_at( pnt );
1039 if( pre_c ) {
1040 add_msg( m_info,
1041 _( "There is already an unfinished construction there, examine it to continue working on it" ) );
1042 return;
1043 }
1044 std::list<item> used;
1045 const construction &con = *valid.find( pnt )->second;
1046 // create the partial construction struct
1047 partial_con pc;
1048 pc.id = con.id;
1049 pc.counter = 0;
1050 // Set the trap that has the examine function
1051 // Special handling for constructions that take place on existing traps.
1052 // Basically just don't add the unfinished construction trap.
1053 // TODO: handle this cleaner, instead of adding a special case to pit iexamine.
1054 if( here.tr_at( pnt ).loadid == tr_null ) {
1055 here.trap_set( pnt, tr_unfinished_construction );

1057 // Use up the components
1058 for( const auto &it : con.requirements->get_components() ) {
1059 std::list<item> tmp = g->u.consume_items( it, 1, is_crafting_component );
1060 used.splice( used.end(), tmp );
1061 }
1062 pc.components = used;
1063 here.partial_con_set( pnt, pc );
1064 for( const auto &it : con.requirements->get_tools() ) {
1065 g->u.consume_tools( it );
1066 }
1067 g->u.assign_activity( ACT_BUILD );
1068 g->u.activity.placement = here.getabs( pnt );
1069}

1072{
1073 if( !all_constructions.is_finalized() ) {
1074 debugmsg( "complete_construction called before finalization" );
1075 return;
1076 }
1077 map &here = get_map();
1078 const tripoint terp = here.getlocal( ch.activity.placement );
1079 partial_con *pc = here.partial_con_at( terp );
1080 if( !pc ) {
1081 debugmsg( "No partial construction found at activity placement in complete_construction()" );
1082 if( here.tr_at( terp ).loadid == tr_unfinished_construction ) {

1103
1104 award_xp( *ch.as_player() );

1129 dump_spots.push_back( pt );

1168 // clear the activity

1170
1171 // This comes after clearing the activity, in case the function interrupts
1172 // activities
1173 built.post_special( terp );
1174 // npcs will automatically resume backlog, players wont.
1175 if( ch.is_avatar() && !ch.backlog.empty() &&
1176 ch.backlog.front().id() == ACT_MULTIPLE_CONSTRUCTION ) {
1177 ch.backlog.clear();
1178 ch.assign_activity( ACT_MULTIPLE_CONSTRUCTION );
1179 }
