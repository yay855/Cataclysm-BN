1014static bool mx_portal( map &m, const tripoint &abs_sub )
1015{
1016 // All points except the borders are valid--we need the 1 square buffer so that we can do a 1 unit radius
1017 // around our chosen portal point without clipping against the edge of the map.
1018 const tripoint_range<tripoint> points =
1019 m.points_in_rectangle( { 1, 1, abs_sub.z }, { SEEX * 2 - 2, SEEY * 2 - 2, abs_sub.z } );
1020
1021 // Get a random point in our collection that does not have a trap and does not have the NO_FLOOR flag.
1022 const cata::optional<tripoint> portal_pos = random_point( points, [&]( const tripoint & p ) {
1023 return !m.has_flag_ter( TFLAG_NO_FLOOR, p ) && m.tr_at( p ).is_null();
1024 } );
1025
1026 // If we can't get a point to spawn the portal (e.g. we're triggered in entirely open air) we're done here.
1027 if( !portal_pos ) {
1028 return false;
1029 }
1030
1031 // For our portal point and every adjacent location, make rubble if it doesn't have the NO_FLOOR flag.
1032 for( const tripoint &p : m.points_in_radius( *portal_pos, 1 ) ) {
1033 if( !m.has_flag_ter( TFLAG_NO_FLOOR, p ) ) {
1034 m.make_rubble( p, f_rubble_rock, true );
1035 }
1036 }
1037
1038 m.trap_set( *portal_pos, tr_portal );
1039
1040 // We'll make between 0 and 4 attempts to spawn monsters here.
1041 int num_monsters = rng( 0, 4 );
1042 for( int i = 0; i < num_monsters; i++ ) {
1043 // Get a random location from our points that is not the portal location, does not have the
1044 // NO_FLOOR flag, and isn't currently occupied by a creature.
1045 const cata::optional<tripoint> mon_pos = random_point( points, [&]( const tripoint & p ) {
1046 /// TODO: wrong: this checks for creatures on the main game map. Not within the map m.
1047 return !m.has_flag_ter( TFLAG_NO_FLOOR, p ) && *portal_pos != p && !g->critter_at( p );
1048 } );
1049
1050 // If we couldn't get a random location, we can't place a monster and we know that there are no
1051 // more possible valid locations, so just bail.
1052 if( !mon_pos ) {
1053 break;
1054 }
1055
1056 // Make rubble here--it's not necessarily a location that is directly adjacent to the portal.
1057 m.make_rubble( *mon_pos, f_rubble_rock, true );
1058

1984{
1985 // This map extra creates small clay deposits using a simple cellular automaton.
1986
1987 constexpr int width = SEEX * 2;
1988 constexpr int height = SEEY * 2;
1989
1990 for( int tries = 0; tries < 5; tries++ ) {
1991 // Generate the cells for our clay deposit.
1992 std::vector<std::vector<int>> current = CellularAutomata::generate_cellular_automaton(
1993 point( width, height ), 35, 5, 4, 3 );
1994
1995 // With our settings for the CA, it's sometimes possible to get a bad generation with not enough
1996 // alive cells (or even 0).

1999 for( int j = 0; j < height; j++ ) {
2000 alive_count += current[i][j];
2001 }

2003

2014 m.furn_set( location, f_null );
2015 m.ter_set( location, t_clay );
2016 }
2017 }
2018 }
2019
2020 // If we got here, it meant we had a successful try and can just break out of
2021 // our retry loop.
2022 return true;
2023 }
