92/* JsonIn
93 * ======
94 *
95 * The JsonIn class provides a wrapper around a std::istream,
96 * with methods for reading JSON data directly from the stream.
97 *
98 * JsonObject and JsonArray provide higher-level wrappers,
99 * and are a little easier to use in most cases,
100 * but have the small overhead of indexing the members or elements before use.
101 *
102 * Typical JsonIn usage might be something like the following:
103 *
104 * JsonIn jsin(myistream);
105 * // expecting an array of objects
106 * jsin.start_array(); // throws JsonError if array not found
107 * while (!jsin.end_array()) { // end_array returns false if not the end
108 * JsonObject jo = jsin.get_object();
109 * ... // load object using JsonObject methods
110 * }
111 *
112 * The array could have been loaded into a JsonArray for convenience.
113 * Not doing so saves one full pass of the data,
114 * as the element positions don't have to be read beforehand.
115 *
116 * If the JSON structure is not as expected,
117 * verbose error messages are provided, indicating the problem,
118 * and the exact line number and byte offset within the istream.
119 *
120 *
121 * Single-Pass Loading
122 * -------------------
123 *
124 * A JsonIn can also be used for single-pass loading,
125 * by passing off members as they arrive according to their names.
126 *
127 * Typical usage might be:
128 *
129 * JsonIn jsin(&myistream);
130 * // expecting an array of objects, to be mapped by id
131 * std::map<std::string,my_data_type> myobjects;
132 * jsin.start_array();
133 * while (!jsin.end_array()) {
134 * my_data_type myobject;
135 * jsin.start_object();
136 * while (!jsin.end_object()) {
137 * std::string name = jsin.get_member_name();
138 * if (name == "id") {
139 * myobject.id = jsin.get_string();
140 * } else if (name == "name") {
141 * myobject.name = _(jsin.get_string());
142 * } else if (name == "description") {
143 * myobject.description = _(jsin.get_string());
144 * } else if (name == "points") {
145 * myobject.points = jsin.get_int();
146 * } else if (name == "flags") {
147 * if (jsin.test_string()) { // load single string tag
148 * myobject.tags.insert(jsin.get_string());
149 * } else { // load tag array
150 * jsin.start_array();
151 * while (!jsin.end_array()) {
152 * myobject.tags.insert(jsin.get_string());
153 * }
154 * }
155 * } else {
156 * jsin.skip_value();
157 * }
158 * }
159 * myobjects[myobject.id] = myobject;
160 * }
161 *
162 * The get_* methods automatically verify and skip separators and whitespace.
163 *
164 * Using this method, unrecognized members are silently ignored,
165 * allowing for things like "comment" members,
166 * but the user must remember to provide an "else" clause,
167 * explicitly skipping them.
168 *
169 * If an if;else if;... is missing the "else", it /will/ cause bugs,
170 * so preindexing as a JsonObject is safer, as well as tidier.
171 */

754 * ==========
755 *
756 * The JsonObject class provides easy access to incoming JSON object data.
757 *
758 * JsonObject maps member names to the byte offset of the paired value,
759 * given an underlying JsonIn stream.
760 *
761 * It provides data by seeking the stream to the relevant position,
762 * and calling the correct JsonIn method to read the value from the stream.
763 *
764 *
765 * General Usage
766 * -------------
767 *
768 * JsonObject jo(jsin);
769 * std::string id = jo.get_string("id");

956/* JsonArray
957 * =========
958 *
959 * The JsonArray class provides easy access to incoming JSON array data.
960 *
961 * JsonArray stores only the byte offset of each element in the JsonIn stream,
962 * and iterates or accesses according to these offsets.
963 *
964 * It provides data by seeking the stream to the relevant position,
965 * and calling the correct JsonIn method to read the value from the stream.
966 *
967 * Arrays can be iterated over,
968 * or accessed directly by element index.
969 *
970 * Elements can be returned as standard data types,
971 * or automatically read into a compatible container.
972 *
973 *
974 * Iterative Access
975 * ----------------
976 *
977 * JsonArray ja = jo.get_array("some_array_member");
978 * std::vector<int> myarray;
979 * while (ja.has_more()) {
980 * myarray.push_back(ja.next_int());
981 * }
982 *
983 * If the array member is not found, get_array() returns an empty array,
984 * and has_more() is always false, so myarray will remain empty.
985 *
986 * next_int() and the other next_* methods advance the array iterator,
987 * so after the final element has been read,
988 * has_more() will return false and the loop will terminate.
989 *
990 * If the next element is not an integer,
991 * JsonArray will throw a JsonError indicating the problem,
992 * and the position in the input stream.
993 *
994 * To handle arrays with elements of indeterminate type,
995 * the test_* methods can be used, calling next_* according to the result.
996 *
997 * Note that this style of iterative access requires an element to be read,
998 * or else the index will not be incremented, resulting in an infinite loop.
999 * Unwanted elements can be skipped with JsonArray::skip_value().
1000 *
1001 *
1002 * Positional Access
1003 * -----------------
1004 *
1005 * JsonArray ja = jo.get_array("xydata");
1006 * point xydata(ja.get_int(0), ja.get_int(1));
1007 *
1008 * Arrays also provide has_int(index) etc., for positional type testing,
1009 * and read(index, &container) for automatic deserialization.
1010 *
1011 *
1012 * Automatic Deserialization
1013 * -------------------------
1014 *
1015 * Elements can also be automatically read into compatible containers,
1016 * such as maps, sets, vectors, and classes implementing JsonDeserializer,
1017 * using the read_next() and read() methods.
1018 *
1019 * JsonArray ja = jo.get_array("custom_datatype_array");
1020 * while (ja.has_more()) {
1021 * MyDataType mydata; // MyDataType implementing JsonDeserializer
1022 * ja.read_next(mydata);

1024 * }
