478template <typename T>
479void inline_requirements( std::vector<std::vector<T>> &list,
480 const std::function<const std::vector<std::vector<T>> & ( const requirement_data & )> &getter )
481{
482 // add a single component to the vector. If component already exists, chooses min count
483 const auto add_component = []( const T & comp, std::vector<T> &accum ) {
484 auto iter = std::find_if( accum.begin(), accum.end(), [&]( const T & req ) {
485 return !req.requirement && req.type == comp.type;
486 } );
487 if( iter == accum.end() ) {
488 accum.push_back( comp ); // component doesn't exist yet, adding it
489 } else {
490 // this is the list of alternatives (a OR b OR c), so if there are two identical components
491 // with the different counts as an alternatives, the one with the lowest count is selected
492 iter->count = std::min( comp.count, iter->count );
493 }
494 };
495
496 // if `comp` is a component, adds it to the `accum`
497 // if it's a requirement, recursively expands it into simple components and adds them to the vector
498 const auto inline_one = [&]( const T & comp, std::vector<T> &accum ) {
499 std::set<requirement_id> already_nested;
500 std::list<requirement_id> stack;
501
502 std::function<void( const T &comp )> rec;
503 rec = [&]( const T & comp ) {
504 // add simple component to the vector
505 if( !comp.requirement ) {
506 add_component( comp, accum );
507 return;
508 }
509 // otherwise expand component as requirement
510 const requirement_id r( comp.type.str() );
511 if( !r.is_valid() ) {
512 debugmsg( "Tried to inline unknown requirement %s", r.c_str() );
513 return;
514 }
515 // stack just holds the current path of inlining for debug purposes
516 stack.push_back( r );
517 if( already_nested.count( r ) ) {
518 // print debug msg and skip just this one requirement
519 debugmsg( "Tried to inline requirement %s which forms a cycle", r.c_str() );
520 stack.pop_back();
521 return;
522 }
523 already_nested.insert( r );
524
525 const auto &req = r.obj();
526 const requirement_data multiplied = req * comp.count;
527
528 const std::vector<std::vector<T>> &to_inline = getter( multiplied );
529 if( !to_inline.empty() ) {
530 const std::vector<T> &nested = to_inline.front();
531 for( const T &nested_comp : nested ) {
532 rec( nested_comp );
533 }
534 }
535 already_nested.erase( r );
536 stack.pop_back();
537 };
538
539 rec( comp );
540 };

542 for( std::vector<T> &vec : list ) {
543 const std::vector<T> vec_copy = vec;
544 vec.clear();
545 for( const T &comp : vec_copy ) {

551{
552 for( auto &r : const_cast<std::map<requirement_id, requirement_data> &>( all() ) ) {
553 inline_requirements<tool_comp>( r.second.tools, []( const requirement_data & d ) -> const auto & {

562 alter_tool_comp_vector new_vec;
563 for( const std::vector<tool_comp> &list : vec ) {
564 std::vector<tool_comp> new_list;

567 if( comp.subtype_expanded ) {
568 new_list.push_back( comp );
569 continue;
570 }
571 const std::list<itype_id> replacements = item_controller->subtype_replacement( comp.type );
572 for( const itype_id &replacing_type : replacements ) {
573 // Don't replace if replacement is already in list (e.g. it was explicitly specified)

870bool requirement_data::check_enough_materials( const item_comp &comp, const inventory &crafting_inv,
871 const std::function<bool( const item & )> &filter, int batch ) const
872{
873 if( comp.available != available_status::a_true ) {
874 return false;
875 }
876 const int cnt = std::abs( comp.count ) * batch;
877 const tool_comp *tq = find_by_type( tools, comp.type );
878 if( tq != nullptr && tq->available == available_status::a_true ) {
879 // The very same item type is also needed as tool!
880 // Use charges of it, or use it by count?
881 const int tc = tq->by_charges() ? 1 : std::abs( tq->count );
882 // Check for components + tool count. Check item amount (excludes
883 // pseudo items) and tool amount (includes pseudo items)
884 // Imagine: required = 1 welder (component) + 1 welder (tool),
885 // available = 1 welder (real item), 1 welding rig (creates
886 // a pseudo welder item). has_components(welder,2) returns false
887 // as there is only one real welder available, but has_tools(welder,2)
888 // returns true.
889 // Keep in mind that both requirements (tool+component) are checked
890 // before this. That assures that one real item is actually available,
891 // two welding rigs (and no real welder) would make this component
892 // non-available even before this function is called.
893 // Only ammo and (some) food is counted by charges, both are unlikely
894 // to appear as tool, but it's possible /-:
895 const item_comp i_tmp( comp.type, cnt + tc );
896 const tool_comp t_tmp( comp.type, -( cnt + tc ) ); // not by charges!
897 // batch factor is explicitly 1, because it's already included in the count.
898 if( !i_tmp.has( crafting_inv, filter, 1 ) && !t_tmp.has( crafting_inv, filter, 1 ) ) {
899 comp.available = available_status::a_insufficent;
900 }
901 }
902 const itype *it = &*comp.type;
903 for( const auto &ql : it->qualities ) {
904 const quality_requirement *qr = find_by_type( qualities, ql.first );
905 if( qr == nullptr || qr->level > ql.second ) {
906 continue;
907 }
908 // This item can be used for the quality requirement, same as above for specific
909 // tools applies.
910 if( !crafting_inv.has_quality( qr->type, qr->level, qr->count + std::abs( comp.count ) ) ) {
911 comp.available = available_status::a_insufficent;
912 }
913 }
914 return comp.available == available_status::a_true;
915}

917template <typename T>
918static bool apply_blacklist( std::vector<std::vector<T>> &vec, const itype_id &id )
919{
920 // remove all instances of @id type from each of the options
921 for( auto &opts : vec ) {
922 opts.erase( std::remove_if( opts.begin(), opts.end(), [&id]( const T & e ) {
923 return e.type == id;
924 } ), opts.end() );
925 }
926
927 // did we remove the last instance of an option group?
928 const bool blacklisted = std::any_of( vec.begin(), vec.end(), []( const std::vector<T> &e ) {
929 return e.empty();
930 } );
931
932 // if an option group is left empty then it can be removed
933 vec.erase( std::remove_if( vec.begin(), vec.end(), []( const std::vector<T> &e ) {
934 return e.empty();
935 } ), vec.end() );
936
937 return blacklisted;
938}

946template <typename T>
947static void apply_replacement( std::vector<std::vector<T>> &vec, const itype_id &id,
948 const itype_id &replacement )
949{
950 // If the target and replacement are both present, remove the target.
951 // If only the target is present, replace it.
952 for( auto &opts : vec ) {
953 typename std::vector<T>::iterator target = opts.end();
954 typename std::vector<T>::iterator replacement_target = opts.end();
955 for( typename std::vector<T>::iterator iter = opts.begin(); iter != opts.end(); ++iter ) {
956 if( iter->type == id ) {
957 target = iter;
958 } else if( iter->type == replacement ) {
959 replacement_target = iter;
960 }
961 }
962 // No target to replace, do nothing.
963 if( target == opts.end() ) {
964 continue;
965 }
966 // Target but no replacement, replace.
967 if( replacement_target == opts.end() ) {
968 target->type = replacement;
969 continue;
970 }
971 // Both target and replacement, remove the target entry and leave the existing replacement.
972 opts.erase( target );
973 }

1009 requirement_data ret = *this;
1010 auto new_qualities = std::vector<quality_requirement>();
1011 bool remove_fire = false;
1012 for( auto &it : ret.tools ) {
1013 bool replaced = false;
1014 for( const auto &tool : it ) {

1016
1017 // If crafting required a welder or forge then disassembly requires metal sawing

1019 type == itype_forge || type == itype_char_forge ) {

1026 type == itype_mold_plastic ) {
1027 new_qualities.emplace_back( quality_id( "CUT" ), 1, 1 );
1028 replaced = true;
1029 break;
1030 }
1031
1032 if( type == itype_crucible ) {
1033 replaced = true;
1034 break;
1035 }
1036 //This ensures that you don't need a hand press to break down reloaded ammo.
1037 if( type == itype_press ) {
1038 replaced = true;
1039 remove_fire = true;
1040 new_qualities.emplace_back( quality_id( "PULL" ), 1, 1 );
1041 break;
1042 }
1043 if( type == itype_fire && remove_fire ) {
1044 replaced = true;
1045 break;
1046 }
1047 }

1051 // This avoids the pesky integrated toolset
1052 it.clear();

1054 }
1055
1056 // Warning: This depends on the fact that tool qualities
1057 // are all mandatory (don't use variants)
1058 // If that ever changes, this will be wrong!
1059 if( ret.qualities.empty() ) {
1060 ret.qualities.resize( 1 );

1073 replaced = true;

1075 //This only happens in case of a reversible recipe, an explicit
1076 //deconstruction recipe can still specify glare protection
1077 break;
1078 }
1079 if( quality.type == quality_id( "KNIT" ) ) {
1080 replaced = true;
1081 //Ditto for knitting needles
1082 break;
1083 }
1084 }
1085 if( replaced ) {
1086 it.clear();
1087 }
1088 }
1089 }
1090
1091 auto &qualities = ret.qualities[0];
1092 qualities.insert( qualities.end(), new_qualities.begin(), new_qualities.end() );

1094 {
1095 const auto itr = std::unique( qualities.begin(), qualities.end(),
1096 []( const quality_requirement & a, const quality_requirement & b ) {
1097 return a.type == b.type;
1098 } );
1099 qualities.resize( std::distance( qualities.begin(), itr ) );

1102 // Remove empty variant sections
1103 ret.tools.erase( std::remove_if( ret.tools.begin(), ret.tools.end(),
1104 []( const std::vector<tool_comp> &tcv ) {

1113 } ), cov.end() );

1115 } ), ret.components.end() );

1120requirement_data requirement_data::continue_requirements( const std::vector<item_comp>
1121 &required_comps, const std::list<item> &remaining_comps )
1122{
1123 // Create an empty requirement_data
1124 requirement_data ret;
1125
1126 // For items we can't change what alternative we selected half way through
1127 for( const item_comp &it : required_comps ) {
1128 ret.components.emplace_back( std::vector<item_comp>( {it} ) );
1129 }
1130
1131 inventory craft_components;
1132 craft_components += remaining_comps;
1133
1134 // Remove requirements that are completely fulfilled by current craft components
1135 // For each requirement that isn't completely fulfilled, reduce the requirement by the amount
1136 // that we still have
1137 // We also need to consume whatever charges we use in case two requirements share a common type
1138 ret.components.erase( std::remove_if( ret.components.begin(), ret.components.end(),
1139 [&craft_components]( std::vector<item_comp> &comps ) {
1140 item_comp &comp = comps.front();
1141 if( item::count_by_charges( comp.type ) && comp.count > 0 ) {
1142 int qty = craft_components.charges_of( comp.type, comp.count );
1143 comp.count -= qty;
1144 // This is terrible but inventory doesn't have a use_charges() function so...
1145 std::vector<item *> del;
1146 craft_components.visit_items( [&comp, &qty, &del]( item * e ) {
1147 std::list<item> used;
1148 if( e->use_charges( comp.type, qty, used, tripoint_zero ) ) {
1149 del.push_back( e );
1150 }
1151 return qty > 0 ? VisitResponse::SKIP : VisitResponse::ABORT;
1152 } );
1153 craft_components.remove_items_with( [&del]( const item & e ) {
1154 for( const item *it : del ) {
1155 if( it == &e ) {
1156 return true;
1157 }
1158 }
1159 return false;
1160 } );
1161 } else {

1163 comp.count -= amount;
1164 craft_components.use_amount( comp.type, amount );
1165 }

1167 } ), ret.components.end() );

1172template<typename T, typename Accum>
1173static std::vector<std::vector<T>> consolidate( std::vector<std::vector<T>> old_vec,
1174 const Accum &accum )
1175{
1176 const auto type_lt = []( const T & lhs, const T & rhs ) -> bool {
1177 //TODO change to use localized sorting
1178 return std::forward_as_tuple( lhs.type.str(), lhs.requirement )
1179 < std::forward_as_tuple( rhs.type.str(), rhs.requirement );
1180 };
1181 // in order to simplify blueprint requirements, we merge a longer requirement
1182 // list into a shorter requirement list whose types are a subsequence of the
1183 // longer list's types. However, this operation is not symmetric and depends
1184 // on the order of the requirement lists. Thus we sort the lists first, to
1185 // ensure consistent results when the order of construction requirements changes.
1186 for( std::vector<T> &old_inner : old_vec ) {
1187 std::sort( old_inner.begin(), old_inner.end(), type_lt );
1188 }
1189 std::sort( old_vec.begin(), old_vec.end(),
1190 [&type_lt]( const std::vector<T> &lhs, const std::vector<T> &rhs ) -> bool {
1191 return std::lexicographical_compare( lhs.begin(), lhs.end(), rhs.begin(), rhs.end(),
1192 type_lt );
1193 } );
1194
1195 std::vector<std::vector<T>> new_vec;
1196 for( std::vector<T> &old_inner : old_vec ) {
1197 bool match = false;
1198 for( std::vector<T> &new_inner : new_vec ) {
1199 // in order to simplify blueprint requirements, we merge a longer
1200 // requirement list into a shorter requirement list whose types are
1201 // a subsequence of the longer list's types.
1202 //
1203 // note this actually may make a requirement stricter.
1204 // for example, if the item requirement was
1205 // [ [ [ "a", 1 ], [ "b", 1 ], [ "c", 1 ] ],
1206 // [ [ "a", 1 ], [ "b", 1 ] ] ]
1207 // then you could satisfy it by having one "a" and one "b", one
1208 // "c" and one "a", two "a", or two "b", etc.
1209 //
1210 // but after consolidation, it becomes
1211 // [ [ [ "a", 2 ], [ "b", 2 ] ] ]
1212 // then you can only satisfy it by having either 2 "a" or 2 "b"
1213 if( std::includes( new_inner.begin(), new_inner.end(),
1214 old_inner.begin(), old_inner.end(),
1215 type_lt ) ) {
1216 // old_inner is a subsequence of new_inner
1217 match = true;
1218 std::swap( old_inner, new_inner );
1219 } else if( std::includes( old_inner.begin(), old_inner.end(),
1220 new_inner.begin(), new_inner.end(),
1221 type_lt ) ) {
1222 // new_inner is a subsequence of old_inner
1223 match = true;
1224 }
1225 if( match ) {
1226 for( auto it1 = new_inner.begin(), it2 = old_inner.begin();
1227 it1 < new_inner.end(); ++it2 ) {
1228 if( !type_lt( *it2, *it1 ) ) {
1229 // which means *it2 and *it1 have the same type, since
1230 // we know new_inner is a subsequence of old_inner
1231 *it1 = accum( *it1, *it2 );
1232 ++it1;
1233 }
1234 }
1235 break;
1236 }
1237 }
1238 if( !match ) {
1239 new_vec.emplace_back( old_inner );
1240 }
1241 }
1242 return new_vec;
1243}

1245void requirement_data::consolidate()
1246{
1247 qualities = ::consolidate( qualities,
1248 []( const quality_requirement & lhs, const quality_requirement & rhs ) {
1249 quality_requirement ret = lhs;
1250 ret.count = std::max( ret.count, rhs.count );
1251 ret.level = std::max( ret.level, rhs.level );
1252 return ret;
1253 } );
1254
1255 tools = ::consolidate( tools,
1256 []( const tool_comp & lhs, const tool_comp & rhs ) {
1257 tool_comp ret = lhs;
1258 if( ret.count < 0 && rhs.count < 0 ) {
1259 ret.count = std::min( ret.count, rhs.count );
1260 } else if( ret.count > 0 && rhs.count > 0 ) {
1261 ret.count += rhs.count;
1262 } else {
1263 debugmsg( "required counts of the same tool have different signs" );
1264 }
1265 return ret;
1266 } );
1267
1268 components = ::consolidate( components,
1269 []( const item_comp & lhs, const item_comp & rhs ) {
1270 item_comp ret = lhs;
1271 ret.count += rhs.count;
1272 return ret;
1273 } );
1274}

1276template<typename T>
1277static bool sorted_equal( std::vector<std::vector<T>> lhs, std::vector<std::vector<T>> rhs )
1278{
1279 if( lhs.size() != rhs.size() ) {
1280 return false;
1281 }
1282 for( auto &inner : lhs ) {
1283 std::sort( inner.begin(), inner.end() );
1284 }
1285 for( auto &inner : rhs ) {
1286 std::sort( inner.begin(), inner.end() );
1287 }
1288 std::sort( lhs.begin(), lhs.end() );
1289 std::sort( rhs.begin(), rhs.end() );
1290 return lhs == rhs;
1291}

1329/// Helper function for deduped_requirement_data constructor below.
1330///
1331/// The goal of this function is to consolidate a particular item_comp that
1332/// would otherwise be duplicated between two requirements.
1333///
1334/// It operates recursively (increasing @p index with the depth of recursion),
1335/// searching for another item_comp to merge @p leftover with. For each
1336/// compatible item_comp found it performs that merger and writes out a
1337/// suitably updated form of the overall requirements to @p result.
1338///
1339/// If it chooses *not* to merge with any particular item_comp, then it deletes
1340/// that item_comp from the options, to avoid the duplication.
1341///
1342/// Lastly, it also outputs a version of the requirements where @p leftover
1343/// remains where it was, and all other compatible item_comp entries have been
1344/// deleted.
1345///
1346/// @param leftover The item_comp needing to be dealt with.
1347/// @param req_prefix The requirements considered so far; more will be appended
1348/// to this.
1349/// @param to_expand The original requirements we are working through to look
1350/// for a duplicate.
1351/// @param orig_index The index into the alter_item_comp_vector where @p
1352/// leftover was originally to be found. If it isn't merged with another item,
1353/// then it will be re-inserted at this position.
1354/// @param index The position within @p to_expand where we will next look for
1355/// duplicates of @p leftover to merge with.
1356/// @param result The finished requirements should be appended to this.
1357static void expand_item_in_reqs(
1358 const item_comp &leftover, requirement_data::alter_item_comp_vector req_prefix,
1359 const requirement_data::alter_item_comp_vector &to_expand, size_t orig_index, size_t index,
1360 std::vector<requirement_data::alter_item_comp_vector> &result )
1361{
1362 assert( req_prefix.size() >= orig_index );
1363 assert( orig_index < index );
1364
1365 if( index == to_expand.size() ) {
1366 // We reached the end without using the leftovers. So need to add them
1367 // as their own requirement, separate from everything else.
1368 req_prefix.insert( req_prefix.begin() + orig_index, { leftover } );
1369 result.push_back( req_prefix );
1370 return;
1371 }
1372
1373 std::vector<item_comp> this_requirement = to_expand[index];
1374 auto duplicate = std::find_if( this_requirement.begin(), this_requirement.end(),
1375 [&]( const item_comp & c ) {
1376 return c.type == leftover.type;
1377 } );
1378 if( duplicate == this_requirement.end() ) {
1379 // No match in this one; proceed to next
1380 req_prefix.push_back( this_requirement );
1381 expand_item_in_reqs( leftover, req_prefix, to_expand, orig_index, index + 1, result );
1382 return;
1383 }
1384 // First option: amalgamate the leftovers into this requirement, which
1385 // forces us to pick that specific option:
1386 requirement_data::alter_item_comp_vector req = req_prefix;
1387 req.push_back( { item_comp( leftover.type, leftover.count + duplicate->count ) } );
1388 req.insert( req.end(), to_expand.begin() + index + 1, to_expand.end() );
1389 result.push_back( req );
1390
1391 // Second option: use a separate option for this requirement, which means
1392 // we need to recurse further to find something into which to amalgamate
1393 // the original requirement
1394 this_requirement.erase( duplicate );
1395 if( !this_requirement.empty() ) {
1396 req_prefix.push_back( this_requirement );
1397 expand_item_in_reqs( leftover, req_prefix, to_expand, orig_index, index + 1, result );
1398 }
1399}

1401deduped_requirement_data::deduped_requirement_data( const requirement_data &in,
1402 const recipe_id &context )
1403{
1404 // This constructor works through a requirement_data, converting it into an
1405 // equivalent set of requirement_data alternatives, where each alternative
1406 // has the property that no item type appears more than once.
1407 //
1408 // We only deal with item requirements. Tool requirements could be handled
1409 // similarly, but no examples where they are a problem have yet been
1410 // raised.
1411 //
1412 // We maintain a queue of requirement_data component info to be split.
1413 // Each to_check struct has a vector of component requirements, and an
1414 // index. The index is the position within the vector to be checked next.
1415 struct to_check {
1416 alter_item_comp_vector components;
1417 size_t index;
1418 };
1419 std::stack<to_check, std::vector<to_check>> pending;
1420 pending.push( { in.get_components(), 0 } );
1421
1422 while( !pending.empty() ) {
1423 to_check next = pending.top();
1424 pending.pop();
1425
1426 if( next.index == next.components.size() ) {
1427 alternatives_.emplace_back( in.get_tools(), in.get_qualities(), next.components );
1428 continue;
1429 }
1430
1431 // Build a set of all the itypes used in later stages of this set of
1432 // requirements.
1433 std::unordered_set<itype_id> later_itypes;
1434 for( size_t i = next.index + 1; i != next.components.size(); ++i ) {
1435 std::transform( next.components[i].begin(), next.components[i].end(),
1436 std::inserter( later_itypes, later_itypes.end() ),
1437 []( const item_comp & c ) {
1438 return c.type;
1439 } );
1440 }
1441
1442 std::vector<item_comp> this_requirement = next.components[next.index];
1443
1444 auto first_duplicated = std::stable_partition(
1445 this_requirement.begin(), this_requirement.end(),
1446 [&]( const item_comp & c ) {
1447 return !later_itypes.count( c.type );
1448 }
1449 );
1450
1451 for( auto comp_it = first_duplicated; comp_it != this_requirement.end(); ++comp_it ) {
1452 // Factor this requirement out into its own separate case
1453
1454 alter_item_comp_vector req_prefix( next.components.begin(),
1455 next.components.begin() + next.index );
1456 std::vector<alter_item_comp_vector> result;
1457 expand_item_in_reqs( *comp_it, req_prefix, next.components, next.index, next.index + 1,
1458 result );
1459 for( const alter_item_comp_vector &v : result ) {
1460 // When v is smaller, that means the current requirement was
1461 // deleted, in which case we don't advance index.
1462 size_t index_inc = v.size() == next.components.size() ? 1 : 0;
1463 pending.push( { v, next.index + index_inc } );
1464 }
1465 }
1466
1467 // Deal with all the remaining, non-duplicated ones
1468 this_requirement.erase( first_duplicated, this_requirement.end() );
1469 if( !this_requirement.empty() ) {
1470 alter_item_comp_vector without_dupes = next.components;
1471 without_dupes[next.index] = this_requirement;
1472 pending.push( { without_dupes, next.index + 1 } );
1473 }
1474
1475 // Because this algorithm is super-exponential in the worst case, add a
1476 // sanity check to prevent things getting too far out of control.
1477 // The worst case in the core game currently is chainmail_suit_faraday
1478 // with 63 alternatives.
1479 static constexpr size_t max_alternatives = 100;
1480 if( alternatives_.size() + pending.size() > max_alternatives ) {
1481 debugmsg( "Construction of deduped_requirement_data generated too many alternatives. "
1482 "The recipe %s should be simplified. See the Recipe section in "
1483 "doc/JSON_INFO.md for more details.", context.str() );
1484 is_too_complex_ = true;
1485 alternatives_ = { in };
1486 return;
1487 }
1488 }
1489}
