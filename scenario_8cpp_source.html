104// Strategy: a third of the time, return the generic scenario. Otherwise, return a scenario,
105// weighting 0 cost scenario more likely--the weight of a scenario with cost n is 2/(|n|+2),
106// e.g., cost 1 is 2/3rds as likely, cost -2 is 1/2 as likely.
107const scenario *scenario::weighted_random()
108{
109 if( one_in( 3 ) ) {
110 return generic();
111 }
112
113 const auto &list = all_scenarios.get_all();
114 while( true ) {
115 const scenario &scen = random_entry_ref( list );
116
117 if( x_in_y( 2, std::abs( scen.point_cost() ) + 2 ) ) {
118 return &scen;
119 }
120 // else reroll in the while loop.
121 }
122}

248{

268void scen_blacklist::finalize()
269{
270 std::vector<string_id<scenario>> all_scens;
271 for( const scenario &scen : scenario::get_all() ) {
272 all_scens.emplace_back( scen.ident() );
273 }
274 for( const string_id<scenario> &sc : sc_blacklist.scenarios ) {
275 if( std::find( all_scens.begin(), all_scens.end(), sc ) == all_scens.end() ) {
276 debugmsg( "Scenario blacklist contains invalid scenario" );
277 }
278 }
279
280 if( sc_blacklist.whitelist ) {
281 std::set<string_id<scenario>> listed_scenarios = sc_blacklist.scenarios;
282 sc_blacklist.scenarios.clear();
283 for( const scenario &scen : scenario::get_all() ) {
284 sc_blacklist.scenarios.insert( scen.ident() );
285 }
286 for( auto i = sc_blacklist.scenarios.begin(); i != sc_blacklist.scenarios.end(); ) {
287 if( listed_scenarios.count( *i ) != 0 ) {
288 i = sc_blacklist.scenarios.erase( i );
289 } else {
290 ++i;
291 }
292 }
293 }

301std::vector<profession_id> scenario::permitted_professions() const
302{
303 if( !cached_permitted_professions.empty() ) {
304 return cached_permitted_professions;
305 }
306
307 const auto all = profession::get_all();
308 std::vector<profession_id> &res = cached_permitted_professions;
309 for( const profession &p : all ) {
310 const bool present = std::find( professions.begin(), professions.end(),
311 p.ident() ) != professions.end();
312
313 bool conflicting_traits = scenario_traits_conflict_with_profession_traits( p );
314
315 if( blacklist || professions.empty() ) {
316 if( !present && !p.has_flag( "SCEN_ONLY" ) && !conflicting_traits ) {
317 res.push_back( p.ident() );
318 }
319 } else if( present ) {
320 if( !conflicting_traits ) {
321 res.push_back( p.ident() );
322 } else {
323 debugmsg( "Scenario %s and profession %s have conflicting trait requirements",
324 id.c_str(), p.ident().c_str() );
325 }
326 } else if( extra_professions ) {
327 if( !p.has_flag( "SCEN_ONLY" ) && !conflicting_traits ) {
328 res.push_back( p.ident() );
329 }
330 }
331 }
332
333 if( res.empty() ) {
334 debugmsg( "Scenario %s blacklists all professions.", id );
335 res.push_back( profession::generic() );
336 }
337 return res;
338}

340bool scenario::scenario_traits_conflict_with_profession_traits( const profession &p ) const
341{
342 for( auto &pt : p.get_forbidden_traits() ) {
343 if( is_locked_trait( pt ) ) {
344 return true;
345 }
346 }
347
348 for( auto &pt : p.get_locked_traits() ) {
349 if( is_forbidden_trait( pt ) ) {
350 return true;
351 }
352 }
353
354 // check if:
355 // locked traits for scenario prevent taking locked traits for professions
356 // locked traits for professions prevent taking locked traits for scenario
357 for( auto &st : get_locked_traits() ) {
358 for( auto &pt : p.get_locked_traits() ) {
359 if( are_conflicting_traits( st, pt ) || are_conflicting_traits( pt, st ) ) {
360 return true;
361 }
362 }
363 }
364 return false;

367const profession_id &scenario::weighted_random_profession() const
368{
369 // Strategy: 1/3 of the time, return the generic profession (if it's permitted).
370 // Otherwise, the weight of each permitted profession is 2 / ( |point cost| + 2 )
371 const auto choices = permitted_professions();
372 if( one_in( 3 ) && choices.front() == profession::generic()->ident() ) {
373 return profession::generic();
374 }
375
376 while( true ) {
377 const profession_id &candidate = random_entry_ref( choices );
378 if( x_in_y( 2, 2 + std::abs( candidate->point_cost() ) ) ) {
379 return candidate->ident();
380 }
381 }
382 return profession::generic(); // Suppress warnings
383}
