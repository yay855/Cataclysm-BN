24static std::string get_achievements_text( const achievements_tracker &achievements )
25{
26 std::string os;
27 std::vector<const achievement *> valid_achievements = achievements.valid_achievements();
28 valid_achievements.erase(
29 std::remove_if( valid_achievements.begin(), valid_achievements.end(),
30 [&]( const achievement * a ) {
31 return achievements.is_hidden( a );
32 } ), valid_achievements.end() );
33 using sortable_achievement =
34 std::tuple<achievement_completion, std::string, const achievement *>;
35 std::vector<sortable_achievement> sortable_achievements;
36 std::transform( valid_achievements.begin(), valid_achievements.end(),
37 std::back_inserter( sortable_achievements ),
38 [&]( const achievement * ach ) {
39 achievement_completion comp = achievements.is_completed( ach->id );
40 return std::make_tuple( comp, ach->name().translated(), ach );
41 } );
42 std::sort( sortable_achievements.begin(), sortable_achievements.end(), localized_compare );
43 for( const sortable_achievement &ach : sortable_achievements ) {
44 os += achievements.ui_text_for( std::get<const achievement *>( ach ) ) + "\n";
45 }
46 if( valid_achievements.empty() ) {
47 os += _( "This game has no valid achievements.\n" );
48 }
49 os += _( "Note that only achievements that existed when you started this game and still "
50 "exist now will appear here." );
51 return os;
52}
