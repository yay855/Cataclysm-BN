9/**
10 * Runtime localization system for Cataclysm: Bright Nights.
11 *
12 * Implements translated string lookup within gettext MO files
13 * with full support for integer plural forms.
14 *
15 * Key differences from GNU libintl:
16 * 1. Does not support translation domains
17 * 2. Does not support encoding conversion, operates on UTF-8 only
18 * 3. Does not depend on locale / environment variables
19 * 4. Supports loading MO files from arbitrary paths
20 * 5. Supports loading multiple MO files into a single "domain"
21 *
22 * For MO file structure, see GNU gettext manual:
23 * https://www.gnu.org/software/gettext/manual/
24 */
25

35enum class PlfOp : int {
