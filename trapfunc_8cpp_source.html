64// A pit becomes less effective as it fills with corpses.
65static float pit_effectiveness( const tripoint &p )
66{
67 units::volume corpse_volume = 0_ml;
68 for( auto &pit_content : g->m.i_at( p ) ) {
69 if( pit_content.is_corpse() ) {
70 corpse_volume += pit_content.volume();
71 }
72 }
73
74 // 10 zombies; see item::volume
75 const units::volume filled_volume = 10 * units::from_milliliter<float>( 62500 );
76
77 return std::max( 0.0f, 1.0f - corpse_volume / filled_volume );
78}

1031
