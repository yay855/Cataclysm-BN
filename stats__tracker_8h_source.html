26// The stats_tracker is intended to keep a summary of events that have occured.
27// For each event_type it stores an event_multiset.
28// Within the event_tracker, counts are kept. The events are partitioned
29// according to their data (an event::data_type object, which is a map of keys
30// to values).
31// The stats_tracker can be queried in various ways to get summary statistics
32// about events that have occured.
33

50 // count returns the number of events matching given criteria that have
51 // occured.
52 // total returns the sum of some integer-valued field across every
53 // event satisfying certain criteria.
54 // maximum and minimum return the max and min respectively of some
55 // integer-valued field across every event satisfying certain criteria.
56 // For example, count might return the number of times the avatar has
57 // taken damage, total might return the total damage taken in all
58 // those cases, and maximum might return the single largest amount
59 // of damage taken in a single event across all those cases.
60 // The criteria takes the form of an event::data_type map specifying
61 // some values that must be matched in the events of that type. You can
62 // provide just a subset of the relevant keys from the event_type in
63 // your criteria.
64 int count() const;
65 int count( const cata::event::data_type &criteria ) const;
66 int total( const std::string &field ) const;

76 private:
77 event_type type_;
78 counts_type counts_;
79};

123 template<typename Class, typename... FnArgs, typename... Args>
124 void send_to_all( void ( Class::*mem_fn )( FnArgs... ), Args &&... args ) const {
125 static_assert( std::is_base_of<Class, Watcher>::value,
126 "Watcher must be derived from Class" );
127 // Sending an event to a watcher can cause it to be erased, so we
128 // need to always ensure we have the next iterator prepared in
129 // advance.
130 auto current = watchers_.begin();
131
132 while( current != watchers_.end() ) {
133 auto next = current;
134 ++next;
135 ( static_cast<Watcher *>( *current )->*mem_fn )( args... );
136 current = next;
137 }
138 }
