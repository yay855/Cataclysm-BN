696bool monexamine::give_items_to( monster &z )
697{
698 std::string pet_name = z.get_name();
699 if( !z.storage_item ) {
700 add_msg( _( "There is no container on your %s to put things in!" ), pet_name );
701 return true;
702 }
703
704 item &storage = *z.storage_item;
705 units::mass max_weight = z.weight_capacity() - z.get_carried_weight();
706 units::volume max_volume = storage.get_storage() - z.get_carried_volume();
707 avatar &you = get_avatar();
708 drop_locations items = game_menus::inv::multidrop( you );
709 drop_locations to_move;
710 for( const drop_location &itq : items ) {
711 item it_copy = *itq.loc;
712 if( it_copy.count_by_charges() ) {
713 it_copy.charges = itq.count;
714 }
715 units::volume item_volume = it_copy.volume();
716 units::mass item_weight = it_copy.weight();
717 if( max_weight < item_weight ) {
718 add_msg( _( "The %1$s is too heavy for the %2$s to carry." ), it_copy.tname(), pet_name );
719 continue;
720 } else if( max_volume < item_volume ) {
721 add_msg( _( "The %1$s is too big to fit in the %2$s." ), it_copy.tname(), storage.tname() );
722 continue;
723 } else {
724 max_weight -= item_weight;
725 max_volume -= item_volume;
726 to_move.insert( to_move.end(), itq );
727 }
728 }
729 z.add_effect( effect_ai_waiting, 5_turns );
730 you.drop( to_move, z.pos(), true );
731
732 return false;
