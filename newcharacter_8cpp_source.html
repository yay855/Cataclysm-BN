318 int r = rng( 1, 9 );

321 case 1:
322 case 2:
323 case 3:
324 case 4:

338 case 5:

341 case 1:
342 if( str_max < HIGH_STAT ) {
343 str_max++;
344 points.stat_points--;
345 } else if( points.stat_points_left() >= 2 && str_max < max_stat_points ) {
346 str_max++;

352 dex_max++;
353 points.stat_points--;
354 } else if( points.stat_points_left() >= 2 && dex_max < max_stat_points ) {
355 dex_max++;
356 points.stat_points = points.stat_points - 2;
357 }
358 break;
359 case 3:
360 if( int_max < HIGH_STAT ) {
361 int_max++;
362 points.stat_points--;
363 } else if( points.stat_points_left() >= 2 && int_max < max_stat_points ) {
364 int_max++;
365 points.stat_points = points.stat_points - 2;
366 }
367 break;
368 case 4:
369 if( per_max < HIGH_STAT ) {
370 per_max++;
371 points.stat_points--;
372 } else if( points.stat_points_left() >= 2 && per_max < max_stat_points ) {
373 per_max++;
374 points.stat_points = points.stat_points - 2;
375 }
376 break;
377 }
378 break;
379 }
380 /* fallthrough */
381 case 6:
382 case 7:
383 case 8:
384 case 9:
385 const skill_id aSkill = Skill::random_skill();
386 const int level = get_skill_level( aSkill );
387
388 if( level < points.skill_points_left() && level < MAX_SKILL && loops > 10000 ) {

390 // For balance reasons, increasing a skill from level 0 gives you 1 extra level for free
391 set_skill_level( aSkill, ( level == 0 ? 2 : level + 1 ) );
392 }
393 break;
394 }
395 loops++;
396 }
397 set_body();
