182/*
183Function: add_field
184Inserts the given field_type_id into the field list for a given tile if it does not already exist.
185Returns false if the field_type_id already exists, true otherwise.
186If the field already exists, it will return false BUT it will add the intensity/age to the current values for upkeep.
187If you wish to modify an already existing field use find_field and modify the result.
188Intensity defaults to 1, and age to 0 (permanent) if not specified.
189*/
190bool field::add_field( const field_type_id &field_type_to_add, const int new_intensity,
191 const time_duration &new_age )
192{
193 // sanity check, we don't want to store fd_null
194 if( !field_type_to_add ) {
195 debugmsg( "Tried to add null field" );
196 return false;
197 }
198 auto it = _field_type_list.find( field_type_to_add );
199 if( it != _field_type_list.end() ) {
200 // Most fields stack intensities, but some add duration instead
201 if( it->first->stacking_type == fields::stacking_type::intensity ) {
202 it->second.set_field_intensity( it->second.get_field_intensity() + new_intensity );
203 } else {
204 time_duration half_life = field_type_to_add->half_life;
205 if( new_age < half_life ) {
206 it->second.mod_field_age( new_age - half_life );
207 }
208 }
209 return false;
210 }
211 if( !_displayed_field_type ||
212 field_type_to_add.obj().priority >= _displayed_field_type.obj().priority ) {
213 _displayed_field_type = field_type_to_add;
214 }
215 _field_type_list[field_type_to_add] = field_entry( field_type_to_add, new_intensity, new_age );
216 return true;
217}
