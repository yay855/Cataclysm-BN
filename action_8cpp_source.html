660action_id handle_action_menu()
661{
662 const input_context ctxt = get_default_mode_input_context();
663 std::string catgname;
664
665#define REGISTER_ACTION( name ) entries.emplace_back( name, true, hotkey_for_action(name), \
666 ctxt.get_action_name( action_ident( name ) ) );
667#define REGISTER_CATEGORY( name ) categories_by_int[last_category] = name; \
668 catgname = name; \
669 catgname += ""; \
670 entries.emplace_back( last_category, true, -1, catgname ); \
671 last_category++;
672
673 // Calculate weightings for the various actions to give the player suggestions
674 // Weight >= 200: Special action only available right now
675 std::map<action_id, int> action_weightings;
676
677 // Check if we're in a potential combat situation, if so, sort a few actions to the top.
678 if( !g->u.get_hostile_creatures( 60 ).empty() ) {
679 // Only prioritize movement options if we're not driving.
680 if( !g->u.controlling_vehicle ) {
681 action_weightings[ACTION_CYCLE_MOVE] = 400;
682 }
683 // Only prioritize fire weapon options if we're wielding a ranged weapon.
684 if( g->u.weapon.is_gun() || g->u.weapon.has_flag( flag_REACH_ATTACK ) ) {
685 action_weightings[ACTION_FIRE] = 350;
686 }
687 }
688
689 // If we're already running, make it simple to toggle running to off.
690 if( g->u.movement_mode_is( CMM_RUN ) ) {
691 action_weightings[ACTION_TOGGLE_RUN] = 300;
692 }
693 // If we're already crouching, make it simple to toggle crouching to off.
694 if( g->u.movement_mode_is( CMM_CROUCH ) ) {
695 action_weightings[ACTION_TOGGLE_CROUCH] = 300;
696 }
697
698 map &here = get_map();
