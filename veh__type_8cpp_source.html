41// GENERAL GUIDELINES
42// To determine mount position for parts (dx, dy), check this scheme:
43// orthogonal dir left: (Y-)
44// ^
45// back: X- -------> forward dir: X+
46// v
47// orthogonal dir right (Y+)
48//
49// i.e, if you want to add a part to the back from the center of vehicle,
50// use dx = -1, dy = 0;
51// for the part 1 tile forward and two tiles left from the center of vehicle,
52// use dx = 1, dy = -2.
53//
54// Internal parts should be added after external on the same mount point, i.e:
55// part {"x": 0, "y": 1, "part": "seat"}, // put a seat (it's external)
56// part {"x": 0, "y": 1, "part": "controls"}, // put controls for driver here
57// part {"x": 0, "y": 1, "seatbelt"} // also, put a seatbelt here
58// To determine, what parts can be external, and what can not, check
59// vehicle_parts.json
60// If you use wrong config, installation of part will fail

161static void parse_vp_reqs( const JsonObject &obj, const std::string &id, const std::string &key,
162 std::vector<std::pair<requirement_id, int>> &reqs,
163 std::map<skill_id, int> &skills, int &moves )
164{
165
166 if( !obj.has_object( key ) ) {
167 return;
168 }
169 JsonObject src = obj.get_object( key );
170
171 auto sk = src.get_array( "skills" );
172 if( !sk.empty() ) {
173 skills.clear();
174 }
175 for( JsonArray cur : sk ) {
176 skills.emplace( skill_id( cur.get_string( 0 ) ), cur.size() >= 2 ? cur.get_int( 1 ) : 1 );
177 }
178
179 if( src.has_int( "time" ) ) {
180 moves = src.get_int( "time" );
181 } else if( src.has_string( "time" ) ) {
182 moves = to_moves<int>( read_from_json_string<time_duration>( *src.get_raw( "time" ),
183 time_duration::units ) );
184 }
185
186 if( src.has_string( "using" ) ) {
187 reqs = { { requirement_id( src.get_string( "using" ) ), 1 } };
188 } else if( src.has_array( "using" ) ) {
189 reqs.clear();
190 for( JsonArray cur : src.get_array( "using" ) ) {
191 reqs.emplace_back( requirement_id( cur.get_string( 0 ) ), cur.get_int( 1 ) );
192 }
193 } else {
194 reqs.clear();
195 // Construct a requirement to capture "components", "qualities", and
196 // "tools" that might be listed.
197 const requirement_id req_id( string_format( "inline_%s_%s", key.c_str(), id.c_str() ) );
198 requirement_data::load_requirement( src, req_id );
199 reqs.emplace_back( req_id, 1 );
200 }

807static int scale_time( const std::map<skill_id, int> &sk, int mv, const player &p )
808{
809 if( sk.empty() ) {
810 return mv;
811 }
812
813 const int lvl = std::accumulate( sk.begin(), sk.end(), 0, [&p]( int lhs,
814 const std::pair<skill_id, int> &rhs ) {
815 return lhs + std::max( std::min( p.get_skill_level( rhs.first ), MAX_SKILL ) - rhs.second,
816 0 );
817 } );
818 // 10% per excess level (reduced proportionally if >1 skill required) with max 50% reduction
819 // 10% reduction per assisting NPC
820 return mv * ( 1.0 - std::min( static_cast<double>( lvl ) / sk.size() / 10.0, 0.5 ) )
821 * ( 10 - character_funcs::get_crafting_helpers( p, 3 ).size() ) / 10;
822}

962 */
963void vehicle_prototype::load( const JsonObject &jo )
964{
965 vehicle_prototype &vproto = vtypes[ vproto_id( jo.get_string( "id" ) ) ];
966 // If there are already parts defined, this vehicle prototype overrides an existing one.
967 // If the json contains a name, it means a completely new prototype (replacing the
968 // original one), therefore the old data has to be cleared.
969 // If the json does not contain a name (the prototype would have no name), it means appending
970 // to the existing prototype (the parts are not cleared).
971 if( !vproto.parts.empty() && jo.has_string( "name" ) ) {
972 vproto = vehicle_prototype();
973 }
974 if( vproto.parts.empty() ) {
975 vproto.name = jo.get_string( "name" );
976 }
977
978 vgroups[vgroup_id( jo.get_string( "id" ) )].add_vehicle( vproto_id( jo.get_string( "id" ) ), 100 );
979
980 const auto add_part_obj = [&]( const JsonObject & part, point pos ) {
981 part_def pt;
982 pt.pos = pos;
983 pt.part = vpart_id( part.get_string( "part" ) );
984
985 assign( part, "ammo", pt.with_ammo, true, 0, 100 );
986 assign( part, "ammo_types", pt.ammo_types, true );
987 assign( part, "ammo_qty", pt.ammo_qty, true, 0 );
988 assign( part, "fuel", pt.fuel, true );
989
990 vproto.parts.push_back( pt );
991 };
992
993 const auto add_part_string = [&]( const std::string & part, point pos ) {
994 part_def pt;
995 pt.pos = pos;
996 pt.part = vpart_id( part );
997 vproto.parts.push_back( pt );
998 };
999
1000 if( jo.has_member( "blueprint" ) ) {
1001 // currently unused, read to suppress unvisited members warning
1002 jo.get_array( "blueprint" );
1003 }
1004
1005 for( JsonObject part : jo.get_array( "parts" ) ) {
1006 point pos = point( part.get_int( "x" ), part.get_int( "y" ) );
1007
1008 if( part.has_string( "part" ) ) {
1009 add_part_obj( part, pos );
1010 } else if( part.has_array( "parts" ) ) {
1011 for( const JsonValue entry : part.get_array( "parts" ) ) {
1012 if( entry.test_string() ) {
1013 std::string part_name = entry.get_string();
1014 add_part_string( part_name, pos );
1015 } else {
1016 JsonObject subpart = entry.get_object();
1017 add_part_obj( subpart, pos );
1018 }
1019 }
1020 }
1021 }
1022
1023 for( JsonObject spawn_info : jo.get_array( "items" ) ) {
1024 vehicle_item_spawn next_spawn;
1025 next_spawn.pos.x = spawn_info.get_int( "x" );
1026 next_spawn.pos.y = spawn_info.get_int( "y" );
1027
1028 next_spawn.chance = spawn_info.get_int( "chance" );
1029 if( next_spawn.chance <= 0 || next_spawn.chance > 100 ) {
1030 debugmsg( "Invalid spawn chance in %s (%d, %d): %d%%",
1031 vproto.name, next_spawn.pos.x, next_spawn.pos.y, next_spawn.chance );
1032 }
1033
1034 // constrain both with_magazine and with_ammo to [0-100]
1035 next_spawn.with_magazine = std::max( std::min( spawn_info.get_int( "magazine",
1036 next_spawn.with_magazine ), 100 ), 0 );
1037 next_spawn.with_ammo = std::max( std::min( spawn_info.get_int( "ammo", next_spawn.with_ammo ),
1038 100 ), 0 );
1039
1040 if( spawn_info.has_array( "items" ) ) {
1041 //Array of items that all spawn together (i.e. jack+tire)
1042 spawn_info.read( "items", next_spawn.item_ids, true );
1043 } else if( spawn_info.has_string( "items" ) ) {
1044 //Treat single item as array
1045 next_spawn.item_ids.push_back( itype_id( spawn_info.get_string( "items" ) ) );
1046 }
1047 if( spawn_info.has_array( "item_groups" ) ) {
1048 //Pick from a group of items, just like map::place_items
1049 for( const std::string line : spawn_info.get_array( "item_groups" ) ) {
1050 next_spawn.item_groups.push_back( item_group_id( line ) );
1051 }
1052 } else if( spawn_info.has_string( "item_groups" ) ) {
1053 next_spawn.item_groups.push_back( item_group_id( spawn_info.get_string( "item_groups" ) ) );
1054 }
1055 vproto.item_spawns.push_back( std::move( next_spawn ) );
1056 }
