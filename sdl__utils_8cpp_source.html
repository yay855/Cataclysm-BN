48std::vector<SDL_Color> color_linear_interpolate( const SDL_Color &start_color,
49 const SDL_Color &end_color, const unsigned additional_steps )
50{
51 const unsigned steps = additional_steps + 2;
52 std::vector<SDL_Color> colors;
53 colors.reserve( steps );
54 colors.push_back( start_color );
55
56 const float step = 1.0f / ( steps - 1 );
57 for( unsigned i = 0; i < steps - 2; i++ ) {
58 const float percent = ( i + 1 ) * step;
59 const Uint8 r = static_cast<Uint8>( start_color.r + static_cast<Uint8>( percent *
60 ( end_color.r - start_color.r ) ) );
61 const Uint8 g = static_cast<Uint8>( start_color.g + static_cast<Uint8>( percent *
62 ( end_color.g - start_color.g ) ) );
63 const Uint8 b = static_cast<Uint8>( start_color.b + static_cast<Uint8>( percent *
64 ( end_color.b - start_color.b ) ) );
65 colors.push_back( { r, g, b, 255 } );
66 }
67 colors.push_back( end_color );
68 return colors;
69}

81{
82 const float inner_ratio = static_cast<float>( inner.w ) / inner.h;

87
88 const int w = factor * inner.w;
