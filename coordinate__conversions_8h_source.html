7/**
8 * This file defines legacy coordinate conversion functions. We should be
9 * migrating to the new functions defined in coordinates.h.
10 *
11 * For documentation on coordinate systems in general see
12 * doc/POINTS_COORDINATES.md.
13 *
14 * TODO: type for mmr coords, also add them to doc/POINTS_COORDINATES.md
15 * memory map region (mmr): Memory map region is a unit of tile memory saved to a directory.
16 * Each region contains MM_REG_SIZExMM_REG_SIZE memorized submaps, and is used only for
17 * saving/loading memorized submaps, see map_memory.cpp.
18 * Translation from sm to mmr:
19 * sm.x /= MM_REG_SIZE
20 * sm.y /= MM_REG_SIZE
21 * (with special handling for negative values).
22 *
23 * This file provides static translation functions, named like this:
24@code
25 static point <from>_to_<to>_copy(int x, int y);
26 static point <from>_to_<to>_copy(const point& p);
27 static tripoint <from>_to_<to>_copy(const tripoint& p);
28 static void <from>_to_<to>(int &x, int &y);
29 static void <from>_to_<to>(point& p);
30 static void <from>_to_<to>(tripoint& p);
31 static point <from>_to_<to>_remain(int &x, int &y);
32 static point <from>_to_<to>_remain(point& p);
33@endcode
34 * Functions ending with _copy return the translated coordinates,
35 * other functions change the parameters itself and don't return anything.
36 * Functions ending with _remain return the translated coordinates and
37 * store the remainder in the parameters.
38 */
