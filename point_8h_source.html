83 /**
84 * Rotate point clockwise @param turns times, 90 degrees per turn,
85 * around the center of a rectangle with the dimensions specified
86 * by @param dim
87 * By default rotates around the origin (0, 0).
88 * NOLINTNEXTLINE(cata-use-named-point-constants) */
89 point rotate( int turns, const point &dim = { 1, 1 } ) const {
90 assert( turns >= 0 );
91 assert( turns <= 4 );
92
93 switch( turns ) {
94 case 1:
95 return { dim.y - y - 1, x };
96 case 2:
97 return { dim.x - x - 1, dim.y - y - 1 };
98 case 3:
99 return { y, dim.x - x - 1 };
100 }
101
102 return *this;
103 }
