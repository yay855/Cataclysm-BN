67item vehicle_part::properties_to_item() const
68{
69 map &here = get_map();
70 item tmp = base;
71 tmp.unset_flag( "VEHICLE" );
72
73 // Cables get special handling: their target coordinates need to remain
74 // stored, and if a cable actually drops, it should be half-connected.
75 // Except grid-connected ones, for now.
76 if( tmp.has_flag( "CABLE_SPOOL" ) && !tmp.has_flag( "TOW_CABLE" ) ) {
77 if( has_flag( targets_grid ) ) {
78 // Ideally, we'd drop the cable on the charger instead
79 tmp.erase_var( "source_x" );
80 tmp.erase_var( "source_y" );
81 tmp.erase_var( "source_z" );
82 tmp.erase_var( "state" );
83 tmp.active = false;
84 tmp.charges = tmp.type->maximum_charges();
85 } else {
86 const tripoint local_pos = here.getlocal( target.first );
87 if( !here.veh_at( local_pos ) ) {
88 // That vehicle ain't there no more.
89 tmp.set_flag( "NO_DROP" );
90 }
91
92 tmp.set_var( "source_x", target.first.x );
93 tmp.set_var( "source_y", target.first.y );
94 tmp.set_var( "source_z", target.first.z );
95 tmp.set_var( "state", "pay_out_cable" );
96 tmp.active = true;
97 }
98 }
99
100 // force rationalization of damage values to the middle value of each damage level so
101 // that parts will stack nicely
102 tmp.set_damage( tmp.damage_level( 4 ) * itype::damage_scale );
103 return tmp;

260 if( is_tank() && liquid->phase >= LIQUID && qty != 0 ) {

303double vehicle_part::consume_energy( const itype_id &ftype, double energy_j )
304{
305 if( base.contents.empty() || !is_fuel_store() ) {
306 return 0.0f;
307 }
308
309 item &fuel = base.contents.back();
310 if( fuel.typeId() == ftype ) {
311 assert( fuel.is_fuel() );
312 // convert energy density in MJ/L to J/ml
313 const double energy_p_mL = fuel.fuel_energy() * 1000;
314 const int ml_to_use = static_cast<int>( std::floor( energy_j / energy_p_mL ) );
315 int charges_to_use = fuel.charges_per_volume( ml_to_use * 1_ml );
316
317 if( !charges_to_use ) {
318 return 0.0;
319 }
320 if( charges_to_use >= fuel.charges ) {
321 charges_to_use = fuel.charges;
322 base.contents.clear_items();
323 } else {
324 fuel.charges -= charges_to_use;
325 }
326 item fuel_consumed( ftype, calendar::turn, charges_to_use );
327 return energy_p_mL * units::to_milliliter<int>( fuel_consumed.volume( true ) );
328 }
329 return 0.0;

332bool vehicle_part::can_reload( const item &obj ) const
333{
334 // first check part is not destroyed and can contain ammo
335 if( !is_fuel_store() ) {
336 return false;
337 }
338
339 if( !obj.is_null() ) {
340 const itype_id obj_type = obj.typeId();
341 if( is_reactor() ) {
342 return base.is_reloadable_with( obj_type );
343 }
344
345 // forbid filling tanks with solids or non-material things
346 if( is_tank() && ( obj.made_of( SOLID ) || obj.made_of( PNULL ) ) ) {
347 return false;
348 }
349 // forbid putting liquids, gasses, and plasma in things that aren't tanks
350 else if( !obj.made_of( SOLID ) && !is_tank() ) {
351 return false;
352 }
353 // prevent mixing of different ammo
354 if( !ammo_current().is_null() && ammo_current() != obj_type ) {
355 return false;
356 }
357 // For storage with set type, prevent filling with different types
358 if( info().fuel_type != fuel_type_none && info().fuel_type != obj_type ) {
359 return false;
360 }
361 // don't fill magazines with inappropriate fuel
362 if( !is_tank() && !base.is_reloadable_with( obj_type ) ) {
363 return false;
364 }
365 }
366
367 return ammo_remaining() < ammo_capacity();
368}
