12/**
13* Calculates the number of alive neighbors by looking at the Moore neighborhood (3x3 grid of cells).
14* @param cells The cells to look at. Assumed to be of a consistent width/height equal to the specified
15* values.
16* @param size The width+height of the cells. Specified up front to avoid checked it each time.
17* @param p
18* @returns The number of neighbors that are alive, a value between 0 and 8.
19*/
20inline int neighbor_count( const std::vector<std::vector<int>> &cells, const point &size,
21 const point &p )
22{
23 int neighbors = 0;
24 for( int ni = -1; ni <= 1; ni++ ) {
25 for( int nj = -1; nj <= 1; nj++ ) {
26 const point n( p + point( ni, nj ) );
27
28 // These neighbors are outside the bounds, so they can't contribute.
29 if( n.x < 0 || n.x >= size.x || n.y < 0 || n.y >= size.y ) {
30 continue;
31 }
32
33 neighbors += cells[n.x][n.y];
34 }
35 }
36 // Because we included ourself in the loop above, subtract ourselves back out.
37 neighbors -= cells[p.x][p.y];
38
39 return neighbors;
40}

42/**
43* Generate a cellular automaton using the provided parameters.
44* Basic rules are as follows:
45* - alive% of cells start alive
46* - Run for specified number of iterations
47* - Dead cells with > birth_limit neighbors become alive
48* - Alive cells with > statis_limit neighbors stay alive
49* - The rest die
50* @param size Dimensions of the grid.
51* @param alive Number between 0 and 100, used to roll an x_in_y check for the cell to start alive.
52* @param iterations Number of iterations to run the cellular automaton.
53* @param birth_limit Dead cells with > birth_limit neighbors become alive.
54* @param stasis_limit Alive cells with > statis_limit neighbors stay alive
55* @returns The width x height grid of cells. Each cell is a 0 if dead or a 1 if alive.
56*/
57inline std::vector<std::vector<int>> generate_cellular_automaton( const point &size,
58 const int alive, const int iterations, const int birth_limit, const int stasis_limit )
59{
60 std::vector<std::vector<int>> current( size.x, std::vector<int>( size.y, 0 ) );
61 std::vector<std::vector<int>> next( size.x, std::vector<int>( size.y, 0 ) );
62
63 // Initialize our initial set of cells.
64 for( int i = 0; i < size.x; i++ ) {
65 for( int j = 0; j < size.y; j++ ) {
66 current[i][j] = x_in_y( alive, 100 );
67 }
68 }
69
70 for( int iteration = 0; iteration < iterations; iteration++ ) {
71 for( int i = 0; i < size.x; i++ ) {
72 for( int j = 0; j < size.y; j++ ) {
73 // Skip the edges--no need to complicate this with more complex neighbor
74 // calculations, just keep them constant.
75 if( i == 0 || i == size.x - 1 || j == 0 || j == size.y - 1 ) {
76 next[i][j] = 0;
77 continue;
78 }
79
80 // Count our neighors.
81 const int neighbors = neighbor_count( current, size, point( i, j ) );
82
83 // Dead and > birth_limit neighbors, so become alive.
84 if( ( current[i][j] == 0 ) && ( neighbors > birth_limit ) ) {
85 next[i][j] = 1;
86 }
87 // Alive and > statis_limit neighbors, so stay alive.
88 else if( ( current[i][j] == 1 ) && ( neighbors > stasis_limit ) ) {
89 next[i][j] = 1;
90 }
91 // Else, die.
92 else {
93 next[i][j] = 0;
94 }
95 }
96 }
97
98 // Swap our current and next vectors and repeat.
99 std::swap( current, next );
100 }
101
102 return current;
103}
