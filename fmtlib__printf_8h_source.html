21// Checks if a value fits in int - used to avoid warnings about comparing
22// signed and unsigned integers.
23template <bool IsSigned> struct int_checker {
24 template <typename T> static bool fits_in_int( T value ) {
25 unsigned max = max_value<int>();
26 return value <= max;
27 }
28 static bool fits_in_int( bool ) {
29 return true;
30 }
31};
32
33template <> struct int_checker<true> {
34 template <typename T> static bool fits_in_int( T value ) {
35 return value >= ( std::numeric_limits<int>::min )() &&
36 value <= max_value<int>();
37 }
38 static bool fits_in_int( int ) {
39 return true;
40 }
41};

43class printf_precision_handler
44{
45 public:
46 template <typename T, FMT_ENABLE_IF( std::is_integral<T>::value )>
47 int operator()( T value ) {
48 if( !int_checker<std::numeric_limits<T>::is_signed>::fits_in_int( value ) ) {
49 FMT_THROW( format_error( "number is too big" ) );
50 }
51 return ( std::max )( static_cast<int>( value ), 0 );
52 }
53
54 template < typename T, FMT_ENABLE_IF( !std::is_integral<T>::value ) >
55 int operator()( T ) {
56 FMT_THROW( format_error( "precision is not integer" ) );
57 return 0;
58 }

82template <typename T, typename Context> class arg_converter
83{
84 private:
85 using char_type = typename Context::char_type;
86
87 basic_format_arg<Context> &arg_;
88 char_type type_;
89
90 public:
91 arg_converter( basic_format_arg<Context> &arg, char_type type )
92 : arg_( arg ), type_( type ) {}
93
94 void operator()( bool value ) {
95 if( type_ != 's' ) {
96 operator()<bool>( value );
97 }
98 }
99
100 template <typename U, FMT_ENABLE_IF( std::is_integral<U>::value )>
101 void operator()( U value ) {
102 bool is_signed = type_ == 'd' || type_ == 'i';
103 using target_type = conditional_t<std::is_same<T, void>::value, U, T>;
104 if( const_check( sizeof( target_type ) <= sizeof( int ) ) ) {
105 // Extra casts are used to silence warnings.
106 if( is_signed ) {
107 arg_ = detail::make_arg<Context>(
108 static_cast<int>( static_cast<target_type>( value ) ) );
109 } else {
110 using unsigned_type = typename make_unsigned_or_bool<target_type>::type;
111 arg_ = detail::make_arg<Context>(
112 static_cast<unsigned>( static_cast<unsigned_type>( value ) ) );
113 }
114 } else {
115 if( is_signed ) {
116 // glibc's printf doesn't sign extend arguments of smaller types:
117 // std::printf("%lld", -42); // prints "4294967254"
118 // but we don't have to do the same because it's a UB.
119 arg_ = detail::make_arg<Context>( static_cast<long long>( value ) );
120 } else {
121 arg_ = detail::make_arg<Context>(
122 static_cast<typename make_unsigned_or_bool<U>::type>( value ) );
123 }
124 }
125 }
126
127 template < typename U, FMT_ENABLE_IF( !std::is_integral<U>::value ) >
128 void operator()( U ) {} // No conversion needed for non-integral types.
129};

131// Converts an integer argument to T for printf, if T is an integral type.
132// If T is void, the argument is converted to corresponding signed or unsigned
133// type depending on the type specifier: 'd' and 'i' - signed, other -
134// unsigned).
135template <typename T, typename Context, typename Char>
136void convert_arg( basic_format_arg<Context> &arg, Char type )
137{
138 visit_format_arg( arg_converter<T, Context>( arg, type ), arg );
139}

171// Checks if an argument is a valid printf width specifier and sets
172// left alignment if it is negative.
173template <typename Char> class printf_width_handler
174{
175 private:
176 using format_specs = basic_format_specs<Char>;
177
178 format_specs &specs_;
179
180 public:
181 explicit printf_width_handler( format_specs &specs ) : specs_( specs ) {}
182
183 template <typename T, FMT_ENABLE_IF( std::is_integral<T>::value )>
184 unsigned operator()( T value ) {
185 auto width = static_cast<uint32_or_64_or_128_t<T>>( value );
186 if( detail::is_negative( value ) ) {
187 specs_.align = align::left;
188 width = 0 - width;
189 }
190 unsigned int_max = max_value<int>();
191 if( width > int_max ) {
192 FMT_THROW( format_error( "number is too big" ) );
193 }
194 return static_cast<unsigned>( width );
195 }
196
197 template < typename T, FMT_ENABLE_IF( !std::is_integral<T>::value ) >
198 unsigned operator()( T ) {
199 FMT_THROW( format_error( "width is not integer" ) );
200 return 0;
201 }
202};

234template <typename OutputIt, typename Char>
235class printf_arg_formatter : public detail::arg_formatter_base<OutputIt, Char>

240 private:

247 void write_null_pointer( char ) {
248 this->specs()->type = 0;
249 this->write( "(nil)" );
250 }
251
252 void write_null_pointer( wchar_t ) {
253 this->specs()->type = 0;
254 this->write( L"(nil)" );
255 }
256
257 public:
258 using format_specs = typename base::format_specs;

260 /**
261 \rst
262 Constructs an argument formatter object.
263 *buffer* is a reference to the output buffer and *specs* contains format
264 specifier information for standard argument types.

270 template <typename T, FMT_ENABLE_IF( fmt::detail::is_integral<T>::value )>
271 iterator operator()( T value ) {
272 // MSVC2013 fails to compile separate overloads for bool and char_type so
273 // use std::is_same instead.
274 if( std::is_same<T, bool>::value ) {
275 format_specs &fmt_specs = *this->specs();
276 if( fmt_specs.type != 's' ) {
277 return base::operator()( value ? 1 : 0 );
278 }
279 fmt_specs.type = 0;
280 this->write( value != 0 );
281 } else if( std::is_same<T, char_type>::value ) {
282 format_specs &fmt_specs = *this->specs();
283 if( fmt_specs.type && fmt_specs.type != 'c' ) {
284 return ( *this )( static_cast<int>( value ) );
285 }
286 fmt_specs.sign = sign::none;
287 fmt_specs.alt = false;
288 fmt_specs.fill[0] = ' '; // Ignore '0' flag for char types.
289 // align::numeric needs to be overwritten here since the '0' flag is
290 // ignored for non-numeric types
291 if( fmt_specs.align == align::none || fmt_specs.align == align::numeric ) {
292 fmt_specs.align = align::right;
293 }
294 return base::operator()( value );
295 } else {
296 return base::operator()( value );
297 }
298 return this->out();
299 }
300
301 template <typename T, FMT_ENABLE_IF( std::is_floating_point<T>::value )>
302 iterator operator()( T value ) {
303 return base::operator()( value );
304 }
305
306 /** Formats a null-terminated C string. */
307 iterator operator()( const char *value ) {
308 if( value ) {
309 base::operator()( value );
310 } else if( this->specs()->type == 'p' ) {
311 write_null_pointer( char_type() );
312 } else {
313 this->write( "(null)" );
314 }
315 return this->out();
316 }
317
318 /** Formats a null-terminated wide C string. */
319 iterator operator()( const wchar_t *value ) {
320 if( value ) {
321 base::operator()( value );
322 } else if( this->specs()->type == 'p' ) {
323 write_null_pointer( char_type() );
324 } else {
325 this->write( L"(null)" );
326 }
327 return this->out();
328 }
329
330 iterator operator()( basic_string_view<char_type> value ) {
331 return base::operator()( value );
332 }
333
334 iterator operator()( monostate value ) {
335 return base::operator()( value );
336 }
337
338 /** Formats a pointer. */
339 iterator operator()( const void *value ) {
340 if( value ) {
341 return base::operator()( value );
342 }
343 this->specs()->type = 0;
344 write_null_pointer( char_type() );
345 return this->out();
346 }
347
348 /** Formats an argument of a custom (user-defined) type. */
349 iterator operator()( typename basic_format_arg<context_type>::handle handle ) {
350 handle.format( context_.parse_context(), context_ );
351 return this->out();
352 }
353};

439template <typename OutputIt, typename Char>
440void basic_printf_context<OutputIt, Char>::parse_flags( format_specs &specs,
441 const Char *&it,
442 const Char *end )
443{
444 for( ; it != end; ++it ) {
445 switch( *it ) {
446 case '-':
447 specs.align = align::left;
448 break;
449 case '+':
450 specs.sign = sign::plus;
451 break;
452 case '0':
453 specs.fill[0] = '0';
454 break;
455 case ' ':
456 if( specs.sign != sign::plus ) {
457 specs.sign = sign::space;
458 }
459 break;
460 case '#':
461 specs.alt = true;
462 break;
463 default:
464 return;
465 }
466 }
467}

469template <typename OutputIt, typename Char>
470typename basic_printf_context<OutputIt, Char>::format_arg
471basic_printf_context<OutputIt, Char>::get_arg( int arg_index )
472{
473 if( arg_index < 0 ) {
474 arg_index = parse_ctx_.next_arg_id();
475 } else {
476 parse_ctx_.check_arg_id( --arg_index );
477 }
478 return detail::get_arg( *this, arg_index );

481template <typename OutputIt, typename Char>
482int basic_printf_context<OutputIt, Char>::parse_header( const Char *&it,
483 const Char *end,
484 format_specs &specs )
485{
486 int arg_index = -1;
487 char_type c = *it;
488 if( c >= '0' && c <= '9' ) {
489 // Parse an argument index (if followed by '$') or a width possibly
490 // preceded with '0' flag(s).
491 detail::error_handler eh;
492 int value = parse_nonnegative_int( it, end, eh );
493 if( it != end && *it == '$' ) { // value is an argument index
494 ++it;
495 arg_index = value;
496 } else {
497 if( c == '0' ) {
498 specs.fill[0] = '0';
499 }
500 if( value != 0 ) {
501 // Nonzero value means that we parsed width and don't need to
502 // parse it or flags again, so return now.
503 specs.width = value;
504 return arg_index;
505 }
506 }
507 }
508 parse_flags( specs, it, end );
509 // Parse width.
510 if( it != end ) {
511 if( *it >= '0' && *it <= '9' ) {
512 detail::error_handler eh;
513 specs.width = parse_nonnegative_int( it, end, eh );
514 } else if( *it == '*' ) {
515 ++it;
516 specs.width = static_cast<int>( visit_format_arg(
517 detail::printf_width_handler<char_type>( specs ), get_arg() ) );
518 }
519 }
520 return arg_index;
521}

523template <typename OutputIt, typename Char>
524template <typename ArgFormatter>
525OutputIt basic_printf_context<OutputIt, Char>::format()
526{
527 auto out = this->out();
528 const Char *start = parse_ctx_.begin();
529 const Char *end = parse_ctx_.end();
530 auto it = start;
531 while( it != end ) {
532 char_type c = *it++;
533 if( c != '%' ) {
534 continue;
535 }
536 if( it != end && *it == c ) {
537 out = std::copy( start, it, out );
538 start = ++it;
539 continue;
540 }
541 out = std::copy( start, it - 1, out );
542
543 format_specs specs;
544 specs.align = align::right;
545
546 // Parse argument index, flags and width.
547 int arg_index = parse_header( it, end, specs );
548 if( arg_index == 0 ) {
549 on_error( "argument not found" );
550 }
551
552 // Parse precision.
553 if( it != end && *it == '.' ) {
554 ++it;
555 c = it != end ? *it : 0;
556 if( '0' <= c && c <= '9' ) {
557 detail::error_handler eh;
558 specs.precision = parse_nonnegative_int( it, end, eh );
559 } else if( c == '*' ) {
560 ++it;
561 specs.precision = static_cast<int>(
562 visit_format_arg( detail::printf_precision_handler(), get_arg() ) );
563 } else {
564 specs.precision = 0;
565 }
566 }
567
568 format_arg arg = get_arg( arg_index );
569 // For d, i, o, u, x, and X conversion specifiers, if a precision is
570 // specified, the '0' flag is ignored
571 if( specs.precision >= 0 && arg.is_integral() )
572 specs.fill[0] =
573 ' '; // Ignore '0' flag for non-numeric types or if '-' present.
574 if( specs.precision >= 0 && arg.type() == detail::type::cstring_type ) {
575 auto str = visit_format_arg( detail::get_cstring<Char>(), arg );
576 auto str_end = str + specs.precision;
577 auto nul = std::find( str, str_end, Char() );
578 arg = detail::make_arg<basic_printf_context>( basic_string_view<Char>(
579 str,
580 detail::to_unsigned( nul != str_end ? nul - str : specs.precision ) ) );
581 }
582 if( specs.alt && visit_format_arg( detail::is_zero_int(), arg ) ) {
583 specs.alt = false;
584 }
585 if( specs.fill[0] == '0' ) {
586 if( arg.is_arithmetic() && specs.align != align::left ) {
587 specs.align = align::numeric;
588 } else {
589 specs.fill[0] = ' '; // Ignore '0' flag for non-numeric types or if '-'
590 }
591 // flag is also present.
592 }
593
594 // Parse length and convert the argument to the required type.
595 c = it != end ? *it++ : 0;
596 char_type t = it != end ? *it : 0;
597 using detail::convert_arg;
598 switch( c ) {
599 case 'h':
600 if( t == 'h' ) {
601 ++it;
602 t = it != end ? *it : 0;
603 convert_arg<signed char>( arg, t );
604 } else {
605 convert_arg<short>( arg, t );
606 }
607 break;
608 case 'l':
609 if( t == 'l' ) {
610 ++it;
611 t = it != end ? *it : 0;
612 convert_arg<long long>( arg, t );
613 } else {
614 convert_arg<long>( arg, t );
615 }
616 break;
617 case 'j':
618 convert_arg<intmax_t>( arg, t );
619 break;
620 case 'z':
621 convert_arg<size_t>( arg, t );
622 break;
623 case 't':
624 convert_arg<std::ptrdiff_t>( arg, t );
625 break;
626 case 'L':
627 // printf produces garbage when 'L' is omitted for long double, no
628 // need to do the same.
629 break;
630 default:
631 --it;
632 convert_arg<void>( arg, c );
633 }
634
635 // Parse type.
636 if( it == end ) {
637 FMT_THROW( format_error( "invalid format string" ) );
638 }
639 specs.type = static_cast<char>( *it++ );
640 if( arg.is_integral() ) {
641 // Normalize type.
642 switch( specs.type ) {
643 case 'i':
644 case 'u':
645 specs.type = 'd';
646 break;
647 case 'c':
648 visit_format_arg( detail::char_converter<basic_printf_context>( arg ),
649 arg );
650 break;
651 }
652 }
653
654 start = it;
655
656 // Format argument.
657 out = visit_format_arg( ArgFormatter( out, specs, *this ), arg );
658 }
659 return std::copy( start, it, out );
660}

727 std::FILE *f, const S &format,
728 basic_format_args<basic_printf_context_t<type_identity_t<Char>>> args )

730 basic_memory_buffer<Char> buffer;

735 : static_cast<int>( size );
