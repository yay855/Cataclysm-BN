425template <typename T>
426advanced_inv_area::itemstack advanced_inv_area::i_stacked( T items )
427{
428 //create a new container for our stacked items
429 advanced_inv_area::itemstack stacks;
430 // // make a list of the items first, so we can add non stacked items back on
431 // std::list<item> items(things.begin(), things.end());
432 // used to recall indices we stored `itype_id' item at in itemstack
433 std::unordered_map<itype_id, std::set<int>> cache;
434 // iterate through and create stacks
435 for( auto &elem : items ) {
436 const auto id = elem.typeId();
437 auto iter = cache.find( id );
438 bool got_stacked = false;
439 // cache entry exists
440 if( iter != cache.end() ) {
441 // check to see if it stacks with each item in a stack, not just front()
442 for( auto &idx : iter->second ) {
443 for( auto &it : stacks[idx] ) {
444 if( ( got_stacked = it->display_stacked_with( elem ) ) ) {
445 stacks[idx].push_back( &elem );
446 break;
447 }
448 }
449 if( got_stacked ) {
450 break;
451 }
452 }
453 }
454 if( !got_stacked ) {
455 cache[id].insert( stacks.size() );
456 stacks.push_back( { &elem } );
457 }
458 }
459 return stacks;
460}
