83// Figure out encumbrance penalties this clothing is involved in
84item_penalties get_item_penalties( std::list<item>::const_iterator worn_item_it,
85 const Character &c, int tabindex )
86{
87 layer_level layer = worn_item_it->get_layer();
88
89 std::vector<body_part> body_parts_with_stacking_penalty;
90 std::vector<body_part> body_parts_with_out_of_order_penalty;
91 std::vector<std::set<std::string>> lists_of_bad_items_within;
92
93 for( body_part bp : all_body_parts ) {
94 if( bp != tabindex && num_bp != tabindex ) {
95 continue;
96 }
97 if( !worn_item_it->covers( bp ) ) {
98 continue;
99 }
100 const int num_items = std::count_if( c.worn.begin(), c.worn.end(),
101 [layer, bp]( const item & i ) {
102 return i.get_layer() == layer && i.covers( bp ) && !( i.has_flag( flag_SEMITANGIBLE ) ||
103 i.has_flag( flag_COMPACT ) );
104 } );
105 if( num_items > 1 ) {
106 body_parts_with_stacking_penalty.push_back( bp );
107 }
108
109 std::set<std::string> bad_items_within;
110 for( auto it = c.worn.begin(); it != worn_item_it; ++it ) {
111 if( it->get_layer() > layer && it->covers( bp ) ) {
112 bad_items_within.insert( it->type_name() );
113 }
114 }
115 if( !bad_items_within.empty() ) {
116 body_parts_with_out_of_order_penalty.push_back( bp );
117 lists_of_bad_items_within.push_back( bad_items_within );
118 }
119 }
120
121 // We intersect all the lists_of_bad_items_within so that if there is one
122 // common bad item we're wearing this one over it can be mentioned in the
123 // message explaining the penalty.
124 while( lists_of_bad_items_within.size() > 1 ) {
125 std::set<std::string> intersection_of_first_two;
126 std::set_intersection(
127 lists_of_bad_items_within[0].begin(), lists_of_bad_items_within[0].end(),
128 lists_of_bad_items_within[1].begin(), lists_of_bad_items_within[1].end(),
129 std::inserter( intersection_of_first_two, intersection_of_first_two.begin() )
130 );
131 lists_of_bad_items_within.erase( lists_of_bad_items_within.begin() );
132 lists_of_bad_items_within[0] = std::move( intersection_of_first_two );
133 }
134
135 if( lists_of_bad_items_within.empty() ) {
136 lists_of_bad_items_within.emplace_back();
137 }
138
139 return { std::move( body_parts_with_stacking_penalty ),
140 std::move( body_parts_with_out_of_order_penalty ),
141 std::move( lists_of_bad_items_within[0] ) };
142}

144std::string body_part_names( const std::vector<body_part> &parts )
145{
146 if( parts.empty() ) {
147 debugmsg( "Asked for names of empty list" );
148 return {};
149 }
150
151 std::vector<std::string> names;
152 names.reserve( parts.size() );
153 for( size_t i = 0; i < parts.size(); ++i ) {
154 const body_part part = parts[i];
155 if( i + 1 < parts.size() && parts[i + 1] == static_cast<body_part>( bp_aiOther[part] ) ) {
156 // Can combine two body parts (e.g. arms)
157 names.push_back( body_part_name_accusative( part, 2 ) );
158 ++i;
159 } else {
160 names.push_back( body_part_name_accusative( part ) );
161 }
162 }
163
164 return enumerate_as_string( names );
165}

386struct layering_item_info {
387 item_penalties penalties;
388 int encumber;
389 std::string name;
390
391 // Operator overload required to leverage vector equality operator.
392 bool operator ==( const layering_item_info &o ) const {
393 // This is used to merge e.g. both arms into one entry when their items
394 // are equivalent. For that purpose we don't care about the exact
395 // penalities because they will list different body parts; we just
396 // check that the badness is the same (which is all that matters for
397 // rendering the right-hand list).
398 return this->penalties.badness() == o.penalties.badness() &&
399 this->encumber == o.encumber &&
400 this->name == o.name;
401 }
402};
