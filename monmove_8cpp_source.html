83bool monster::is_immune_field( const field_type_id &fid ) const
84{
85 if( fid == fd_fungal_haze ) {
86 return has_flag( MF_NO_BREATHE ) || type->in_species( FUNGUS );
87 }
88 if( fid == fd_fungicidal_gas ) {
89 return !type->in_species( FUNGUS );
90 }
91 if( fid == fd_insecticidal_gas ) {
92 return !type->in_species( INSECT ) && !type->in_species( SPIDER );
93 }
94 const field_type &ft = fid.obj();
95 if( ft.has_fume ) {
96 return has_flag( MF_NO_BREATHE );
97 }
98 if( ft.has_acid ) {
99 return has_flag( MF_ACIDPROOF ) || flies();
100 }
101 if( ft.has_fire ) {
102 return has_flag( MF_FIREPROOF );
103 }
104 if( ft.has_elec ) {
105 return has_flag( MF_ELECTRIC );
106 }
107 if( ft.immune_mtypes.count( type->id ) > 0 ) {
108 return true;
109 }
110 // No specific immunity was found, so fall upwards
111 return Creature::is_immune_field( fid );
112}

119bool monster::will_move_to( const tripoint &p ) const
120{
121 if( g->m.impassable( p ) ) {
122 if( digging() ) {
123 if( !g->m.has_flag( "BURROWABLE", p ) ) {
124 return false;
125 }
126 } else if( !( can_climb() && g->m.has_flag( "CLIMBABLE", p ) ) ) {
127 return false;
128 }
129 }
130
131 if( ( !can_submerge() && !flies() ) && g->m.has_flag( TFLAG_DEEP_WATER, p ) ) {
132 return false;
133 }
134
135 if( digs() && !g->m.has_flag( "DIGGABLE", p ) && !g->m.has_flag( "BURROWABLE", p ) ) {
136 return false;
137 }
138
139 if( has_flag( MF_AQUATIC ) && !g->m.has_flag( "SWIMMABLE", p ) ) {
140 return false;
141 }
142
143 if( has_flag( MF_SUNDEATH ) && g->is_in_sunlight( p ) ) {
144 return false;
145 }
146
147 if( get_size() > MS_MEDIUM && g->m.has_flag_ter( TFLAG_SMALL_PASSAGE, p ) ) {
148 return false; // if a large critter, can't move through tight passages
149 }
150
151 // Various avoiding behaviors.
152
153 bool avoid_fire = has_flag( MF_AVOID_FIRE );
154 bool avoid_fall = has_flag( MF_AVOID_FALL );
155 bool avoid_simple = has_flag( MF_AVOID_DANGER_1 );
156 bool avoid_complex = has_flag( MF_AVOID_DANGER_2 );
157 /*
158 * Because some avoidance behaviors are supersets of others,
159 * we can cascade through the implications. Complex implies simple,
160 * and simple implies fire and fall.
161 * unfortunately, fall does not necessarily imply fire, nor the converse.
162 */
163 if( avoid_complex ) {
164 avoid_simple = true;
165 }
166 if( avoid_simple ) {
167 avoid_fire = true;
168 avoid_fall = true;
169 }
170
171 // technically this will shortcut in evaluation from fire or fall
172 // before hitting simple or complex but this is more explicit
173 if( avoid_fire || avoid_fall || avoid_simple || avoid_complex ) {
174 const ter_id target = g->m.ter( p );
175
176 // Don't enter lava if we have any concept of heat being bad
177 if( avoid_fire && target == t_lava ) {
178 return false;
179 }
180
181 if( avoid_fall ) {
182 // Don't throw ourselves off cliffs if we have a concept of falling
183 if( !g->m.has_floor( p ) && !flies() ) {
184 return false;
185 }
186
187 // Don't enter open pits ever unless tiny, can fly or climb well
188 if( !( type->size == MS_TINY || can_climb() ) &&
189 ( target == t_pit || target == t_pit_spiked || target == t_pit_glass ) ) {
190 return false;
191 }
192 }
193
194 // Some things are only avoided if we're not attacking
195 if( attitude( &g->u ) != MATT_ATTACK ) {
196 // Sharp terrain is ignored while attacking
197 if( avoid_simple && g->m.has_flag( "SHARP", p ) &&
198 !( type->size == MS_TINY || flies() ) ) {
199 return false;
200 }
201 }
202
203 const field &target_field = g->m.field_at( p );
204
205 // Higher awareness is needed for identifying these as threats.
206 if( avoid_complex ) {
207 const trap &target_trap = g->m.tr_at( p );
208 // Don't enter any dangerous fields
209 if( is_dangerous_fields( target_field ) ) {
210 return false;
211 }
212 // Don't step on any traps (if we can see)
213 if( has_flag( MF_SEES ) && !target_trap.is_benign() && g->m.has_floor( p ) ) {
214 return false;
215 }
216 }
217
218 // Without avoid_complex, only fire and electricity are checked for field avoidance.
219 if( avoid_fire && target_field.find_field( fd_fire ) ) {
220 return false;
221 }
222 if( avoid_simple && target_field.find_field( fd_electricity ) ) {
223 return false;
224 }
225 }
226
227 return true;
228}

230bool monster::can_reach_to( const tripoint &p ) const
231{
232 map &here = get_map();
233 if( p.z > pos().z && z_is_valid( pos().z ) ) {
234 if( here.has_flag( TFLAG_RAMP_UP, tripoint( p.xy(), p.z - 1 ) ) ) {
235 return true;
236 }
237 if( !here.has_flag( TFLAG_GOES_UP, pos() ) && !here.has_flag( TFLAG_NO_FLOOR, p ) ) {
238 // can't go through the roof
239 return false;
240 }
241 } else if( p.z < pos().z && z_is_valid( pos().z ) ) {
242 if( !here.has_flag( TFLAG_GOES_DOWN, pos() ) ) {
243 // can't go through the floor
244 // you would fall anyway if there was no floor, so no need to check for that here
245 return false;
246 }
247 }
248 return true;
249}

282float monster::rate_target( Creature &c, float best, bool smart ) const
283{
284 const auto d = rl_dist_fast( pos(), c.pos() );
285 if( d <= 0 ) {
286 return FLT_MAX;
287 }
288
289 // Check a very common and cheap case first
290 if( !smart && d >= best ) {
291 return FLT_MAX;
292 }
293
294 if( !sees( c ) ) {
295 return FLT_MAX;
296 }
297
298 if( !smart ) {
299 return int( d );
300 }
301
302 float power = c.power_rating();
303 monster *mon = dynamic_cast< monster * >( &c );
304 // Their attitude to us and not ours to them, so that bobcats won't get gunned down
305 if( mon != nullptr && mon->attitude_to( *this ) == Attitude::A_HOSTILE ) {
306 power += 2;
307 }
308
309 if( power > 0 ) {
310 return int( d ) / power;
311 }
312
313 return FLT_MAX;
314}

316void monster::plan()
317{
318 const auto &factions = g->critter_tracker->factions();
319
320 // Bots are more intelligent than most living stuff
321 bool smart_planning = has_flag( MF_PRIORITIZE_TARGETS );
322 Creature *target = nullptr;
323 int max_sight_range = std::max( type->vision_day, type->vision_night );
324 // 8.6f is rating for tank drone 60 tiles away, moose 16 or boomer 33
325 float dist = !smart_planning ? max_sight_range : 8.6f;
326 bool fleeing = false;
327 bool docile = friendly != 0 && has_effect( effect_docile );
328 bool waiting = has_effect( effect_ai_waiting );
329
330 const bool angers_hostile_weak = type->has_anger_trigger( mon_trigger::HOSTILE_WEAK );
331 const int angers_hostile_near = type->has_anger_trigger( mon_trigger::HOSTILE_CLOSE ) ? 5 : 0;
332 const int angers_mating_season = type->has_anger_trigger( mon_trigger::MATING_SEASON ) ? 3 : 0;
333 const int angers_cub_threatened = type->has_anger_trigger( mon_trigger::PLAYER_NEAR_BABY ) ? 8 : 0;
334 const int fears_hostile_near = type->has_fear_trigger( mon_trigger::HOSTILE_CLOSE ) ? 5 : 0;
335
336 bool group_morale = has_flag( MF_GROUP_MORALE ) && morale < type->morale;
337 bool swarms = has_flag( MF_SWARMS );
338 auto mood = attitude();
339
340 // If we can see the player, move toward them or flee, simpleminded animals are too dumb to follow the player.
341 if( friendly == 0 && sees( g->u ) && !waiting ) {
342 dist = rate_target( g->u, dist, smart_planning );
343 fleeing = fleeing || is_fleeing( g->u );
344 target = &g->u;
345 if( dist <= 5 ) {
346 anger += angers_hostile_near;
347 morale -= fears_hostile_near;
348 if( angers_mating_season > 0 ) {
349 bool mating_angry = false;
350 season_type season = season_of_year( calendar::turn );
351 for( auto &elem : type->baby_flags ) {
352 if( ( season == SUMMER && elem == "SUMMER" ) ||
353 ( season == WINTER && elem == "WINTER" ) ||
354 ( season == SPRING && elem == "SPRING" ) ||
355 ( season == AUTUMN && elem == "AUTUMN" ) ) {
356 mating_angry = true;
357 break;
358 }
359 }
360 if( mating_angry ) {
361 anger += angers_mating_season;
362 }
363 }
364 }
365 if( angers_cub_threatened > 0 ) {
366 for( monster &tmp : g->all_monsters() ) {
367 if( type->baby_monster == tmp.type->id ) {
368 // baby nearby; is the player too close?
369 dist = tmp.rate_target( g->u, dist, smart_planning );
370 if( dist <= 3 ) {
371 //proximity to baby; monster gets furious and less likely to flee
372 anger += angers_cub_threatened;
373 morale += angers_cub_threatened / 2;
374 }
375 }
376 }
377 }
378 } else if( friendly != 0 && !docile && !waiting ) {
379 for( monster &tmp : g->all_monsters() ) {
380 if( tmp.friendly == 0 ) {
381 float rating = rate_target( tmp, dist, smart_planning );
382 if( rating < dist ) {
383 target = &tmp;
384 dist = rating;
385 }
386 }
387 }
388 }
389
390 if( waiting ) {
391 set_dest( pos() );
392 return;
393 }
394
395 if( docile ) {
396 if( friendly != 0 && target != nullptr ) {
397 set_dest( target->pos() );
398 }
399
400 return;
401 }
402
403 int valid_targets = ( target == nullptr ) ? 1 : 0;
404 for( npc &who : g->all_npcs() ) {
405 auto faction_att = faction.obj().attitude( who.get_monster_faction() );
406 if( faction_att == MFA_NEUTRAL || faction_att == MFA_FRIENDLY ) {
407 continue;
408 }
409
410 float rating = rate_target( who, dist, smart_planning );
411 bool fleeing_from = is_fleeing( who );
412 if( rating == dist && ( fleeing || attitude( &who ) == MATT_ATTACK ) ) {
413 ++valid_targets;
414 if( one_in( valid_targets ) ) {
415 target = &who;
416 }
417 }
418 // Switch targets if closer and hostile or scarier than current target
419 if( ( rating < dist && fleeing ) ||
420 ( faction_att == MFA_HATE ) ||
421 ( rating < dist && attitude( &who ) == MATT_ATTACK ) ||
422 ( !fleeing && fleeing_from ) ) {
423 target = &who;
424 dist = rating;
425 valid_targets = 1;
426 }
427 fleeing = fleeing || fleeing_from;
428 if( rating <= 5 ) {
429 anger += angers_hostile_near;
430 morale -= fears_hostile_near;
431 if( angers_mating_season > 0 ) {
432 bool mating_angry = false;
433 season_type season = season_of_year( calendar::turn );
434 for( auto &elem : type->baby_flags ) {
435 if( ( season == SUMMER && elem == "SUMMER" ) ||
436 ( season == WINTER && elem == "WINTER" ) ||
437 ( season == SPRING && elem == "SPRING" ) ||
438 ( season == AUTUMN && elem == "AUTUMN" ) ) {
439 mating_angry = true;
440 break;
441 }
442 }
443 if( mating_angry ) {
444 anger += angers_mating_season;
445 }
446 }
447 }
448 }
449
450 fleeing = fleeing || ( mood == MATT_FLEE );
451 if( friendly == 0 ) {
452 for( const auto &fac : factions ) {
453 auto faction_att = faction.obj().attitude( fac.first );
454 if( faction_att == MFA_NEUTRAL || faction_att == MFA_FRIENDLY ) {
455 continue;
456 }
457
458 for( const weak_ptr_fast<monster> &weak : fac.second ) {
459 const shared_ptr_fast<monster> shared = weak.lock();
460 if( !shared ) {
461 continue;
462 }
463 monster &mon = *shared;
464 float rating = rate_target( mon, dist, smart_planning );
465 if( rating == dist ) {
466 ++valid_targets;
467 if( one_in( valid_targets ) ) {
468 target = &mon;
469 }
470 }
471 if( rating < dist ) {
472 target = &mon;
473 dist = rating;
474 valid_targets = 1;
475 }
476 if( rating <= 5 ) {
477 anger += angers_hostile_near;
478 morale -= fears_hostile_near;
479 }
480 }
481 }
482 }
483
484 // Friendly monsters here
485 // Avoid for hordes of same-faction stuff or it could get expensive
486 const auto actual_faction = friendly == 0 ? faction : mfaction_str_id( "player" );
487 const auto &myfaction_iter = factions.find( actual_faction );
488 if( myfaction_iter == factions.end() ) {
489 DebugLog( DL::Error, DC::Game ) << disp_name() << " tried to find faction "
490 << actual_faction.id().str()
491 << " which wasn't loaded in game::monmove";
492 swarms = false;
493 group_morale = false;
494 }
495 swarms = swarms && target == nullptr; // Only swarm if we have no target
496 if( group_morale || swarms ) {
497 for( const weak_ptr_fast<monster> &weak : myfaction_iter->second ) {
498 const shared_ptr_fast<monster> shared = weak.lock();
499 if( !shared ) {
500 continue;
501 }
502 monster &mon = *shared;
503 float rating = rate_target( mon, dist, smart_planning );
504 if( group_morale && rating <= 10 ) {
505 morale += 10 - rating;
506 }
507 if( swarms ) {
508 if( rating < 5 ) { // Too crowded here
509 wander_pos.x = posx() * rng( 1, 3 ) - mon.posx();
510 wander_pos.y = posy() * rng( 1, 3 ) - mon.posy();
511 wandf = 2;
512 target = nullptr;
513 // Swarm to the furthest ally you can see
514 } else if( rating < FLT_MAX && rating > dist && wandf <= 0 ) {
515 target = &mon;
516 dist = rating;
517 }
518 }
519 }
520 }
521
522 // Operating monster keep you safe while they operate, how nice....
523 if( type->has_special_attack( "OPERATE" ) ) {
524 int prev_friendlyness = friendly;
525 if( has_effect( effect_operating ) ) {
526 friendly = 100;
527 for( auto critter : g->m.get_creatures_in_radius( pos(), 6 ) ) {
528 monster *mon = dynamic_cast<monster *>( critter );
529 if( mon != nullptr && mon->type->in_species( ZOMBIE ) ) {
530 anger = 100;
531 } else {
532 anger = 0;
533 }
534 }
535 } else {
536 friendly = prev_friendlyness;
537 }
538 }
539
540 if( has_effect( effect_dragging ) ) {
541
542 if( type->has_special_attack( "OPERATE" ) ) {
543
544 bool found_path_to_couch = false;
545 tripoint tmp( pos() + point( 12, 12 ) );
546 tripoint couch_loc;
547 for( const auto &couch_pos : g->m.find_furnitures_with_flag_in_radius( pos(), 10,
548 flag_AUTODOC_COUCH ) ) {
549 if( g->m.clear_path( pos(), couch_pos, 10, 0, 100 ) ) {
550 if( rl_dist( pos(), couch_pos ) < rl_dist( pos(), tmp ) ) {
551 tmp = couch_pos;
552 found_path_to_couch = true;
553 couch_loc = couch_pos;
554 }
555 }
556 }
557
558 if( !found_path_to_couch ) {
559 anger = 0;
560 remove_effect( effect_dragging );
561 } else {
562 set_dest( couch_loc );
563 }
564 }
565
566 } else if( target != nullptr ) {
567
568 tripoint dest = target->pos();
569 auto att_to_target = attitude_to( *target );
570 if( att_to_target == Attitude::A_HOSTILE && !fleeing ) {
571 set_dest( dest );
572 } else if( fleeing ) {
573 set_dest( tripoint( posx() * 2 - dest.x, posy() * 2 - dest.y, posz() ) );
574 }
575 if( angers_hostile_weak && att_to_target != Attitude::A_FRIENDLY ) {
576 int hp_per = target->hp_percentage();
577 if( hp_per <= 70 ) {
578 anger += 10 - static_cast<int>( hp_per / 10 );
579 }
580 }
581 } else if( friendly > 0 && one_in( 3 ) ) {
582 // Grow restless with no targets
583 friendly--;
584 } else if( friendly < 0 && sees( g->u ) && !has_flag( MF_PET_WONT_FOLLOW ) ) {
585 if( rl_dist( pos(), g->u.pos() ) > 2 ) {
586 set_dest( g->u.pos() );
587 } else {
588 unset_dest();
589 }
590 }
591
592 // being led by a leash override other movements decisions
593 if( has_effect( effect_led_by_leash ) && friendly != 0 ) {
594 // if we have an hostile target adjacent to the payer, and we're not fleeing, we can potentially attack it
595 if( target != nullptr && rl_dist( g->u.pos(), target->pos() ) < 2 &&
596 target->attitude_to( g->u ) == Attitude::A_HOSTILE && !fleeing ) {
597 // if we're too far from the player, go back to it
598 if( rl_dist( pos(), g->u.pos() ) > 5 ) {
599 set_dest( g->u.pos() );
600 }
601 } else if( rl_dist( pos(), g->u.pos() ) > 1 ) {
602 set_dest( g->u.pos() );
603 } else {
604 unset_dest();
605 }
606 }
607}

609/**
610 * Method to make monster movement speed consistent in the face of staggering behavior and
611 * differing distance metrics.
612 * It works by scaling the cost to take a step by
613 * how much that step reduces the distance to your goal.
614 * Since it incorporates the current distance metric,
615 * it also scales for diagonal vs orthogonal movement.
616 **/
617static float get_stagger_adjust( const tripoint &source, const tripoint &destination,
618 const tripoint &next_step )
619{
620 // TODO: push this down into rl_dist
621 const float initial_dist =
622 trigdist ? trig_dist( source, destination ) : rl_dist( source, destination );
623 const float new_dist =
624 trigdist ? trig_dist( next_step, destination ) : rl_dist( next_step, destination );
625 // If we return 0, it wil cancel the action.
626 return std::max( 0.01f, initial_dist - new_dist );
627}

651// General movement.
652// Currently, priority goes:
653// 1) Special Attack
654// 2) Sight-based tracking
655// 3) Scent-based tracking
656// 4) Sound-based tracking
657void monster::move()
658{
659 // We decrement wandf no matter what. We'll save our wander_to plans until
660 // after we finish out set_dest plans, UNLESS they time out first.
661 if( wandf > 0 ) {
662 wandf--;
663 }
664
665 //Hallucinations have a chance of disappearing each turn
666 if( is_hallucination() && one_in( 25 ) ) {
667 die( nullptr );
668 return;
669 }
670 map &here = get_map();
671
672 behavior::monster_oracle_t oracle( this );
673 behavior::tree goals;
674 goals.add( type->get_goals() );
675 std::string action = goals.tick( &oracle );
676 //The monster can consume objects it stands on. Check if there are any.
677 //If there are. Consume them.
678 // TODO: Stick this in a map and dispatch to it via the action string.
679 if( action == "consume_items" ) {
680 if( g->u.sees( *this ) ) {
681 add_msg( _( "The %s flows around the objects on the floor and they are quickly dissolved!" ),
682 name() );
683 }
684 static const auto volume_per_hp = 250_ml;
685 for( auto &elem : g->m.i_at( pos() ) ) {
686 hp += elem.volume() / volume_per_hp; // Yeah this means it can get more HP than normal.
687 if( has_flag( MF_ABSORBS_SPLITS ) ) {
688 while( hp / 2 > type->hp ) {
689 monster *const spawn = g->place_critter_around( type->id, pos(), 1 );
690 if( !spawn ) {
691 break;
692 }
693 hp -= type->hp;
694 //this is a new copy of the monster. Ideally we should copy the stats/effects that affect the parent
695 spawn->make_ally( *this );
696 if( g->u.sees( *this ) ) {
697 add_msg( _( "The %s splits in two!" ), name() );
698 }
699 }
700 }
701 }
702 g->m.i_clear( pos() );
703 }
704 // record position before moving to put the player there if we're dragging
705 tripoint drag_to = g->m.getabs( pos() );
706
707 const bool pacified = has_effect( effect_pacified );
708
709 // First, use the special attack, if we can!
710 // The attack may change `monster::special_attacks` (e.g. by transforming
711 // this into another monster type). Therefore we can not iterate over it
712 // directly and instead iterate over the map from the monster type
713 // (properties of monster types should never change).
714 for( const auto &sp_type : type->special_attacks ) {
715 const std::string &special_name = sp_type.first;
716 const auto local_iter = special_attacks.find( special_name );
717 if( local_iter == special_attacks.end() ) {
718 continue;
719 }
720 mon_special_attack &local_attack_data = local_iter->second;
721 if( !local_attack_data.enabled ) {
722 continue;
723 }
724
725 // Cooldowns are decremented in monster::process_turn
726
727 if( local_attack_data.cooldown == 0 && !pacified && !is_hallucination() ) {
728 if( !sp_type.second->call( *this ) ) {
729 continue;
730 }
731
732 // `special_attacks` might have changed at this point. Sadly `reset_special`
733 // doesn't check the attack name, so we need to do it here.
734 if( special_attacks.count( special_name ) == 0 ) {
735 continue;
736 }
737 reset_special( special_name );
738 }
739 }
740
741 // Check if they're dragging a foe and find their hapless victim
742 player *dragged_foe = find_dragged_foe();
743
744 // Give nursebots a chance to do surgery.
745 nursebot_operate( dragged_foe );
746
747 // The monster can sometimes hang in air due to last fall being blocked
748 if( !flies() && g->m.has_flag( TFLAG_NO_FLOOR, pos() ) ) {
749 g->m.creature_on_trap( *this, false );
750 if( is_dead() ) {
751 return;
752 }
753 }
754
755 // if the monster is in a deep water tile, it has a chance to drown
756 if( die_if_drowning( pos(), 10 ) ) {
757 return;
758 }
759
760 if( moves < 0 ) {
761 return;
762 }
763
764 // TODO: Move this to attack_at/move_to/etc. functions
765 bool attacking = false;
766 if( !move_effects( attacking ) ) {
767 moves = 0;
768 return;
769 }
770 if( has_flag( MF_IMMOBILE ) || has_flag( MF_RIDEABLE_MECH ) ) {
771 moves = 0;
772 return;
773 }
774 if( has_effect( effect_stunned ) ) {
775 stumble();
776 moves = 0;
777 return;
778 }
779 if( friendly > 0 ) {
780 --friendly;
781 }
782 if( has_effect( effect_ai_waiting ) ) {
783 moves = 0;
784 return;
785 }
786
787 // don't move if a passenger in a moving vehicle
788 auto vp = g->m.veh_at( pos() );
789 bool harness_part = static_cast<bool>( g->m.veh_at( pos() ).part_with_feature( "ANIMAL_CTRL",
790 true ) );
791 if( vp && vp->vehicle().is_moving() && vp->vehicle().get_pet( vp->part_index() ) ) {
792 moves = 0;
793 return;
794 // Don't move if harnessed, even if vehicle is stationary
795 } else if( vp && has_effect( effect_harnessed ) ) {
796 moves = 0;
797 return;
798 // If harnessed monster finds itself moved from the harness point, the harness probably broke!
799 } else if( !harness_part && has_effect( effect_harnessed ) ) {
800 remove_effect( effect_harnessed );
801 }
802 // Set attitude to attitude to our current target
803 monster_attitude current_attitude = attitude( nullptr );
804 if( !wander() ) {
805 if( goal == g->u.pos() ) {
806 current_attitude = attitude( &g->u );
807 } else {
808 for( const npc &guy : g->all_npcs() ) {
809 if( goal == guy.pos() ) {
810 current_attitude = attitude( &guy );
811 }
812 }
813 }
814 }
815
816 if( current_attitude == MATT_IGNORE ||
817 ( current_attitude == MATT_FOLLOW && rl_dist( pos(), goal ) <= MONSTER_FOLLOW_DIST ) ) {
818 moves -= 100;
819 stumble();
820 return;
821 }
822
823 bool moved = false;
824 tripoint destination;
825
826 bool try_to_move = false;
827 for( const tripoint &dest : g->m.points_in_radius( pos(), 1 ) ) {
828 if( dest != pos() ) {
829 if( can_move_to( dest ) && can_squeeze_to( dest ) &&
830 g->critter_at( dest, true ) == nullptr ) {
831 try_to_move = true;
832 break;
833 }
834 }
835 }
836 // If true, don't try to greedily avoid locally bad paths
837 bool pathed = false;
838 if( try_to_move ) {
839 if( !wander() ) {
840 while( !path.empty() && path.front() == pos() ) {
841 path.erase( path.begin() );
842 }
843
844 const auto &pf_settings = get_pathfinding_settings();
845 if( pf_settings.max_dist >= rl_dist( pos(), goal ) &&
846 ( path.empty() || rl_dist( pos(), path.front() ) >= 2 || path.back() != goal ) ) {
847 // We need a new path
848 path = g->m.route( pos(), goal, pf_settings, get_path_avoid() );
849 }
850
851 // Try to respect old paths, even if we can't pathfind at the moment
852 if( !path.empty() && path.back() == goal ) {
853 destination = path.front();
854 moved = true;
855 pathed = true;
856 } else {
857 // Straight line forward, probably because we can't pathfind (well enough)
858 destination = goal;
859 moved = true;
860 }
861 }
862 }
863 if( !moved && has_flag( MF_SMELLS ) ) {
864 // No sight... or our plans are invalid (e.g. moving through a transparent, but
865 // solid, square of terrain). Fall back to smell if we have it.
866 unset_dest();
867 tripoint tmp = scent_move();
868 if( tmp.x != -1 ) {
869 destination = tmp;
870 moved = true;
871 }
872 }
873 if( wandf > 0 && !moved && friendly == 0 ) { // No LOS, no scent, so as a fall-back follow sound
874 unset_dest();
875 if( wander_pos != pos() ) {
876 destination = wander_pos;
877 moved = true;
878 }
879 }
880
881 if( !g->m.has_zlevels() ) {
882 // Otherwise weird things happen
883 destination.z = posz();
884 }
885
886 point new_d( destination.xy() - pos().xy() );
887
888 // toggle facing direction for sdl flip
889 if( !tile_iso ) {
890 if( new_d.x < 0 ) {
891 facing = FD_LEFT;
892 } else if( new_d.x > 0 ) {
893 facing = FD_RIGHT;
894 }
895 } else {
896 if( new_d.y <= 0 && new_d.x <= 0 ) {
897 facing = FD_LEFT;
898 }
899 if( new_d.x >= 0 && new_d.y >= 0 ) {
900 facing = FD_RIGHT;
901 }
902 }
903
904 tripoint next_step;
905 const bool can_open_doors = has_flag( MF_CAN_OPEN_DOORS );
906 const bool staggers = has_flag( MF_STUMBLES );
907 if( moved ) {
908 // Implement both avoiding obstacles and staggering.
909 moved = false;
910 float switch_chance = 0.0;
911 const bool can_bash = bash_skill() > 0;
912 // This is a float and using trig_dist() because that Does the Right Thing(tm)
913 // in both circular and roguelike distance modes.
914 const float distance_to_target = trig_dist( pos(), destination );
915 for( tripoint &candidate : squares_closer_to( pos(), destination ) ) {
916 bool via_ramp = false;
917 if( here.has_flag( TFLAG_RAMP_UP, candidate ) ) {
918 via_ramp = true;
919 candidate.z += 1;
920 } else if( here.has_flag( TFLAG_RAMP_DOWN, candidate ) ) {
921 via_ramp = true;
922 candidate.z -= 1;
923 }
924 tripoint candidate_abs = g->m.getabs( candidate );
925
926 if( candidate.z != posz() ) {
927 bool can_z_move = true;
928 if( !here.valid_move( pos(), candidate, false, true, via_ramp ) ) {
929 // Can't phase through floor
930 can_z_move = false;
931 }
932
933 // If we're trying to go up but can't fly, check if we can climb. If we can't, then don't
934 // This prevents non-climb/fly enemies running up walls
935 if( candidate.z > posz() && !( via_ramp || flies() ) ) {
936 if( !can_climb() || !here.has_floor_or_support( candidate ) ) {
937 // Can't "jump" up a whole z-level
938 can_z_move = false;
939 }
940 }
941
942 // Last chance - we can still do the z-level stair teleport bullshit that isn't removed yet
943 // TODO: Remove z-level stair bullshit teleport after aligning all stairs
944 if( !can_z_move &&
945 posx() / ( SEEX * 2 ) == candidate.x / ( SEEX * 2 ) &&
946 posy() / ( SEEY * 2 ) == candidate.y / ( SEEY * 2 ) ) {
947 const tripoint &upper = candidate.z > posz() ? candidate : pos();
948 const tripoint &lower = candidate.z > posz() ? pos() : candidate;
949 if( g->m.has_flag( TFLAG_GOES_DOWN, upper ) && g->m.has_flag( TFLAG_GOES_UP, lower ) ) {
950 can_z_move = true;
951 }
952 }
953
954 if( !can_z_move ) {
955 continue;
956 }
957 }
958
959 // A flag to allow non-stumbling critters to stumble when the most direct choice is bad.
960 bool bad_choice = false;
961
962 const Creature *target = g->critter_at( candidate, is_hallucination() );
963 if( target != nullptr ) {
964 const Creature::Attitude att = attitude_to( *target );
965 if( att == A_HOSTILE ) {
966 // When attacking an adjacent enemy, we're direct.
967 moved = true;
968 next_step = candidate_abs;
969 break;
970 } else if( att == A_FRIENDLY && ( target->is_player() || target->is_npc() ) ) {
971 continue; // Friendly firing the player or an NPC is illegal for gameplay reasons
972 } else if( !has_flag( MF_ATTACKMON ) && !has_flag( MF_PUSH_MON ) ) {
973 // Bail out if there's a non-hostile monster in the way and we're not pushy.
974 continue;
975 }
976 // Friendly fire and pushing are always bad choices - they take a lot of time
977 bad_choice = true;
978 }
979
980 map &here = g->m;
981 // is there an openable door?
982 if( can_open_doors &&
983 here.open_door( candidate, !here.is_outside( pos() ), true ) ) {
984 moved = true;
985 next_step = candidate_abs;
986 continue;
987 }
988
989 // Try to shove vehicle out of the way
990 shove_vehicle( destination, candidate );
991 // Bail out if we can't move there and we can't bash.
992 if( !pathed && ( !can_move_to( candidate ) || !can_squeeze_to( candidate ) ) ) {
993 if( !can_bash ) {
994 continue;
995 }
996 // Don't bash if we're just tracking a noise.
997 if( wander() && destination == wander_pos ) {
998 continue;
999 }
1000 const int estimate = here.bash_rating( bash_estimate(), candidate );
1001 if( estimate <= 0 ) {
1002 continue;
1003 }
1004
1005 if( estimate < 5 ) {
1006 bad_choice = true;
1007 }
1008 }
1009
1010 const float progress = distance_to_target - trig_dist( candidate, destination );
1011 // The x2 makes the first (and most direct) path twice as likely,
1012 // since the chance of switching is 1/1, 1/4, 1/6, 1/8
1013 switch_chance += progress * 2;
1014 // Randomly pick one of the viable squares to move to weighted by distance.
1015 if( progress > 0 && ( !moved || x_in_y( progress, switch_chance ) ) ) {
1016 moved = true;
1017 next_step = candidate_abs;
1018 // If we stumble, pick a random square, otherwise take the first one,
1019 // which is the most direct path.
1020 // Except if the direct path is bad, then check others
1021 // Or if the path is given by pathfinder
1022 if( !staggers && ( !bad_choice || pathed ) ) {
1023 break;
1024 }
1025 }
1026 }
1027 }
1028 // Finished logic section. By this point, we should have chosen a square to
1029 // move to (moved = true).
1030 const tripoint local_next_step = g->m.getlocal( next_step );
1031 if( moved ) { // Actual effects of moving to the square we've chosen
1032 const bool did_something =
1033 ( !pacified && attack_at( local_next_step ) ) ||
1034 ( !pacified && can_open_doors && g->m.open_door( local_next_step, !g->m.is_outside( pos() ) ) ) ||
1035 ( !pacified && bash_at( local_next_step ) ) ||
1036 ( !pacified && push_to( local_next_step, 0, 0 ) ) ||
1037 move_to( local_next_step, false, false, get_stagger_adjust( pos(), destination, local_next_step ) );
1038
1039 if( !did_something ) {
1040 moves -= 100; // If we don't do this, we'll get infinite loops.
1041 }
1042 if( has_effect( effect_dragging ) && dragged_foe != nullptr ) {
1043
1044 if( !dragged_foe->has_effect( effect_grabbed ) ) {
1045 dragged_foe = nullptr;
1046 remove_effect( effect_dragging );
1047 } else if( g->m.getlocal( drag_to ) != pos() &&
1048 g->critter_at( g->m.getlocal( drag_to ) ) == nullptr ) {
1049 dragged_foe->setpos( g->m.getlocal( drag_to ) );
1050 }
1051 }
1052 } else {
1053 moves -= 100;
1054 stumble();
1055 path.clear();
1056 }
1057
1058 if( has_effect( effect_led_by_leash ) ) {
1059 if( rl_dist( pos(), g->u.pos() ) > 8 ) {
1060 // Either failed to keep up with the player or moved away
1061 remove_effect( effect_led_by_leash );
1062 add_msg( m_info, _( "You lose hold of a leash." ) );
1063 }
1064 }
1065}

1124

1187tripoint monster::scent_move()
1188{
1189 // TODO: Remove when scentmap is 3D
1190 if( std::abs( posz() - g->get_levz() ) > SCENT_MAP_Z_REACH ) {
1191 return { -1, -1, INT_MIN };
1192 }
1193
1194 const std::set<scenttype_id> &tracked_scents = type->scents_tracked;
1195 const std::set<scenttype_id> &ignored_scents = type->scents_ignored;
1196
1197 std::vector<tripoint> smoves;
1198
1199 int bestsmell = 10; // Squares with smell 0 are not eligible targets.
1200 int smell_threshold = 200; // Squares at or above this level are ineligible.
1201 if( has_flag( MF_KEENNOSE ) ) {
1202 bestsmell = 1;
1203 smell_threshold = 400;
1204 }
1205
1206 const bool fleeing = is_fleeing( g->u );
1207 if( fleeing ) {
1208 bestsmell = g->scent.get( pos() );
1209 }
1210
1211 tripoint next( -1, -1, posz() );
1212 if( ( !fleeing && g->scent.get( pos() ) > smell_threshold ) ||
1213 ( fleeing && bestsmell == 0 ) ) {
1214 return next;
1215 }
1216 const bool can_bash = bash_skill() > 0;
1217 for( const auto &dest : g->m.points_in_radius( pos(), 1, SCENT_MAP_Z_REACH ) ) {
1218 int smell = g->scent.get( dest );
1219 const scenttype_id &type_scent = g->scent.get_type( dest );
1220
1221 bool right_scent = false;
1222 // is the monster tracking this scent
1223 if( !tracked_scents.empty() ) {
1224 right_scent = tracked_scents.find( type_scent ) != tracked_scents.end();
1225 }
1226 //is this scent recognised by the monster species
1227 if( !type_scent.is_empty() ) {
1228 const std::set<species_id> &receptive_species = type_scent->receptive_species;
1229 const std::set<species_id> &monster_species = type->species;
1230 std::vector<species_id> v_intersection;
1231 std::set_intersection( receptive_species.begin(), receptive_species.end(), monster_species.begin(),
1232 monster_species.end(), std::back_inserter( v_intersection ) );
1233 if( !v_intersection.empty() ) {
1234 right_scent = true;
1235 }
1236 }
1237 // is the monster actually ignoring this scent
1238 if( !ignored_scents.empty() && ( ignored_scents.find( type_scent ) != ignored_scents.end() ) ) {
1239 right_scent = false;
1240 }
1241
1242 if( ( !fleeing && smell < bestsmell ) || ( fleeing && smell > bestsmell ) || !right_scent ) {
1243 continue;
1244 }
1245 if( g->m.valid_move( pos(), dest, can_bash, true ) &&
1246 ( ( can_move_to( dest ) && !get_map().obstructed_by_vehicle_rotation( pos(), dest ) ) ||
1247 ( dest == g->u.pos() ) ||
1248 ( can_bash && g->m.bash_rating( bash_estimate(), dest ) > 0 ) ) ) {
1249 if( ( !fleeing && smell > bestsmell ) || ( fleeing && smell < bestsmell ) ) {
1250 smoves.clear();
1251 smoves.push_back( dest );
1252 bestsmell = smell;
1253 } else if( ( !fleeing && smell == bestsmell ) || ( fleeing && smell == bestsmell ) ) {
1254 smoves.push_back( dest );
1255 }
1256 }
1257 }
1258
1259 return random_entry( smoves, next );
1260}

1262int monster::calc_movecost( const tripoint &f, const tripoint &t ) const
1263{
1264 int movecost = 0;
1265
1266 const int source_cost = g->m.move_cost( f );
1267 const int dest_cost = g->m.move_cost( t );
1268 // Digging and flying monsters ignore terrain cost
1269 if( flies() || ( digging() && g->m.has_flag( "DIGGABLE", t ) ) ) {
1270 movecost = 100;
1271 // Swimming monsters move super fast in water
1272 } else if( swims() ) {
1273 if( g->m.has_flag( "SWIMMABLE", f ) ) {
1274 movecost += 25;
1275 } else {
1276 movecost += 50 * g->m.move_cost( f );
1277 }
1278 if( g->m.has_flag( "SWIMMABLE", t ) ) {
1279 movecost += 25;
1280 } else {
1281 movecost += 50 * g->m.move_cost( t );
1282 }
1283 } else if( can_submerge() ) {
1284 // No-breathe monsters have to walk underwater slowly
1285 if( g->m.has_flag( "SWIMMABLE", f ) ) {
1286 movecost += 250;
1287 } else {
1288 movecost += 50 * g->m.move_cost( f );
1289 }
1290 if( g->m.has_flag( "SWIMMABLE", t ) ) {
1291 movecost += 250;
1292 } else {
1293 movecost += 50 * g->m.move_cost( t );
1294 }
1295 movecost /= 2;
1296 } else if( climbs() ) {
1297 if( g->m.has_flag( "CLIMBABLE", f ) ) {
1298 movecost += 150;
1299 } else {
1300 movecost += 50 * g->m.move_cost( f );
1301 }
1302 if( g->m.has_flag( "CLIMBABLE", t ) ) {
1303 movecost += 150;
1304 } else {
1305 movecost += 50 * g->m.move_cost( t );
1306 }
1307 movecost /= 2;
1308 } else {
1309 movecost = ( ( 50 * source_cost ) + ( 50 * dest_cost ) ) / 2.0;
1310 }
1311

1331/*
1332 * Return points of an area extending 1 tile to either side and
1333 * (maxdepth) tiles behind basher.
1334 */
1335static std::vector<tripoint> get_bashing_zone( const tripoint &bashee, const tripoint &basher,
1336 int maxdepth )
1337{
1338 std::vector<tripoint> direction;
1339 direction.push_back( bashee );
1340 direction.push_back( basher );
1341 // Draw a line from the target through the attacker.
1342 std::vector<tripoint> path = continue_line( direction, maxdepth );
1343 // Remove the target.
1344 path.insert( path.begin(), basher );
1345 std::vector<tripoint> zone;
1346 // Go ahead and reserve enough room for all the points since
1347 // we know how many it will be.
1348 zone.reserve( 3 * maxdepth );
1349 tripoint previous = bashee;
1350 for( const tripoint &p : path ) {
1351 std::vector<point> swath = squares_in_direction( previous.xy(), p.xy() );
1352 for( const point &q : swath ) {
1353 zone.push_back( tripoint( q, bashee.z ) );
1354 }
1355
1356 previous = p;
1357 }
1358 return zone;
1359}

1420int monster::group_bash_skill( const tripoint &target )
1421{
1422 if( !has_flag( MF_GROUP_BASH ) ) {
1423 return bash_skill();
1424 }
1425 int bashskill = 0;
1426
1427 // pileup = more bash skill, but only help bashing mob directly in front of target
1428 const int max_helper_depth = 5;
1429 const std::vector<tripoint> bzone = get_bashing_zone( target, pos(), max_helper_depth );
1430
1431 for( const tripoint &candidate : bzone ) {
1432 // Drawing this line backwards excludes the target and includes the candidate.
1433 std::vector<tripoint> path_to_target = line_to( target, candidate, 0, 0 );
1434 bool connected = true;
1435 monster *mon = nullptr;
1436 for( const tripoint &in_path : path_to_target ) {
1437 // If any point in the line from zombie to target is not a cooperating zombie,
1438 // it can't contribute.
1439 mon = g->critter_at<monster>( in_path );
1440 if( !mon ) {
1441 connected = false;
1442 break;
1443 }
1444 monster &helpermon = *mon;
1445 if( !helpermon.has_flag( MF_GROUP_BASH ) || helpermon.is_hallucination() ) {
1446 connected = false;
1447 break;
1448 }
1449 }
1450 if( !connected || !mon ) {
1451 continue;
1452 }
1453 // If we made it here, the last monster checked was the candidate.
1454 monster &helpermon = *mon;
1455 // Contribution falls off rapidly with distance from target.
1456 bashskill += helpermon.bash_skill() / rl_dist( candidate, target );
1457 }
1458
1459 return bashskill;
1460}

1533 const bool going_up = p.z > pos().z;

1609 // Nothing, no need to spam

1736bool monster::push_to( const tripoint &p, const int boost, const size_t depth )
1737{
1738 if( is_hallucination() ) {
1739 // Don't let hallucinations push, not even other hallucinations
1740 return false;
1741 }
1742
1743 if( !has_flag( MF_PUSH_MON ) || depth > 2 || has_effect( effect_pushed ) ) {
1744 return false;
1745 }
1746
1747 // TODO: Generalize this to Creature
1748 monster *const critter = g->critter_at<monster>( p );
1749 if( critter == nullptr || critter == this ||
1750 p == pos() || critter->movement_impaired() ) {
1751 return false;
1752 }
1753
1754 if( !can_move_to( p ) ) {
1755 return false;
1756 }
1757
1758 if( critter->is_hallucination() ) {
1759 // Kill the hallu, but return false so that the regular move_to is uses instead
1760 critter->die( nullptr );
1761 return false;
1762 }
1763
1764 // Stability roll of the pushed critter
1765 const int defend = critter->stability_roll();
1766 // Stability roll of the pushing zed
1767 const int attack = stability_roll() + boost;
1768 if( defend > attack ) {
1769 return false;
1770 }
1771
1772 const int movecost_from = 50 * g->m.move_cost( p );
1773 const int movecost_attacker = std::max( movecost_from, 200 - 10 * ( attack - defend ) );
1774 const tripoint dir = p - pos();
1775
1776 // Mark self as pushed to simplify recursive pushing
1777 add_effect( effect_pushed, 1_turns );
1778
1779 for( size_t i = 0; i < 6; i++ ) {
1780 const point d{ rng( -1, 1 ), rng( -1, 1 ) };
1781 if( d.x == 0 && d.y == 0 ) {
1782 continue;
1783 }
1784
1785 // Pushing forward is easier than pushing aside
1786 const int direction_penalty = std::abs( d.x - dir.x ) + std::abs( d.y - dir.y );
1787 if( direction_penalty > 2 ) {
1788 continue;
1789 }
1790
1791 tripoint dest( p + d );
1792 const int dest_movecost_from = 50 * g->m.move_cost( dest );
1793
1794 // Pushing into cars/windows etc. is harder
1795 const int movecost_penalty = g->m.move_cost( dest ) - 2;
1796 if( movecost_penalty <= -2 || get_map().obstructed_by_vehicle_rotation( p, dest ) ) {
1797 // Can't push into unpassable terrain
1798 continue;
1799 }
1800
1801 int roll = attack - ( defend + direction_penalty + movecost_penalty );
1802 if( roll < 0 ) {
1803 continue;
1804 }
1805
1806 Creature *critter_recur = g->critter_at( dest );
1807 if( !( critter_recur == nullptr || critter_recur->is_hallucination() ) ) {
1808 // Try to push recursively
1809 monster *mon_recur = dynamic_cast< monster * >( critter_recur );
1810 if( mon_recur == nullptr ) {
1811 continue;
1812 }
1813
1814 if( critter->push_to( dest, roll, depth + 1 ) ) {
1815 // The tile isn't necessarily free, need to check
1816 if( !g->critter_at( p ) ) {
1817 move_to( p );
1818 }
1819
1820 moves -= movecost_attacker;
1821
1822 // Don't knock down a creature that successfully
1823 // pushed another creature, just reduce moves
1824 critter->moves -= dest_movecost_from;
1825 return true;
1826 } else {
1827 return false;
1828 }
1829 }
1830
1831 critter_recur = g->critter_at( dest );
1832 if( critter_recur != nullptr ) {
1833 if( critter_recur->is_hallucination() ) {
1834 critter_recur->die( nullptr );
1835 }
1836 } else if( !critter->has_flag( MF_IMMOBILE ) ) {
1837 critter->setpos( dest );
1838 move_to( p );
1839 moves -= movecost_attacker;
1840 critter->add_effect( effect_downed, time_duration::from_turns( movecost_from / 100 + 1 ) );
1841 }
1842 return true;
1843 }
1844
1845 // Try to trample over a much weaker zed (or one with worse rolls)
1846 // Don't allow trampling with boost
1847 if( boost > 0 || attack < 2 * defend ) {
1848 return false;
1849 }
1850
1851 g->swap_critters( *critter, *this );
1852 critter->add_effect( effect_stunned, rng( 0_turns, 2_turns ) );
1853 // Only print the message when near player or it can get spammy
1854 if( rl_dist( g->u.pos(), pos() ) < 4 && g->u.sees( *critter ) ) {
1855 add_msg( m_warning, _( "The %1$s tramples %2$s" ),
1856 name(), critter->disp_name() );
1857 }
1858
1859 moves -= movecost_attacker;
1860 if( movecost_from > 100 ) {
1861 critter->add_effect( effect_downed, time_duration::from_turns( movecost_from / 100 + 1 ) );
1862 } else {
1863 critter->moves -= movecost_from;
1864 }
1865
1866 return true;
1867}
