902void talk_function::start_training( npc &p )
903{
904 int cost;
905 time_duration time = 0_turns;
906 std::string name;
907 const skill_id &skill = p.chatbin.skill;
908 const matype_id &style = p.chatbin.style;
909 const spell_id &sp_id = p.chatbin.dialogue_spell;
910 int expert_multiplier = 1;
911 if( skill.is_valid() && g->u.get_skill_level( skill ) < p.get_skill_level( skill ) ) {
912 cost = calc_skill_training_cost( p, skill );
913 time = calc_skill_training_time( p, skill );
914 name = skill.str();
915 } else if( p.chatbin.style.is_valid() && !g->u.martial_arts_data->has_martialart( style ) ) {
916 cost = calc_ma_style_training_cost( p, style );
917 time = calc_ma_style_training_time( p, style );
918 name = p.chatbin.style.str();
919 // already checked if can learn this spell in npctalk.cpp
920 } else if( p.chatbin.dialogue_spell.is_valid() ) {
921 const spell &temp_spell = p.magic->get_spell( sp_id );
922 const bool knows = g->u.magic->knows_spell( sp_id );
923 cost = p.calc_spell_training_cost( knows, temp_spell.get_difficulty(), temp_spell.get_level() );
924 name = temp_spell.id().str();
925 expert_multiplier = knows ? temp_spell.get_level() - g->u.magic->get_spell( sp_id ).get_level() : 1;
926 // quicker to learn with instruction as opposed to books.
927 // if this is a known spell, then there is a set time to gain some exp.
928 // if player doesn't know this spell, then the NPC will teach all of it
929 // which takes max 6 hours, min 3 hours.
930 // TODO: a system for NPCs to train new stuff in bits and pieces
931 // and remember the progress.
932 if( knows ) {
933 time = 1_hours;
934 } else {
935 int seconds = g->u.magic->time_to_learn_spell( g->u, sp_id ) / 50;
936 time = time_duration::from_seconds( clamp( seconds, 7200, 21600 ) );
937 }
938 } else {
939 debugmsg( "start_training with no valid skill or style set" );
940 return;
941 }
942
943 mission *miss = p.chatbin.mission_selected;
944 if( miss != nullptr && miss->get_assigned_player_id() == g->u.getID() &&
945 miss->is_complete( g->u.getID() ) ) {
946 clear_mission( p );
947 } else if( !npc_trading::pay_npc( p, cost ) ) {
948 return;
949 }
950 player_activity act = player_activity( ACT_TRAIN, to_moves<int>( time ),
951 p.getID().get_value(), 0, name );
952 act.values.push_back( expert_multiplier );
953 g->u.assign_activity( act );
954
955 p.add_effect( effect_asked_to_train, 6_hours );
956}
