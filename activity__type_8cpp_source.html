64void activity_type::check_consistency()
65{
66 for( const auto &pair : activity_type_all ) {
67 if( pair.second.verb_.empty() ) {
68 debugmsg( "%s doesn't have a verb", pair.first.c_str() );
69 }
70 const bool has_actor = activity_actors::deserialize_functions.find( pair.second.id_ ) !=
71 activity_actors::deserialize_functions.end();
72 const bool has_turn_func = activity_handlers::do_turn_functions.find( pair.second.id_ ) !=
73 activity_handlers::do_turn_functions.end();
74
75 if( pair.second.based_on_ == based_on_type::NEITHER && !( has_turn_func || has_actor ) ) {
76 debugmsg( "%s needs a do_turn function or activity actor if it's not based on time or speed.",
77 pair.second.id_.c_str() );
