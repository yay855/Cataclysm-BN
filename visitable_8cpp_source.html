111/** Sums the two terms, being careful to not trigger overflow.
112 * Doesn't handle underflow.
113 *
114 * @param a The first addend.
115 * @param b The second addend.
116 * @return the sum of the addends, but truncated to std::numeric_limits<int>::max().
117 */
118template <typename T>
119static T sum_no_wrap( T a, T b )
120{
121 if( a > std::numeric_limits<T>::max() - b ||
122 b > std::numeric_limits<T>::max() - a ) {
123 return std::numeric_limits<T>::max();
124 }
125 return a + b;
126}

380static VisitResponse visit_internal( const std::function<VisitResponse( item *, item * )> &func,
381 item *node, item *parent = nullptr )
382{
383 switch( func( node, parent ) ) {
384 case VisitResponse::ABORT:
385 return VisitResponse::ABORT;
386
387 case VisitResponse::NEXT:
388 if( node->is_gun() || node->is_magazine() ) {
389 // Content of guns and magazines are accessible only via their specific accessors
390 return VisitResponse::NEXT;
391 }
392
393 if( node->contents.visit_contents( func, node ) == VisitResponse::ABORT ) {
394 return VisitResponse::ABORT;
395 }
396 /* intentional fallthrough */
397
398 case VisitResponse::SKIP:
399 return VisitResponse::NEXT;
400 }
401
402 /* never reached but suppresses GCC warning */
403 return VisitResponse::ABORT;
404}

611/** @relates visitable */
612template <>
613std::list<item> visitable<inventory>::remove_items_with( const
614 std::function<bool( const item &e )> &filter, int count )
615{
616 auto inv = static_cast<inventory *>( this );
617 std::list<item> res;
618
619 if( count <= 0 ) {
620 // nothing to do
621 return res;
622 }
623
624 for( auto stack = inv->items.begin(); stack != inv->items.end() && count > 0; ) {
625 std::list<item> &istack = *stack;
626 const auto original_invlet = istack.front().invlet;
627
628 for( auto istack_iter = istack.begin(); istack_iter != istack.end() && count > 0; ) {
629 if( filter( *istack_iter ) ) {
630 count--;
631 res.splice( res.end(), istack, istack_iter++ );
632 // The non-first items of a stack may have different invlets, the code
633 // in inventory only ever checks the invlet of the first item. This
634 // ensures that the first item of a stack always has the same invlet, even
635 // after the original first item was removed.
636 if( istack_iter == istack.begin() && istack_iter != istack.end() ) {
637 istack_iter->invlet = original_invlet;
638 }
639
640 } else {
641 istack_iter->contents.remove_internal( filter, count, res );
642 ++istack_iter;
643 }
644 }
645
646 if( istack.empty() ) {
647 stack = inv->items.erase( stack );
648 } else {
649 ++stack;
650 }
651 }
652
653 // Invalidate binning cache
654 inv->binned = false;
655 inv->items_type_cached = false;
656
657 return res;
658}

661template <>
662std::list<item> visitable<Character>::remove_items_with( const
663 std::function<bool( const item &e )> &filter, int count )
664{
665 auto ch = static_cast<Character *>( this );
666 std::list<item> res;
667
668 if( count <= 0 ) {
669 // nothing to do
670 return res;
671 }
672
673 // first try and remove items from the inventory
674 res = ch->inv.remove_items_with( filter, count );
675 count -= res.size();
676 if( count == 0 ) {
677 return res;
678 }
679
680 // then try any worn items
681 for( auto iter = ch->worn.begin(); iter != ch->worn.end(); ) {
682 if( filter( *iter ) ) {
683 iter->on_takeoff( *ch );
684 res.splice( res.end(), ch->worn, iter++ );
685 if( --count == 0 ) {
686 return res;
687 }
688 } else {
689 iter->contents.remove_internal( filter, count, res );
690 if( count == 0 ) {

694 }
695 }
696
697 // finally try the currently wielded item (if any)
698 if( filter( ch->weapon ) ) {
699 res.push_back( ch->remove_weapon() );
700 count--;
701 } else {
702 ch->weapon.contents.remove_internal( filter, count, res );
703 }
704

808 return res;

811 }

815 // if we removed any items then invalidate the cached mass
816 cur->veh.invalidate_mass();

899template <typename T, typename M>
900static int charges_of_internal( const T &self, const M &main, const itype_id &id, int limit,
901 const std::function<bool( const item & )> &filter,
902 std::function<void( int )> visitor )
903{
904 int qty = 0;
905
906 bool found_tool_with_UPS = false;
907 self.visit_items( [&]( const item * e ) {
908 if( filter( *e ) ) {
909 if( e->is_tool() ) {
910 if( e->typeId() == id ) {
911 // includes charges from any included magazine.
912 qty = sum_no_wrap( qty, e->ammo_remaining() );
913 if( e->has_flag( "USE_UPS" ) ) {
914 found_tool_with_UPS = true;
915 }
916 }
917 return qty < limit ? VisitResponse::SKIP : VisitResponse::ABORT;
918
919 } else if( e->count_by_charges() ) {
920 if( e->typeId() == id ) {
921 qty = sum_no_wrap( qty, e->charges );
922 }
923 // items counted by charges are not themselves expected to be containers
924 return qty < limit ? VisitResponse::SKIP : VisitResponse::ABORT;
925 }
926 }
927 // recurse through any nested containers
928 return qty < limit ? VisitResponse::NEXT : VisitResponse::ABORT;
929 } );
930
931 if( qty < limit && found_tool_with_UPS ) {
932 qty += main.charges_of( itype_UPS, limit - qty );
933 if( visitor ) {
934 visitor( qty );
935 }
936 }
937
938 return std::min( qty, limit );
939}
