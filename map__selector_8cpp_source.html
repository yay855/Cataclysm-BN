39cata::optional<tripoint> random_point( const tripoint_range<tripoint> &range,
40 const std::function<bool( const tripoint & )> &predicate )
41{
42 // Optimist approach: just assume there are plenty of suitable places and a randomly
43 // chosen point will have a good chance to hit one of them.
44 // If there are only few suitable places, we have to find them all, otherwise this loop may never finish.
45 for( int tries = 0; tries < 10; ++tries ) {
46 const tripoint p( rng( range.min().x, range.max().x ), rng( range.min().y, range.max().y ),
47 rng( range.min().z, range.max().z ) );
48 if( predicate( p ) ) {
49 return p;
50 }
51 }
52 std::vector<tripoint> suitable;
53 for( const auto &p : range ) {
54 if( predicate( p ) ) {
55 suitable.push_back( p );
56 }
57 }
58 if( suitable.empty() ) {

60 }
