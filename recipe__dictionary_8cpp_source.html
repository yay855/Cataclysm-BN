158std::vector<const recipe *> recipe_subset::search( const std::string &txt,
159 const search_type key ) const
160{
161 std::vector<const recipe *> res;
162
163 std::copy_if( recipes.begin(), recipes.end(), std::back_inserter( res ), [&]( const recipe * r ) {
164 if( !*r || r->obsolete ) {
165 return false;
166 }
167 switch( key ) {
168 case search_type::name:
169 return lcmatch( r->result_name(), txt );
170
171 case search_type::skill:
172 return lcmatch( r->required_skills_string( nullptr, true, false ), txt );
173
174 case search_type::primary_skill:
175 return lcmatch( r->skill_used->name(), txt );
176
177 case search_type::component:
178 return search_reqs( r->simple_requirements().get_components(), txt );
179
180 case search_type::tool:
181 return search_reqs( r->simple_requirements().get_tools(), txt );
182
183 case search_type::quality:
184 return search_reqs( r->simple_requirements().get_qualities(), txt );
185
186 case search_type::quality_result: {
187 const auto &quals = r->result()->qualities;
188 return std::any_of( quals.begin(), quals.end(), [&]( const std::pair<quality_id, int> &e ) {
189 return lcmatch( e.first->name, txt );
190 } );
191 }
192
193 case search_type::description_result: {
194 const item result = r->create_result();
195 return lcmatch( remove_color_tags( result.info( true ) ), txt );
196 }

199 return false;

378void recipe_dictionary::find_items_on_loops()
379{
380 // Check for infinite recipe loops in food (which are problematic for
381 // nutrient calculations).
382 //
383 // Start by building a directed graph of itypes to potential components of
384 // those itypes.
385 items_on_loops.clear();
386 std::unordered_map<itype_id, std::vector<itype_id>> potential_components_of;
387 for( const itype *i : item_controller->all() ) {
388 if( !i->comestible || i->has_flag( "NUTRIENT_OVERRIDE" ) ) {
389 continue;
390 }
391 std::vector<itype_id> &potential_components = potential_components_of[i->get_id()];
392 for( const recipe_id &rec : i->recipes ) {
393 const requirement_data requirements = rec->simple_requirements();
394 const requirement_data::alter_item_comp_vector &component_requirements =
395 requirements.get_components();
396
397 for( const std::vector<item_comp> &component_options : component_requirements ) {
398 for( const item_comp &component_option : component_options ) {
399 potential_components.push_back( component_option.type );
400 }
401 }
402 }
403 }
404
405 // Now check that graph for loops
406 std::vector<std::vector<itype_id>> loops = cata::find_cycles( potential_components_of );
407 for( const std::vector<itype_id> &loop : loops ) {
408 std::string error_message =
409 "loop in comestible recipes detected: " + loop.back().str();
410 for( const itype_id &i : loop ) {
411 error_message += " -> " + i.str();
412 items_on_loops.insert( i );
413 }
414 error_message += ". Such loops can be broken by either removing or altering "
415 "recipes or marking one of the items involved with the NUTRIENT_OVERRIDE "
416 "flag";
417 debugmsg( error_message );
418 }

421void recipe_dictionary::finalize()
422{
423 DynamicDataLoader::get_instance().load_deferred( deferred );
424
425 // remove abstract recipes
426 delete_if( []( const recipe & element ) {
427 return element.abstract;
428 } );
429
430 finalize_internal( recipe_dict.recipes );
431 finalize_internal( recipe_dict.uncraft );
432
433 for( auto &e : recipe_dict.recipes ) {
434 auto &r = e.second;
435
436 if( r.obsolete ) {
437 continue;
438 }
439
440 for( const auto &bk : r.booksets ) {
441 const itype *booktype = &*bk.first;
442 int req = bk.second > 0 ? bk.second : std::max( booktype->book->req, r.difficulty );
443 islot_book::recipe_with_description_t desc{ &r, req, r.result_name(), false };
444 const_cast<islot_book &>( *booktype->book ).recipes.insert( desc );
445 }
446
447 // if reversible and no specific uncraft recipe exists use this recipe
448 if( r.is_reversible() && !recipe_dict.uncraft.count( recipe_id( r.result().str() ) ) ) {
449 recipe_dict.uncraft[ recipe_id( r.result().str() ) ] = r;
450 }
451 }
452
453 // add pseudo uncrafting recipes
454 for( const itype *e : item_controller->all() ) {
455 const itype_id id = e->get_id();
456 const recipe_id rid = recipe_id( id.str() );
457
458 // books that don't already have an uncrafting recipe
459 if( e->book && !recipe_dict.uncraft.count( rid ) && e->volume > 0_ml ) {
460 int pages = e->volume / 12.5_ml;
461 auto &bk = recipe_dict.uncraft[rid];
462 bk.ident_ = rid;
463 bk.result_ = id;
464 bk.reversible = true;
465 bk.requirements_ = *requirement_id( "uncraft_book" ) * pages;
466 // TODO: allow specifying time in requirement_data
467 bk.time = pages * 10;
468 }
469 }
470
471 // Cache auto-learn recipes and blueprints
472 for( const auto &e : recipe_dict.recipes ) {
473 if( e.second.autolearn ) {
474 recipe_dict.autolearn.insert( &e.second );
475 }
476 if( e.second.is_blueprint() ) {
477 recipe_dict.blueprints.insert( &e.second );
478 }
479 }
480
481 recipe_dict.find_items_on_loops();
482}

500void recipe_subset::include( const recipe *r, int custom_difficulty )
501{
502 if( custom_difficulty < 0 ) {
503 custom_difficulty = r->difficulty;
504 }
505 // We always prefer lower difficulty for the subset, but we save it only if it's not default
506 if( recipes.count( r ) > 0 ) {
507 const auto iter = difficulties.find( r );
508 // See if we need to lower the difficulty of the existing recipe
509 if( iter != difficulties.end() && custom_difficulty < iter->second ) {
510 if( custom_difficulty != r->difficulty ) {
511 iter->second = custom_difficulty; // Added again with lower difficulty
512 } else {
513 difficulties.erase( iter ); // No need to keep the default difficulty. Free some memory
514 }
515 } else if( custom_difficulty < r->difficulty ) {
516 difficulties[r] = custom_difficulty; // Added again with lower difficulty
517 }
518 } else {
519 // add recipe to category and component caches
520 for( const auto &opts : r->simple_requirements().get_components() ) {
521 for( const item_comp &comp : opts ) {
522 component[comp.type].insert( r );
523 }
524 }
525 category[r->category].insert( r );
526 // Set the difficulty is it's not the default
527 if( custom_difficulty != r->difficulty ) {
528 difficulties[r] = custom_difficulty;
529 }
530 // insert the recipe
531 recipes.insert( r );
532 ids.insert( r->ident() );
533 }
534}
