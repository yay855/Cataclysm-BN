134shared_ptr_fast<std::istream> DynamicDataLoader::get_cached_stream( const std::string &path )
135{
136 assert( !finalized && "Cannot open data file after finalization." );
137 assert( stream_cache && "Stream cache is only available during finalization" );
138 shared_ptr_fast<std::istringstream> cached = stream_cache->cache.get( path, nullptr );
139 // Create a new stream if the file is not opened yet, or if some code is still
140 // using the previous stream (in such case, `cached` and `stream_cache` have
141 // two references to the stream, hence the test for > 2).
142 if( !cached ) {
143 cached = make_shared_fast<std::istringstream>( read_entire_file( path ) );
144 } else if( cached.use_count() > 2 ) {
145 cached = make_shared_fast<std::istringstream>( cached->str() );
146 }
147 stream_cache->cache.insert( 8, path, cached );
148 return cached;

151void DynamicDataLoader::load_deferred( deferred_json &data )
152{
153 while( !data.empty() ) {
154 const size_t n = data.size();
155 auto it = data.begin();
156 for( size_t idx = 0; idx != n; ++idx ) {
157 if( !it->first.path ) {
158 debugmsg( "JSON source location has null path, data may load incorrectly" );
159 } else {
160 try {
161 shared_ptr_fast<std::istream> stream = get_cached_stream( *it->first.path );
162 JsonIn jsin( *stream, it->first );
163 JsonObject jo = jsin.get_object();
164 load_object( jo, it->second );
165 } catch( const JsonError &err ) {
166 debugmsg( "(json-error)\n%s", err.what() );
167 }
168 }
169 ++it;
170 }
171 data.erase( data.begin(), it );
172 if( data.size() == n ) {
173 for( const auto &elem : data ) {
174 if( !elem.first.path ) {
175 debugmsg( "JSON source location has null path when reporting circular dependency" );
176 } else {
177 try {
178 shared_ptr_fast<std::istream> stream = get_cached_stream( *it->first.path );
179 JsonIn jsin( *stream, elem.first );
180 jsin.error( "JSON contains circular dependency, this object is discarded" );
181 } catch( const JsonError &err ) {
182 debugmsg( "(json-error)\n%s", err.what() );
183 }
184 }
185 }
186 data.clear();
187 return; // made no progress on this cycle so abort
188 }
189 }

454void DynamicDataLoader::load_data_from_path( const std::string &path, const std::string &src,
455 loading_ui &ui )
456{
457 assert( !finalized && "Can't load additional data after finalization. Must be unloaded first." );
458 // We assume that each folder is consistent in itself,
459 // and all the previously loaded folders.
460 // E.g. the core might provide a vpart "frame-x"
461 // the first loaded mode might provide a vehicle that uses that frame
462 // But not the other way round.
463
464 // get a list of all files in the directory
465 str_vec files = get_files_from_path( ".json", path, true, true );
466 if( files.empty() ) {
467 std::ifstream tmp( path.c_str(), std::ios::in );
468 if( tmp ) {
469 // path is actually a file, don't checking the extension,
470 // assume we want to load this file anyway
471 files.push_back( path );
472 }
473 }
474 // iterate over each file
475 for( auto &files_i : files ) {
476 const std::string &file = files_i;
477 // open the file as a stream
478 cata_ifstream infile = std::move( cata_ifstream().mode( cata_ios_mode::binary ).open( file ) );
479 // and stuff it into ram
480 std::istringstream iss(
481 std::string(
482 ( std::istreambuf_iterator<char>( *infile ) ),
483 std::istreambuf_iterator<char>()
484 )
485 );
486 try {
487 // parse it
488 JsonIn jsin( iss, file );
489 load_all_from_json( jsin, src, ui, path, file );
490 } catch( const JsonError &err ) {
491 throw std::runtime_error( err.what() );
492 }
493 }
494}

496void DynamicDataLoader::load_all_from_json( JsonIn &jsin, const std::string &src, loading_ui &,
497 const std::string &base_path, const std::string &full_path )
498{
499 // TEMPORARY until 0.G: Remove single object support for consistency
500 if( jsin.test_object() ) {
501 // find type and dispatch single object
502 JsonObject jo = jsin.get_object();
503 load_object( jo, src, base_path, full_path );
504 jo.finish();
505 // if there's anything else in the file, it's an error.
506 jsin.eat_whitespace();
507 if( jsin.good() ) {
508 jsin.error( string_format( "expected single-object file but found '%c'", jsin.peek() ) );
509 }
510 } else if( jsin.test_array() ) {
511 jsin.start_array();
512 // find type and dispatch each object until array close
513 while( !jsin.end_array() ) {
514 JsonObject jo = jsin.get_object();
515 load_object( jo, src, base_path, full_path );
516 jo.finish();
517 }
518 } else {
519 // not an object or an array?
520 jsin.error( "expected object or array" );
521 }
522}
