141shared_ptr_fast<std::istream> DynamicDataLoader::get_cached_stream( const std::string &path )
142{
143 assert( !finalized && "Cannot open data file after finalization." );
144 assert( stream_cache && "Stream cache is only available during finalization" );
145 shared_ptr_fast<std::istringstream> cached = stream_cache->cache.get( path, nullptr );
146 // Create a new stream if the file is not opened yet, or if some code is still
147 // using the previous stream (in such case, `cached` and `stream_cache` have
148 // two references to the stream, hence the test for > 2).
149 if( !cached ) {
150 cached = make_shared_fast<std::istringstream>( read_entire_file( path ) );
151 } else if( cached.use_count() > 2 ) {
152 cached = make_shared_fast<std::istringstream>( cached->str() );
153 }
154 stream_cache->cache.insert( 8, path, cached );
155 return cached;
156}

158void DynamicDataLoader::load_deferred( deferred_json &data )
159{
160 while( !data.empty() ) {
161 const size_t n = data.size();
162 auto it = data.begin();
163 for( size_t idx = 0; idx != n; ++idx ) {
164 if( !it->first.path ) {
165 debugmsg( "JSON source location has null path, data may load incorrectly" );
166 } else {
167 try {
168 shared_ptr_fast<std::istream> stream = get_cached_stream( *it->first.path );
169 JsonIn jsin( *stream, it->first );
170 JsonObject jo = jsin.get_object();
171 load_object( jo, it->second );
172 } catch( const JsonError &err ) {
173 debugmsg( "(json-error)\n%s", err.what() );
174 }
175 }
176 ++it;
177 inp_mngr.pump_events();
178 }
179 data.erase( data.begin(), it );
180 if( data.size() == n ) {
181 for( const auto &elem : data ) {
182 if( !elem.first.path ) {
183 debugmsg( "JSON source location has null path when reporting circular dependency" );
184 } else {
185 try {
186 shared_ptr_fast<std::istream> stream = get_cached_stream( *it->first.path );
187 JsonIn jsin( *stream, elem.first );
188 jsin.error( "JSON contains circular dependency, this object is discarded" );
189 } catch( const JsonError &err ) {
190 debugmsg( "(json-error)\n%s", err.what() );
191 }
192 }
193 inp_mngr.pump_events();
194 }
195 data.clear();
196 return; // made no progress on this cycle so abort
197 }
198 }

465void DynamicDataLoader::load_data_from_path( const std::string &path, const std::string &src,
466 loading_ui &ui )
467{
468 assert( !finalized && "Can't load additional data after finalization. Must be unloaded first." );
469 // We assume that each folder is consistent in itself,
470 // and all the previously loaded folders.
471 // E.g. the core might provide a vpart "frame-x"
472 // the first loaded mode might provide a vehicle that uses that frame
473 // But not the other way round.
474
475 // get a list of all files in the directory
476 str_vec files = get_files_from_path( ".json", path, true, true );
477 if( files.empty() ) {
478 std::ifstream tmp( path.c_str(), std::ios::in );
479 if( tmp ) {
480 // path is actually a file, don't checking the extension,
481 // assume we want to load this file anyway
482 files.push_back( path );
483 }
484 }
485 // iterate over each file
486 for( auto &files_i : files ) {
487 const std::string &file = files_i;
488 // open the file as a stream
489 cata_ifstream infile = std::move( cata_ifstream().mode( cata_ios_mode::binary ).open( file ) );
490 // and stuff it into ram
491 std::istringstream iss(
492 std::string(
493 ( std::istreambuf_iterator<char>( *infile ) ),
494 std::istreambuf_iterator<char>()
495 )
496 );
497 try {
498 // parse it
499 JsonIn jsin( iss, file );
500 load_all_from_json( jsin, src, ui, path, file );
501 } catch( const JsonError &err ) {
502 throw std::runtime_error( err.what() );
503 }
504 }
505}

507void DynamicDataLoader::load_all_from_json( JsonIn &jsin, const std::string &src, loading_ui &,
508 const std::string &base_path, const std::string &full_path )
509{
510 // TEMPORARY until 0.G: Remove single object support for consistency
511 if( jsin.test_object() ) {
512 // find type and dispatch single object
513 JsonObject jo = jsin.get_object();
514 load_object( jo, src, base_path, full_path );
515 jo.finish();
516 // if there's anything else in the file, it's an error.
517 jsin.eat_whitespace();
518 if( jsin.good() ) {
519 jsin.error( string_format( "expected single-object file but found '%c'", jsin.peek() ) );
520 }
521 } else if( jsin.test_array() ) {
522 jsin.start_array();
523 // find type and dispatch each object until array close
524 while( !jsin.end_array() ) {
525 JsonObject jo = jsin.get_object();
526 load_object( jo, src, base_path, full_path );
527 jo.finish();
528 }
529 } else {
530 // not an object or an array?
531 jsin.error( "expected object or array" );
532 }
533 inp_mngr.pump_events();
534}

845 ui.proceed();
846 }

848 loader.finalize_loaded_data( ui );

872void init::load_world_modfiles( loading_ui &ui, const std::string &artifacts_file )
873{
874 clear_loaded_data();
875
876 mod_management::t_mod_list &mods = world_generator->active_world->active_mod_order;
877
878 // remove any duplicates whilst preserving order (fixes #19385)
879 std::set<mod_id> found;
880 mods.erase( std::remove_if( mods.begin(), mods.end(), [&found]( const mod_id & e ) {
881 if( found.count( e ) ) {
882 return true;
883 } else {
884 found.insert( e );
885 return false;
886 }
887 } ), mods.end() );
888
889 // require at least one core mod (saves before version 6 may implicitly require dda pack)
890 if( std::none_of( mods.begin(), mods.end(), []( const mod_id & e ) {
891 return e->core;
892} ) ) {
893 mods.insert( mods.begin(), mod_management::get_default_core_content_pack() );
894 }
895
896 // TODO: get rid of artifacts
897 load_artifacts( artifacts_file );
898
899 // this code does not care about mod dependencies,
900 // it assumes that those dependencies are static and
901 // are resolved during the creation of the world.
902 // That means world->active_mod_order contains a list
903 // of mods in the correct order.
904 load_and_finalize_packs( ui, _( "Loading files" ), mods );

907bool init::check_mods_for_errors( loading_ui &ui, const std::vector<mod_id> &opts )
908{
909 const dependency_tree &tree = world_generator->get_mod_manager().get_tree();
910
911 // Deduplicated list of mods to check
912 std::set<mod_id> to_check;
913
914 for( const mod_id &id : opts ) {
915 if( !id.is_valid() ) {
916 std::cerr << string_format( "Unknown mod: [%s]\n", id );
917 return false;
918 }
919
920 if( !tree.is_available( id ) ) {
921 std::cerr << string_format(
922 "Missing dependencies: %s %s\n",
923 id, tree.get_node( id )->s_errors()
924 );
925 return false;
926 }
927
928 to_check.emplace( id );
929 }
930
931 // If no specific mods specified check all non-obsolete mods
932 if( to_check.empty() ) {
933 for( const mod_id &e : world_generator->get_mod_manager().all_mods() ) {
934 if( !e->obsolete ) {
935 to_check.emplace( e );
936 }
937 }
938 }
939 // If no mods are available then test core data only
940 if( to_check.empty() ) {
941 to_check.emplace( mod_management::get_default_core_content_pack() );
942 }
943
944 for( const mod_id &id : to_check ) {
945 clear_loaded_data();
946
947 world_generator->set_active_world( nullptr );
948 world_generator->init();
949 const std::vector<mod_id> mods_empty;
950 WORLDPTR test_world = world_generator->make_new_world( mods_empty );
951 if( !test_world ) {
952 std::cerr << "Failed to generate test world." << std::endl;
953 return false;
954 }
955 world_generator->set_active_world( test_world );
956
957 std::cout << string_format( "Checking mod %s [%s]\n", id->name(), id );
958
959 // Load all dependencies first
960 std::vector<mod_id> mods_list = tree.get_dependencies_of_X_as_strings( id );
961 // Load the mod itself
962 mods_list.push_back( id );
963
964 try {
965 load_and_finalize_packs( ui, _( "Checking mods" ), mods_list );
966 } catch( const std::exception &err ) {
967 std::cerr << "Error loading data: " << err.what() << std::endl;
968 }
969
970 std::string world_name = world_generator->active_world->world_name;
971 world_generator->delete_world( world_name, true );
972
973 // TODO: Why would we need these calls?
974 MAPBUFFER.reset();
975 overmap_buffer.clear();
976 }
977
978 return !debug_has_error_been_observed();
979}
