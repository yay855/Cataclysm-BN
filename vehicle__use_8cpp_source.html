529 is_alarm_on = false;

880 // TODO: take longer to fold bigger vehicles
881 // TODO: make this interruptible
882 g->u.moves -= 500;

886double vehicle::engine_cold_factor( const int e ) const
887{
888 if( !part_info( engines[e] ).has_flag( "E_COLD_START" ) ) {
889 return 0.0;
890 }
891
892 int eff_temp = get_weather().get_temperature( g->u.pos() );
893 if( !parts[ engines[ e ] ].faults().count( fault_glowplug ) ) {
894 eff_temp = std::min( eff_temp, 20 );
895 }
896
897 return 1.0 - ( std::max( 0, std::min( 30, eff_temp ) ) / 30.0 );

900int vehicle::engine_start_time( const int e ) const
901{
902 if( !is_engine_on( e ) || part_info( engines[e] ).has_flag( "E_STARTS_INSTANTLY" ) ||
903 !engine_fuel_left( e ) ) {
904 return 0;
905 }
906
907 const double dmg = parts[engines[e]].damage_percent();
908
909 // non-linear range [100-1000]; f(0.0) = 100, f(0.6) = 250, f(0.8) = 500, f(0.9) = 1000
910 // diesel engines with working glow plugs always start with f = 0.6 (or better)
911 const double cold = 100 / tanh( 1 - std::min( engine_cold_factor( e ), 0.9 ) );
912
913 // watts to old vhp = watts / 373
914 // divided by magic 16 = watts / 6000
915 const double watts_per_time = 6000;
916 return part_vpower_w( engines[ e ], true ) / watts_per_time + 100 * dmg + cold;
917}

919bool vehicle::start_engine( const int e )
920{
921 if( !is_engine_on( e ) ) {
922 return false;
923 }
924
925 const vpart_info &einfo = part_info( engines[e] );
926 vehicle_part &eng = parts[ engines[ e ] ];
927
928 bool out_of_fuel = false;
929 if( einfo.fuel_type != fuel_type_none && engine_fuel_left( e ) <= 0 ) {
930 for( const itype_id &fuel_id : einfo.engine_fuel_opts() ) {
931 if( fuel_left( fuel_id ) > 0 ) {
932 eng.fuel_set( fuel_id );
933 break;
934 }
935 }
936 out_of_fuel = true;
937 }
938
939 if( out_of_fuel ) {
940 if( einfo.fuel_type == fuel_type_muscle ) {
941 // Muscle engines cannot start with broken limbs
942 if( einfo.has_flag( "MUSCLE_ARMS" ) && ( g->u.get_working_arm_count() < 2 ) ) {
943 add_msg( _( "You cannot use %s with a broken arm." ), eng.name() );
944 return false;
945 } else if( einfo.has_flag( "MUSCLE_LEGS" ) && ( g->u.get_working_leg_count() < 2 ) ) {
946 add_msg( _( "You cannot use %s with a broken leg." ), eng.name() );
947 return false;
948 }
949 } else {
950 add_msg( _( "Looks like the %1$s is out of %2$s." ), eng.name(),
951 item::nname( einfo.fuel_type ) );
952 return false;
953 }
954 }
955
956 const double dmg = parts[engines[e]].damage_percent();
957 const int engine_power = std::abs( part_epower_w( engines[e] ) );
958 const double cold_factor = engine_cold_factor( e );
959 const int start_moves = engine_start_time( e );
960
961 const tripoint pos = global_part_pos3( engines[e] );
962 if( einfo.engine_backfire_threshold() ) {
963 if( ( 1 - dmg ) < einfo.engine_backfire_threshold() &&
964 one_in( einfo.engine_backfire_freq() ) ) {
965 backfire( e );
966 } else {
967 sounds::sound( pos, start_moves / 10, sounds::sound_t::movement,
968 string_format( _( "the %s bang as it starts" ), eng.name() ), true, "vehicle",
969 "engine_bangs_start" );
970 }
971 }
972
973 // Immobilizers need removing before the vehicle can be started
974 if( eng.faults().count( fault_immobiliser ) ) {
975 sounds::sound( pos, 5, sounds::sound_t::alarm,
976 string_format( _( "the %s making a long beep" ), eng.name() ), true, "vehicle",
977 "fault_immobiliser_beep" );
978 return false;
979 }
980
981 // Engine with starter motors can fail on both battery and starter motor
982 if( eng.faults_potential().count( fault_starter ) ) {
983 if( eng.faults().count( fault_starter ) ) {
984 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::alarm,
985 string_format( _( "the %s clicking once" ), eng.name() ), true, "vehicle",
986 "engine_single_click_fail" );
987 return false;
988 }
989 // TODO: start_moves is in moves, but it's an integer, convert it to some time class
990 const int start_draw_bat = power_to_energy_bat( engine_power *
991 ( 1.0 + dmg / 2 + cold_factor / 5 ) * 10,
992 1_turns * start_moves / 100 );
993 if( discharge_battery( start_draw_bat, true ) != 0 ) {
994 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::alarm,
995 string_format( _( "the %s rapidly clicking" ), eng.name() ), true, "vehicle",
996 "engine_multi_click_fail" );
997 return false;
998 }
999 }
1000
1001 // Engines always fail to start with faulty fuel pumps
1002 if( eng.faults().count( fault_pump ) || eng.faults().count( fault_diesel ) ) {
1003 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::movement,
1004 string_format( _( "the %s quickly stuttering out." ), eng.name() ), true, "vehicle",
1005 "engine_stutter_fail" );
1006 return false;
1007 }
1008
1009 // Damaged non-electric engines have a chance of failing to start
1010 if( !is_engine_type( e, fuel_type_battery ) && x_in_y( dmg * 100, 120 ) ) {
1011 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::movement,
1012 string_format( _( "the %s clanking and grinding" ), eng.name() ), true, "vehicle",
1013 "engine_clanking_fail" );
1014 return false;
1015 }
1016 sounds::sound( pos, eng.info().engine_noise_factor(), sounds::sound_t::movement,
1017 string_format( _( "the %s starting" ), eng.name() ) );
1018
1019 if( sfx::has_variant_sound( "engine_start", eng.info().get_id().str() ) ) {
1020 sfx::play_variant_sound( "engine_start", eng.info().get_id().str(),
1021 eng.info().engine_noise_factor() );
1022 } else if( is_engine_type( e, fuel_type_muscle ) ) {
1023 sfx::play_variant_sound( "engine_start", "muscle", eng.info().engine_noise_factor() );
1024 } else if( is_engine_type( e, fuel_type_wind ) ) {
1025 sfx::play_variant_sound( "engine_start", "wind", eng.info().engine_noise_factor() );
1026 } else if( is_engine_type( e, fuel_type_battery ) ) {
1027 sfx::play_variant_sound( "engine_start", "electric", eng.info().engine_noise_factor() );
1028 } else {
1029 sfx::play_variant_sound( "engine_start", "combustion", eng.info().engine_noise_factor() );
1030 }
1031 return true;
1032}

1056void vehicle::start_engines( const bool take_control, const bool autodrive )
1057{
1058 bool has_engine = std::any_of( engines.begin(), engines.end(), [&]( int idx ) {
1059 return parts[ idx ].enabled && !parts[ idx ].is_broken();
1060 } );
1061
1062 // if no engines enabled then enable all before trying to start the vehicle
1063 if( !has_engine ) {
1064 for( auto idx : engines ) {
1065 if( !parts[ idx ].is_broken() ) {
1066 parts[ idx ].enabled = true;
1067 }
1068 }
1069 }
1070
1071 int start_time = 0;
1072 // record the first usable engine as the referenced position checked at the end of the engine starting activity
1073 bool has_starting_engine_position = false;
1074 tripoint starting_engine_position;
1075 for( size_t e = 0; e < engines.size(); ++e ) {
1076 if( !has_starting_engine_position && !parts[ engines[ e ] ].is_broken() &&
1077 parts[ engines[ e ] ].enabled ) {
1078 starting_engine_position = global_part_pos3( engines[ e ] );
1079 has_starting_engine_position = true;
1080 }
1081 has_engine = has_engine || is_engine_on( e );
1082 start_time = std::max( start_time, engine_start_time( e ) );
1083 }
1084
1085 if( !has_starting_engine_position ) {
1086 starting_engine_position = global_pos3();
1087 }
1088
1089 if( !has_engine ) {
1090 add_msg( m_info, _( "The %s doesn't have an engine!" ), name );
1091 return;
1092 }
1093
1094 if( take_control && !g->u.controlling_vehicle ) {
1095 g->u.controlling_vehicle = true;
1096 add_msg( _( "You take control of the %s." ), name );
1097 }
1098 if( !autodrive ) {
1099 g->u.assign_activity( ACT_START_ENGINES, start_time );
1100 g->u.activity.placement = starting_engine_position - g->u.pos();
1101 g->u.activity.values.push_back( take_control );
1102 }
