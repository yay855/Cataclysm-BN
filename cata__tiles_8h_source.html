92 /// Returns the width (first) and height (second) of the stored texture.
93 std::pair<int, int> dimension() const {
94 return std::make_pair( srcrect.w, srcrect.h );
95 }
96 /// Interface to @ref SDL_RenderCopyEx, using this as the texture, and
97 /// null as source rectangle (render the whole texture). Other parameters
98 /// are simply passed through.
99 int render_copy_ex( const SDL_Renderer_Ptr &renderer, const SDL_Rect *const dstrect,
100 const double angle,
101 const SDL_Point *const center, const SDL_RendererFlip flip ) const {
102 return SDL_RenderCopyEx( renderer.get(), sdl_texture_ptr.get(), &srcrect, dstrect, angle, center,
103 flip );
104 }
105
106 int set_alpha_mod( int mod ) const {
107 return SDL_SetTextureAlphaMod( sdl_texture_ptr.get(), mod );
108 }
