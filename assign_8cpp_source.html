598 name );

602 const damage_unit &lo_dmg = *lo_iter;

614 err.throw_error(

617 name );
618 }

627void assign_dmg_proportional( const JsonObject &jo,
628 const std::string &name,
629 damage_instance &out,
630 const damage_instance &val,
631 damage_instance proportional,
632 bool &strict )
633{
634 for( const damage_unit &val_dmg : val.damage_units ) {
635 for( damage_unit &scalar : proportional.damage_units ) {
636 if( scalar.type != val_dmg.type ) {
637 continue;
638 }
639
640 // Do not require strict parsing for relative and proportional
641 // values as rules such as +10% are well-formed independent of
642 // whether they affect base value
643 strict = false;
644
645 // Can't have negative percent, and 100% is pointless
646 // If it's 0, it wasn't loaded
647 if( scalar.amount == 1 || scalar.amount < 0 ) {
648 jo.throw_error(
649 "Proportional damage multiplier must be a positive number "
650 "other than 1",
651 name );
652 }
653
654 // If it's 0, it wasn't loaded
655 if( scalar.res_pen < 0 || scalar.res_pen == 1 ) {
656 jo.throw_error(
657 "Proportional armor penetration multiplier must be a "
658 "positive number other than 1",
659 name );
660 }
661
662 // It wasn't loaded, so set it 100%
663 if( scalar.res_pen == 0 ) {
664 scalar.res_pen = 1.0f;
665 }
666
667 // Ditto
668 if( scalar.amount == 0 ) {
669 scalar.amount = 1.0f;
670 }
671
672 // If it's 1, it wasn't loaded (or was loaded as 1)
673 if( scalar.res_mult <= 0 ) {
674 jo.throw_error(
675 "Proportional armor penetration multiplier must be a "
676 "positive number",
677 name );
678 }
679
680 // If it's 1, it wasn't loaded (or was loaded as 1)
681 if( scalar.damage_multiplier <= 0 ) {
682 jo.throw_error(
683 "Proportional damage multiplier must be a positive number",
684 name );
685 }
686
687 damage_unit out_dmg( scalar.type, 0.0f );
688
689 out_dmg.amount = val_dmg.amount * scalar.amount;
690 out_dmg.res_pen = val_dmg.res_pen * scalar.res_pen;
691
692 out_dmg.res_mult = val_dmg.res_mult * scalar.res_mult;
693 out_dmg.damage_multiplier =
694 val_dmg.damage_multiplier * scalar.damage_multiplier;
695
696 out.add( out_dmg );
697 }
698 }
