103void mapbuffer::save( bool delete_after_save )
104{
105 assure_dir_exist( g->get_world_base_save_path() + "/maps" );
106
107 int num_saved_submaps = 0;
108 int num_total_submaps = submaps.size();
109
110 map &here = get_map();
111 const tripoint map_origin = sm_to_omt_copy( here.get_abs_sub() );
112 const bool map_has_zlevels = g != nullptr && here.has_zlevels();
113
114 static_popup popup;
115
116 // A set of already-saved submaps, in global overmap coordinates.
117 std::set<tripoint> saved_submaps;
118 std::list<tripoint> submaps_to_delete;
119 static constexpr std::chrono::milliseconds update_interval( 500 );
120 auto last_update = std::chrono::steady_clock::now();
121
122 for( auto &elem : submaps ) {
123 auto now = std::chrono::steady_clock::now();
124 if( last_update + update_interval < now ) {
125 popup.message( _( "Please wait as the map saves [%d/%d]" ),
126 num_saved_submaps, num_total_submaps );
127 ui_manager::redraw();
128 refresh_display();
129 last_update = now;
130 }
131 // Whatever the coordinates of the current submap are,
132 // we're saving a 2x2 quad of submaps at a time.
133 // Submaps are generated in quads, so we know if we have one member of a quad,
134 // we have the rest of it, if that assumption is broken we have REAL problems.
135 const tripoint om_addr = sm_to_omt_copy( elem.first );
136 if( saved_submaps.count( om_addr ) != 0 ) {
137 // Already handled this one.
138 continue;
139 }
140 saved_submaps.insert( om_addr );
141
142 // A segment is a chunk of 32x32 submap quads.
143 // We're breaking them into subdirectories so there aren't too many files per directory.
144 // Might want to make a set for this one too so it's only checked once per save().
145 const std::string dirname = find_dirname( om_addr );
146 const std::string quad_path = find_quad_path( dirname, om_addr );
147
148 // delete_on_save deletes everything, otherwise delete submaps
149 // outside the current map.
150 const bool zlev_del = !map_has_zlevels && om_addr.z != g->get_levz();
151 save_quad( dirname, quad_path, om_addr, submaps_to_delete,
152 delete_after_save || zlev_del ||
153 om_addr.x < map_origin.x || om_addr.y < map_origin.y ||
154 om_addr.x > map_origin.x + HALF_MAPSIZE ||
155 om_addr.y > map_origin.y + HALF_MAPSIZE );
156 num_saved_submaps += 4;
157 }
158 for( auto &elem : submaps_to_delete ) {
159 remove_submap( elem );
160 }
161
162 get_distribution_grid_tracker().on_saved();
163}

165void mapbuffer::save_quad( const std::string &dirname, const std::string &filename,
166 const tripoint &om_addr, std::list<tripoint> &submaps_to_delete,
167 bool delete_after_save )
168{
169 std::vector<point> offsets;
170 std::vector<tripoint> submap_addrs;
171 offsets.push_back( point_zero );
172 offsets.push_back( point_south );
173 offsets.push_back( point_east );
174 offsets.push_back( point_south_east );
175
176 bool all_uniform = true;
177 for( auto &offsets_offset : offsets ) {
178 tripoint submap_addr = omt_to_sm_copy( om_addr );
179 submap_addr.x += offsets_offset.x;
180 submap_addr.y += offsets_offset.y;
181 submap_addrs.push_back( submap_addr );
182 submap *sm = submaps[submap_addr];
183 if( sm != nullptr && !sm->is_uniform ) {
184 all_uniform = false;
185 }
186 }
187
188 if( all_uniform ) {
189 // Nothing to save - this quad will be regenerated faster than it would be re-read
190 if( delete_after_save ) {
191 for( auto &submap_addr : submap_addrs ) {
192 if( submaps.count( submap_addr ) > 0 && submaps[submap_addr] != nullptr ) {
193 submaps_to_delete.push_back( submap_addr );
194 }
195 }
196 }
197
198 return;
199 }
200
201 // Don't create the directory if it would be empty
202 assure_dir_exist( dirname );
203 write_to_file( filename, [&]( std::ostream & fout ) {
204 JsonOut jsout( fout );
205 jsout.start_array();
206 for( auto &submap_addr : submap_addrs ) {
207 if( submaps.count( submap_addr ) == 0 ) {
208 continue;
209 }
210
211 submap *sm = submaps[submap_addr];
212
213 if( sm == nullptr ) {
214 continue;
215 }
216
217 jsout.start_object();
218
219 jsout.member( "version", savegame_version );
220 jsout.member( "coordinates" );
221
222 jsout.start_array();
223 jsout.write( submap_addr.x );
224 jsout.write( submap_addr.y );
225 jsout.write( submap_addr.z );
226 jsout.end_array();
227
228 sm->store( jsout );
229
230 jsout.end_object();
231
232 if( delete_after_save ) {
233 submaps_to_delete.push_back( submap_addr );
234 }
235 }
236
237 jsout.end_array();
238 } );
239}

243submap *mapbuffer::unserialize_submaps( const tripoint &p )
244{
245 // Map the tripoint to the submap quad that stores it.
246 const tripoint om_addr = sm_to_omt_copy( p );
247 const std::string dirname = find_dirname( om_addr );
248 std::string quad_path = find_quad_path( dirname, om_addr );
249
250 if( !file_exist( quad_path ) ) {
251 // Fix for old saves where the path was generated using std::stringstream, which
252 // did format the number using the current locale. That formatting may insert
253 // thousands separators, so the resulting path is "map/1,234.7.8.map" instead
254 // of "map/1234.7.8.map".
255 std::ostringstream buffer;
256 buffer << dirname << "/" << om_addr.x << "." << om_addr.y << "." << om_addr.z << ".map";
257 if( file_exist( buffer.str() ) ) {
258 quad_path = buffer.str();
259 }
260 }
261
262 using namespace std::placeholders;
263 if( !read_from_file_optional_json( quad_path, std::bind( &mapbuffer::deserialize, this, _1 ) ) ) {
264 // If it doesn't exist, trigger generating it.
265 return nullptr;
266 }
267 if( submaps.count( p ) == 0 ) {
268 debugmsg( "file %s did not contain the expected submap %d,%d,%d",
269 quad_path, p.x, p.y, p.z );
270 return nullptr;
271 }
272 return submaps[ p ];
273}

275void mapbuffer::deserialize( JsonIn &jsin )
276{
277 jsin.start_array();
278 while( !jsin.end_array() ) {
279 std::unique_ptr<submap> sm = std::make_unique<submap>();
280 tripoint submap_coordinates;
281 jsin.start_object();
282 int version = 0;
283 while( !jsin.end_object() ) {
284 std::string submap_member_name = jsin.get_member_name();
285 if( submap_member_name == "version" ) {
286 version = jsin.get_int();
287 } else if( submap_member_name == "coordinates" ) {
288 jsin.start_array();
289 tripoint loc{ jsin.get_int(), jsin.get_int(), jsin.get_int() };
290 jsin.end_array();
291 submap_coordinates = loc;
292 } else {
293 sm->load( jsin, submap_member_name, version );
294 }
295 }
296
297 if( !add_submap( submap_coordinates, sm ) ) {
298 debugmsg( "submap %d,%d,%d was already loaded", submap_coordinates.x, submap_coordinates.y,
299 submap_coordinates.z );
300 }
301 }
