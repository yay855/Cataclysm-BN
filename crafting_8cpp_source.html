131float lighting_crafting_speed_multiplier( const Character &who, const recipe &rec )
132{
133 if( character_funcs::can_see_fine_details( who ) ) {
134 return 1.0f;
135 }
136
137 const SkillLevelMap &char_skills = who.get_all_skills();
138 int skill_bonus = char_skills.exceeds_recipe_requirements( rec );
139
140 // This value whould be within [0,1]
141 const float darkness =
142 (
143 character_funcs::fine_detail_vision_mod( who ) -
144 character_funcs::FINE_VISION_THRESHOLD
145 ) / 7.0f;
146
147 if( rec.has_flag( flag_BLIND_EASY ) ) {
148 // 100% speed in well lit area at skill+0
149 // 25% speed in pitch black at skill+0
150 // skill+2 removes speed penalty
151 return 1.0f - darkness * 0.75f * std::max( 0, 2 - skill_bonus ) / 2.0f;
152 } else if( rec.has_flag( flag_BLIND_HARD ) && skill_bonus >= 2 ) {
153 // 100% speed in well lit area at skill+2
154 // 25% speed in pitch black at skill+2
155 // skill+8 removes speed penalty
156 return 1.0f - darkness * 0.75f * std::max( 0, 8 - skill_bonus ) / 6.0f;
157 } else {
158 // Needs proper vision or the character is not skilled enough
159 return 0.0f;
160 }

163float morale_crafting_speed_multiplier( const Character &who, const recipe &rec )
164{
165 int morale = who.get_morale_level();
166 if( morale >= 0 ) {
167 // No bonus for being happy yet
168 return 1.0f;
169 }
170
171 // Harder jobs are more frustrating, even when skilled
172 // For each skill where skill=difficulty, multiply effective morale by 200%
173 float morale_mult = std::max( 1.0f, 2.0f * rec.difficulty / std::max( 1,
174 who.get_skill_level( rec.skill_used ) ) );
175 for( const std::pair<const skill_id, int> &pr : rec.required_skills ) {
176 morale_mult *= std::max( 1.0f, 2.0f * pr.second / std::max( 1, who.get_skill_level( pr.first ) ) );
177 }
178
179 // Halve speed at -50 effective morale, quarter at -150
180 float morale_effect = 1.0f + ( morale_mult * morale ) / -50.0f;
181
182 return 1.0f / morale_effect;
183}

185template<typename T>
186static float lerped_multiplier( const T &value, const T &low, const T &high )
187{
188 // No effect if less than allowed value
189 if( value < low ) {
190 return 1.0f;
191 }
192 // Bottom out at 25% speed
193 if( value > high ) {
194 return 0.25f;
195 }
196 // Linear interpolation between high and low
197 // y = y0 + ( x - x0 ) * ( y1 - y0 ) / ( x1 - x0 )
198 return 1.0f + ( value - low ) * ( 0.25f - 1.0f ) / ( high - low );
199}

282float crafting_speed_multiplier( const Character &who, const item &craft,
283 const bench_location &bench )
284{
285 if( !craft.is_craft() ) {
286 debugmsg( "Can't calculate crafting speed multiplier of non-craft '%s'", craft.tname() );
287 return 1.0f;
288 }
289
290 const recipe &rec = craft.get_making();
291
292 const float light_multi = lighting_crafting_speed_multiplier( who, rec );
293 const float bench_multi = workbench_crafting_speed_multiplier( craft, bench );
294 const float morale_multi = morale_crafting_speed_multiplier( who, rec );
295
296 const float total_multi = light_multi * bench_multi * morale_multi;
297
298 if( light_multi <= 0.0f ) {
299 who.add_msg_if_player( m_bad, _( "You can no longer see well enough to keep crafting." ) );
300 return 0.0f;
301 }
302 if( bench_multi <= 0.1f || ( bench_multi <= 0.33f && total_multi <= 0.2f ) ) {
303 who.add_msg_if_player( m_bad, _( "The %s is too large and/or heavy to work on. You may want to"
304 " use a workbench or a smaller batch size" ), craft.tname() );
305 return 0.0f;
306 }
307 if( morale_multi <= 0.2f || ( morale_multi <= 0.33f && total_multi <= 0.2f ) ) {
308 who.add_msg_if_player( m_bad, _( "Your morale is too low to continue crafting." ) );
309 return 0.0f;
310 }
311
312 // If we're working below 20% speed, just give up
313 if( total_multi <= 0.2f ) {
314 who.add_msg_if_player( m_bad, _( "You are too frustrated to continue and just give up." ) );
315 return 0.0f;
316 }
317
318 if( calendar::once_every( 1_hours ) && total_multi < 0.75f ) {
319 if( light_multi <= 0.5f ) {
320 who.add_msg_if_player( m_bad, _( "You can't see well and are working slowly." ) );
321 }
322 if( bench_multi <= 0.5f ) {
323 who.add_msg_if_player( m_bad,
324 _( "The %s is to large and/or heavy to work on comfortably. You are"
325 " working slowly." ), craft.tname() );
326 }
327 if( morale_multi <= 0.5f ) {
328 who.add_msg_if_player( m_bad, _( "You can't focus and are working slowly." ) );
329 }
330 }
331
332 return total_multi;

413bool player::check_eligible_containers_for_crafting( const recipe &rec, int batch_size ) const
414{
415 std::vector<const item *> conts = get_eligible_containers_for_crafting();
416 const std::vector<item> res = rec.create_results( batch_size );
417 const std::vector<item> bps = rec.create_byproducts( batch_size );
418 std::vector<item> all;
419 all.reserve( res.size() + bps.size() );
420 all.insert( all.end(), res.begin(), res.end() );
421 all.insert( all.end(), bps.begin(), bps.end() );
422
423 map &here = get_map();
424 for( const item &prod : all ) {
425 if( !prod.made_of( LIQUID ) ) {
426 continue;
427 }
428
429 // we go through half-filled containers first, then go through empty containers if we need
430 std::sort( conts.begin(), conts.end(), item_ptr_compare_by_charges );
431
432 int charges_to_store = prod.charges;
433 for( const item *cont : conts ) {
434 if( charges_to_store <= 0 ) {
435 break;
436 }
437
438 if( !cont->is_container_empty() ) {
439 if( cont->contents.front().typeId() == prod.typeId() ) {
440 charges_to_store -= cont->get_remaining_capacity_for_liquid( cont->contents.front(), true );
441 }
442 } else {
443 charges_to_store -= cont->get_remaining_capacity_for_liquid( prod, true );
444 }
445 }
446
447 // also check if we're currently in a vehicle that has the necessary storage
448 if( charges_to_store > 0 ) {
449 if( optional_vpart_position vp = here.veh_at( pos() ) ) {
450 const itype_id &ftype = prod.typeId();
451 int fuel_cap = vp->vehicle().fuel_capacity( ftype );
452 int fuel_amnt = vp->vehicle().fuel_left( ftype );
453
454 if( fuel_cap >= 0 ) {
455 int fuel_space_left = fuel_cap - fuel_amnt;
456 charges_to_store -= fuel_space_left;
457 }
458 }
459 }
460
461 if( charges_to_store > 0 ) {
462 if( !query_yn(
463 _( "You don't have anything in which to store %s and may have to pour it out or consume it as soon as it is prepared! Proceed?" ),
464 prod.tname() ) ) {
465 return false;
466 }
467 }
468 }
469
470 return true;
471}

624// @param offset is the index of the created item in the range [0, batch_size-1],
625// it makes sure that the used items are distributed equally among the new items.
626static void set_components( std::list<item> &components, const std::list<item> &used,
627 const int batch_size, const size_t offset )
628{
629 if( batch_size <= 1 ) {
630 components.insert( components.begin(), used.begin(), used.end() );
631 return;
632 }
633 // This count does *not* include items counted by charges!
634 size_t non_charges_counter = 0;
635 for( auto &tmp : used ) {
636 if( tmp.count_by_charges() ) {
637 components.push_back( tmp );
638 // This assumes all (count-by-charges) items of the same type have been merged into one,
639 // which has a charges value that can be evenly divided by batch_size.
640 components.back().charges = tmp.charges / batch_size;
641 } else {
642 if( ( non_charges_counter + offset ) % batch_size == 0 ) {
643 components.push_back( tmp );
644 }
645 non_charges_counter++;
646 }
647 }

817double player::crafting_success_roll( const recipe &making ) const
818{
819 int secondary_dice = 0;
820 int secondary_difficulty = 0;
821 for( const auto &pr : making.required_skills ) {
822 secondary_dice += get_skill_level( pr.first );
823 secondary_difficulty += pr.second;
824 }
825
826 // # of dice is 75% primary skill, 25% secondary (unless secondary is null)
827 int skill_dice;
828 if( secondary_difficulty > 0 ) {
829 skill_dice = get_skill_level( making.skill_used ) * 3 + secondary_dice;
830 } else {
831 skill_dice = get_skill_level( making.skill_used ) * 4;
832 }
833
834 for( const npc *np : get_crafting_helpers() ) {
835 if( np->get_skill_level( making.skill_used ) >=
836 get_skill_level( making.skill_used ) ) {
837 // NPC assistance is worth half a skill level
838 skill_dice += 2;
839 add_msg_if_player( m_info, _( "%s helps with craftingâ€¦" ), np->name );
840 break;
841 }
842 }
843
844 // farsightedness can impose a penalty on electronics and tailoring success
845 // it's equivalent to a 2-rank electronics penalty, 1-rank tailoring
846 if( has_trait( trait_HYPEROPIC ) && !worn_with_flag( flag_FIX_FARSIGHT ) &&
847 !has_effect( effect_contacts ) ) {
848 int main_rank_penalty = 0;
849 if( making.skill_used == skill_electronics ) {
850 main_rank_penalty = 2;
851 } else if( making.skill_used == skill_tailor ) {
852 main_rank_penalty = 1;
853 }
854 skill_dice -= main_rank_penalty * 4;
855 }
856
857 // It's tough to craft with paws. Fortunately it's just a matter of grip and fine-motor,
858 // not inability to see what you're doing
859 for( const trait_id &mut : get_mutations() ) {
860 for( const std::pair<const skill_id, int> &skib : mut->craft_skill_bonus ) {
861 if( making.skill_used == skib.first ) {
862 skill_dice += skib.second;
863 }
864 }
865 }
866
867 // Sides on dice is 16 plus your current intelligence
868 ///\EFFECT_INT increases crafting success chance
869 const int skill_sides = 16 + int_cur;
870
871 int diff_dice;
872 if( secondary_difficulty > 0 ) {
873 diff_dice = making.difficulty * 3 + secondary_difficulty;
874 } else {
875 // Since skill level is * 4 also
876 diff_dice = making.difficulty * 4;
877 }
878
879 const int diff_sides = 24; // 16 + 8 (default intelligence)
880
881 const double skill_roll = dice( skill_dice, skill_sides );
882 const double diff_roll = dice( diff_dice, diff_sides );
883
884 if( diff_roll == 0 ) {
885 // Automatic success
886 return 2;
887 }
888
889 return skill_roll / diff_roll;
890}

927bool item::handle_craft_failure( player &crafter )
928{
929 if( !is_craft() ) {
930 debugmsg( "handle_craft_failure() called on non-craft '%s.' Aborting.", tname() );
931 return false;
932 }
933
934 const double success_roll = crafter.crafting_success_roll( get_making() );
935 const int starting_components = this->components.size();
936 // Destroy at most 75% of the components, always a chance of losing 1 though
937 const size_t max_destroyed = std::max<size_t>( 1, components.size() * 3 / 4 );
938 for( size_t i = 0; i < max_destroyed; i++ ) {
939 // This shouldn't happen
940 if( components.empty() ) {
941 break;
942 }
943 // If we roll success, skip destroying a component
944 if( x_in_y( success_roll, 1.0 ) ) {
945 continue;
946 }
947 destroy_random_component( *this, crafter );
948 }
949 if( starting_components > 0 && this->components.empty() ) {
950 // The craft had components and all of them were destroyed.
951 return true;
952 }
953
954 // Minimum 25% progress lost, average 35%. Falls off exponentially
955 // Loss is scaled by the success roll
956 const double percent_progress_loss = rng_exponential( 0.25, 0.35 ) *
957 ( 1.0 - std::min( success_roll, 1.0 ) );
958 const int progess_loss = item_counter * percent_progress_loss;
959 crafter.add_msg_player_or_npc( _( "You mess up and lose %d%% progress." ),
960 _( "<npcname> messes up and loses %d%% progress." ), progess_loss / 100000 );
961 item_counter = clamp( item_counter - progess_loss, 0, 10000000 );
962
963 set_next_failure_point( crafter );
964
965 // Check if we can consume a new component and continue
966 if( !crafter.can_continue_craft( *this ) ) {
967 crafter.cancel_activity();
968 }
969 return false;
970}

1016void complete_craft( player &p, item &craft, const bench_location & )
1017{
1018 if( !craft.is_craft() ) {
1019 debugmsg( "complete_craft() called on non-craft '%s.' Aborting.", craft.tname() );
1020 return;
1021 }
1022
1023 const recipe &making = craft.get_making();
1024 const int batch_size = craft.charges;
1025 std::list<item> &used = craft.components;
1026 const double relative_rot = craft.get_relative_rot();
1027
1028 // Set up the new item, and assign an inventory letter if available
1029 std::vector<item> newits = making.create_results( batch_size );
1030
1031 const bool should_heat = making.hot_result();
1032
1033 bool first = true;
1034 size_t newit_counter = 0;
1035 for( item &newit : newits ) {
1036
1037 // Points to newit unless newit is a non-empty container, then it points to newit's contents.
1038 // Necessary for things like canning soup; sometimes we want to operate on the soup, not the can.
1039 item &food_contained = ( newit.is_container() && !newit.contents.empty() ) ?
1040 newit.contents.back() : newit;
1041
1042 // messages, learning of recipe, food spoilage calculation only once
1043 if( first ) {
1044 first = false;
1045 // TODO: reconsider recipe memorization
1046 if( p.knows_recipe( &making ) ) {
1047 add_msg( _( "You craft %s from memory." ), making.result_name() );
1048 } else {
1049 add_msg( _( "You craft %s using a book as a reference." ), making.result_name() );
1050 // If we made it, but we don't know it,
1051 // we're making it from a book and have a chance to learn it.
1052 // Base expected time to learn is 1000*(difficulty^4)/skill/int moves.
1053 // This means time to learn is greatly decreased with higher skill level,
1054 // but also keeps going up as difficulty goes up.
1055 // Worst case is lvl 10, which will typically take
1056 // 10^4/10 (1,000) minutes, or about 16 hours of crafting it to learn.
1057 int difficulty = p.has_recipe( &making, p.crafting_inventory(), p.get_crafting_helpers() );
1058 ///\EFFECT_INT increases chance to learn recipe when crafting from a book
1059 const double learning_speed =
1060 std::max( p.get_skill_level( making.skill_used ), 1 ) *
1061 std::max( p.get_int(), 1 );
1062 const double time_to_learn = 1000 * 8 * std::pow( difficulty, 4 ) / learning_speed;
1063 if( x_in_y( making.time, time_to_learn ) ) {
1064 p.learn_recipe( &making );
1065 add_msg( m_good, _( "You memorized the recipe for %s!" ),
1066 making.result_name() );
1067 }
1068 }
1069 }
1070
1071 food_contained.inherit_flags( used, making );
1072
1073 for( const std::string &flag : making.flags_to_delete ) {
1074 food_contained.unset_flag( flag );
1075 }
1076
1077 // Don't store components for things made by charges,
1078 // Don't store components for things that can't be uncrafted.
1079 if( recipe_dictionary::get_uncraft( making.result() ) && !food_contained.count_by_charges() &&
1080 making.is_reversible() ) {
1081 // Setting this for items counted by charges gives only problems:
1082 // those items are automatically merged everywhere (map/vehicle/inventory),
1083 // which would either lose this information or merge it somehow.
1084 set_components( food_contained.components, used, batch_size, newit_counter );
1085 newit_counter++;
1086 } else if( food_contained.is_food() && !food_contained.has_flag( flag_NUTRIENT_OVERRIDE ) ) {
1087 // if a component item has "cooks_like" it will be replaced by that item as a component
1088 for( item &comp : used ) {
1089 // only comestibles have cooks_like. any other type of item will throw an exception, so filter those out
1090 if( comp.is_comestible() && !comp.get_comestible()->cooks_like.is_empty() ) {
1091 comp = item( comp.get_comestible()->cooks_like, comp.birthday(), comp.charges );
1092 }
1093 // If this recipe is cooked, components are no longer raw.
1094 if( should_heat ) {
1095 comp.set_flag_recursive( flag_COOKED );
1096 }
1097 }
1098
1099 // use a copy of the used list so that the byproducts don't build up over iterations (#38071)
1100 std::list<item> usedbp = used;
1101 // byproducts get stored as a "component" but with a byproduct flag for consumption purposes
1102 if( making.has_byproducts() ) {
1103 for( item &byproduct : making.create_byproducts( batch_size ) ) {
1104 byproduct.set_flag( flag_BYPRODUCT );
1105 usedbp.push_back( byproduct );
1106 }
1107 }
1108 // store components for food recipes that do not have the override flag
1109 set_components( food_contained.components, usedbp, batch_size, newit_counter );
1110
1111 // store the number of charges the recipe would create with batch size 1.
1112 if( &newit != &food_contained ) { // If a canned/contained item was craftedâ€¦
1113 // â€¦ the container holds exactly one completion of the recipe, no matter the batch size.
1114 food_contained.recipe_charges = food_contained.charges;
1115 } else { // Otherwise, the item is already stacked so we need to divide by batch size.
1116 newit.recipe_charges = newit.charges / batch_size;
1117 }
1118 newit_counter++;
1119 }
1120
1121 if( food_contained.goes_bad() ) {
1122 food_contained.set_relative_rot( relative_rot );
1123 }
1124
1125 newit.set_owner( p.get_faction()->id );
1126 // If these aren't equal, newit is a container, so finalize its contents too.
1127 if( &newit != &food_contained ) {
1128 food_contained.set_owner( p.get_faction()->id );
1129 }
1130
1131 if( newit.made_of( LIQUID ) ) {
1132 liquid_handler::handle_all_liquid( newit, PICKUP_RANGE );
1133 } else {
1134 set_item_inventory( p, newit );
1135 }
1136 }
1137
1138 if( making.has_byproducts() ) {
1139 std::vector<item> bps = making.create_byproducts( batch_size );
1140 for( auto &bp : bps ) {
1141 if( bp.goes_bad() ) {
1142 bp.set_relative_rot( relative_rot );
1143 }
1144 bp.set_owner( p.get_faction()->id );
1145 if( bp.made_of( LIQUID ) ) {
1146 liquid_handler::handle_all_liquid( bp, PICKUP_RANGE );
1147 } else {
1148 set_item_inventory( p, bp );
1149 }
1150 }
1151 }
1152
1153 p.inv.restack( p );
1154}

1156bool player::can_continue_craft( item &craft )
1157{
1158 if( !craft.is_craft() ) {
1159 debugmsg( "complete_craft() called on non-craft '%s.' Aborting.", craft.tname() );
1160 return false;
1161 }
1162 if( has_trait( trait_DEBUG_HS ) ) {
1163 return true;
1164 }
1165
1166 const recipe &rec = craft.get_making();
1167
1168 const requirement_data continue_reqs = craft.get_continue_reqs();
1169
1170 // Avoid building an inventory from the map if we don't have to, as it is expensive
1171 if( !continue_reqs.is_empty() ) {
1172
1173 std::function<bool( const item & )> filter = rec.get_component_filter();
1174 const std::function<bool( const item & )> no_rotten_filter =
1175 rec.get_component_filter( recipe_filter_flags::no_rotten );
1176 // continue_reqs are for all batches at once
1177 const int batch_size = 1;
1178
1179 if( !continue_reqs.can_make_with_inventory( crafting_inventory(), filter, batch_size ) ) {
1180 std::string buffer = _( "You don't have the required components to continue crafting!" );
1181 buffer += "\n";
1182 buffer += continue_reqs.list_missing();
1183 popup( buffer, PF_NONE );
1184 return false;
1185 }
1186
1187 std::string buffer = _( "Consume the missing components and continue crafting?" );
1188 buffer += "\n";
1189 buffer += continue_reqs.list_all();
1190 if( !query_yn( buffer ) ) {
1191 return false;
1192 }
1193
1194 if( continue_reqs.can_make_with_inventory( crafting_inventory(), no_rotten_filter,
1195 batch_size ) ) {
1196 filter = no_rotten_filter;
1197 } else {
1198 if( !query_yn( _( "Some components required to continue are rotten.\n"
1199 "Continue crafting anyway?" ) ) ) {
1200 return false;
1201 }
1202 }
1203
1204 inventory map_inv;
1205 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1206
1207 std::vector<comp_selection<item_comp>> item_selections;
1208 for( const auto &it : continue_reqs.get_components() ) {
1209 comp_selection<item_comp> is = select_item_component( it, batch_size, map_inv, true, filter );
1210 if( is.use_from == cancel ) {
1211 cancel_activity();
1212 add_msg( _( "You stop crafting." ) );
1213 return false;
1214 }
1215 item_selections.push_back( is );
1216 }
1217 for( const auto &it : item_selections ) {
1218 craft.components.splice( craft.components.end(), consume_items( it, batch_size, filter ) );
1219 }
1220 }
1221
1222 if( !craft.has_tools_to_continue() ) {
1223
1224 const std::vector<std::vector<tool_comp>> &tool_reqs = rec.simple_requirements().get_tools();
1225 const int batch_size = craft.charges;
1226
1227 std::vector<std::vector<tool_comp>> adjusted_tool_reqs;
1228 for( const std::vector<tool_comp> &alternatives : tool_reqs ) {
1229 std::vector<tool_comp> adjusted_alternatives;
1230 for( const tool_comp &alternative : alternatives ) {
1231 tool_comp adjusted_alternative = alternative;
1232 if( adjusted_alternative.count > 0 ) {
1233 adjusted_alternative.count *= batch_size;
1234 // Only for the next 5% progress
1235 adjusted_alternative.count = std::max(
1236 crafting::charges_for_continuing( adjusted_alternative.count ), 1 );
1237 }
1238 adjusted_alternatives.push_back( adjusted_alternative );
1239 }
1240 adjusted_tool_reqs.push_back( adjusted_alternatives );
1241 }
1242
1243 const requirement_data tool_continue_reqs( adjusted_tool_reqs,

1307/* selection of component if a recipe requirement has multiple options (e.g. 'duct tap' or 'welder') */
1308comp_selection<item_comp> player::select_item_component( const std::vector<item_comp> &components,
1309 int batch, inventory &map_inv, bool can_cancel,
1310 const std::function<bool( const item & )> &filter, bool player_inv )
1311{
1312 std::vector<item_comp> player_has;
1313 std::vector<item_comp> map_has;
1314 std::vector<item_comp> mixed;
1315
1316 comp_selection<item_comp> selected;
1317
1318 for( const auto &component : components ) {
1319 itype_id type = component.type;
1320 int count = ( component.count > 0 ) ? component.count * batch : std::abs( component.count );
1321
1322 if( item::count_by_charges( type ) && count > 0 ) {
1323 int map_charges = map_inv.charges_of( type, INT_MAX, filter );
1324
1325 // If map has infinite charges, just use them
1326 if( map_charges == item::INFINITE_CHARGES ) {
1327 selected.use_from = use_from_map;
1328 selected.comp = component;
1329 return selected;
1330 }
1331 if( player_inv ) {
1332 int player_charges = charges_of( type, INT_MAX, filter );
1333 bool found = false;
1334 if( player_charges >= count ) {
1335 player_has.push_back( component );
1336 found = true;
1337 }
1338 if( map_charges >= count ) {
1339 map_has.push_back( component );
1340 found = true;
1341 }
1342 if( !found && player_charges + map_charges >= count ) {
1343 mixed.push_back( component );
1344 }
1345 } else {
1346 if( map_charges >= count ) {
1347 map_has.push_back( component );
1348 }
1349 }
1350 } else { // Counting by units, not charges
1351
1352 // Can't use pseudo items as components
1353 if( player_inv ) {
1354 bool found = false;
1355 if( has_amount( type, count, false, filter ) ) {
1356 player_has.push_back( component );
1357 found = true;
1358 }
1359 if( map_inv.has_components( type, count, filter ) ) {
1360 map_has.push_back( component );
1361 found = true;
1362 }
1363 if( !found &&
1364 amount_of( type, false, std::numeric_limits<int>::max(), filter ) +
1365 map_inv.amount_of( type, false, std::numeric_limits<int>::max(), filter ) >= count ) {
1366 mixed.push_back( component );
1367 }
1368 } else {
1369 if( map_inv.has_components( type, count, filter ) ) {
1370 map_has.push_back( component );
1371 }
1372 }
1373 }
1374 }
1375
1376 /* select 1 component to use */
1377 if( player_has.size() + map_has.size() + mixed.size() == 1 ) { // Only 1 choice
1378 if( player_has.size() == 1 ) {
1379 selected.use_from = use_from_player;
1380 selected.comp = player_has[0];
1381 } else if( map_has.size() == 1 ) {
1382 selected.use_from = use_from_map;
1383 selected.comp = map_has[0];
1384 } else {
1385 selected.use_from = use_from_both;
1386 selected.comp = mixed[0];
1387 }
1388 } else if( is_npc() ) {
1389 if( !player_has.empty() ) {
1390 selected.use_from = use_from_player;
1391 selected.comp = player_has[0];
1392 } else if( !map_has.empty() ) {
1393 selected.use_from = use_from_map;
1394 selected.comp = map_has[0];
1395 } else {
1396 debugmsg( "Attempted a recipe with no available components!" );
1397 selected.use_from = cancel;
1398 return selected;
1399 }
1400 } else { // Let the player pick which component they want to use
1401 uilist cmenu;
1402 // Populate options with the names of the items
1403 for( auto &map_ha : map_has ) { // Index 0-(map_has.size()-1)
1404 std::string tmpStr = string_format( _( "%s (%d/%d nearby)" ),
1405 item::nname( map_ha.type ),
1406 ( map_ha.count * batch ),
1407 item::count_by_charges( map_ha.type ) ?
1408 map_inv.charges_of( map_ha.type, INT_MAX, filter ) :
1409 map_inv.amount_of( map_ha.type, false, INT_MAX, filter ) );
1410 cmenu.addentry( tmpStr );
1411 }
1412 for( auto &player_ha : player_has ) { // Index map_has.size()-(map_has.size()+player_has.size()-1)
1413 std::string tmpStr = string_format( _( "%s (%d/%d on person)" ),
1414 item::nname( player_ha.type ),
1415 ( player_ha.count * batch ),
1416 item::count_by_charges( player_ha.type ) ?
1417 charges_of( player_ha.type, INT_MAX, filter ) :
1418 amount_of( player_ha.type, false, INT_MAX, filter ) );
1419 cmenu.addentry( tmpStr );
1420 }
1421 for( auto &component : mixed ) {
1422 // Index player_has.size()-(map_has.size()+player_has.size()+mixed.size()-1)
1423 int available = item::count_by_charges( component.type ) ?
1424 map_inv.charges_of( component.type, INT_MAX, filter ) +
1425 charges_of( component.type, INT_MAX, filter ) :
1426 map_inv.amount_of( component.type, false, INT_MAX, filter ) +
1427 amount_of( component.type, false, INT_MAX, filter );
1428 std::string tmpStr = string_format( _( "%s (%d/%d nearby & on person)" ),
1429 item::nname( component.type ),
1430 component.count * batch,
1431 available );
1432 cmenu.addentry( tmpStr );
1433 }
1434
1435 // Unlike with tools, it's a bad thing if there aren't any components available
1436 if( cmenu.entries.empty() ) {
1437 if( player_inv ) {
1438 if( has_trait( trait_DEBUG_HS ) ) {
1439 selected.use_from = use_from_player;
1440 return selected;
1441 }
1442 }
1443 debugmsg( "Attempted a recipe with no available components!" );
1444 selected.use_from = cancel;
1445 return selected;
1446 }
1447
1448 cmenu.allow_cancel = can_cancel;
1449
1450 // Get the selection via a menu popup
1451 cmenu.title = _( "Use which component?" );
1452 cmenu.query();
1453
1454 if( cmenu.ret < 0 ||
1455 static_cast<size_t>( cmenu.ret ) >= map_has.size() + player_has.size() + mixed.size() ) {
1456 selected.use_from = cancel;
1457 return selected;
1458 }
1459
1460 size_t uselection = static_cast<size_t>( cmenu.ret );
1461 if( uselection < map_has.size() ) {
1462 selected.use_from = usage::use_from_map;
1463 selected.comp = map_has[uselection];
1464 } else if( uselection < map_has.size() + player_has.size() ) {
1465 uselection -= map_has.size();
1466 selected.use_from = usage::use_from_player;
1467 selected.comp = player_has[uselection];
1468 } else {
1469 uselection -= map_has.size() + player_has.size();
1470 selected.use_from = usage::use_from_both;
1471 selected.comp = mixed[uselection];
1472 }
1473 }
1474
1475 return selected;
1476}

1513std::list<item> player::consume_items( map &m, const comp_selection<item_comp> &is, int batch,
1514 const std::function<bool( const item & )> &filter,
1515 const tripoint &origin, int radius )
1516{
1517 std::list<item> ret;
1518
1519 if( has_trait( trait_DEBUG_HS ) ) {
1520 return ret;
1521 }
1522
1523 item_comp selected_comp = is.comp;
1524
1525 const tripoint &loc = origin;
1526 const bool by_charges = item::count_by_charges( selected_comp.type ) && selected_comp.count > 0;
1527 // Count given to use_amount/use_charges, changed by those functions!
1528 int real_count = ( selected_comp.count > 0 ) ? selected_comp.count * batch : std::abs(
1529 selected_comp.count );
1530 // First try to get everything from the map, than (remaining amount) from player
1531 if( is.use_from & use_from_map ) {
1532 if( by_charges ) {
1533 std::list<item> tmp = m.use_charges( loc, radius, selected_comp.type, real_count, filter );
1534 ret.splice( ret.end(), tmp );
1535 } else {
1536 std::list<item> tmp = g->m.use_amount( loc, radius, selected_comp.type, real_count, filter );
1537 remove_ammo( tmp, *this );
1538 ret.splice( ret.end(), tmp );
1539 }
1540 }
1541 if( is.use_from & use_from_player ) {
1542 if( by_charges ) {
1543 std::list<item> tmp = use_charges( selected_comp.type, real_count, filter );
1544 ret.splice( ret.end(), tmp );
1545 } else {
1546 std::list<item> tmp = use_amount( selected_comp.type, real_count, filter );
1547 remove_ammo( tmp, *this );
1548 ret.splice( ret.end(), tmp );
1549 }
1550 }
1551 // condense those items into one
1552 if( by_charges && ret.size() > 1 ) {
1553 std::list<item>::iterator b = ret.begin();
1554 b++;
1555 while( ret.size() > 1 ) {
1556 ret.front().charges += b->charges;
1557 b = ret.erase( b );
1558 }
1559 }
1560 lastconsumed = selected_comp.type;
1561 empty_buckets( *this );
1562 return ret;
1563}

1592static std::vector<avail_tool_comp>
1593find_tool_component( const Character *player_with_inv, const std::vector<tool_comp> &tools,
1594 int batch,
1595 const inventory &map_inv, cost_adjustment charge_mod )
1596{
1597 std::vector<avail_tool_comp> available_tools;
1598
1599 auto calc_charges = [&]( const tool_comp & t ) {
1600 const int full_craft_charges = std::max( 1, t.count * batch );
1601 switch( charge_mod ) {
1602 case cost_adjustment::none:
1603 return std::make_pair( charges_for_complete( full_craft_charges ),
1604 charges_for_complete( full_craft_charges ) );
1605 case cost_adjustment::start_only:
1606 return std::make_pair( charges_for_starting( full_craft_charges ),
1607 charges_for_complete( full_craft_charges ) );
1608 case cost_adjustment::continue_only:
1609 return std::make_pair( charges_for_continuing( full_craft_charges ),
1610 charges_for_complete( full_craft_charges ) );
1611 }
1612
1613 debugmsg( "Invalid tool_charge_mod" );
1614 return std::make_pair( INT_MAX, INT_MAX );
1615 };
1616
1617 bool found_nocharge = false;
1618 // Use charges of any tools that require charges used
1619 for( auto it = tools.begin(); it != tools.end() && !found_nocharge; ++it ) {
1620 itype_id type = it->type;
1621 if( it->count > 0 ) {
1622 const std::pair<int, int> &expected_count = calc_charges( *it );
1623 const int count = expected_count.first;
1624 const int ideal = expected_count.second;
1625 if( player_with_inv ) {
1626 if( player_with_inv->has_charges( type, count ) ) {
1627 int total_charges = player_with_inv->charges_of( type );
1628 comp_selection<tool_comp> sel( use_from_player, *it );
1629 available_tools.emplace_back( sel, total_charges, ideal );
1630 }
1631 }
1632 if( map_inv.has_charges( type, count ) ) {
1633 int total_charges = map_inv.charges_of( type );
1634 comp_selection<tool_comp> sel( use_from_map, *it );
1635 available_tools.emplace_back( sel, total_charges, ideal );
1636 }
1637 } else if( ( player_with_inv && player_with_inv->has_amount( type, 1 ) )
1638 || map_inv.has_tools( type, 1 ) ) {
1639 comp_selection<tool_comp> sel( use_from_none, *it );
1640 available_tools.emplace_back( sel, 0, 0 );
1641 }
1642 }
1643
1644 std::sort( available_tools.begin(), available_tools.end(),
1645 []( const avail_tool_comp & lhs, const avail_tool_comp & rhs ) {
1646 if( lhs.comp.use_from == use_from_none && rhs.comp.use_from != use_from_none ) {
1647 return true;
1648 }
1649 if( rhs.comp.use_from == use_from_none ) {
1650 return false;
1651 }
1652 if( lhs.charges >= lhs.ideal && rhs.charges < rhs.ideal ) {
1653 return true;
1654 }
1655 if( lhs.charges < lhs.ideal && rhs.charges >= rhs.ideal ) {
1656 return false;
1657 }
1658 // We want "bigger" tools first because those are likely to be vehicle mounted
1659 return static_cast<float>( lhs.ideal ) / lhs.charges > static_cast<float>
1660 ( rhs.ideal ) / rhs.charges;
1661 } );
1662 return available_tools;
1663}

1750bool player::craft_consume_tools( item &craft, int mulitplier, bool start_craft )
1751{
1752 if( !craft.is_craft() ) {
1753 debugmsg( "craft_consume_tools() called on non-craft '%s.' Aborting.", craft.tname() );
1754 return false;
1755 }
1756 if( has_trait( trait_DEBUG_HS ) ) {
1757 return true;
1758 }
1759 if( start_craft && mulitplier > 1 ) {
1760 debugmsg( "start_craft is true, but multiplier is %d > 1", mulitplier );
1761 return false;
1762 }
1763
1764 const auto calc_charges = [&craft, &start_craft, &mulitplier]( int charges ) {
1765 int ret = charges;
1766
1767 if( charges <= 0 ) {
1768 return ret;
1769 }
1770
1771 // Account for batch size
1772 int full_cost = charges * craft.charges;
1773
1774 if( start_craft ) {
1775 return crafting::charges_for_starting( full_cost );
1776 } else {
1777 // In case more than 5% progress was accomplished in one turn
1778 return crafting::charges_for_continuing( full_cost ) * mulitplier;
1779 }
1780 };
1781
1782 // First check if we still have our cached selections
1783 const std::vector<comp_selection<tool_comp>> &cached_tool_selections =
1784 craft.get_cached_tool_selections();
1785
1786 inventory map_inv;
1787 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1788
1789 for( const comp_selection<tool_comp> &tool_sel : cached_tool_selections ) {
1790 itype_id type = tool_sel.comp.type;
1791 if( tool_sel.comp.count > 0 ) {
1792 const int count = calc_charges( tool_sel.comp.count );
1793 switch( tool_sel.use_from ) {
1794 case use_from_player:
1795 if( !has_charges( type, count ) ) {
1796 add_msg_player_or_npc(
1797 _( "You have insufficient %s charges and can't continue crafting" ),
1798 _( "<npcname> has insufficient %s charges and can't continue crafting" ),
1799 item::nname( type ) );
1800 craft.set_tools_to_continue( false );
1801 return false;
1802 }
1803 break;
1804 case use_from_map:
1805 if( !map_inv.has_charges( type, count ) ) {
1806 add_msg_player_or_npc(
1807 _( "You have insufficient %s charges and can't continue crafting" ),
1808 _( "<npcname> has insufficient %s charges and can't continue crafting" ),
1809 item::nname( type ) );
1810 craft.set_tools_to_continue( false );
1811 return false;
1812 }
1813 break;
1814 case use_from_both:
1815 case use_from_none:
1816 case cancel:
1817 case num_usages:
1818 break;
1819 }
1820 } else if( !has_amount( type, 1 ) && !map_inv.has_tools( type, 1 ) ) {
1821 add_msg_player_or_npc(
1822 _( "You no longer have a %s and can't continue crafting" ),
1823 _( "<npcname> no longer has a %s and can't continue crafting" ),
1824 item::nname( type ) );
1825 craft.set_tools_to_continue( false );
1826 return false;
1827 }
1828 }
1829
1830 // We have the selections, so consume them
1831 for( const comp_selection<tool_comp> &tool : cached_tool_selections ) {
1832 comp_selection<tool_comp> to_consume = tool;
1833 to_consume.comp.count = calc_charges( to_consume.comp.count );
1834 consume_tools( to_consume, 1 );
1835 }
1836 return true;
1837}

1875ret_val<bool> crafting::can_disassemble( const Character &who, const item &obj,
1876 const inventory &inv )
1877{
1878 const auto &r = recipe_dictionary::get_uncraft( obj.typeId() );
1879
1880 if( !r || obj.has_flag( flag_ETHEREAL_ITEM ) ) {
1881 return ret_val<bool>::make_failure( _( "You cannot disassemble this." ) );
1882 }
1883
1884 // check sufficient light
1885 if( lighting_crafting_speed_multiplier( who, r ) == 0.0f ) {
1886 return ret_val<bool>::make_failure( _( "You can't see to craft!" ) );
1887 }
1888
1889 // refuse to disassemble rotten items
1890 const item *food = obj.get_food();
1891 if( ( obj.goes_bad() && obj.rotten() ) || ( food && food->goes_bad() && food->rotten() ) ) {
1892 return ret_val<bool>::make_failure( _( "It's rotten, I'm not taking that apart." ) );
1893 }
1894
1895 // refuse to disassemble items containing monsters/pets
1896 std::string monster = obj.get_var( "contained_name" );
1897 if( !monster.empty() ) {
1898 return ret_val<bool>::make_failure( _( "You must remove the %s before you can disassemble this." ),
1899 monster );
1900 }
1901
1902 if( obj.count_by_charges() ) {
1903 int batch_size = r.disassembly_batch_size();
1904 if( obj.charges < batch_size ) {
1905 auto msg = vgettext( "You need at least %d charge of %s.",
1906 "You need at least %d charges of %s.", batch_size );
1907 return ret_val<bool>::make_failure( msg, batch_size, obj.tname() );
1908 }
1909 }
1910 const auto &dis = r.disassembly_requirements();
1911
1912 for( const auto &opts : dis.get_qualities() ) {
1913 for( const auto &qual : opts ) {
1914 if( !qual.has( inv, return_true<item> ) ) {
1915 // Here should be no dot at the end of the string as 'to_string()' provides it.
1916 return ret_val<bool>::make_failure( _( "You need %s" ), qual.to_string() );
1917 }
1918 }
1919 }
1920
1921 for( const auto &opts : dis.get_tools() ) {
1922 const bool found = std::any_of( opts.begin(), opts.end(),
1923 [&]( const tool_comp & tool ) {
1924 return ( tool.count <= 0 && inv.has_tools( tool.type, 1 ) ) ||
1925 ( tool.count > 0 && inv.has_charges( tool.type, tool.count ) );
1926 } );
1927
1928 if( !found ) {
1929 const tool_comp &tool_required = opts.front();
1930 if( tool_required.count <= 0 ) {
1931 return ret_val<bool>::make_failure( _( "You need %s." ),
1932 item::nname( tool_required.type ) );
1933 } else {
1934 //~ %1$s: tool name, %2$d: needed charges
1935 return ret_val<bool>::make_failure( vgettext( "You need a %1$s with %2$d charge.",
1936 "You need a %1$s with %2$d charges.", tool_required.count ),
1937 item::nname( tool_required.type ),
1938 tool_required.count );
1939 }
1940 }
1941 }
1942
1943 return ret_val<bool>::make_success();
1944}

1969 if( !query_yn( _( "Disassembling the %s may anger the people who own it, continue?" ),

1990 }
1991 }
1992 }
1993
1994 int batch_size = r.disassembly_batch_size();
1995
1996 if( interactive && get_option<bool>( "QUERY_DISASSEMBLE" ) ) {
1997 std::string msg;
1998 msg += string_format( _( "Disassembling the %s may yield:\n" ),
1999 colorize( obj.tname(), obj.color_in_inventory() ) );
2000 const std::vector<item_comp> components = obj.get_uncraft_components();
2001 for( const item_comp &component : components ) {
2002 msg += "- " + component.to_string() + "\n";

2005 msg += string_format( _( "(per batch of %d)\n" ), batch_size );

2008 msg += _( "Really disassemble?\n" );

2020 } else if( num_batches == 1 || autoselect_max ) {
2021 // Skip prompt if can only do 1, or if we're doing all
2022 res.batches = num_batches;
2023 } else {
2024 string_input_popup popup_input;
2025 std::string title;
2026 std::string descr;
2027 if( batch_size != 1 ) {
2028 descr = string_format( _( "Disassembly will be done in batches of %d." ), batch_size );
2029 title = string_format(
2030 _( "Disassemble how many batches of %s [MAX: %d]: " ),
2031 obj.type_name( 1 ), num_batches );
2032 } else {
2033 title = string_format( _( "Disassemble how many %s [MAX: %d]: " ),
2034 obj.type_name( 1 ), num_batches );
2035 }
2036 int result = num_batches;
2037 popup_input.title( title ).description( descr ).edit( result );
2038 if( popup_input.canceled() || result <= 0 ) {
2039 add_msg( _( "Never mind." ) );
2040 return res;
2041 }
2042 res.batches = std::min( result, num_batches );
2043 }

2115
2116 // Get the proper recipe - the one for disassembly, not assembly
2117 const auto dis_requirements = dis.disassembly_requirements();
2118 const bool filthy = org_item.is_filthy();
2119
2120 // Make a copy to keep its data (damage/components) even after it
2121 // has been removed.
2122 item dis_item = org_item;
2123
2124 float component_success_chance = std::min( std::pow( 0.8, dis_item.damage_level( 4 ) ), 1.0 );

2137 target.loc.remove_item();

2210 if( dis.difficulty != 0 && !comp_success ) {

2213 }
