83{
84 all_sequences.finalize();

86 // Validate explicit sequences
87 for( const construction_sequence &seq : all_sequences.get_all() ) {
88 if( seq.blacklisted ) {
89 continue;
90 }
91 if( !seq.post_furniture.is_empty() && !seq.post_furniture.is_valid() ) {
92 debugmsg( R"(Construction sequence "%s" defines invalid post_furniture "%s".)",
93 seq.id, seq.post_furniture );
94 continue;
95 }
96 if( !seq.post_terrain.is_empty() && !seq.post_terrain.is_valid() ) {
97 debugmsg( R"(Construction sequence "%s" defines invalid post_terrain "%s".)",
98 seq.id, seq.post_terrain );
99 continue;
100 }
101 if( seq.post_furniture.is_empty() == seq.post_terrain.is_empty() ) {
102 debugmsg( R"(Construction sequence "%s" must define post_furniture OR post_terrain.)",
103 seq.id );
104 continue;
105 }
106 if( !seq.elems.empty() ) {
107 bool constr_ok = true;
108 for( const construction_str_id &c_id : seq.elems ) {
109 if( !c_id.is_valid() ) {
110 debugmsg( R"(Construction sequence "%s" contains invalid construction id "%s".)",
111 seq.id, c_id );
112 constr_ok = false;
113 break;
114 }
115 if( c_id->is_blacklisted() ) {
116 debugmsg( R"(Construction sequence "%s" contains blacklisted construction id "%s".)",
117 seq.id, c_id );
118 constr_ok = false;
119 break;
120 }
121 if( !has_valid_category( *c_id ) ) {
122 debugmsg( R"(Construction sequence "%s" should no contain constructions from REPAIR category ("%s").)",
123 seq.id, c_id );
124 constr_ok = false;
125 break;
126 }
127 if( !has_valid_result( *c_id ) ) {
128 debugmsg( R"(In construction sequence "%s" step "%s" produces no furniture or terrain.)",
129 seq.id, c_id
130 );
131 constr_ok = false;
132 break;
133 }
134 }
135 if( !constr_ok ) {
136 continue;
137 }
138 const construction &first = *seq.elems.front();
139 if( !is_valid_first_step( first ) ) {
140 debugmsg( R"(First entry "%s" in construction sequence "%s" )"
141 R"(must have t_dirt as valid prerequisite.)",
142 first.id, seq.id );
143 continue;
144 }
145 const construction &last = *seq.elems.back();
146 if( last.post_terrain != seq.post_terrain || last.post_furniture != seq.post_furniture ) {
147 debugmsg( R"(Last entry "%s" in construction sequence "%s" )"
148 R"(must produce terrain or furniture specified by the sequence.)",
149 last.id, seq.id );
150 }
151 for( size_t i = 1; i < seq.elems.size(); i++ ) {
152 const construction &c1 = *seq.elems[i - 1];
153 const construction &c2 = *seq.elems[i];
154
155 if( c1.post_furniture != c2.pre_furniture ) {
156 debugmsg( R"(In construction sequence "%s" post_furniture and pre_furniture )"
157 R"(don't match when going from "%s" to "%s".)",
158 seq.id, c1.id, c2.id
159 );
160 constr_ok = false;
161 break;
162 }
163 if( c1.post_terrain != c2.pre_terrain ) {
164 debugmsg( R"(In construction sequence "%s" post_terrain and pre_terrain )"
165 R"(don't match when going from "%s" to "%s".)",
166 seq.id, c1.id, c2.id
167 );
168 constr_ok = false;
169 break;
170 }
171 }
172 if( !constr_ok ) {
173 continue;
174 }
175 }
176 cata::optional<construction_sequence_id> conflict_id;
177 if( !seq.post_terrain.is_empty() ) {
178 auto res = sequences_for_ter.emplace( seq.post_terrain, seq.id.id() );
179 if( !res.second ) {
180 conflict_id = res.first->second.id();
181 }
182 } else {
183 auto res = sequences_for_furn.emplace( seq.post_furniture, seq.id.id() );
184 if( !res.second ) {
185 conflict_id = res.first->second.id();
186 }
187 }
188 if( conflict_id ) {
189 debugmsg( R"(Construction sequence "%s" defines same result as another existing sequence "%s".)",
190 seq.id, *conflict_id
191 );
192 }
193 }
194
195 // Generate auto 1-element sequences
196 std::map<ter_str_id, std::vector<construction_sequence>> ter_map;
197 std::map<furn_str_id, std::vector<construction_sequence>> furn_map;
198 for( const construction_id &c_id : constructions::get_all_sorted() ) {
199 const construction &c = *c_id;
200 if( !is_valid_first_step( c ) || !has_valid_result( c ) || !has_valid_category( c ) ) {
201 // Recipe must be a valid first construction step to be turned into sequence
202 continue;
203 }
204 std::vector<construction_sequence> *seq_vec = nullptr;
205 if( !c.post_terrain.is_empty() ) {
206 auto it = ter_map.emplace( c.post_terrain, std::vector<construction_sequence>() );
207 seq_vec = &it.first->second;
208 } else {
209 auto it = furn_map.emplace( c.post_furniture, std::vector<construction_sequence>() );
210 seq_vec = &it.first->second;
211 }
212
213 construction_sequence seq;
214 seq.was_loaded = true;
215 seq.id = construction_sequence_id( "#auto#_" + c.id.str() );
216 seq.elems.push_back( c.id );
217 seq_vec->push_back( seq );
218 }
219 for( const auto &it : ter_map ) {
220 if( it.second.size() != 1 ) {
221 // Don't autogenerate sequence if multiple constructions
222 // can be turned into a sequence, let user solve conflict.
223 continue;
224 }
225 // Don't autogenerate sequence if there is already an explicitly defined sequence
226 if( lookup_sequence( it.first ) ) {
227 continue;
228 }
229 const construction_sequence &ref = all_sequences.insert( it.second.front() );
230 sequences_for_ter.emplace( it.first, ref.id.id() );
231 }
232 for( const auto &it : furn_map ) {
233 if( it.second.size() != 1 ) {
234 // Don't autogenerate sequence if multiple constructions
235 // can be turned into a sequence, let user solve conflict.
236 continue;
237 }
238 // Don't autogenerate sequence if there is already an explicitly defined sequence
239 if( lookup_sequence( it.first ) ) {
240 continue;
241 }
242 const construction_sequence &ref = all_sequences.insert( it.second.front() );
243 sequences_for_furn.emplace( it.first, ref.id.id() );
244 }
