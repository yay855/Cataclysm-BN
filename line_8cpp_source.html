18double iso_tangent( double distance, units::angle vertex )
19{
20 // we can use the cosine formula (a² = b² + c² - 2bc⋅cosθ) to calculate the tangent
21 return std::sqrt( 2 * std::pow( distance, 2 ) * ( 1 - cos( vertex ) ) );

24void bresenham( const point &p1, const point &p2, int t,
25 const std::function<bool( const point & )> &interact )
26{
27 // The slope components.
28 const point d = p2 - p1;
29 // Signs of slope values.
30 const point s( ( d.x == 0 ) ? 0 : sgn( d.x ), ( d.y == 0 ) ? 0 : sgn( d.y ) );
31 // Absolute values of slopes x2 to avoid rounding errors.
32 const point a = d.abs() * 2;
33
34 point cur = p1;
35
36 if( a.x == a.y ) {
37 while( cur.x != p2.x ) {
38 cur.y += s.y;
39 cur.x += s.x;
40 if( !interact( cur ) ) {
41 break;
42 }
43 }
44 } else if( a.x > a.y ) {
45 while( cur.x != p2.x ) {
46 if( t > 0 ) {
47 cur.y += s.y;
48 t -= a.x;
49 }
50 cur.x += s.x;
51 t += a.y;
52 if( !interact( cur ) ) {
53 break;
54 }
55 }
56 } else {
57 while( cur.y != p2.y ) {
58 if( t > 0 ) {
59 cur.x += s.x;
60 t -= a.y;
61 }
62 cur.y += s.y;
63 t += a.x;
64 if( !interact( cur ) ) {
65 break;
66 }
67 }
68 }

258std::vector<point> line_to( const point &p1, const point &p2, int t )
259{
260 std::vector<point> line;
261 // Preallocate the number of cells we need instead of allocating them piecewise.
262 const int numCells = square_dist( p1, p2 );
263 if( numCells == 0 ) {
264 line.push_back( p1 );
265 } else {
266 line.reserve( numCells );
267 bresenham( p1, p2, t, [&line]( const point & new_point ) {
268 line.push_back( new_point );
269 return true;
270 } );
271 }
272 return line;

326// This more general version of this function gives correct values for larger values.
327unsigned make_xyz( const tripoint &p )
328{
329 static constexpr double sixteenth_arc = M_PI / 8;
330 int vertical_position = ( ( p.z > 0 ) ? 2u : ( p.z < 0 ) ? 1u : 0u ) * 9u;
331 if( p.xy() == point_zero ) {
332 return vertical_position;
333 }
334 // Get the arctan of the angle and divide by approximately 22.5 deg to get the octant.
335 // the angle is in, then truncate it and map to the right direction.
336 // You can read 'octant' as being "number of 22.5 degree sections away from due south".
337 // FIXME: atan2 normally takes arguments in ( y, x ) order. This is
338 // passing ( x, y ).
339 int octant = atan2( p.x, p.y ) / sixteenth_arc;
340 switch( octant ) {
341 case 0:
342 return direction::SOUTH + vertical_position;
343 case 1:
344 case 2:
345 return direction::SOUTHEAST + vertical_position;
346 case 3:
347 case 4:
348 return direction::EAST + vertical_position;
349 case 5:
350 case 6:
351 return direction::NORTHEAST + vertical_position;
352 case -1:
353 case -2:

355 case -3:
356 case -4:
357 return direction::WEST + vertical_position;
358 case -5:
359 case -6:

361 case 7:
362 case 8:
363 case -7:
364 case -8:

370// returns the normalized dx, dy, dz for the current line vector.
371static std::tuple<double, double, double> slope_of( const std::vector<tripoint> &line )
372{
373 assert( !line.empty() && line.front() != line.back() );
374 const double len = trig_dist( line.front(), line.back() );
375 double normDx = ( line.back().x - line.front().x ) / len;
376 double normDy = ( line.back().y - line.front().y ) / len;
377 double normDz = ( line.back().z - line.front().z ) / len;
378 // slope of <x, y, z>
379 return std::make_tuple( normDx, normDy, normDz );
380}

382float get_normalized_angle( const point &start, const point &end )
383{
384 // Taking the abs value of the difference puts the values in the first quadrant.
385 const float absx = std::abs( std::max( start.x, end.x ) - std::min( start.x, end.x ) );
386 const float absy = std::abs( std::max( start.y, end.y ) - std::min( start.y, end.y ) );
387 const float max = std::max( absx, absy );
388 if( max == 0 ) {
389 return 0;
390 }
391 const float min = std::min( absx, absy );
392 return min / max;
393}

547// Cardinals are cardinals. Result is cardinal and adjacent sub-cardinals.
548// Sub-Cardinals are sub-cardinals && abs(x) == abs(y). Result is sub-cardinal and adjacent cardinals.
549// Sub-sub-cardinals are direction && abs(x) > abs(y) or vice versa.
550// Result is adjacent cardinal and sub-cardinals, plus the nearest other cardinal.
551// e.g. if the direction is NNE, also include E.
552std::vector<tripoint> squares_closer_to( const tripoint &from, const tripoint &to )
553{
554 std::vector<tripoint> adjacent_closer_squares;
555 const tripoint d( -from + to );
556 const point a( std::abs( d.x ), std::abs( d.y ) );
557 if( d.z != 0 ) {
558 adjacent_closer_squares.push_back( from + tripoint( sgn( d.x ), sgn( d.y ), sgn( d.z ) ) );
559 }
560 if( a.x > a.y ) {
561 // X dominant.
562 adjacent_closer_squares.push_back( from + point( sgn( d.x ), 0 ) );
563 adjacent_closer_squares.push_back( from + point( sgn( d.x ), 1 ) );
564 adjacent_closer_squares.push_back( from + point( sgn( d.x ), -1 ) );
565 if( d.y != 0 ) {
566 adjacent_closer_squares.push_back( from + point( 0, sgn( d.y ) ) );
567 }
568 } else if( a.x < a.y ) {
569 // Y dominant.
570 adjacent_closer_squares.push_back( from + point( 0, sgn( d.y ) ) );
571 adjacent_closer_squares.push_back( from + point( 1, sgn( d.y ) ) );
572 adjacent_closer_squares.push_back( from + point( -1, sgn( d.y ) ) );
573 if( d.x != 0 ) {
574 adjacent_closer_squares.push_back( from + point( sgn( d.x ), 0 ) );
575 }
576 } else if( d.x != 0 ) {
577 // Pure diagonal.
578 adjacent_closer_squares.push_back( from + point( sgn( d.x ), sgn( d.y ) ) );
579 adjacent_closer_squares.push_back( from + point( sgn( d.x ), 0 ) );
580 adjacent_closer_squares.push_back( from + point( 0, sgn( d.y ) ) );
581 }
582
583 return adjacent_closer_squares;
584}

586// Returns a vector of the adjacent square in the direction of the target,
587// and the two squares flanking it.
588std::vector<point> squares_in_direction( const point &p1, const point &p2 )
589{
590 int junk = 0;
591 point center_square = line_to( p1, p2, junk )[0];
592 std::vector<point> adjacent_squares;
593 adjacent_squares.push_back( center_square );
594 if( p1.x == center_square.x ) {
595 // Horizontally adjacent.
596 adjacent_squares.push_back( point( p1.x + 1, center_square.y ) );
597 adjacent_squares.push_back( point( p1.x - 1, center_square.y ) );
598 } else if( p1.y == center_square.y ) {
599 // Vertically adjacent.
600 adjacent_squares.push_back( point( center_square.x, p1.y + 1 ) );
601 adjacent_squares.push_back( point( center_square.x, p1.y - 1 ) );
602 } else {
603 // Diagonally adjacent.
604 adjacent_squares.push_back( point( p1.x, center_square.y ) );
605 adjacent_squares.push_back( point( center_square.x, p1.y ) );
606 }
607 return adjacent_squares;
608}

751void calc_ray_end( units::angle angle, const int range, const tripoint &p, tripoint &out )
752{
753 // forces input angle to be between 0 and 360, calculated from actual input
754 angle = fmod( angle, 360_degrees );
755 if( angle < 0_degrees ) {
756 angle += 360_degrees;
757 }
758 out.z = p.z;
759 if( trigdist ) {
760 out.x = p.x + range * cos( angle );
761 out.y = p.y + range * sin( angle );
762 } else {
763 int mult = 0;
764 if( angle >= 135_degrees && angle <= 315_degrees ) {
765 mult = -1;
766 } else {
767 mult = 1;
768 }
769
770 if( angle <= 45_degrees || ( 135_degrees <= angle && angle <= 215_degrees ) ||
771 315_degrees < angle ) {
772 out.x = p.x + range * mult;
773 out.y = p.y + range * tan( angle ) * mult;
774 } else {
775 out.x = p.x + range * 1 / tan( angle ) * mult;
776 out.y = p.y + range * mult;
777 }
778 }
779}
