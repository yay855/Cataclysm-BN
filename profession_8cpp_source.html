386std::list<item> profession::items( bool male, const std::vector<trait_id> &traits ) const
387{
388 std::list<item> result;
389 auto add_legacy_items = [&result]( const itypedecvec & vec ) {
390 for( const itypedec &elem : vec ) {
391 item it( elem.type_id, advanced_spawn_time(), item::default_charges_tag {} );
392 if( !elem.snip_id.is_null() ) {
393 it.set_snippet( elem.snip_id );
394 }
395 it = it.in_its_container();
396 result.push_back( it );
397 }
398 };
399
400 add_legacy_items( legacy_starting_items );
401 add_legacy_items( male ? legacy_starting_items_male : legacy_starting_items_female );
402
403 const std::vector<item> group_both = item_group::items_from( _starting_items,
404 advanced_spawn_time() );
405 const std::vector<item> group_gender = item_group::items_from( male ? _starting_items_male :
406 _starting_items_female, advanced_spawn_time() );
407 result.insert( result.begin(), group_both.begin(), group_both.end() );
408 result.insert( result.begin(), group_gender.begin(), group_gender.end() );
409
410 std::vector<itype_id> bonus = item_substitutions.get_bonus_items( traits );
411 for( const itype_id &elem : bonus ) {
412 if( elem != no_bonus ) {
413 result.push_back( item( elem, advanced_spawn_time(), item::default_charges_tag {} ) );
414 }
415 }
416 for( auto iter = result.begin(); iter != result.end(); ) {
417 const auto sub = item_substitutions.get_substitution( *iter, traits );
418 if( !sub.empty() ) {
419 result.insert( result.begin(), sub.begin(), sub.end() );
420 iter = result.erase( iter );
421 } else {
422 ++iter;
423 }
424 }
425 for( item &it : result ) {
426 clear_faults( it );
427 if( it.is_holster() && it.contents.num_item_stacks() == 1 ) {
428 clear_faults( it.contents.front() );
429 }
430 if( it.has_flag( "VARSIZE" ) ) {
431 it.set_flag( "FIT" );
432 }
433 }
434
435 if( result.empty() ) {
436 // No need to do the below stuff. Plus it would cause said below stuff to crash
437 return result;
438 }
439
440 // Merge charges for items that stack with each other
441 for( auto outer = result.begin(); outer != result.end(); ++outer ) {
442 if( !outer->count_by_charges() ) {
443 continue;
444 }
445 for( auto inner = std::next( outer ); inner != result.end(); ) {
446 if( outer->stacks_with( *inner ) ) {
447 outer->merge_charges( *inner );
448 inner = result.erase( inner );
449 } else {
450 ++inner;
451 }
452 }
453 }
454
455 result.sort( []( const item & first, const item & second ) {
456 return first.get_layer() < second.get_layer();
457 } );
458 return result;
459}

554void json_item_substitution::load( const JsonObject &jo )
555{
556 const bool item_mode = jo.has_string( "item" );
557 const std::string title = jo.get_string( item_mode ? "item" : "trait" );
558
559 auto check_duplicate_item = [&]( const itype_id & it ) {
560 return substitutions.find( it ) != substitutions.end() ||
561 std::find_if( bonuses.begin(), bonuses.end(),
562 [&it]( const std::pair<itype_id, trait_requirements> &p ) {
563 return p.first == it;
564 } ) != bonuses.end();
565 };
566 if( item_mode && check_duplicate_item( itype_id( title ) ) ) {
567 jo.throw_error( "Duplicate definition of item" );
568 }
569
570 if( item_mode ) {
571 if( jo.has_member( "bonus" ) ) {
572 bonuses.emplace_back( itype_id( title ),
573 trait_requirements( jo.get_object( "bonus" ) ) );
574 }
575
576 for( const JsonValue sub : jo.get_array( "sub" ) ) {
577 substitution s;
578 JsonObject obj = sub.get_object();
579 s.trait_reqs = trait_requirements( obj );
580 for( const JsonValue info : obj.get_array( "new" ) ) {
581 s.infos.emplace_back( info );
582 }
583 substitutions[itype_id( title )].push_back( s );
584 }
585 } else {
586 for( const JsonObject sub : jo.get_array( "sub" ) ) {
587 substitution s;
588 itype_id old_it;
589 sub.read( "item", old_it, true );
590 if( check_duplicate_item( old_it ) ) {
591 sub.throw_error( "Duplicate definition of item" );
592 }
593 s.trait_reqs.present.push_back( trait_id( title ) );
594 for( const JsonValue info : sub.get_array( "new" ) ) {
595 s.infos.emplace_back( info );
596 }
597 substitutions[old_it].push_back( s );
598 }
599 }

602void json_item_substitution::check_consistency()
603{
604 auto check_if_trait = []( const trait_id & t ) {
605 if( !t.is_valid() ) {
606 debugmsg( "%s is not a trait", t.c_str() );
607 }
608 };
609 auto check_if_itype = []( const itype_id & i ) {
610 if( !i.is_valid() ) {
611 debugmsg( "%s is not an itype_id", i.c_str() );
612 }
613 };
614 auto check_trait_reqs = [&check_if_trait]( const trait_requirements & tr ) {
615 for( const trait_id &str : tr.present ) {
616 check_if_trait( str );
617 }
618 for( const trait_id &str : tr.absent ) {
619 check_if_trait( str );
620 }
621 };
622
623 for( const auto &pair : substitutions ) {
624 check_if_itype( pair.first );
625 for( const substitution &s : pair.second ) {
626 check_trait_reqs( s.trait_reqs );
627 for( const substitution::info &inf : s.infos ) {
628 check_if_itype( inf.new_item );
629 }
630 }
631 }
632 for( const auto &pair : bonuses ) {
633 check_if_itype( pair.first );
634 check_trait_reqs( pair.second );
635 }

638bool json_item_substitution::trait_requirements::meets_condition( const std::vector<trait_id>
639 &traits ) const
640{
641 const auto pred = [&traits]( const trait_id & s ) {
642 return std::find( traits.begin(), traits.end(), s ) != traits.end();
643 };
644 return std::all_of( present.begin(), present.end(), pred ) &&
645 std::none_of( absent.begin(), absent.end(), pred );
646}

648std::vector<item> json_item_substitution::get_substitution( const item &it,
649 const std::vector<trait_id> &traits ) const
650{
651 auto iter = substitutions.find( it.typeId() );
652 std::vector<item> ret;
653 if( iter == substitutions.end() ) {
654 for( const item *con : it.contents.all_items_top() ) {
655 const auto sub = get_substitution( *con, traits );
656 ret.insert( ret.end(), sub.begin(), sub.end() );
657 }
658 return ret;
659 }
660
661 const auto sub = std::find_if( iter->second.begin(),
662 iter->second.end(), [&traits]( const substitution & s ) {
663 return s.trait_reqs.meets_condition( traits );
664 } );
665 if( sub == iter->second.end() ) {
666 return ret;
667 }
668
669 const int old_amt = it.count();
670 for( const substitution::info &inf : sub->infos ) {
671 item result( inf.new_item, advanced_spawn_time() );
672 const int new_amt = std::max( 1, static_cast<int>( std::round( inf.ratio * old_amt ) ) );
673
674 if( !result.count_by_charges() ) {
675 for( int i = 0; i < new_amt; i++ ) {
676 ret.push_back( result.in_its_container() );
677 }
678 } else {
679 result.mod_charges( -result.charges + new_amt );
680 while( result.charges > 0 ) {
681 const item pushed = result.in_its_container();
682 ret.push_back( pushed );
683 result.mod_charges( pushed.contents.empty() ? -pushed.charges : -pushed.contents.back().charges );
684 }
685 }
686 }
