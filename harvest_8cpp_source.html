134void harvest_list::check_consistency()
135{
136 for( const auto &pr : harvest_all ) {
137 const auto &hl = pr.second;
138 const std::string hl_id = hl.id().c_str();
139 auto error_func = [&]( const harvest_entry & entry ) {
140 std::string errorlist;
141 bool item_valid = true;
142 if( ( entry.type == "bionic" || entry.type == "bionic_group" ) && entry.mass_ratio != 0.0f ) {
143 errorlist += string_format( "\"mass_ratio\" != 0.0, but type %s", entry.type.c_str() );
144 }
145 if( !( itype_id( entry.drop ).is_valid() || ( entry.type == "bionic_group" &&
146 item_group::group_is_defined( item_group_id( entry.drop ) ) ) ) ) {
147 item_valid = false;
148 errorlist += entry.drop;
149 }
150
151 // non butchery harvests need to be excluded
152 if( hl_id.substr( 0, 14 ) != "harvest_inline" ) {
153 if( entry.type == "null" ) {
154 if( !item_valid ) {
155 errorlist += ", ";
156 }
157 errorlist += "null type";
158 } else if( !( entry.type == "flesh" || entry.type == "bone" || entry.type == "skin" ||
159 entry.type == "offal" || entry.type == "bionic" || entry.type == "bionic_group" ) ) {
160 if( !item_valid ) {
161 errorlist += ", ";
162 }
163 errorlist += entry.type;
164 }
165 }
166 return errorlist;
167 };
168 const std::string errors = enumerate_as_string( hl.entries_.begin(), hl.entries_.end(),
169 error_func );
170 if( !errors.empty() ) {
171 debugmsg( "Harvest list %s has invalid entry: %s", hl_id, errors );

197std::string harvest_list::describe( int at_skill ) const
198{
199 if( empty() ) {
200 return "";
201 }
202
203 return enumerate_as_string( entries().begin(), entries().end(),
204 [at_skill]( const harvest_entry & en ) {
205 float min_f = en.base_num.first;
206 float max_f = en.base_num.second;
207 if( at_skill >= 0 ) {
208 min_f += en.scale_num.first * at_skill;
209 max_f += en.scale_num.second * at_skill;
210 } else {
211 max_f = en.max;
212 }
213 // TODO: Avoid repetition here by making a common harvest drop function
214 int max_drops = std::min<int>( en.max, std::round( std::max( 0.0f, max_f ) ) );
215 int min_drops = std::max<int>( 0.0f, std::round( std::min( min_f, max_f ) ) );
216 if( max_drops <= 0 ) {
217 return std::string();
218 }
219
220 std::string ss;
221 ss += "<bold>" + item::nname( itype_id( en.drop ), max_drops ) + "</bold>";
222 // If the number is unspecified, just list the type
223 if( max_drops >= 1000 && min_drops <= 0 ) {
224 return ss;
225 }
226 ss += ": ";
227 if( min_drops == max_drops ) {
228 ss += string_format( "<stat>%d</stat>", min_drops );
229 } else if( max_drops < 1000 ) {
230 ss += string_format( "<stat>%d-%d</stat>", min_drops, max_drops );
231 } else {
232 ss += string_format( "<stat>%d+</stat>", min_drops );
233 }
234
235 return ss;
236 } );
237}
