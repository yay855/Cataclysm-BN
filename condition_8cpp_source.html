63void read_condition( const JsonObject &jo, const std::string &member_name,
64 std::function<bool( const T & )> &condition, bool default_val )
65{
66 const auto null_function = [default_val]( const T & ) {
67 return default_val;
68 };
69
70 if( !jo.has_member( member_name ) ) {
71 condition = null_function;
72 } else if( jo.has_string( member_name ) ) {
73 const std::string type = jo.get_string( member_name );
74 conditional_t<T> sub_condition( type );
75 condition = [sub_condition]( const T & d ) {
76 return sub_condition( d );

78 } else if( jo.has_object( member_name ) ) {
79 JsonObject con_obj = jo.get_object( member_name );
80 conditional_t<T> sub_condition( con_obj );
81 condition = [sub_condition]( const T & d ) {
82 return sub_condition( d );
83 };
84 } else {
85 jo.throw_error( "invalid condition syntax", member_name );

421 bool is_npc )

424 const std::string &op = jo.get_string( "op" );
425 const int value = jo.get_int( "value" );

431

434 if( !var.empty() ) {

436 }
437
438 if( op == "==" ) {
439 return stored_value == value;
440
441 } else if( op == "!=" ) {
442 return stored_value != value;
443
444 } else if( op == "<=" ) {
445 return stored_value <= value;
446
447 } else if( op == ">=" ) {
448 return stored_value >= value;
449
450 } else if( op == "<" ) {
451 return stored_value < value;
452
453 } else if( op == ">" ) {
454 return stored_value > value;
455 }
456
457 return false;
