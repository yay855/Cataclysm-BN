493SDL_Rect get_android_render_rect( float DisplayBufferWidth, float DisplayBufferHeight )
494{
495 // If the display buffer aspect ratio is wider than the display,
496 // draw it at the top of the screen so it doesn't get covered up
497 // by the virtual keyboard. Otherwise just center it.
498 SDL_Rect dstrect;
499 float DisplayBufferAspect = DisplayBufferWidth / ( float )DisplayBufferHeight;
500 float WindowHeightLessShortcuts = ( float )WindowHeight;
501 if( !get_option<bool>( "ANDROID_SHORTCUT_OVERLAP" ) && quick_shortcuts_enabled ) {
502 WindowHeightLessShortcuts -= get_option<int>( "ANDROID_SHORTCUT_HEIGHT" );
503 }
504 float WindowAspect = WindowWidth / ( float )WindowHeightLessShortcuts;
505 if( WindowAspect < DisplayBufferAspect ) {
506 dstrect.x = 0;
507 dstrect.y = 0;
508 dstrect.w = WindowWidth;
509 dstrect.h = WindowWidth / DisplayBufferAspect;
510 } else {
511 dstrect.x = 0.5f * ( WindowWidth - ( WindowHeightLessShortcuts * DisplayBufferAspect ) );
512 dstrect.y = 0;
513 dstrect.w = WindowHeightLessShortcuts * DisplayBufferAspect;
514 dstrect.h = WindowHeightLessShortcuts;
515 }
516
517 // Make sure the destination rectangle fits within the visible area
518 if( get_option<bool>( "ANDROID_KEYBOARD_SCREEN_SCALE" ) && has_visible_display_frame ) {
519 int vdf_right = visible_display_frame.x + visible_display_frame.w;
520 int vdf_bottom = visible_display_frame.y + visible_display_frame.h;
521 if( vdf_right < dstrect.x + dstrect.w ) {
522 dstrect.w = vdf_right - dstrect.x;
523 }
524 if( vdf_bottom < dstrect.y + dstrect.h ) {
525 dstrect.h = vdf_bottom - dstrect.y;
526 }
527 }
528 return dstrect;
529}

685static cata::optional<std::pair<tripoint_abs_omt, std::string>> get_mission_arrow(
686 const inclusive_cuboid<tripoint> &overmap_area, const tripoint_abs_omt &center )
687{
688 if( get_avatar().get_active_mission() == nullptr ) {
689 return cata::nullopt;
690 }
691 if( !get_avatar().get_active_mission()->has_target() ) {
692 return cata::nullopt;
693 }
694 const tripoint_abs_omt mission_target = get_avatar().get_active_mission_target();
695
696 std::string mission_arrow_variant;
697 if( overmap_area.contains( mission_target.raw() ) ) {
698 mission_arrow_variant = "mission_cursor";
699 return std::make_pair( mission_target, mission_arrow_variant );
700 }
701
702 inclusive_rectangle<point> area_flat( overmap_area.p_min.xy(), overmap_area.p_max.xy() );
703 if( area_flat.contains( mission_target.raw().xy() ) ) {
704 int area_z = center.z();
705 if( mission_target.z() > area_z ) {
706 mission_arrow_variant = "mission_arrow_up";
707 } else {
708 mission_arrow_variant = "mission_arrow_down";
709 }
710 return std::make_pair( tripoint_abs_omt( mission_target.xy(), area_z ), mission_arrow_variant );
711 }
712
713 const std::vector<tripoint> traj = line_to( center.raw(),
714 tripoint( mission_target.raw().xy(), center.raw().z ) );
715
716 if( traj.empty() ) {
717 debugmsg( "Failed to gen overmap mission trajectory %s %s",
718 center.to_string(), mission_target.to_string() );
719 return cata::nullopt;
720 }
721
722 tripoint arr_pos = traj[0];
723 for( auto it = traj.rbegin(); it != traj.rend(); it++ ) {
724 if( overmap_area.contains( *it ) ) {
725 arr_pos = *it;
726 break;
727 }
728 }
729
730 const int north_border_y = ( overmap_area.p_max.y - overmap_area.p_min.y ) / 3;
731 const int south_border_y = north_border_y * 2;
732 const int west_border_x = ( overmap_area.p_max.x - overmap_area.p_min.x ) / 3;
733 const int east_border_x = west_border_x * 2;
734
735 tripoint north_pmax( overmap_area.p_max );
736 north_pmax.y = overmap_area.p_min.y + north_border_y;
737 tripoint south_pmin( overmap_area.p_min );
738 south_pmin.y += south_border_y;
739 tripoint west_pmax( overmap_area.p_max );
740 west_pmax.x = overmap_area.p_min.x + west_border_x;
741 tripoint east_pmin( overmap_area.p_min );
742 east_pmin.x += east_border_x;
743
744 const inclusive_cuboid<tripoint> north_sector( overmap_area.p_min, north_pmax );
745 const inclusive_cuboid<tripoint> south_sector( south_pmin, overmap_area.p_max );
746 const inclusive_cuboid<tripoint> west_sector( overmap_area.p_min, west_pmax );
747 const inclusive_cuboid<tripoint> east_sector( east_pmin, overmap_area.p_max );
748
749 mission_arrow_variant = "mission_arrow_";
750 if( north_sector.contains( arr_pos ) ) {
751 mission_arrow_variant += 'n';
752 } else if( south_sector.contains( arr_pos ) ) {
753 mission_arrow_variant += 's';
754 }
755 if( west_sector.contains( arr_pos ) ) {
756 mission_arrow_variant += 'w';
757 } else if( east_sector.contains( arr_pos ) ) {
758 mission_arrow_variant += 'e';
759 }
760
761 return std::make_pair( tripoint_abs_omt( arr_pos ), mission_arrow_variant );
762}

764std::string cata_tiles::get_omt_id_rotation_and_subtile(
765 const tripoint_abs_omt &omp, int &rota, int &subtile )
766{
767 auto oter_at = []( const tripoint_abs_omt & p ) {
768 const oter_id &cur_ter = overmap_buffer.ter( p );
769
770 if( !uistate.overmap_show_forest_trails &&
771 is_ot_match( "forest_trail", cur_ter, ot_match_type::type ) ) {
772 return oter_id( "forest" );
773 }
774
775 return cur_ter;
776 };
777
778 oter_id ot_id = oter_at( omp );
779 const oter_t &ot = *ot_id;
780 oter_type_id ot_type_id = ot.get_type_id();
781 oter_type_t ot_type = *ot_type_id;
782
783 if( ot_type.has_connections() ) {
784 // This would be for connected terrain
785
786 // get terrain neighborhood
787 const oter_type_id neighborhood[4] = {
788 oter_at( omp + point_south )->get_type_id(),
789 oter_at( omp + point_east )->get_type_id(),
790 oter_at( omp + point_west )->get_type_id(),
791 oter_at( omp + point_north )->get_type_id()
792 };
793
794 char val = 0;
795
796 // populate connection information
797 for( int i = 0; i < 4; ++i ) {
798 if( ot_type.connects_to( neighborhood[i] ) ) {
799 val += 1 << i;
800 }
801 }
802
803 get_rotation_and_subtile( val, rota, subtile );
804 } else {
805 // 'Regular', nonlinear terrain only needs to worry about rotation, not
806 // subtile
807 ot.get_rotation_and_subtile( rota, subtile );
808 }
809
810 return ot_type_id.id().str();
811}

813static point draw_string( Font &font,
814 const SDL_Renderer_Ptr &renderer,
815 const GeometryRenderer_Ptr &geometry,
816 const std::string &str,
817 point p,
818 const unsigned char color )
819{
820 const char *cstr = str.c_str();
821 int len = str.length();
822 while( len > 0 ) {
823 const uint32_t ch32 = UTF8_getch( &cstr, &len );
824 const std::string ch = utf32_to_utf8( ch32 );
825 font.OutputChar( renderer, geometry, ch, p, color );
826 p.x += mk_wcwidth( ch32 ) * font.width;
827 }
828 return p;
829}

1234static bool draw_window( Font_Ptr &font, const catacurses::window &w, const point &offset )
1235{
1236 if( scaling_factor > 1 ) {
1237 SDL_RenderSetLogicalSize( renderer.get(), WindowWidth / scaling_factor,
1238 WindowHeight / scaling_factor );
1239 }
1240
1241 cata_cursesport::WINDOW *const win = w.get<cata_cursesport::WINDOW>();
1242 //Keeping track of the last drawn window
1243 const cata_cursesport::WINDOW *winBuffer = static_cast<cata_cursesport::WINDOW *>
1244 ( ::winBuffer.lock().get() );
1245 if( !fontScaleBuffer ) {
1246 fontScaleBuffer = tilecontext->get_tile_width();
1247 }
1248 const int fontScale = tilecontext->get_tile_width();
1249 //This creates a problem when map_font is different from the regular font
1250 //Specifically when showing the overmap
1251 //And in some instances of screen change, i.e. inventory.
1252 bool oldWinCompatible = false;
1253

1255 invalidate_framebuffer_proportion( win );
1256
1257 // use the oversize buffer when dealing with windows that can have a different font than the main text font
1258 bool use_oversized_framebuffer = g && ( w == g->w_terrain || w == g->w_overmap );
1259
1260 std::vector<curseline> &framebuffer = use_oversized_framebuffer ? oversized_framebuffer :
1261 terminal_framebuffer;
1262
1263 /*

1265 A number of windows are coexisting on the screen, so don't have to interfere.
1266
1267 g->w_terrain, g->w_minimap, g->w_HP, g->w_status, g->w_status2, g->w_messages,
1268 g->w_location, and g->w_minimap, can be buffered if either of them was
1269 the previous window.
1270
1271 g->w_overmap and g->w_omlegend are likewise.
1272
1273 Everything else works on strict equality because there aren't yet IDs for some of them.

1275 if( g && ( w == g->w_terrain || w == g->w_minimap ) ) {
1276 if( winBuffer == g->w_terrain || winBuffer == g->w_minimap ) {
1277 oldWinCompatible = true;
1278 }
1279 } else if( g && ( w == g->w_overmap || w == g->w_omlegend ) ) {
1280 if( winBuffer == g->w_overmap || winBuffer == g->w_omlegend ) {
1281 oldWinCompatible = true;
1282 }
1283 } else {
1284 if( win == winBuffer ) {
1285 oldWinCompatible = true;
1286 }
1287 }
1288
1289 // TODO: Get this from UTF system to make sure it is exactly the kind of space we need
1290 static const std::string space_string = " ";
1291
1292 bool update = false;
1293 for( int j = 0; j < win->height; j++ ) {
1294 if( !win->line[j].touched ) {
1295 continue;
1296 }
1297
1298 const int fby = win->pos.y + j;
1299 if( fby >= static_cast<int>( framebuffer.size() ) ) {
1300 // prevent indexing outside the frame buffer. This might happen for some parts of the window. FIX #28953.
1301 break;
1302 }
1303
1304 update = true;
1305 win->line[j].touched = false;
1306 for( int i = 0; i < win->width; i++ ) {
1307 const int fbx = win->pos.x + i;
1308 if( fbx >= static_cast<int>( framebuffer[fby].chars.size() ) ) {
1309 // prevent indexing outside the frame buffer. This might happen for some parts of the window.
1310 break;
1311 }
1312
1313 const cursecell &cell = win->line[j].chars[i];
1314
1315 const int drawx = offset.x + i * font->width;
1316 const int drawy = offset.y + j * font->height;
1317 if( drawx + font->width > WindowWidth || drawy + font->height > WindowHeight ) {
1318 // Outside of the display area, would not render anyway
1319 continue;
1320 }
1321
1322 // Avoid redrawing an unchanged tile by checking the framebuffer cache
1323 // TODO: handle caching when drawing normal windows over graphical tiles
1324 cursecell &oldcell = framebuffer[fby].chars[fbx];
1325
1326 if( oldWinCompatible && cell == oldcell && fontScale == fontScaleBuffer ) {
1327 continue;
1328 }

1332 continue; // second cell of a multi-cell character
1333 }

1336 if( cell.ch == space_string ) {

1340 }

1342 const catacurses::base_color FG = cell.FG;
1343 const catacurses::base_color BG = cell.BG;
1344 int cw = ( codepoint == UNKNOWN_UNICODE ) ? 1 : utf8_width( cell.ch );
1345 if( cw < 1 ) {
1346 // utf8_width() may return a negative width
1347 continue;
1348 }
1349 bool use_draw_ascii_lines_routine = get_option<bool>( "USE_DRAW_ASCII_LINES_ROUTINE" );
1350 unsigned char uc = static_cast<unsigned char>( cell.ch[0] );
1351 switch( codepoint ) {
1352 case LINE_XOXO_UNICODE:
1353 uc = LINE_XOXO_C;
1354 break;
1355 case LINE_OXOX_UNICODE:
1356 uc = LINE_OXOX_C;
1357 break;
1358 case LINE_XXOO_UNICODE:
1359 uc = LINE_XXOO_C;
1360 break;
1361 case LINE_OXXO_UNICODE:
1362 uc = LINE_OXXO_C;
1363 break;
1364 case LINE_OOXX_UNICODE:
1365 uc = LINE_OOXX_C;
1366 break;
1367 case LINE_XOOX_UNICODE:
1368 uc = LINE_XOOX_C;
1369 break;
1370 case LINE_XXXO_UNICODE:
1371 uc = LINE_XXXO_C;
1372 break;
1373 case LINE_XXOX_UNICODE:
1374 uc = LINE_XXOX_C;
1375 break;
1376 case LINE_XOXX_UNICODE:
1377 uc = LINE_XOXX_C;
1378 break;
1379 case LINE_OXXX_UNICODE:
1380 uc = LINE_OXXX_C;
1381 break;
1382 case LINE_XXXX_UNICODE:
1383 uc = LINE_XXXX_C;
1384 break;
1385 case UNKNOWN_UNICODE:
1386 use_draw_ascii_lines_routine = true;
1387 break;
1388 default:
1389 use_draw_ascii_lines_routine = false;
1390 break;
1391 }
1392 geometry->rect( renderer, point( drawx, drawy ), font->width * cw, font->height,

1394 if( use_draw_ascii_lines_routine ) {
1395 font->draw_ascii_lines( renderer, geometry, uc, point( drawx, drawy ), FG );

1399 }

1401 win->draw = false; //We drew the window, mark it as so
1402 //Keeping track of last drawn window and tilemode zoom level

1597static int HandleDPad()
1598{
1599 // Check if we have a gamepad d-pad event.
1600 if( SDL_JoystickGetHat( joystick, 0 ) != SDL_HAT_CENTERED ) {
1601 // When someone tries to press a diagonal, they likely will
1602 // press a single direction first. Wait a few milliseconds to
1603 // give them time to press both of the buttons for the diagonal.
1604 int button = SDL_JoystickGetHat( joystick, 0 );
1605 int lc = ERR;
1606 if( button == SDL_HAT_LEFT ) {
1607 lc = JOY_LEFT;
1608 } else if( button == SDL_HAT_DOWN ) {
1609 lc = JOY_DOWN;
1610 } else if( button == SDL_HAT_RIGHT ) {
1611 lc = JOY_RIGHT;
1612 } else if( button == SDL_HAT_UP ) {
1613 lc = JOY_UP;
1614 } else if( button == SDL_HAT_LEFTUP ) {
1615 lc = JOY_LEFTUP;
1616 } else if( button == SDL_HAT_LEFTDOWN ) {
1617 lc = JOY_LEFTDOWN;
1618 } else if( button == SDL_HAT_RIGHTUP ) {
1619 lc = JOY_RIGHTUP;
1620 } else if( button == SDL_HAT_RIGHTDOWN ) {
1621 lc = JOY_RIGHTDOWN;
1622 }
1623
1624 if( delaydpad == std::numeric_limits<Uint32>::max() ) {
1625 delaydpad = SDL_GetTicks() + dpad_delay;
1626 queued_dpad = lc;
1627 }
1628
1629 // Okay it seems we're ready to process.
1630 if( SDL_GetTicks() > delaydpad ) {
1631
1632 if( lc != ERR ) {
1633 if( dpad_continuous && ( lc & lastdpad ) == 0 ) {
1634 // Continuous movement should only work in the same or similar directions.
1635 dpad_continuous = false;
1636 lastdpad = lc;
1637 return 0;
1638 }
1639
1640 last_input = input_event( lc, CATA_INPUT_GAMEPAD );
1641 lastdpad = lc;
1642 queued_dpad = ERR;
1643
1644 if( !dpad_continuous ) {
1645 delaydpad = SDL_GetTicks() + 200;
1646 dpad_continuous = true;
1647 } else {
1648 delaydpad = SDL_GetTicks() + 60;
1649 }
1650 return 1;
1651 }
1652 }
1653 } else {
1654 dpad_continuous = false;
1655 delaydpad = std::numeric_limits<Uint32>::max();
1656
1657 // If we didn't hold it down for a while, just
1658 // fire the last registered press.
1659 if( queued_dpad != ERR ) {
1660 last_input = input_event( queued_dpad, CATA_INPUT_GAMEPAD );
1661 queued_dpad = ERR;
1662 return 1;
1663 }
1664 }
1665
1666 return 0;
1667}

1741/**
1742 * Translate SDL key codes to key identifiers used by ncurses, this
1743 * allows the input_manager to only consider those.
1744 * @return 0 if the input can not be translated (unknown key?),
1745 * -1 when a ALT+number sequence has been started,
1746 * or something that a call to ncurses getch would return.
1747 */
1748static int sdl_keysym_to_curses( const SDL_Keysym &keysym )
1749{
1750
1751 const std::string diag_mode = get_option<std::string>( "DIAG_MOVE_WITH_MODIFIERS_MODE" );
1752
1753 if( diag_mode == "mode1" ) {
1754 if( keysym.mod & KMOD_CTRL && sdl_keycode_is_arrow( keysym.sym ) ) {
1755 return handle_arrow_combo( keysym.sym );
1756 } else {
1757 end_arrow_combo();
1758 }
1759 }
1760
1761 if( diag_mode == "mode2" ) {
1762 //Shift + Cursor Arrow (diagonal clockwise)
1763 if( keysym.mod & KMOD_SHIFT ) {
1764 switch( keysym.sym ) {
1765 case SDLK_LEFT:
1766 return inp_mngr.get_first_char_for_action( "LEFTUP" );
1767 case SDLK_RIGHT:
1768 return inp_mngr.get_first_char_for_action( "RIGHTDOWN" );
1769 case SDLK_UP:
