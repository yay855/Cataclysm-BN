288hint_rating rate_action_use( const avatar &you, const item &it )
289{
290 if( it.is_tool() ) {
291 return it.ammo_sufficient() ? hint_rating::good : hint_rating::iffy;
292 } else if( it.is_gunmod() ) {
293 /** @EFFECT_GUN >0 allows rating estimates for gun modifications */
294 if( you.get_skill_level( skill_id( "gun" ) ) == 0 ) {
295 return hint_rating::iffy;
296 } else {
297 return hint_rating::good;
298 }
299 } else if( it.is_food() || it.is_medication() || it.is_book() || it.is_armor() ) {
300 // The rating is subjective, could be argued as hint_rating::cant or hint_rating::good as well
301 return hint_rating::iffy;
302 } else if( it.type->has_use() ) {
303 return hint_rating::good;
304 } else if( !it.is_container_empty() ) {
305 return rate_action_use( you, it.get_contained() );
306 }
307
308 return hint_rating::cant;
309}
