352/*
353 * Complex conversion of outdated overmap terrain ids.
354 * This is used when loading saved games with old oter_ids.
355 */
356void overmap::convert_terrain(
357 const std::unordered_map<tripoint_om_omt, std::string> &needs_conversion )
358{
359 for( const auto &convert : needs_conversion ) {
360 const tripoint_om_omt pos = convert.first;
361 const std::string old = convert.second;
362
363 struct convert_nearby {
364 point offset;
365 std::string x_id;
366 std::string y_id;
367 std::string new_id;
368 };
369
370 std::vector<convert_nearby> nearby;
371 std::vector<std::pair<tripoint, std::string>> convert_unrelated_adjacent_tiles;
372
373 if( old == "fema" || old == "fema_entrance" || old == "fema_1_3" ||
374 old == "fema_2_1" || old == "fema_2_2" || old == "fema_2_3" ||
375 old == "fema_3_1" || old == "fema_3_2" || old == "fema_3_3" ||
376 old == "mine_entrance" ) {
377 ter_set( pos, oter_id( old + "_north" ) );
378 } else if( old.compare( 0, 10, "mass_grave" ) == 0 ) {
379 ter_set( pos, oter_id( "field" ) );
380 } else if( old == "mine_shaft" ) {
381 ter_set( pos, oter_id( "mine_shaft_middle_north" ) );
382 }
383
384 for( const auto &conv : nearby ) {
385 const auto x_it = needs_conversion.find( pos + point( conv.offset.x, 0 ) );
386 const auto y_it = needs_conversion.find( pos + point( 0, conv.offset.y ) );
387 if( x_it != needs_conversion.end() && x_it->second == conv.x_id &&
388 y_it != needs_conversion.end() && y_it->second == conv.y_id ) {
389 ter_set( pos, oter_id( conv.new_id ) );
390 break;
391 }
392 }
393
394 for( const std::pair<tripoint, std::string> &conv : convert_unrelated_adjacent_tiles ) {
395 ter_set( pos + conv.first, oter_id( conv.second ) );
396 }
397 }
