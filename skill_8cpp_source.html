223void SkillLevel::train( int amount, bool skip_scaling )
224{
225 // Working off rust to regain levels goes twice as fast as reaching levels in the first place
226 if( _level < _highestLevel ) {
227 amount *= 2;
228 }
229
230 if( skip_scaling ) {
231 _exercise += amount;
232 } else {
233 const double scaling = get_option<float>( "SKILL_TRAINING_SPEED" );
234 if( scaling > 0.0 ) {
235 _exercise += roll_remainder( amount * scaling );
236 }
237 }
238
239 if( _exercise >= 100 * ( _level + 1 ) * ( _level + 1 ) ) {
240 _exercise = 0;
241 ++_level;
242 if( _level > _highestLevel ) {
243 _highestLevel = _level;
244 }
245 }
246}

269bool SkillLevel::rust( bool charged_bio_mem, int character_rate )
270{
271 const time_duration delta = calendar::turn - _lastPracticed;
272 const float char_rate = character_rate / 100.0;
273 const time_duration skill_rate = rustRate( _level );
274 if( to_turns<int>( skill_rate ) * char_rate <= 0 || delta <= 0_turns ||
275 delta % ( skill_rate * char_rate ) != 0_turns ) {
276 return false;
277 }
278
279 if( charged_bio_mem ) {
280 return one_in( 5 );
281 }
282
283 _exercise -= _level;
284 const std::string &rust_type = get_option<std::string>( "SKILL_RUST" );
285 if( _exercise < 0 ) {
286 if( rust_type == "vanilla" || rust_type == "int" ) {
287 _exercise = ( 100 * _level * _level ) - 1;
288 --_level;
289 } else {
290 _exercise = 0;
291 }
292 }
293
294 return false;
295}
