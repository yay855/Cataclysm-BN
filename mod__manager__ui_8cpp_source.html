116static void check_conflict( const mod_id &what, const mod_id &with,
117 std::vector<conflict_pair> &ret )
118{
119 auto it = std::find( what->conflicts.begin(), what->conflicts.end(), with );

139ret_val<bool> mod_ui::try_add( const mod_id &mod_to_add, std::vector<mod_id> &active_list )
140{
141 if( std::find( active_list.begin(), active_list.end(), mod_to_add ) != active_list.end() ) {
142 // The same mod can not be added twice. That makes no sense.
143 return ret_val<bool>::make_failure( _( "The mod is already on the list." ) );
144 }
145 if( !mod_to_add.is_valid() ) {
146 return ret_val<bool>::make_failure( _( "Unable to find mod with id \"%s\"." ), mod_to_add );
147 }
148 const MOD_INFORMATION &mod = *mod_to_add;
149 dependency_node *checknode = mm_tree.get_node( mod.ident );
150 if( !checknode ) {
151 return ret_val<bool>::make_failure( _( "Failed to build dependency tree for the mod." ) );
152 }
153 if( checknode->has_errors() ) {
154 return ret_val<bool>::make_failure( _( "The mod has dependency problem(s):\n\n%s" ),
155 checknode->s_errors() );
156 }
157
158 // get dependencies of selection in the order that they would appear from the top of the active list
159 std::vector<mod_id> dependencies = mm_tree.get_dependencies_of_X_as_strings( mod.ident );
160
161 std::vector<conflict_pair> conflicts;
162 check_conflicts( mod_to_add, active_list, conflicts );
163 for( const auto &dep : dependencies ) {
164 check_conflicts( dep, active_list, conflicts );
165 }
166 if( !conflicts.empty() ) {
167 return ret_val<bool>::make_failure(
168 _( "The mod or some of its dependencies has conflict(s) with active mods:\n\n%s" ),
169 fmt_conflicts( conflicts ) );
170 }
171
172 // check to see if mod is a core, and if so check to see if there is already a core in the mod list
173 if( mod.core ) {
174 // (more than 0 active elements) && (active[0] is a CORE) && active[0] is not the add candidate
175 if( !active_list.empty() && active_list[0]->core && active_list[0] != mod_to_add ) {
176 // remove existing core
177 try_rem( 0, active_list );
178 }
179
180 // add to start of active_list if it doesn't already exist in it
181 active_list.insert( active_list.begin(), mod_to_add );
182 } else {
183 // now check dependencies and add them as necessary
184 std::vector<mod_id> mods_to_add;
185 bool new_core = false;
186 for( auto &i : dependencies ) {
187 if( std::find( active_list.begin(), active_list.end(), i ) == active_list.end() ) {
188 if( i->core ) {
189 mods_to_add.insert( mods_to_add.begin(), i );
190 new_core = true;
191 } else {
192 mods_to_add.push_back( i );
193 }
194 }
195 }
196
197 if( new_core && !active_list.empty() ) {
198 try_rem( 0, active_list );
199 active_list.insert( active_list.begin(), mods_to_add[0] );
200 mods_to_add.erase( mods_to_add.begin() );
201 }
202 // now add the rest of the dependencies serially to the end
203 for( auto &i : mods_to_add ) {
204 active_list.push_back( i );
205 }
206 // and finally add the one we are trying to add!
207 active_list.push_back( mod.ident );
208 }
209
210 return ret_val<bool>::make_success();
211}

213void mod_ui::try_rem( size_t selection, std::vector<mod_id> &active_list )
214{
215 // first make sure that what we are looking for exists in the list
216 if( selection >= active_list.size() ) {
217 // trying to access an out of bounds value! quit early
218 return;
219 }
220 const mod_id sel_string = active_list[selection];
221
222 const MOD_INFORMATION &mod = *active_list[selection];
223
224 std::vector<mod_id> dependents = mm_tree.get_dependents_of_X_as_strings( mod.ident );
225
226 // search through the rest of the active list for mods that depend on this one
227 for( auto &i : dependents ) {
228 auto rem = std::find( active_list.begin(), active_list.end(), i );
229 if( rem != active_list.end() ) {
230 active_list.erase( rem );
231 }
232 }
233 std::vector<mod_id>::iterator rem = std::find( active_list.begin(), active_list.end(),
234 sel_string );
235 if( rem != active_list.end() ) {
236 active_list.erase( rem );

240void mod_ui::try_shift( char direction, size_t &selection, std::vector<mod_id> &active_list )
241{
242 // error catch for out of bounds
243 if( selection >= active_list.size() ) {
244 return;
245 }
246
247 // eliminates 'uninitialized variable' warning
248 size_t newsel = 0;
249 size_t oldsel = 0;
250 int selshift = 0;
251
252 // shift up (towards 0)
253 if( direction == '+' && can_shift_up( selection, active_list ) ) {
254 // see if the mod at selection-1 is a) a core, or b) is depended on by this mod
255 newsel = selection - 1;
256 oldsel = selection;
257
258 selshift = -1;
259 }
260 // shift down (towards active_list.size()-1)
261 else if( direction == '-' && can_shift_down( selection, active_list ) ) {
262 newsel = selection;
263 oldsel = selection + 1;
264
265 selshift = +1;
266 } else {
267 return;
268 }
269
270 mod_id modstring = active_list[newsel];
271 mod_id selstring = active_list[oldsel];
272
273 // we can shift!
274 // switch values!
275 active_list[newsel] = selstring;
276 active_list[oldsel] = modstring;
277
278 selection += selshift;

281bool mod_ui::can_shift_up( size_t selection, const std::vector<mod_id> &active_list )
282{
283 // error catch for out of bounds
284 if( selection >= active_list.size() ) {
285 return false;
286 }
287 // dependencies of this active element
288 std::vector<mod_id> dependencies = mm_tree.get_dependencies_of_X_as_strings(
289 active_list[selection] );
290
291 // figure out if we can move up!
292 if( selection == 0 ) {
293 // can't move up, don't bother trying
294 return false;
295 }
296 // see if the mod at selection-1 is a) a core, or b) is depended on by this mod
297 int newsel = selection - 1;
298
299 mod_id newsel_id = active_list[newsel];
300 bool newsel_is_dependency =
301 std::find( dependencies.begin(), dependencies.end(), newsel_id ) != dependencies.end();
302
303 return !newsel_id->core && !newsel_is_dependency;
