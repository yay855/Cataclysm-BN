251void distraction_manager_gui::deserialize( JsonIn &jsin )
252{
253 jsin.start_array();
254 while( !jsin.end_array() ) {
255 JsonObject jo = jsin.get_object();
256
257 if( !jo.has_string( "Distraction Type" ) ) {
258 continue;
259 }
260
261 const distraction_type type_id = jo.get_enum_value<distraction_type>( "Distraction Type" );
262 const bool boolean = jo.get_bool( "Bool" );
263
264 distractions[type_id] = boolean;
265 }
