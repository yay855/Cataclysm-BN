487cata::optional<zone_type_id> zone_manager::query_type() const
488{
489 const auto &types = get_manager().get_types();
490 std::vector<std::pair<zone_type_id, zone_type>> types_vec;
491 std::copy( types.begin(), types.end(),
492 std::back_inserter<std::vector<std::pair<zone_type_id, zone_type>>>( types_vec ) );
493 std::sort( types_vec.begin(), types_vec.end(),
494 []( const std::pair<zone_type_id, zone_type> &lhs, const std::pair<zone_type_id, zone_type> &rhs ) {
495 return localized_compare( lhs.second.name(), rhs.second.name() );
496 } );
497
498 uilist as_m;
499 as_m.desc_enabled = true;
500 as_m.text = _( "Select zone type:" );
501
502 size_t i = 0;
503 for( const auto &pair : types_vec ) {
504 const auto &type = pair.second;
505
506 as_m.addentry_desc( i++, true, MENU_AUTOASSIGN, type.name(), type.desc() );
507 }
508
509 as_m.query();
510 if( as_m.ret < 0 ) {
511 return {};
512 }
513 size_t index = as_m.ret;
514
515 auto iter = types_vec.begin();
516 std::advance( iter, index );
517
518 return iter->first;
519}

1052void zone_manager::rotate_zones( map &target_map, const int turns )
1053{
1054 if( turns == 0 ) {
1055 return;
1056 }
1057 const tripoint a_start = target_map.getabs( tripoint_zero );
1058 const tripoint a_end = target_map.getabs( tripoint( 23, 23, 0 ) );
1059 const point dim( 24, 24 );
1060 for( zone_data &zone : zones ) {
1061 const tripoint z_start = zone.get_start_point();
1062 const tripoint z_end = zone.get_end_point();
1063 if( ( a_start.x <= z_start.x && a_start.y <= z_start.y ) &&
1064 ( a_end.x > z_start.x && a_end.y >= z_start.y ) &&
1065 ( a_start.x <= z_end.x && a_start.y <= z_end.y ) &&
1066 ( a_end.x >= z_end.x && a_end.y >= z_end.y ) &&
1067 ( a_start.z == z_start.z )
1068 ) {
1069 tripoint z_l_start3 = target_map.getlocal( z_start );
1070 tripoint z_l_end3 = target_map.getlocal( z_end );
1071 // don't rotate centered squares
1072 if( z_l_start3.x == z_l_start3.y && z_l_end3.x == z_l_end3.y && z_l_start3.x + z_l_end3.x == 23 ) {
1073 continue;
1074 }
1075 point z_l_start = z_l_start3.xy().rotate( turns, dim );
1076 point z_l_end = z_l_end3.xy().rotate( turns, dim );
1077 point new_z_start = target_map.getabs( z_l_start );
1078 point new_z_end = target_map.getabs( z_l_end );
1079 tripoint first = tripoint( std::min( new_z_start.x, new_z_end.x ),
1080 std::min( new_z_start.y, new_z_end.y ), a_start.z );
1081 tripoint second = tripoint( std::max( new_z_start.x, new_z_end.x ),
1082 std::max( new_z_start.y, new_z_end.y ), a_end.z );
1083 zone.set_position( std::make_pair( first, second ), false );
1084 }
1085 }
