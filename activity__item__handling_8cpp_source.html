411static std::list<pickup::act_item> convert_to_items( Character &p,
412 const drop_locations &drop,
413 std::function<bool( item_location loc )> filter )
414{
415 std::list<pickup::act_item> res;
416
417 for( const drop_location &rec : drop ) {
418 const item_location loc = rec.loc;
419 const int count = rec.count;
420
421 if( !filter( loc ) ) {
422 continue;
423 } else if( !p.is_worn( *loc ) && !p.is_wielding( *loc ) ) {
424 // Special case. After dropping the first few items, the remaining items are already separated.
425 // That means: `drop` already contains references to each of the items in
426 // `p.inv.const_stack`, and `count` will be 1 for each of them.
427 // If we continued without this check, we iterate over `p.inv.const_stack` multiple times,
428 // but each time stopping after visiting the first item.
429 // In the end, we would add references to the same item (the first one in the stack) multiple times.
430 if( count == 1 ) {
431 res.emplace_back( loc, 1, loc.obtain_cost( p, 1 ) );
432 continue;
433 }
434 int obtained = 0;
435 for( const item &it : p.inv.const_stack( p.get_item_position( &*loc ) ) ) {
436 if( obtained >= count ) {
437 break;
438 }
439 const int qty = it.count_by_charges() ? std::min<int>( it.charges, count - obtained ) : 1;
440 obtained += qty;
441 item_location loc( p, const_cast<item *>( &it ) );
442 res.emplace_back( loc, qty, loc.obtain_cost( p, qty ) );
443 }
444 } else {
445 res.emplace_back( loc, count, p.is_wielding( *loc ) ? 0 : loc.obtain_cost( p ) );
446 }
447 }
448
449 return res;
450}

455// Prepares items for dropping by reordering them so that the drop
456// cost is minimal and "dependent" items get taken off first.
457// Implements the "backpack" logic.
458std::list<act_item> reorder_for_dropping( Character &p, const drop_locations &drop )
459{
460 std::list<act_item> res = convert_to_items( p, drop,
461 [&p]( item_location loc ) {
462 return p.is_wielding( *loc );
463 } );
464 std::list<act_item> inv = convert_to_items( p, drop,
465 [&p]( item_location loc ) {
466 return !p.is_wielding( *loc ) && !p.is_worn( *loc );
467 } );
468 std::list<act_item> worn = convert_to_items( p, drop,
469 [&p]( item_location loc ) {
470 return p.is_worn( *loc );
471 } );
472
473 // Sort inventory items by volume in ascending order
474 inv.sort( []( const act_item & first, const act_item & second ) {
475 return first.loc->volume() < second.loc->volume();
476 } );
477 // Add missing dependent worn items (if any).
478 for( const auto &wait : worn ) {
479 for( item *dit : p.get_dependent_worn_items( *wait.loc ) ) {
480 const auto iter = std::find_if( worn.begin(), worn.end(),
481 [dit]( const act_item & ait ) {
482 return &*ait.loc == dit;
483 } );
484
485 if( iter == worn.end() ) {
486 // TODO: Use a calculated cost
487 const item_location loc( p, dit );
488 act_item act( loc, loc->count(), loc.obtain_cost( p, loc->count() ) );
489 worn.emplace_front( loc, loc->count(), loc.obtain_cost( p ) );
490 }
491 }
492 }
493 // Sort worn items by storage in descending order, but dependent items always go first.
494 worn.sort( []( const act_item & first, const act_item & second ) {
495 return first.loc->is_worn_only_with( *second.loc )
496 || ( first.loc->get_storage() > second.loc->get_storage()
497 && !second.loc->is_worn_only_with( *first.loc ) );
498 } );
499
500 // Avoid tumbling to the ground. Unload cleanly.
501 units::volume dropped_inv_contents = std::accumulate( inv.begin(), inv.end(), 0_ml,
502 []( units::volume acc, const act_item & ait ) {
503 return acc + ait.loc->volume();
504 } );
505 const units::volume dropped_worn_storage = std::accumulate( worn.begin(), worn.end(), 0_ml,
506 []( units::volume acc, const act_item & ait ) {
507 return acc + ait.loc->get_storage();
508 } );
509 std::set<int> inv_indices;
510 std::transform( inv.begin(), inv.end(), std::inserter( inv_indices, inv_indices.begin() ),
511 [&p]( const act_item & ait ) {
512 return p.get_item_position( &*ait.loc );
513 } );
514
515 units::volume excessive_volume = p.volume_carried() - dropped_inv_contents
516 - p.volume_capacity_reduced_by( dropped_worn_storage );
517 if( excessive_volume > 0_ml ) {
518 invslice old_inv = p.inv.slice();
519 for( size_t i = 0; i < old_inv.size() && excessive_volume > 0_ml; i++ ) {
520 // TODO: Reimplement random dropping?
521 if( inv_indices.count( i ) != 0 ) {
522 continue;
523 }
524 std::list<item> &inv_stack = *old_inv[i];
525 for( item &item : inv_stack ) {
526 // Note: zero cost, but won't be contained on drop
527 act_item to_drop = act_item( item_location( p, &item ), item.count(), 0 );
528 inv.push_back( to_drop );
529 excessive_volume -= to_drop.loc->volume();
530 if( excessive_volume <= 0_ml ) {
531 break;
532 }
533 }
534 }
535 // Need to re-sort
536 inv.sort( []( const act_item & first, const act_item & second ) {
537 return first.loc->volume() < second.loc->volume();
538 } );
539 }
540
541 // Cumulatively decreases
542 units::volume remaining_dropped_storage = dropped_worn_storage;
543
544 while( !worn.empty() && !inv.empty() ) {
545 units::volume front_storage = worn.front().loc->get_storage();
546 // Does not fit
547 // TODO: but maybe an item further down the line does
548 if( remaining_dropped_storage < inv.front().loc->volume() ) {
549 break;
550 }
551
552 res.push_back( worn.front() );
553 worn.pop_front();
554 remaining_dropped_storage -= front_storage;
555 while( !inv.empty() ) {
556 units::volume inventory_item_volume = inv.front().loc->volume();
557 if( front_storage < inventory_item_volume ) {
558 break;
559 }
560 front_storage -= inventory_item_volume;
561
562 res.push_back( inv.front() );
563 // Free of charge
564 res.back().consumed_moves = 0;
565
566 inv.pop_front();
567 }
568 }
569
570 // Now insert everything that remains
571 std::copy( inv.begin(), inv.end(), std::back_inserter( res ) );
572 std::copy( worn.begin(), worn.end(), std::back_inserter( res ) );
573
574 return res;
575}

598 units::volume last_storage_volume = items.front().loc->get_storage();

600 act_item &ait = items.front();
601
602 if( !try_rebuild_if_needed( ait.loc ) ) {
603 debugmsg( "Lost target item of ACT_DROP" );
604 items.pop_front();
605 return res;
606 }
607
608 p.mod_moves( -ait.consumed_moves );
609
610 if( p.is_worn( *ait.loc ) ) {
611 if( !p.takeoff( *ait.loc, &res ) ) {
612 // Skip item if failed to take it off
613 debugmsg( "Failed to obtain worn target item of ACT_DROP" );
614 items.pop_front();
615 continue;
616 }
617 } else if( ait.loc->count_by_charges() ) {
618 res.push_back( p.reduce_charges( const_cast<item *>( &*ait.loc ), ait.count ) );
619 } else {
620 res.push_back( p.i_rem( &*ait.loc ) );
621 }
622
623 // TODO: Get the item consistently instead of using back()
624 item &current_drop = res.back();
625
626 // Hack: if it consumes zero moves, it must have been contained
627 // TODO: Properly mark containment somehow
628 *current_drop.drop_token = token_provider.make_next( calendar::turn );;
629 if( ait.consumed_moves == 0 && last_storage_volume >= current_drop.volume() ) {
630 last_storage_volume -= current_drop.volume();
631 current_drop.drop_token->parent_number = last_token.parent_number;

633 last_token = *current_drop.drop_token;
634 last_storage_volume = current_drop.get_storage();

637 items.pop_front();
638 }

640 return res;

825static double get_capacity_fraction( int capacity, int volume )
826{
827 // fraction of capacity the item would occupy
828 // fr = 1 is for capacity smaller than is size of item
829 // in such case, let's assume player does the trip for full cost with item in hands
830 double fr = 1;
831
832 if( capacity > volume ) {
833 fr = static_cast<double>( volume ) / capacity;
834 }
835
836 return fr;

839static int move_cost_inv( const item &it, const tripoint &src, const tripoint &dest )
840{
841 // to prevent potentially ridiculous number
842 const int MAX_COST = 500;
843
844 // it seems that pickup cost is flat 100
845 // in function pick_one_up, variable moves_taken has initial value of 100
846 // and never changes until it is finally used in function
847 // remove_from_map_or_vehicle
848 const int pickup_cost = 100;
849
850 // drop cost for non-tumbling items (from inventory overload) is also flat 100
851 // according to convert_to_items (it does contain todo to use calculated costs)
852 const int drop_cost = 100;
853
854 // typical flat ground move cost
855 const int mc_per_tile = 100;
856
857 // only free inventory capacity
858 const int inventory_capacity = units::to_milliliter( g->u.volume_capacity() -
859 g->u.volume_carried() );
860
861 const int item_volume = units::to_milliliter( it.volume() );
862
863 const double fr = get_capacity_fraction( inventory_capacity, item_volume );
864
865 // approximation of movement cost between source and destination
866 const int move_cost = mc_per_tile * rl_dist( src, dest ) * fr;
867
868 return std::min( pickup_cost + drop_cost + move_cost, MAX_COST );
869}

919static bool vehicle_activity( player &p, const tripoint &src_loc, int vpindex, char type )
920{
921 map &here = get_map();
922 vehicle *veh = veh_pointer_or_null( here.veh_at( src_loc ) );
923 if( !veh ) {
924 return false;
925 }
926 int time_to_take = 0;
927 if( vpindex >= veh->part_count() ) {
928 // if parts got removed during our work, we can't just carry on removing, we want to repair parts!
929 // so just bail out, as we don't know if the next shifted part is suitable for repair.
930 if( type == 'r' ) {
931 return false;
932 } else if( type == 'o' ) {
933 vpindex = veh->get_next_shifted_index( vpindex, p );
934 if( vpindex == -1 ) {
935 return false;
936 }
937 }
938 }
939 const vpart_info &vp = veh->part_info( vpindex );
940 if( type == 'r' ) {
941 const vehicle_part part = veh->part( vpindex );
942 time_to_take = vp.repair_time( p ) * part.damage() / part.max_damage();
943 } else if( type == 'o' ) {
944 time_to_take = vp.removal_time( p );
945 }
946 p.assign_activity( ACT_VEHICLE, time_to_take, static_cast<int>( type ) );
947 // so , NPCs can remove the last part on a position, then there is no vehicle there anymore,
948 // for someone else who stored that position at the start of their activity.
949 // so we may need to go looking a bit further afield to find it , at activities end.
950 for( const auto pt : veh->get_points( true ) ) {
951 p.activity.coord_set.insert( here.getabs( pt ) );
952 }
953 // values[0]
954 p.activity.values.push_back( here.getabs( src_loc ).x );
955 // values[1]
956 p.activity.values.push_back( here.getabs( src_loc ).y );
957 // values[2]
958 p.activity.values.push_back( point_zero.x );
959 // values[3]
960 p.activity.values.push_back( point_zero.y );
961 // values[4]
962 p.activity.values.push_back( -point_zero.x );
963 // values[5]
964 p.activity.values.push_back( -point_zero.y );
965 // values[6]
966 p.activity.values.push_back( veh->index_of_part( &veh->part( vpindex ) ) );
967 p.activity.str_values.push_back( vp.get_id().str() );
968 // values[7]
969 p.activity.values.push_back( here.getabs( src_loc ).z );
970 // this would only be used for refilling tasks
971 item_location target;
972 p.activity.targets.emplace_back( std::move( target ) );
973 p.activity.placement = here.getabs( src_loc );
974 p.activity_vehicle_part_index = -1;
975 return true;

1080 if( !has_skill ) {

1094 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS, false, id );

1117 //first step: try only constructions with the same group

1171 return activity_reason_info::build( do_activity_reason::NO_COMPONENTS, false, id );
1172 }
1173 //only cc failed, no pre-req

1210 check_weight_if( p.backlog.front().id() ) );
1211 bool found_welder = false;
1212 for( item *elem : p.inv_dump() ) {
1213 if( elem->has_quality( qual_WELD ) ) {
1214 found_welder = true;
1215 }
1216 temp_inv += *elem;
1217 }
1218 map &here = get_map();
1219 for( const tripoint &elem : loot_spots ) {
1220 // if we are searching for things to fetch, we can skip certain things.
1221 // if, however they are already near the work spot, then the crafting / inventory functions will have their own method to use or discount them.
1222 if( in_loot_zones ) {
1223 // skip tiles in IGNORE zone and tiles on fire
1224 // (to prevent taking out wood off the lit brazier)
1225 // and inaccessible furniture, like filled charcoal kiln
1226 if( mgr.has( zone_type_LOOT_IGNORE, here.getabs( elem ) ) ||
1227 here.dangerous_field_at( elem ) ||
1228 !here.can_put_items_ter_furn( elem ) ) {
1229 continue;
1230 }
1231 }
1232 for( const auto &elem2 : here.i_at( elem ) ) {
1233 if( in_loot_zones && elem2.made_of( LIQUID ) ) {
1234 continue;
1235 }
1236 if( check_weight ) {
1237 // this fetch task will need to pick up an item. so check for its weight/volume before setting off.
1238 if( in_loot_zones && ( elem2.volume() > volume_allowed ||
1239 elem2.weight() > weight_allowed ) ) {
1240 continue;
1241 }
1242 }
1243 temp_inv += elem2;
1244 }
1245 if( !in_loot_zones ) {
1246 if( const cata::optional<vpart_reference> vp = here.veh_at( elem ).part_with_feature( "CARGO",
1247 false ) ) {
1248 vehicle &src_veh = vp->vehicle();
1249 int src_part = vp->part_index();
1250 for( auto &it : src_veh.get_items( src_part ) ) {
1251 temp_inv += it;
1252 }
1253 }
1254 }
1255 }
1256 // use nearby welding rig without needing to drag it or position yourself on the right side of the vehicle.
1257 if( !found_welder ) {
1258 for( const tripoint &elem : here.points_in_radius( src_loc, PICKUP_RANGE - 1 ) ) {
1259 const optional_vpart_position vp = here.veh_at( elem );
1260 if( vp ) {
1261 vehicle &veh = vp->vehicle();
1262 const cata::optional<vpart_reference> weldpart = vp.part_with_feature( "WELDRIG", true );
1263 if( weldpart ) {
1264 item welder( itype_welder, calendar::start_of_cataclysm );
1265 welder.charges = veh.fuel_left( itype_battery, true );
1266 welder.set_flag( "PSEUDO" );
1267 temp_inv.add_item( welder );
1268 item soldering_iron( itype_soldering_iron, calendar::start_of_cataclysm );
1269 soldering_iron.charges = veh.fuel_left( itype_battery, true );
1270 soldering_iron.set_flag( "PSEUDO" );
1271 temp_inv.add_item( soldering_iron );
1272 }
1273 }
1274 }
1275 }
1276 return needed_things.obj().can_make_with_inventory( temp_inv, is_crafting_component );
1277}

1547 );

1571 const plot_options &options = dynamic_cast<const plot_options &>( zone.get_options() );
1572 const itype_id seed = options.get_seed();
1573 // If its a farm zone with no specified seed, and we've checked for tilling and harvesting.
1574 // then it means no further work can be done here
1575 if( seed.is_empty() ) {
1576 return activity_reason_info::fail( do_activity_reason::ALREADY_DONE );
1577 }

1579 return itm.is_seed();
1580 } );
1581 for( const auto elem : seed_inv ) {
1582 if( elem->typeId() == itype_id( seed ) ) {
1583 return activity_reason_info::ok( do_activity_reason::NEEDS_PLANTING );
1584 }
1585 }
1586 // didn't find the seed, but maybe there are overlapping farm zones
1587 // and another of the zones is for a seed that we have
1588 // so loop again, and return false once all zones done.
1589 }
1590
1591 } else {
1592 // can't plant, till or harvest
1593 return activity_reason_info::fail( do_activity_reason::ALREADY_DONE );
1594 }
1595
1596 }
1597 // looped through all zones, and only got here if its plantable, but have no seeds.
1598 return activity_reason_info::fail( do_activity_reason::NEEDS_PLANTING );
1599 } else if( act == ACT_FETCH_REQUIRED ) {
1600 // we check if its possible to get all the requirements for fetching at two other places.
1601 // 1. before we even assign the fetch activity and;
1602 // 2. when we form the src_set to loop through at the beginning of the fetch activity.
1603 return activity_reason_info::ok( do_activity_reason::CAN_DO_FETCH );
1604 }
1605 // Shouldn't get here because the zones were checked previously. if it does, set enum reason as "no zone"
1606 return activity_reason_info::fail( do_activity_reason::NO_ZONE );
1607}

1626static std::vector<std::tuple<tripoint, itype_id, int>> requirements_map( player &p,
1627 const int distance = ACTIVITY_SEARCH_DISTANCE )
1628{
1629 std::vector<std::tuple<tripoint, itype_id, int>> requirement_map;
1630 if( p.backlog.empty() || p.backlog.front().str_values.empty() ) {
1631 return requirement_map;
1632 }
1633 const requirement_data things_to_fetch = requirement_id( p.backlog.front().str_values[0] ).obj();
1634 const activity_id activity_to_restore = p.backlog.front().id();
1635 // NOLINTNEXTLINE(performance-unnecessary-copy-initialization)
1636 requirement_id things_to_fetch_id = things_to_fetch.id();
1637 std::vector<std::vector<item_comp>> req_comps = things_to_fetch.get_components();
1638 std::vector<std::vector<tool_comp>> tool_comps = things_to_fetch.get_tools();
1639 std::vector<std::vector<quality_requirement>> quality_comps = things_to_fetch.get_qualities();
1640 zone_manager &mgr = zone_manager::get_manager();
1641 const bool pickup_task = p.backlog.front().id() == ACT_MULTIPLE_FARM ||
1642 p.backlog.front().id() == ACT_MULTIPLE_CHOP_PLANKS ||
1643 p.backlog.front().id() == ACT_MULTIPLE_BUTCHER ||
1644 p.backlog.front().id() == ACT_MULTIPLE_CHOP_TREES ||
1645 p.backlog.front().id() == ACT_VEHICLE_DECONSTRUCTION ||
1646 p.backlog.front().id() == ACT_VEHICLE_REPAIR ||
1647 p.backlog.front().id() == ACT_MULTIPLE_FISH ||
1648 p.backlog.front().id() == ACT_MULTIPLE_MINE;
1649 // where it is, what it is, how much of it, and how much in total is required of that item.
1650 std::vector<std::tuple<tripoint, itype_id, int>> final_map;
1651 std::vector<tripoint> loot_spots;
1652 std::vector<tripoint> already_there_spots;
1653 std::vector<tripoint> combined_spots;
1654 std::map<itype_id, int> total_map;
1655 map &here = get_map();
1656 tripoint src_loc = here.getlocal( p.backlog.front().placement );
1657 for( const tripoint &elem : here.points_in_radius( src_loc,
1658 PICKUP_RANGE - 1 ) ) {
1659 already_there_spots.push_back( elem );
1660 combined_spots.push_back( elem );
1661 }
1662 for( const tripoint &elem : mgr.get_point_set_loot( here.getabs( p.pos() ), distance,
1663 p.is_npc() ) ) {
1664 // if there is a loot zone that's already near the work spot, we don't want it to be added twice.
1665 if( std::find( already_there_spots.begin(), already_there_spots.end(),
1666 elem ) != already_there_spots.end() ) {
1667 // construction tasks don't need the loot spot *and* the already_there/combined spots both added.
1668 // but a farming task will need to go and fetch the tool no matter if its near the work spot.
1669 // whereas the construction will automatically use what's nearby anyway.
1670 if( pickup_task ) {
1671 loot_spots.push_back( elem );
1672 } else {
1673 continue;
1674 }
1675 } else {
1676 loot_spots.push_back( elem );
1677 combined_spots.push_back( elem );
1678 }
1679 }
1680 add_basecamp_storage_to_loot_zone_list( mgr, src_loc, p, loot_spots, combined_spots );
1681 // if the requirements aren't available, then stop.
1682 if( !are_requirements_nearby( pickup_task ? loot_spots : combined_spots, things_to_fetch_id, p,
1683 activity_to_restore, pickup_task, src_loc ) ) {
1684 return requirement_map;
1685 }
1686 // if the requirements are already near the work spot and its a construction/crafting task, then no need to fetch anything more.
1687 if( !pickup_task &&
1688 are_requirements_nearby( already_there_spots, things_to_fetch_id, p, activity_to_restore,
1689 false, src_loc ) ) {
1690 return requirement_map;
1691 }
1692 // a vector of every item in every tile that matches any part of the requirements.
1693 // will be filtered for amounts/charges afterwards.
1694 for( const tripoint &point_elem : pickup_task ? loot_spots : combined_spots ) {
1695 std::map<itype_id, int> temp_map;
1696 for( const item &stack_elem : here.i_at( point_elem ) ) {
1697 for( std::vector<item_comp> &elem : req_comps ) {
1698 for( item_comp &comp_elem : elem ) {
1699 if( comp_elem.type == stack_elem.typeId() ) {

1702 // that is fine, we will choose which "line" to fulfill later, and the decrement will count towards that then.

1704 std::find( already_there_spots.begin(), already_there_spots.end(),
1705 point_elem ) != already_there_spots.end() ) {
1706 comp_elem.count -= stack_elem.count();
1707 }
1708 temp_map[stack_elem.typeId()] += stack_elem.count();
1709 }
1710 }
1711 }
1712 for( std::vector<tool_comp> &elem : tool_comps ) {
1713 for( tool_comp &comp_elem : elem ) {
1714 if( comp_elem.type == stack_elem.typeId() ) {
1715 if( !pickup_task &&
1716 std::find( already_there_spots.begin(), already_there_spots.end(),
1717 point_elem ) != already_there_spots.end() ) {
1718 comp_elem.count -= stack_elem.count();
1719 }
1720 if( comp_elem.by_charges() ) {
1721 // we don't care if there are 10 welders with 5 charges each
1722 // we only want the one welder that has the required charge.
1723 if( stack_elem.ammo_remaining() >= comp_elem.count ) {
1724 temp_map[stack_elem.typeId()] += stack_elem.ammo_remaining();
1725 }
1726 } else {
1727 temp_map[stack_elem.typeId()] += stack_elem.count();
1728 }
1729 }
1730 }
1731 }
1732
1733 for( std::vector<quality_requirement> &elem : quality_comps ) {
1734 for( quality_requirement &comp_elem : elem ) {
1735 const quality_id tool_qual = comp_elem.type;
1736 const int qual_level = comp_elem.level;
1737 if( stack_elem.has_quality( tool_qual, qual_level ) ) {
1738 if( !pickup_task &&
1739 std::find( already_there_spots.begin(), already_there_spots.end(),
1740 point_elem ) != already_there_spots.end() ) {
1741 comp_elem.count -= stack_elem.count();
1742 }
1743 temp_map[stack_elem.typeId()] += stack_elem.count();
1744 }
1745 }
1746 }
1747 }
1748 for( const auto &map_elem : temp_map ) {
1749 total_map[map_elem.first] += map_elem.second;
1750 // if its a construction/crafting task, we can discount any items already near the work spot.
1751 // we don't need to fetch those, they will be used automatically in the construction.
1752 // a shovel for tilling, for example, however, needs to be picked up, no matter if its near the spot or not.
1753 if( !pickup_task ) {
1754 if( std::find( already_there_spots.begin(), already_there_spots.end(),
1755 point_elem ) != already_there_spots.end() ) {
1756 continue;
1757 }
1758 }
1759 requirement_map.push_back( std::make_tuple( point_elem, map_elem.first, map_elem.second ) );
1760 }
1761 }
1762 // Ok we now have a list of all the items that match the requirements, their points, and a quantity for each one.
1763 // we need to consolidate them, and winnow it down to the minimum required counts, instead of all matching.
1764 for( const std::vector<item_comp> &elem : req_comps ) {
1765 bool line_found = false;
1766 for( const item_comp &comp_elem : elem ) {
1767 if( line_found || comp_elem.count <= 0 ) {
1768 break;

1779 item_quantity += quantity_here;
1780 }

1782 // it's just this spot that can fulfil the requirement on its own
1783 final_map.push_back( std::make_tuple( pos_here, item_here, std::min<int>( quantity_here,
1784 quantity_required ) ) );
1785 if( quantity_here >= quantity_required ) {
1786 line_found = true;
1787 break;
1788 } else {
1789 remainder = quantity_required - quantity_here;
1790 }
1791 break;
1792 }
1793 it++;
1794 }
1795 if( line_found ) {
1796 while( true ) {
1797 // go back over things
1798 if( it == requirement_map.begin() ) {
1799 break;
1800 }
1801 if( remainder <= 0 ) {
1802 line_found = true;
1803 break;
1804 }

1806 itype_id item_here2 = std::get<1>( *it );

1812 } else {
1813 final_map.push_back( std::make_tuple( pos_here2, item_here2, remainder ) );

1815 }
1816 }
1817 it--;
1818 }
1819 }

1825 if( line_found || comp_elem.count < -1 ) {
1826 break;
1827 }
1828 int quantity_required = std::max( 1, comp_elem.count );
1829 int item_quantity = 0;
1830 auto it = requirement_map.begin();
1831 int remainder = 0;
1832 while( it != requirement_map.end() ) {

1844 line_found = true;

2059 it.charges = pickup_count;

2071 add_msg( _( "%s picks up several items." ), p.disp_name() );

2077 here.add_item_or_charges( src_loc, leftovers );

2080 }
2081 }
2082 }
2083 }

2087 p.set_moves( 0 );
2088 p.activity = player_activity();
2089 p.backlog.clear();
2090 return false;
2091}

2350 }

2356 }

2371 --num_processed;
2372 }
2373 break;
2374 }
2375 }

2377 return;
2378 }
2379 }

2384 }

2389 npc *guy = dynamic_cast<npc *>( &p );
2390 guy->revert_after_activity();
2391 }
2392 p.activity.set_to_null();
