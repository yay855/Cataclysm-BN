59static void drop_or_embed_projectile( const dealt_projectile_attack &attack )
60{

63 if( drop_item.is_null() ) {
64 return;

67 const tripoint &pt = attack.end_point;

74
75 // copies the drop item to spill the contents
76 item( drop_item ).spill_contents( pt );
77

80 sounds::sound( pt, 16, sounds::sound_t::combat, _( "glass breaking!" ), false, "bullet_hit",

84
85 if( proj.has_effect( ammo_effect_BURST ) ) {
86 // Drop the contents, not the thrown item
87 if( g->u.sees( pt ) ) {
88 add_msg( _( "The %s bursts!" ), drop_item.tname() );
89 }
90
91 // copies the drop item to spill the contents
92 item( drop_item ).spill_contents( pt );
93
94 // TODO: Sound
95 return;
96 }
97
98 // Copy the item
99 item dropped_item = drop_item;
100
101 monster *mon = dynamic_cast<monster *>( attack.hit_critter );
102
103 // We can only embed in monsters
104 bool mon_there = mon != nullptr && !mon->is_dead_state();
105 // And if we actually want to embed
106 bool embed = mon_there && !proj.has_effect( ammo_effect_NO_EMBED ) &&
107 !proj.has_effect( ammo_effect_TANGLE );
108 // Don't embed in small creatures
109 if( embed ) {
110 const m_size critter_size = mon->get_size();
111 const units::volume vol = dropped_item.volume();
112 embed = embed && ( critter_size > MS_TINY || vol < 250_ml );
113 embed = embed && ( critter_size > MS_SMALL || vol < 500_ml );
114 // And if we deal enough damage
115 // Item volume bumps up the required damage too
116 embed = embed &&
117 ( attack.dealt_dam.type_damage( DT_CUT ) / 2 ) +
118 attack.dealt_dam.type_damage( DT_STAB ) >
119 attack.dealt_dam.type_damage( DT_BASH ) +
120 vol * 3 / 250_ml + rng( 0, 5 );
121 }
122
123 if( embed ) {
124 mon->add_item( dropped_item );
125 if( g->u.sees( *mon ) ) {
126 add_msg( _( "The %1$s embeds in %2$s!" ), dropped_item.tname(), mon->disp_name() );
127 }
128 } else {
129 bool do_drop = true;
130 // monsters that are able to be tied up will store the item another way
131 // see monexamine.cpp tie_or_untie()
132 // if they aren't friendly they will try and break out of the net/bolas/lasso
133 // players and NPCs just get the downed effect, and item is dropped.
134 // TODO: storing the item on player until they recover from downed
135 if( proj.has_effect( ammo_effect_TANGLE ) && mon_there ) {
136 do_drop = false;
137 }
138 if( proj.has_effect( ammo_effect_ACT_ON_RANGED_HIT ) ) {
139 // Don't drop if it exploded
140 do_drop = !dropped_item.process( nullptr, attack.end_point, true );
141 }
142
143 map &here = get_map();
144 if( do_drop ) {
145 here.add_item_or_charges( attack.end_point, dropped_item );
146 }
147
148 if( proj.has_effect( ammo_effect_HEAVY_HIT ) ) {
149 if( here.has_flag( flag_LIQUID, pt ) ) {
150 sounds::sound( pt, 10, sounds::sound_t::combat, _( "splash!" ), false, "bullet_hit", "hit_water" );
151 } else {
152 sounds::sound( pt, 8, sounds::sound_t::combat, _( "thud." ), false, "bullet_hit", "hit_wall" );
153 }
154 const trap &tr = here.tr_at( pt );
155 if( tr.triggered_by_item( dropped_item ) ) {
156 tr.trigger( pt, nullptr, &dropped_item );
157 }
158 }
159 }

174projectile_attack_aim projectile_attack_roll( const dispersion_sources &dispersion, double range,
175 double target_size )
176{
177 projectile_attack_aim aim;
178
179 // dispersion is a measure of the dispersion of shots due to the gun + shooter characteristics
180 // i.e. it is independent of any particular shot
181
182 // shot_dispersion is the actual dispersion for this particular shot, i.e.
183 // the error angle between where the shot was aimed and where this one actually went
184 // NB: some cases pass dispersion == 0 for a "never misses" shot e.g. bio_magnet,
185 aim.dispersion = dispersion.roll();
186
187 // an isosceles triangle is formed by the intended and actual target tiles
188 aim.missed_by_tiles = iso_tangent( range, units::from_arcmin( aim.dispersion ) );
189
190 // fraction we missed a monster target by (0.0 = perfect hit, 1.0 = miss)
191 if( target_size > 0.0 ) {
192 aim.missed_by = std::min( 1.0, aim.missed_by_tiles / target_size );
193 } else {
194 // Special case 0 size targets, just to be safe from 0.0/0.0 NaNs
195 aim.missed_by = 1.0;
196 }
197
198 return aim;
199}

514double hit_chance( const dispersion_sources &dispersion, double range, double target_size,
515 double missed_by )
516{
517 if( range <= 0 ) {
518 return 1.0;
519 }
520
521 double missed_by_tiles = missed_by * target_size;
522
523 // T = (2*D**2 * (1 - cos V)) ** 0.5 (from iso_tangent)
524 // cos V = 1 - T**2 / (2*D**2)
525 double cosV = 1 - missed_by_tiles * missed_by_tiles / ( 2 * range * range );
526 double needed_dispersion = ( cosV < -1.0 ? M_PI : acos( cosV ) ) * 180 * 60 / M_PI;
527
528 return normal_cdf( needed_dispersion, dispersion.avg(), dispersion.avg() / 2 );
529}
