178 for( JsonObject type : jo.get_array( "spawn_types" ) ) {
179 if( type.has_object( "vehicle_json" ) ) {

188 ( builtin_functions[type.get_string( "vehicle_function" )] ) );
189 } else {
190 type.throw_error( "load_vehicle_spawn: missing required vehicle_json (object) or vehicle_function (string)." );

221static void builtin_jackknifed_semi( map &m, const std::string &terrainid )
222{
223 const VehicleLocation *loc = vplacement_id( terrainid + "_semi" ).obj().pick();
224 if( !loc ) {
225 debugmsg( "builtin_jackknifed_semi unable to get location to place vehicle. placement %s_semi",
226 terrainid );
227 return;
228 }
229
230 const units::angle facing = loc->pick_facing();
231 int facing_degrees = std::lround( to_degrees( facing ) );
232 const point semi_p = loc->pick_point();
233 point trailer_p;
234
235 if( facing_degrees == 0 ) {
236 trailer_p.x = semi_p.x + 4;
237 trailer_p.y = semi_p.y - 10;
238 } else if( facing_degrees == 90 ) {
239 trailer_p.x = semi_p.x + 12;
240 trailer_p.y = semi_p.y + 1;
241 } else if( facing_degrees == 180 ) {
242 trailer_p.x = semi_p.x - 4;
243 trailer_p.y = semi_p.y + 10;

246 trailer_p.y = semi_p.y - 1;
247 }
248
249 m.add_vehicle( vgroup_id( "semi_truck" ), semi_p,
250 units::fmod( facing + 135_degrees, 360_degrees ), -1, 1 );
251 m.add_vehicle( vgroup_id( "truck_trailer" ), trailer_p,
252 units::fmod( facing + 90_degrees, 360_degrees ), -1, 1 );
253}
