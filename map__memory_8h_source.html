30struct mm_submap {

38 /** Whether this mm_submap is empty. Empty submaps are skipped during saving. */

40 return tiles.empty() && symbols.empty();
41 }
42
43 inline const memorized_terrain_tile &tile( const point &p ) const {
44 if( tiles.empty() ) {
45 return default_tile;
46 } else {
47 return tiles[p.y * SEEX + p.x];
48 }
49 }
50
51 inline void set_tile( const point &p, const memorized_terrain_tile &value ) {
52 if( tiles.empty() ) {
53 // call 'reserve' first to force allocation of exact size
54 tiles.reserve( SEEX * SEEY );
55 tiles.resize( SEEX * SEEY, default_tile );
56 }
57 tiles[p.y * SEEX + p.x] = value;
58 }
59
60 inline int symbol( const point &p ) const {
61 if( symbols.empty() ) {
62 return default_symbol;
63 } else {
64 return symbols[p.y * SEEX + p.x];
65 }
66 }
67
68 inline void set_symbol( const point &p, int value ) {
69 if( symbols.empty() ) {
70 // call 'reserve' first to force allocation of exact size
71 symbols.reserve( SEEX * SEEY );
72 symbols.resize( SEEX * SEEY, default_symbol );
73 }
74 symbols[p.y * SEEX + p.x] = value;
75 }

77 void serialize( JsonOut &jsout ) const;
78 void deserialize( JsonIn &jsin );

80 private:
