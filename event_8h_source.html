544class event
545{
546 public:
547 using data_type = std::map<std::string, cata_variant>;
548
549 event( event_type type, time_point time, data_type &&data )
550 : type_( type )
551 , time_( time )
552 , data_( std::move( data ) )
553 {}
554
555 // Call this to construct an event in a type-safe manner. It will
556 // verify that the types you pass match the expected types for the
557 // event_type you pass as a template parameter.
558 template<event_type Type, typename... Args>
559 static event make( Args &&... args ) {
560 using Spec = event_detail::event_spec<Type>;
561 // Using is_empty mostly just to verify that the type is defined at
562 // all, but it so happens that it ought to be empty too.
563 static_assert( std::is_empty<Spec>::value,
564 "spec for this event type must be defined and empty" );
565 static_assert( sizeof...( Args ) == Spec::fields.size(),
566 "wrong number of arguments for event type" );
567
568 return event_detail::make_event_helper <
569 Type, std::make_index_sequence<sizeof...( Args )>
570 > ()( calendar::turn, std::forward<Args>( args )... );
571 }
572
573 using fields_type = std::unordered_map<std::string, cata_variant_type>;
574 static fields_type get_fields( event_type );
575
576 event_type type() const {
577 return type_;
578 }
579 time_point time() const {
580 return time_;
581 }
582
583 cata_variant get_variant( const std::string &key ) const {
584 auto it = data_.find( key );
585 if( it == data_.end() ) {
586 debugmsg( "No such key %s in event of type %s", key,
587 io::enum_to_string( type_ ) );
588 abort();
589 }
590 return it->second;
591 }
592
593 cata_variant get_variant_or_void( const std::string &key ) const {
594 auto it = data_.find( key );
595 if( it == data_.end() ) {
596 return cata_variant();
597 }
598 return it->second;
599 }
600
601 template<cata_variant_type Type>
602 auto get( const std::string &key ) const {
603 return get_variant( key ).get<Type>();
604 }
605
606 template<typename T>
607 auto get( const std::string &key ) const {
608 return get_variant( key ).get<T>();
609 }
610
611 const data_type &data() const {
612 return data_;
613 }
614 private:
615 event_type type_;
616 time_point time_;
617 data_type data_;
618};
