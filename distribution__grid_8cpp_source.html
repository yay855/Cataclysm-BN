182distribution_grid &distribution_grid_tracker::make_distribution_grid_at(
183 const tripoint_abs_sm &sm_pos )
184{
185 if( !get_option<bool>( "ELECTRIC_GRID" ) ) {
186 return empty_grid;
187 }
188 const std::set<tripoint_abs_omt> overmap_positions = overmap_buffer.electric_grid_at(
189 project_to<coords::omt>( sm_pos ) );
190 assert( !overmap_positions.empty() );
191 std::vector<tripoint_abs_sm> submap_positions;
192 for( const tripoint_abs_omt &omp : overmap_positions ) {
193 tripoint_abs_sm tp = project_to<coords::sm>( omp );
194 submap_positions.emplace_back( tp + point_zero );
195 submap_positions.emplace_back( tp + point_east );
196 submap_positions.emplace_back( tp + point_south );
197 submap_positions.emplace_back( tp + point_south_east );
198 }
199 shared_ptr_fast<distribution_grid> dist_grid = make_shared_fast<distribution_grid>
200 ( submap_positions, mb );
201 for( const tripoint_abs_sm &smp : submap_positions ) {
202 shared_ptr_fast<distribution_grid> &old_grid = parent_distribution_grids[smp];
203 if( old_grid != dist_grid ) {
204 grids_requiring_updates.erase( old_grid );
205 old_grid = dist_grid;
206 }
207 }
208
209 if( !dist_grid->empty() ) {
210 grids_requiring_updates.emplace( dist_grid );
211 }
212
213 // This ugly expression + lint suppresion are needed to convince clang-tidy
214 // that we are, in fact, NOT leaking memory.
215 return *parent_distribution_grids[submap_positions.front()];
216 // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDeleteLeaks)
217}

222void distribution_grid_tracker::on_saved()
223{
224 if( !get_option<bool>( "ELECTRIC_GRID" ) ||
225 world_generator->active_world == nullptr ) {
226 return;

228 tripoint_abs_sm min_bounds( bounds.p_min, -OVERMAP_DEPTH );
229 tripoint_abs_sm max_bounds( bounds.p_max, OVERMAP_HEIGHT );
230 tripoint_range<tripoint_abs_sm> bounds_range( min_bounds, max_bounds );
231 // Remove all grids that are no longer in the bounds
232 for( auto iter = parent_distribution_grids.begin(); iter != parent_distribution_grids.end(); ) {
233 if( !bounds_range.is_point_inside( iter->first ) ) {
234 grids_requiring_updates.erase( iter->second );
235 iter = parent_distribution_grids.erase( iter );
236 } else {
237 ++iter;
238 }
239 }
240 for( const tripoint_abs_sm &sm_pos : bounds_range ) {
241 if( parent_distribution_grids.find( sm_pos ) == parent_distribution_grids.end() ) {
242 make_distribution_grid_at( sm_pos );
