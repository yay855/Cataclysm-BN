66 for( auto &entry : color_array ) {
67 entry.invert = get( entry.invert_id );
68
69 if( !entry.name_custom.empty() ) {
70 // Not using name_to_color because we want default color of this name
71 const auto id = name_to_id( entry.name_custom );
72 auto &other = color_array[id];
73 entry.custom = other.color;
74 }
75
76 if( !entry.name_invert_custom.empty() ) {
77 const auto id = name_to_id( entry.name_invert_custom );
78 auto &other = color_array[id];
79 entry.custom = other.color;
80 }
81
82 inverted_map[entry.color] = entry.col_id;
83 inverted_map[entry.invert] = entry.invert_id;
84 }
85
86 // Highlights in a next run, to make sure custom colors are set
87 for( auto &entry : color_array ) {
88 const std::string my_name = get_name( entry.color );
89 const std::string root = my_name.substr( 2, my_name.length() - 2 );
90 const size_t underscore_num = std::count( root.begin(), root.end(), '_' ) -
91 ( root.find( "light_" ) != std::string::npos ) -
92 ( root.find( "dark_" ) != std::string::npos );
93 // do not try to highlight color pairs, highlighted, background, and invalid colors
94 if( my_name.substr( 0, 2 ) == "c_" && root != "unset" && underscore_num < 1 ) {
95 for( size_t j = 0; j < NUM_HL; j++ ) {
96 entry.highlight[j] = highlight_from_names( my_name, hilights[j] );
97 }
98 }
99 }

551/**
552 * Given the name of a foreground color, returns the nc_color value that matches. If
553 * no match is found, c_unset is returned.
554 * Special cases:
555 * {"black" , c_black}, // missing default prefix c_
556 * {"<c|h|i>_black" , h_black}, // has prefix c_ or h_ or i_
557 * {"dark_gray_red" , c_dark_gray_red}, // use dark_ instead of dk as the latter is being deprecated
558 * {"light_blue_red" , c_light_blue_red}, // use light_ instead of lt as the latter is being deprecated
559 * @param color The color to get, as a std::string.
560 * @return The nc_color constant that matches the input.
561 */
562nc_color color_from_string( const std::string &color,
563 const report_color_error color_error )
564{
565 if( color.empty() ) {
566 return c_unset;
567 }
568 std::string new_color = color;
569 if( new_color.substr( 1, 1 ) != "_" ) { //c_ //i_ //h_
570 new_color = "c_" + new_color;
571 }
572
573 const std::pair<std::string, std::string> pSearch[2] = { { "light_", "lt" }, { "dark_", "dk" } };
574 for( const auto &i : pSearch ) {
575 size_t pos = 0;
576 while( ( pos = new_color.find( i.second, pos ) ) != std::string::npos ) {
577 new_color.replace( pos, i.second.length(), i.first );
578 pos += i.first.length();
579 if( color_error == report_color_error::yes ) {
580 debugmsg( "Deprecated foreground color suffix was used: (%s) in (%s). Please update mod that uses that.",
581 i.second, color );
582 }
583 }
584 }
585
586 const nc_color col = all_colors.name_to_color( new_color, color_error );
587 if( col > 0 ) {
588 return col;
589 }
590
591 return c_unset;
592}

601 if( sColor != "c_unset" ) {

639color_tag_parse_result get_color_from_tag( const std::string &s,
640 const report_color_error color_error )
641{
642 if( s.empty() || s[0] != '<' ) {
643 return { color_tag_parse_result::non_color_tag, {} };
644 }
645 if( s.substr( 0, 8 ) == "</color>" ) {
646 return { color_tag_parse_result::close_color_tag, {} };
647 }
648 if( s.substr( 0, 7 ) != "<color_" ) {
649 return { color_tag_parse_result::non_color_tag, {} };
650 }
651 size_t tag_close = s.find( '>' );
652 if( tag_close == std::string::npos ) {
653 return { color_tag_parse_result::non_color_tag, {} };
654 }
655 std::string color_name = s.substr( 7, tag_close - 7 );
656 const nc_color color = color_from_string( color_name, color_error );
657 if( color != c_unset ) {
658 return { color_tag_parse_result::open_color_tag, color };
659 } else {
660 return { color_tag_parse_result::non_color_tag, color };
661 }
662}
