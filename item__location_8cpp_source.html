36template <typename T>
37static int find_index( const T &sel, const item *obj )
38{
39 int idx = -1;
40 sel.visit_items( [&idx, &obj]( const item * e ) {
41 idx++;
42 if( e == obj ) {
43 return VisitResponse::ABORT;
44 }
45 return VisitResponse::NEXT;
46 } );
47 return idx;
48}
49
50template <typename T>
51static item *retrieve_index( const T &sel, int idx )
52{
53 item *obj = nullptr;
54 sel.visit_items( [&idx, &obj]( const item * e ) {
55 if( idx-- == 0 ) {
56 obj = const_cast<item *>( e );
57 return VisitResponse::ABORT;
58 }
59 return VisitResponse::NEXT;
60 } );
61 return obj;
62}

679void item_location::deserialize( JsonIn &js )
680{
681 auto obj = js.get_object();
682 auto type = obj.get_string( "type" );
683
684 int idx = -1;
685 tripoint pos = tripoint_min;
686
687 obj.read( "idx", idx );
688 obj.read( "pos", pos );
689
690 if( type == "character" ) {
691 character_id who_id;
692 if( obj.has_member( "character" ) ) {
693 obj.read( "character", who_id );
694 } else {
695 // This is for migrating saves before npc item locations were supported and all
696 // character item locations were assumed to be on g->u
697 who_id = g->u.getID();
698 }
699 ptr.reset( new impl::item_on_person( who_id, idx ) );
700
701 } else if( type == "map" ) {
702 ptr.reset( new impl::item_on_map( pos, idx ) );
703
704 } else if( type == "vehicle" ) {
705 vehicle *const veh = veh_pointer_or_null( get_map().veh_at( pos ) );
706 int part = obj.get_int( "part" );
707 if( veh && part >= 0 && part < veh->part_count() ) {
708 ptr.reset( new impl::item_on_vehicle( vehicle_cursor( *veh, part ), idx ) );
709 }
710 } else if( type == "in_container" ) {
711 item_location parent;
712 obj.read( "parent", parent );
713 const std::list<item *> parent_contents = parent->contents.all_items_top();
714 auto iter = parent_contents.begin();
715 std::advance( iter, idx );
716 ptr.reset( new impl::item_in_container( parent, *iter ) );
717 }
718}
