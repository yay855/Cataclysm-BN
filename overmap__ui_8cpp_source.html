202std::tuple<char, nc_color, size_t> get_note_display_info( const std::string &note )
203{
204 std::tuple<char, nc_color, size_t> result {'N', c_yellow, 0};
205 bool set_color = false;
206 bool set_symbol = false;
207
208 size_t pos = 0;
209 for( int i = 0; i < 2; ++i ) {
210 // find the first non-whitespace non-delimiter
211 pos = note.find_first_not_of( " :;", pos, 3 );
212 if( pos == std::string::npos ) {
213 return result;
214 }
215
216 // find the first following delimiter
217 const auto end = note.find_first_of( " :;", pos, 3 );
218 if( end == std::string::npos ) {
219 return result;
220 }
221
222 // set color or symbol
223 if( !set_symbol && note[end] == ':' ) {
224 std::get<0>( result ) = note[end - 1];
225 std::get<2>( result ) = end + 1;
226 set_symbol = true;
227 } else if( !set_color && note[end] == ';' ) {
228 std::get<1>( result ) = get_note_color( note.substr( pos, end - pos ) );
229 std::get<2>( result ) = end + 1;
230 set_color = true;
231 }
232
233 pos = end + 1;
234 }
235
236 return result;
237}

768 } else if( nmenu.ret >= 0 && nmenu.ret < static_cast<int>( notes.size() ) ) {
769 result = notes[nmenu.ret].p;
770 quit = true;
771 } else {
772 quit = true;
773 }
774 }

1857 params = overmap_path_params::for_watercraft();

1860 player_veh->average_or_rating() );
1861 const bool tiny = player_veh->get_points().size() <= 3;
1862 params = overmap_path_params::for_land_vehicle( offroad_coeff, tiny, can_float );

1865 }
1866 } else {
1867 params = overmap_path_params::for_player();
