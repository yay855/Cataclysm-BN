147/**
148 * Multiplication and division with scalars. Result is a quantity with the same unit
149 * as the input.
150 * Functions are templated to allow scaling with different types:
151 * \code
152 * quantity<int, foo> a{ 10, foo{} };
153 * auto b = a * 4.52;
154 * static_assert(std::is_same<decltype(b), quantity<double, foo>>::value);
155 * \endcode
156 *
157 * Note that the result for `*=` and `/=` is calculated using the given types, but is
158 * implicitly converted back to `value_type` as it is stored in the operand.
159 * \code
160 * quantity<int, foo> a{ 10, foo{} };
161 * a *= 4.52;
162 * assert( a == quantity<int, foo>( 10 * 4.52, foo{} ) );
163 * assert( a != quantity<int, foo>( 10 * (int)4.52, foo{} ) );
164 * assert( a == quantity<int, foo>( 45, foo{} ) );
165 * \endcode
166 *
167 * Division of a quantity with a quantity of the same unit yields a dimensionless
168 * scalar value, with the same type as the division of the contained `value_type`s:
169 * \code
170 * quantity<int, foo> a{ 10, foo{} };
171 * quantity<double, foo> b{ 20, foo{} };
172 * auto proportion = a / b;
173 * static_assert(std::is_same<decltype(proportion), double>::value);
174 * assert( proportion == 10 / 20.0 );
175 * \endcode
176 *
177 */
178/**@{*/
179// the decltype in the result type ensures the returned type has the same scalar type
180// as you would get when performing the operation directly:
181// `int * double` => `double` and `char * int` => `int`
182// st is scalar type (dimensionless)
183// lvt is the value type (of a quantity) on the left side, rvt is the value type on the right side
184// ut is unit type (same for left and right side)
185// The enable_if ensures no ambiguity, the compiler may otherwise not be able to decide whether
186// "quantity / scalar" or "quantity / other_quanity" is meant.
187
188// scalar * quantity<foo, unit> == quantity<decltype(foo * scalar), unit>
189template<typename lvt, typename ut, typename st, typename = typename std::enable_if<std::is_arithmetic<st>::value>::type>
190inline constexpr quantity<decltype( std::declval<lvt>() * std::declval<st>() ), ut>
191operator*( const st &factor, const quantity<lvt, ut> &rhs )
192{
193 static_assert( quantity_details<ut>::common_zero_point::value,
194 "Units with multiple scales with different zero should not be multiplied/divided/etc. directly." );
195 return { factor * rhs.value(), ut{} };
196}
197
198// same as above only with inverse order of operands: quantity * scalar
199template<typename lvt, typename ut, typename st, typename = typename std::enable_if<std::is_arithmetic<st>::value>::type>
200inline constexpr quantity<decltype( std::declval<st>() * std::declval<lvt>() ), ut>
201operator*( const quantity<lvt, ut> &lhs, const st &factor )
202{
203 static_assert( quantity_details<ut>::common_zero_point::value,
204 "Units with multiple scales with different zero should not be multiplied/divided/etc. directly." );
205 return { lhs.value() *factor, ut{} };
206}
