96float clothing_mod::get_mod_val( const clothing_mod_type &type, const item &it ) const
97{
98 const int thickness = it.get_thickness();
99 const int coverage = it.get_coverage();
100 float result = 0.0f;
101 for( const mod_value &mv : mod_values ) {
102 if( mv.type == type ) {
103 float tmp = mv.value;
104 if( mv.thickness_propotion ) {
105 tmp *= thickness;
106 }
107 if( mv.coverage_propotion ) {
108 tmp *= coverage / 100.0f;
109 }
110 if( mv.round_up ) {
111 tmp = std::ceil( tmp );
112 }
113 result += tmp;
114 }
115 }
116 return result;
