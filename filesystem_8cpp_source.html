156//--------------------------------------------------------------------------------------------------
157// For non-empty path, call function for each file at path.
158//--------------------------------------------------------------------------------------------------
159template <typename Function>
160void for_each_dir_entry( const std::string &path, Function function )
161{
162 using dir_ptr = DIR*;
163
164 if( path.empty() ) {
165 return;
166 }
167
168 const dir_ptr root = opendir( path.c_str() );
169 if( !root ) {
170 const auto e_str = strerror( errno );
171 dbg( DL::Warn ) << "opendir [" << path << "] failed with \"" << e_str << "\".";
172 return;
173 }
174
175 while( const auto entry = readdir( root ) ) {
176 function( *entry );
177 }
178 closedir( root );
179}

271//--------------------------------------------------------------------------------------------------
272// If at_end is true, returns whether entry's name ends in match.
273// Otherwise, returns whether entry's name contains match.
274//--------------------------------------------------------------------------------------------------
275bool name_contains( const dirent &entry, const std::string &match, const bool at_end )
276{
277 const size_t len_fname = strlen( entry.d_name );
278 const size_t len_match = match.length();
279
280 if( len_match > len_fname ) {
281 return false;
282 }
283
284 const size_t offset = at_end ? ( len_fname - len_match ) : 0;
285 return strstr( entry.d_name + offset, match.c_str() ) != nullptr;
286}

288//--------------------------------------------------------------------------------------------------
289// Return every file at root_path matching predicate.
290//
291// If root_path is empty, search the current working directory.
292// If root_path does not exist, returns empty list.
293// If recursive_search is true, search breadth-first into the directory hierarchy.
294//
295// Results are ordered depth-first with directories searched in lexically order. Furthermore,
296// regular files at each level are also ordered lexically by file name.
297//
298// Files ending in ~ are excluded.
299//--------------------------------------------------------------------------------------------------
300template <typename Predicate>
301std::vector<std::string> find_file_if_bfs( const std::string &root_path,
302 const bool recursive_search,
303 Predicate predicate )
304{
305 std::deque<std::string> directories;
306 if( root_path.empty() ) {
307 directories.emplace_back( "./" );
308 } else if( string_ends_with( root_path, "/" ) ) {
309 directories.emplace_back( root_path );
310 } else {
311 directories.emplace_back( root_path + "/" );
312 }
313 std::vector<std::string> results;
314 if( !dir_exist( *directories.begin() ) ) {
315 return results;
316 }
317
318 while( !directories.empty() ) {
319 const auto path = std::move( directories.front() );
320 directories.pop_front();
321
322 const auto n_dirs = static_cast<std::ptrdiff_t>( directories.size() );
323 const auto n_results = static_cast<std::ptrdiff_t>( results.size() );
324
325 for_each_dir_entry( path, [&]( const dirent & entry ) {
326 // exclude special directories.
327 if( is_special_dir( entry ) ) {
328 return;
329 }
330
331 const auto full_path = path + entry.d_name;
332
333 // don't add files ending in '~'.
334 if( full_path.back() == '~' ) {
335 return;
336 }
337
338 // add sub directories to recursive_search if requested
339 const auto is_dir = is_directory( entry, full_path );
340 if( recursive_search && is_dir ) {
341 directories.emplace_back( full_path + "/" );
342 }
343
344 // check the file
345 if( !predicate( entry, is_dir ) ) {
346 return;
347 }
348
349 results.emplace_back( full_path );
350 } );
351
352 // Keep files and directories to recurse ordered consistently
353 // by sorting from the old end to the new end.
354 // NOLINTNEXTLINE(cata-use-localized-sorting)
355 std::sort( std::begin( directories ) + n_dirs, std::end( directories ) );
356 // NOLINTNEXTLINE(cata-use-localized-sorting)
357 std::sort( std::begin( results ) + n_results, std::end( results ) );
358 }
359
360 return results;
361}

374/**
375 * Find directories which containing pattern.
376 * @param pattern Search pattern.
377 * @param root_path Search root.
378 * @param recursive_search Be recurse or not.
379 * @return vector or directories without pattern filename at end.
380 */
381std::vector<std::string> get_directories_with( const std::string &pattern,
382 const std::string &root_path, const bool recursive_search )
383{
384 if( pattern.empty() ) {
385 return std::vector<std::string>();
386 }
387
388 auto files = find_file_if_bfs( root_path, recursive_search, [&]( const dirent & entry, bool ) {
389 return name_contains( entry, pattern, true );
390 } );
391
392 // Chop off the file names. Dir path MUST be splitted by '/'
393 for( auto &file : files ) {
394 file.erase( file.rfind( '/' ), std::string::npos );

396

402/**
403 * Find directories which containing pattern.
404 * @param patterns Search patterns.
405 * @param root_path Search root.
406 * @param recursive_search Be recurse or not.
407 * @return vector or directories without pattern filename at end.
408 */
409std::vector<std::string> get_directories_with( const std::vector<std::string> &patterns,
410 const std::string &root_path, const bool recursive_search )
411{
412 if( patterns.empty() ) {
413 return std::vector<std::string>();
414 }
415
416 const auto ext_beg = std::begin( patterns );
417 const auto ext_end = std::end( patterns );
418

420 return std::any_of( ext_beg, ext_end, [&]( const std::string & ext ) {
421 return name_contains( entry, ext, true );
422 } );
423 } );

427 file.erase( file.rfind( '/' ), std::string::npos );
428 }
429
430 //remove resulting duplicates
431 files.erase( std::unique( std::begin( files ), std::end( files ) ), std::end( files ) );
432
