141 for( auto &elem : task_priorities ) {
142 elem.second = 0;

146 for( auto &elem : task_priorities ) {
147 if( elem.second > 0 ) {
148 return true;
149 }

151 return false;

155 if( it != task_priorities.end() ) {
156 return it->second;
157 } else {
158 return 0;
159 }
160 }

162 std::vector<std::pair<activity_id, int>> pairs( begin( task_priorities ), end( task_priorities ) );

165 sort( begin( pairs ), end( pairs ), []( const std::pair<activity_id, int> &a,
166 const std::pair<activity_id, int> &b ) {
167 return a.second > b.second;
168 } );
169 for( std::pair<activity_id, int> elem : pairs ) {
170 ret.push_back( elem.first );
171 }
172 return ret;
