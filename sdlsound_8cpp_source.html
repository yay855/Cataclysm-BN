103 * Attempt to initialize an audio device. Returns false if initialization fails.

105bool init_sound()
106{
107 int audio_rate = 44100;
108 Uint16 audio_format = AUDIO_S16;
109 int audio_channels = 2;
110 int chunksize = 2048;
111 // Autodetect device
112 const char *audio_device = nullptr;
113 // We only care about sample format and number of channels.
114 // SDL will be allowed to choose a different frequency if it needs to.
115 int flags = SDL_AUDIO_ALLOW_FREQUENCY_CHANGE;
116
117 // We should only need to init once
118 if( !sound_init_success ) {
119 const char *driver = SDL_GetCurrentAudioDriver();
120 DebugLog( DL::Info, DC::Main ) << "Active audio driver: " << driver;
121
122 int open_code = Mix_OpenAudioDevice( audio_rate, audio_format, audio_channels,
123 chunksize, audio_device, flags );
124 if( open_code == 0 ) {
125 int dev_rate = 0;
126 Uint16 dev_format = 0;
127 int dev_channels = 0;
128 if( Mix_QuerySpec( &dev_rate, &dev_format, &dev_channels ) ) {
129 DebugLog( DL::Info, DC::Main )
130 << string_format( "Opened mixer with specs: %d Hz, %d channel(s), format %#x",
131 dev_rate, dev_channels, dev_format );
132 } else {
133 DebugLog( DL::Warn, DC::SDL ) << "Mix_QuerySpec failed: " << Mix_GetError();
134 }
135
136 Mix_AllocateChannels( 128 );
137 Mix_ReserveChannels( static_cast<int>( sfx::channel::MAX_CHANNEL ) );
138
139 // For the sound effects system.
140 Mix_GroupChannels( static_cast<int>( sfx::channel::daytime_outdoors_env ),
141 static_cast<int>( sfx::channel::nighttime_outdoors_env ),
142 static_cast<int>( sfx::group::time_of_day ) );
143 Mix_GroupChannels( static_cast<int>( sfx::channel::underground_env ),
144 static_cast<int>( sfx::channel::outdoor_blizzard ),
145 static_cast<int>( sfx::group::weather ) );
146 Mix_GroupChannels( static_cast<int>( sfx::channel::danger_extreme_theme ),
147 static_cast<int>( sfx::channel::danger_low_theme ),
148 static_cast<int>( sfx::group::context_themes ) );
149 Mix_GroupChannels( static_cast<int>( sfx::channel::stamina_75 ),
150 static_cast<int>( sfx::channel::stamina_35 ),
151 static_cast<int>( sfx::group::fatigue ) );
152
153 sound_init_success = true;
154 } else {
155 dbg( DL::Error ) << "Failed to open audio mixer, sound won't work: " << Mix_GetError();
156 }
157 }
158
159 return sound_init_success;

200/** Callback called when we finish playing music. */
201void musicFinished()
202{
203 if( test_mode ) {
204 return;
205 }
206
207 Mix_HaltMusic();
208 Mix_FreeMusic( current_music );
209 current_music = nullptr;
210
211 const auto iter = playlists.find( current_playlist );
212 if( iter == playlists.end() ) {
213 return;
214 }
215 const music_playlist &list = iter->second;
216 if( list.entries.empty() ) {
217 return;
218 }
219
220 // Load the next file to play.
221 absolute_playlist_at++;
222
223 // Wrap around if we reached the end of the playlist.
224 if( absolute_playlist_at >= list.entries.size() ) {
225 absolute_playlist_at = 0;
226 }
227
228 current_playlist_at = playlist_indexes.at( absolute_playlist_at );
229
230 const music_playlist::entry &next = list.entries[current_playlist_at];
231 play_music_file( next.file, next.volume );
232}

234void play_music( const std::string &playlist )
235{
236 const auto iter = playlists.find( playlist );
237 if( iter == playlists.end() ) {
238 return;
239 }
240 const music_playlist &list = iter->second;
241 if( list.entries.empty() ) {
242 return;
243 }
244
245 // Don't interrupt playlist that's already playing.
246 if( playlist == current_playlist ) {
247 return;
248 }
249
250 for( size_t i = 0; i < list.entries.size(); i++ ) {
251 playlist_indexes.push_back( i );
252 }
253 if( list.shuffle ) {
254 // Son't need to worry about the determinism check here because it only
255 // affects audio, not game logic.
256 // NOLINTNEXTLINE(cata-determinism)
257 static auto eng = cata_default_random_engine(
258 std::chrono::system_clock::now().time_since_epoch().count() );
259 std::shuffle( playlist_indexes.begin(), playlist_indexes.end(), eng );
260 }
261
262 current_playlist = playlist;
263 current_playlist_at = playlist_indexes.at( absolute_playlist_at );
264
265 const music_playlist::entry &next = list.entries[current_playlist_at];
266 play_music_file( next.file, next.volume );
267}

469static Mix_Chunk *do_pitch_shift( Mix_Chunk *s, float pitch )
470{
471 Uint32 s_in = s->alen / 4;
472 Uint32 s_out = static_cast<Uint32>( static_cast<float>( s_in ) * pitch );
473 float pitch_real = static_cast<float>( s_out ) / static_cast<float>( s_in );
474 Mix_Chunk *result = static_cast<Mix_Chunk *>( malloc( sizeof( Mix_Chunk ) ) );
475 result->allocated = 1;
476 result->alen = s_out * 4;
477 result->abuf = static_cast<Uint8 *>( malloc( result->alen * sizeof( Uint8 ) ) );
478 result->volume = s->volume;
479 for( Uint32 i = 0; i < s_out; i++ ) {
480 Sint16 lt = 0;
481 Sint16 rt = 0;
482 Sint16 lt_out = 0;
483 Sint16 rt_out = 0;
484 Sint64 lt_avg = 0;
485 Sint64 rt_avg = 0;
486 Uint32 begin = static_cast<Uint32>( static_cast<float>( i ) / pitch_real );
487 Uint32 end = static_cast<Uint32>( static_cast<float>( i + 1 ) / pitch_real );
488
489 // check for boundary case
490 if( end > 0 && ( end >= ( s->alen / 4 ) ) ) {
491 end = begin;
492 }
493
494 for( Uint32 j = begin; j <= end; j++ ) {
495 lt = ( s->abuf[( 4 * j ) + 1] << 8 ) | ( s->abuf[( 4 * j ) + 0] );
496 rt = ( s->abuf[( 4 * j ) + 3] << 8 ) | ( s->abuf[( 4 * j ) + 2] );
497 lt_avg += lt;
498 rt_avg += rt;
499 }
500 lt_out = static_cast<Sint16>( static_cast<float>( lt_avg ) / static_cast<float>
501 ( end - begin + 1 ) );
502 rt_out = static_cast<Sint16>( static_cast<float>( rt_avg ) / static_cast<float>
503 ( end - begin + 1 ) );
504 result->abuf[( 4 * i ) + 1] = static_cast<Uint8>( ( lt_out >> 8 ) & 0xFF );
505 result->abuf[( 4 * i ) + 0] = static_cast<Uint8>( lt_out & 0xFF );
506 result->abuf[( 4 * i ) + 3] = static_cast<Uint8>( ( rt_out >> 8 ) & 0xFF );
507 result->abuf[( 4 * i ) + 2] = static_cast<Uint8>( rt_out & 0xFF );
508 }
509 return result;
510}

541void sfx::play_variant_sound( const std::string &id, const std::string &variant, int volume,
542 units::angle angle, double pitch_min, double pitch_max )
543{
544 if( test_mode ) {
545 return;
546 }
547
548 add_msg( m_debug, "sound id: %s, variant: %s, volume: %d ", id, variant, volume );
549
550 if( !check_sound( volume ) ) {
551 return;
552 }
553 const sound_effect *eff = find_random_effect( id, variant );
554 if( eff == nullptr ) {
555 return;
556 }
557 const sound_effect &selected_sound_effect = *eff;
558
559 Mix_Chunk *effect_to_play = get_sfx_resource( selected_sound_effect.resource_id );
560 bool is_pitched = ( pitch_min > 0 ) && ( pitch_max > 0 );
561 if( is_pitched ) {
562 double pitch_random = rng_float( pitch_min, pitch_max );
563 effect_to_play = do_pitch_shift( effect_to_play, static_cast<float>( pitch_random ) );
564 }
565 Mix_VolumeChunk( effect_to_play,
566 selected_sound_effect.volume * get_option<int>( "SOUND_EFFECT_VOLUME" ) * volume / ( 100 * 100 ) );
567 int channel = Mix_PlayChannel( static_cast<int>( sfx::channel::any ), effect_to_play, 0 );
568 bool failed = ( channel == -1 );
569 if( !failed && is_pitched ) {
570 if( Mix_RegisterEffect( channel, empty_effect, cleanup_when_channel_finished,
571 effect_to_play ) == 0 ) {
572 // We'll be unable to de-allocate the chunk, stop the playback right now.
573 failed = true;
574 dbg( DL::Warn ) << "Mix_RegisterEffect failed: " << Mix_GetError();
575 Mix_HaltChannel( channel );
576 }
577 }
578 if( !failed ) {
579 if( Mix_SetPosition( channel, static_cast<Sint16>( to_degrees( angle ) ), 1 ) == 0 ) {
580 // Not critical
581 dbg( DL::Info ) << "Mix_SetPosition failed: " << Mix_GetError();
582 }
583 }
584 if( failed ) {
585 dbg( DL::Error ) << "Failed to play sound effect: " << Mix_GetError();
586 if( is_pitched ) {
587 cleanup_when_channel_finished( channel, effect_to_play );
588 }
589 }

629 Mix_HaltChannel( ch );
630 }
631 }
632 if( failed ) {
633 dbg( DL::Error ) << "Failed to play sound effect: " << Mix_GetError();
634 if( is_pitched ) {
635 cleanup_when_channel_finished( ch, effect_to_play );
636 }
637 }
638 current_ambient.id = id;
639 current_ambient.variant = variant;
640 current_ambient.volume = volume;
641 current_ambient.channel = channel;
642 current_ambient.fade_in_duration = fade_in_duration;
