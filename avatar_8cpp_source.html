342int avatar::time_to_read( const item &book, const player &reader, const player *learner ) const
343{
344 const auto &type = book.type->book;
345 const skill_id &skill = type->skill;
346 // The reader's reading speed has an effect only if they're trying to understand the book as they read it
347 // Reading speed is assumed to be how well you learn from books (as opposed to hands-on experience)
348 const bool try_understand = character_funcs::is_fun_to_read( reader, book ) ||
349 reader.get_skill_level( skill ) < type->level;
350 int reading_speed = try_understand ? std::max( reader.read_speed(), read_speed() ) : read_speed();
351 if( learner ) {
352 reading_speed = std::max( reading_speed, learner->read_speed() );
353 }
354
355 int retval = type->time * reading_speed;
356 retval *= std::min( character_funcs::fine_detail_vision_mod( *this ),
357 character_funcs::fine_detail_vision_mod( reader ) );
358
359 const int effective_int = std::min( { get_int(), reader.get_int(), learner ? learner->get_int() : INT_MAX } );
360 if( type->intel > effective_int && !reader.has_trait( trait_PROF_DICEMASTER ) ) {
361 retval += type->time * ( type->intel - effective_int ) * 100;
362 }
363 return retval;

731void avatar::do_read( item_location loc )
732{
733 if( !loc ) {
734 activity.set_to_null();
735 return;
736 }
737
738 item &book = *loc;
739 const auto &reading = book.type->book;
740 if( !reading ) {
741 activity.set_to_null();
742 return;
743 }
744 const skill_id &skill = reading->skill;
745
746 if( !has_identified( book.typeId() ) ) {
747 // Note that we've read the book.
748 items_identified.insert( book.typeId() );
749 skim_book_msg( book, *this );
750 activity.set_to_null();
751 return;
752 }
753
754 const bool allow_recipes = get_option<bool>( "ALLOW_LEARNING_BOOK_RECIPES" );
755
756 //learners and their penalties
757 std::vector<std::pair<player *, double>> learners;
758 for( size_t i = 0; i < activity.values.size(); i++ ) {
759 player *n = g->find_npc( character_id( activity.values[i] ) );
760 if( n != nullptr ) {
761 const std::string &s = activity.get_str_value( i, "1" );
762 learners.push_back( { n, strtod( s.c_str(), nullptr ) } );
763 }
764 // Otherwise they must have died/teleported or something
765 }
766 learners.push_back( { this, 1.0 } );
767 //whether to continue reading or not
768 bool continuous = false;
769 // NPCs who learned a little about the skill
770 std::set<std::string> little_learned;
771 std::set<std::string> cant_learn;
772 std::list<std::string> out_of_chapters;
773
774 for( auto &elem : learners ) {
775 player *learner = elem.first;
776
777 int book_fun_for = character_funcs::get_book_fun_for( *learner, book );
778 if( book_fun_for != 0 ) {
779 learner->add_morale( MORALE_BOOK, book_fun_for * 5, book_fun_for * 15, 1_hours, 30_minutes, true,
780 book.type );
781 }
782
783 book.mark_chapter_as_read( *learner );
784
785 const auto available_recipes = book.get_available_recipes( *learner );
786 std::vector<const recipe *> learnable_recipes;
787 for( const std::pair<const recipe *, int> &p : available_recipes ) {
788 if( allow_recipes && !learner->knows_recipe( p.first ) ) {
789 learnable_recipes.push_back( p.first );
790 learner->learn_recipe( p.first );
791 if( learner->is_player() ) {
792 add_msg( m_info, _( "You memorize a recipe for %s." ), p.first->result_name() );
793 }
794 }
795 }
796
797 if( skill && learner->get_skill_level( skill ) < reading->level &&
798 learner->get_skill_level_object( skill ).can_train() ) {
799 SkillLevel &skill_level = learner->get_skill_level_object( skill );
800 const int originalSkillLevel = skill_level.level();
801
802 // Calculate experience gained
803 /** @EFFECT_INT increases reading comprehension */
804 // Enhanced Memory Banks modestly boosts experience
805 int min_ex = std::max( 1, reading->time / 10 + learner->get_int() / 4 );
806 int max_ex = reading->time / 5 + learner->get_int() / 2 - originalSkillLevel;
807 if( has_active_bionic( bio_memory ) ) {
808 min_ex += 2;
809 }
810
811 min_ex = adjust_for_focus( min_ex );
812 max_ex = adjust_for_focus( max_ex );
813
814 if( max_ex < 2 ) {
815 max_ex = 2;
816 }
817 if( max_ex > 10 ) {
818 max_ex = 10;
819 }
820 if( max_ex < min_ex ) {
821 max_ex = min_ex;
822 }
823
824 min_ex *= ( originalSkillLevel + 1 ) * elem.second;
825 min_ex = std::max( min_ex, 1 );
826 max_ex *= ( originalSkillLevel + 1 ) * elem.second;
827 max_ex = std::max( min_ex, max_ex );
828
829 skill_level.readBook( min_ex, max_ex, reading->level );
830
831 std::string skill_name = skill.obj().name();
832
833 if( skill_level != originalSkillLevel ) {
834 g->events().send<event_type::gains_skill_level>(
835 learner->getID(), skill, skill_level.level() );
836 if( learner->is_player() ) {
837 add_msg( m_good, _( "You increase %s to level %d." ), skill.obj().name(),
838 originalSkillLevel + 1 );
839 } else {
840 add_msg( m_good, _( "%s increases their %s level." ), learner->disp_name(), skill_name );
841 }
842 } else {
843 //skill_level == originalSkillLevel
844 if( activity.index == learner->getID().get_value() ) {
845 continuous = true;
846 }
847 if( learner->is_player() ) {
848 add_msg( m_info, _( "You learn a little about %s! (%d%%)" ), skill_name, skill_level.exercise() );
849 } else {
850 little_learned.insert( learner->disp_name() );
851 }
852 }
853
854 if( learnable_recipes.empty() && ( skill_level == reading->level || !skill_level.can_train() ) ) {
855 if( learner->is_player() ) {
856 add_msg( m_info, _( "You can no longer learn from %s." ), book.type_name() );
857 } else {
858 cant_learn.insert( learner->disp_name() );
859 }
860 }
861 } else if( learnable_recipes.empty() && skill ) {
862 if( learner->is_player() ) {
863 add_msg( m_info, _( "You can no longer learn from %s." ), book.type_name() );
864 } else {
865 cant_learn.insert( learner->disp_name() );
866 }
867 }
868 }
869 //end for all learners
870
871 if( little_learned.size() == 1 ) {
872 add_msg( m_info, _( "%s learns a little about %s!" ), little_learned.begin()->c_str(),
873 skill.obj().name() );
874 } else if( !little_learned.empty() ) {
875 const std::string little_learned_msg = enumerate_as_string( little_learned );
876 add_msg( m_info, _( "%s learn a little about %s!" ), little_learned_msg, skill.obj().name() );
877 }
878
879 if( !cant_learn.empty() ) {
880 const std::string names = enumerate_as_string( cant_learn );
881 add_msg( m_info, _( "%s can no longer learn from %s." ), names, book.type_name() );
882 }
883 if( !out_of_chapters.empty() ) {
884 const std::string names = enumerate_as_string( out_of_chapters );
885 add_msg( m_info, _( "Rereading the %s isn't as much fun for %s." ),
886 book.type_name(), names );
887 if( out_of_chapters.front() == disp_name() && one_in( 6 ) ) {
888 add_msg( m_info, _( "Maybe you should find something new to read" ) );
889 }
890 }
891
892 // NPCs can't learn martial arts from manuals (yet).
893 auto m = book.type->use_methods.find( "MA_MANUAL" );
894 if( m != book.type->use_methods.end() ) {
895 const matype_id style_to_learn = martial_art_learned_from( *book.type );
896 skill_id skill_used = style_to_learn->primary_skill;
897 int difficulty = std::max( 1, style_to_learn->learn_difficulty );
898 difficulty = std::max( 1, 20 + difficulty * 2 - get_skill_level( skill_used ) * 2 );
899 add_msg( m_debug, _( "Chance to learn one in: %d" ), difficulty );
900
901 if( one_in( difficulty ) ) {
902 m->second.call( *this, book, false, pos() );
903 continuous = false;
904 } else {
905 if( activity.index == getID().get_value() ) {
906 continuous = true;
907 switch( rng( 1, 5 ) ) {
908 case 1:
909 add_msg( m_info,
910 _( "You train the moves according to the book, but can't get a grasp of the style, so you start from the beginning." ) );
911 break;
912 case 2:
913 add_msg( m_info,
914 _( "This martial art is not easy to grasp. You start training the moves from the beginning." ) );
915 break;
916 case 3:
917 add_msg( m_info,
918 _( "You decide to read the manual and train even more. In martial arts, patience leads to mastery." ) );
919 break;
920 case 4:
921 case 5:
922 add_msg( m_info, _( "You try again. This training will finally pay off." ) );
923 break;
924 }
925 } else {
926 add_msg( m_info, _( "You train for a while." ) );
927 }
928 }
929 }
930
931 if( continuous ) {
932 activity.set_to_null();
933 read( loc, true );
934 if( activity ) {
935 return;
936 }
937 }
938
939 activity.set_to_null();
940}

1182bool avatar::wield( item &target )
1183{
1184 if( is_wielding( target ) ) {
1185 return true;
1186 }
1187
1188 if( !can_wield( target ).success() ) {
1189 return false;
1190 }
1191
1192 if( !unwield() ) {
1193 return false;
1194 }
1195 cached_info.erase( "weapon_value" );
1196 if( target.is_null() ) {
1197 return true;
1198 }
1199
1200 // Query whether to draw an item from a holster when attempting to wield the holster
1201 if( target.get_use( "holster" ) && !target.contents.empty() ) {
1202 //~ %1$s: weapon name, %2$s: holster name
1203 if( query_yn( pgettext( "holster", "Draw %1$s from %2$s?" ), target.get_contained().tname(),
1204 target.tname() ) ) {
1205 invoke_item( &target );
1206 return false;
1207 }
1208 }
1209
1210 // Wielding from inventory is relatively slow and does not improve with increasing weapon skill.
1211 // Worn items (including guns with shoulder straps) are faster but still slower
1212 // than a skilled player with a holster.
1213 // There is an additional penalty when wielding items from the inventory whilst currently grabbed.
1214
1215 bool worn = is_worn( target );
1216 int mv = item_handling_cost( target, true,
1217 worn ? INVENTORY_HANDLING_PENALTY / 2 : INVENTORY_HANDLING_PENALTY );
1218
1219 add_msg( m_debug, "wielding took %d moves", mv );
1220 moves -= mv;
1221
1222 if( has_item( target ) ) {
1223 weapon = i_rem( &target );
1224 } else {
1225 weapon = target;
1226 }
1227
1228 last_item = weapon.typeId();
1229 recoil = MAX_RECOIL;
1230
1231 weapon.on_wield( *this, mv );
1232
1233 inv.update_invlet( weapon );
1234 inv.update_cache_with_item( weapon );
1235
1236 return true;
1237}

1239bool avatar::invoke_item( item *used, const tripoint &pt )
1240{
1241 const std::map<std::string, use_function> &use_methods = used->type->use_methods;
1242
1243 if( use_methods.empty() ) {
1244 return false;
1245 } else if( use_methods.size() == 1 ) {
1246 return invoke_item( used, use_methods.begin()->first, pt );
1247 }
1248
1249 uilist umenu;
1250
1251 umenu.text = string_format( _( "What to do with your %s?" ), used->tname() );
1252 umenu.hilight_disabled = true;
1253
1254 for( const auto &e : use_methods ) {
1255 const auto res = e.second.can_call( *this, *used, false, pt );
1256 umenu.addentry_desc( MENU_AUTOASSIGN, res.success(), MENU_AUTOASSIGN, e.second.get_name(),
1257 res.str() );
1258 }
1259
1260 umenu.desc_enabled = std::any_of( umenu.entries.begin(),
1261 umenu.entries.end(), []( const uilist_entry & elem ) {
1262 return !elem.desc.empty();
1263 } );
1264
1265 umenu.query();
