235 bool fully_inside = false;

238 // initially: fully_outside=true, fully_inside=false (fast fill)

253 LIGHT_AMBIENT_DIM * 0.8 : LIGHT_AMBIENT_LOW;

306 const auto &prev_transparency_cache = prev_map_cache.transparency_cache;

311 constexpr std::array<point, 5> cardinals = {
312 {point_zero, point_north, point_west, point_east, point_south}

315 {{quadrant::NE, quadrant::NW}},

372{
373 auto &map_cache = get_cache( zlev );
374 auto &lm = map_cache.lm;
375 auto &sm = map_cache.sm;
376 auto &outside_cache = map_cache.outside_cache;
377 auto &prev_floor_cache = get_cache( clamp( zlev + 1, -OVERMAP_DEPTH, OVERMAP_DEPTH ) ).floor_cache;
378 bool top_floor = zlev == OVERMAP_DEPTH;
379 std::memset( lm, 0, sizeof( lm ) );
380 std::memset( sm, 0, sizeof( sm ) );
381
382 /* Bulk light sources wastefully cast rays into neighbors; a burning hospital can produce
383 significant slowdown, so for stuff like fire and lava:
384 * Step 1: Store the position and luminance in buffer via add_light_source, for efficient
385 checking of neighbors.
386 * Step 2: After everything else, iterate buffer and apply_light_source only in non-redundant
387 directions
388 * Step 3: ????
389 * Step 4: Profit!

397 constexpr std::array<std::array<quadrant, 2>, 4> dir_quadrants = { {
398 {{ quadrant::NE, quadrant::NW }},
399 {{ quadrant::SW, quadrant::NW }},
400 {{ quadrant::SE, quadrant::NE }},

405 const float natural_light = g->natural_light_level( zlev );

407 build_sunlight_cache( zlev );
408
409 apply_character_light( get_player_character() );

411 apply_character_light( guy );
412 }

415 // Traverse the submaps in order
416 for( int smx = 0; smx < my_MAPSIZE; ++smx ) {
417 for( int smy = 0; smy < my_MAPSIZE; ++smy ) {
418 const auto cur_submap = get_submap_at_grid( { smx, smy, zlev } );

506 const auto &vp = pt->info();
507 if( vp.has_flag( VPFLAG_CONE_LIGHT ) ||
508 vp.has_flag( VPFLAG_WIDE_CONE_LIGHT ) ) {

531 add_light_source( src, M_SQRT2 ); // Add a little surrounding light
532 apply_light_arc( src, v->face.dir() + pt->direction, veh_luminance,

534 }

537 add_light_source( src, M_SQRT2 ); // Add a little surrounding light

541 const bool odd_turn = calendar::once_every( 2_turns );
542 if( ( odd_turn && vp.has_flag( VPFLAG_ODDTURN ) ) ||
543 ( !odd_turn && vp.has_flag( VPFLAG_EVENTURN ) ) ||
544 ( !( vp.has_flag( VPFLAG_EVENTURN ) || vp.has_flag( VPFLAG_ODDTURN ) ) ) ) {
545

558 continue;
559 }

561 add_light_from_items( pp, v->get_items( static_cast<int>( p ) ).begin(),
562 v->get_items( static_cast<int>( p ) ).end() );

580 lm[elem.first.x][elem.first.y].fill( elem.second );

638map::apparent_light_info map::apparent_light_helper( const level_cache &map_cache,
639 const tripoint &p )
640{
641 const float vis = std::max( map_cache.seen_cache[p.x][p.y], map_cache.camera_cache[p.x][p.y] );
642 const bool obstructed = vis <= LIGHT_TRANSPARENCY_SOLID + 0.1;
643
644 auto is_opaque = [&map_cache]( const point & p ) {
645 return map_cache.transparency_cache[p.x][p.y] <= LIGHT_TRANSPARENCY_SOLID &&
646 map_cache.vision_transparency_cache[p.x][p.y] <= LIGHT_TRANSPARENCY_SOLID;
647 };
648
649 const bool p_opaque = is_opaque( p.xy() );
650 float apparent_light;
651
652 if( p_opaque && vis > 0 ) {
653 // This is the complicated case. We want to check which quadrants the
654 // player can see the tile from, and only count light values from those
655 // quadrants.
656 struct offset_and_quadrants {
657 point offset;
658 std::array<quadrant, 2> quadrants;
659 };
660 static constexpr std::array<offset_and_quadrants, 8> adjacent_offsets = {{
661 { point_south, {{ quadrant::SE, quadrant::SW }} },
662 { point_north, {{ quadrant::NE, quadrant::NW }} },
663 { point_east, {{ quadrant::SE, quadrant::NE }} },
664 { point_south_east, {{ quadrant::SE, quadrant::SE }} },
665 { point_north_east, {{ quadrant::NE, quadrant::NE }} },
666 { point_west, {{ quadrant::SW, quadrant::NW }} },
667 { point_south_west, {{ quadrant::SW, quadrant::SW }} },
668 { point_north_west, {{ quadrant::NW, quadrant::NW }} },
669 }
670 };
671
672 four_quadrants seen_from( 0 );
673 for( const offset_and_quadrants &oq : adjacent_offsets ) {
674 const point neighbour = p.xy() + oq.offset;
675
676 if( !lightmap_boundaries.contains( neighbour ) ) {
677 continue;
678 }
679 if( is_opaque( neighbour ) ) {
680 continue;
681 }
682 if( map_cache.seen_cache[neighbour.x][neighbour.y] == 0 &&
683 map_cache.camera_cache[neighbour.x][neighbour.y] == 0 ) {
684 continue;
685 }
686 // This is a non-opaque visible neighbour, so count visibility from the relevant
687 // quadrants
688 seen_from[oq.quadrants[0]] = vis;
689 seen_from[oq.quadrants[1]] = vis;
690 }
691 apparent_light = ( seen_from * map_cache.lm[p.x][p.y] ).max();
692 } else {
693 // This is the simple case, for a non-opaque tile light from all
694 // directions is equivalent
695 apparent_light = vis * map_cache.lm[p.x][p.y].max();
696 }
697 return { obstructed, apparent_light };
698}

700lit_level map::apparent_light_at( const tripoint &p, const visibility_variables &cache ) const
701{
702 const int dist = rl_dist( g->u.pos(), p );
703
704 // Clairvoyance overrides everything.
705 if( dist <= cache.u_clairvoyance ) {
706 return lit_level::BRIGHT;
707 }
708 const auto &map_cache = get_cache_ref( p.z );
709 const apparent_light_info a = apparent_light_helper( map_cache, p );
710
711 // Unimpaired range is an override to strictly limit vision range based on various conditions,
712 // but the player can still see light sources.
713 if( dist > g->u.unimpaired_range() ) {
714 if( !a.obstructed && map_cache.sm[p.x][p.y] > 0.0 ) {
715 return lit_level::BRIGHT_ONLY;
716 } else {
717 return lit_level::DARK;
718 }
719 }
720 if( a.obstructed ) {
721 if( a.apparent_light > LIGHT_AMBIENT_LIT ) {
722 if( a.apparent_light > cache.g_light_level ) {
723 // This represents too hazy to see detail,
724 // but enough light getting through to illuminate.
725 return lit_level::BRIGHT_ONLY;
726 } else {
727 // If it's not brighter than the surroundings, it just ends up shadowy.
728 return lit_level::LOW;
729 }
730 } else {
731 return lit_level::BLANK;
732 }
733 }
734 // Then we just search for the light level in descending order.
735 if( a.apparent_light > LIGHT_SOURCE_BRIGHT || map_cache.sm[p.x][p.y] > 0.0 ) {
736 return lit_level::BRIGHT;
737 }
738 if( a.apparent_light > LIGHT_AMBIENT_LIT ) {
739 return lit_level::LIT;
740 }
741 if( a.apparent_light >= cache.vision_threshold ) {
742 return lit_level::LOW;
743 } else {
744 return lit_level::BLANK;
745 }
746}

748bool map::pl_sees( const tripoint &t, const int max_range ) const
749{
750 if( !inbounds( t ) ) {
751 return false;
752 }
753
754 if( max_range >= 0 && square_dist( t, g->u.pos() ) > max_range ) {
755 return false; // Out of range!
756 }
757
758 const auto &map_cache = get_cache_ref( t.z );
759 const apparent_light_info a = apparent_light_helper( map_cache, t );
760 const float light_at_player = map_cache.lm[g->u.posx()][g->u.posy()].max();
761 return !a.obstructed &&
762 ( a.apparent_light >= g->u.get_vision_threshold( light_at_player ) ||
763 map_cache.sm[t.x][t.y] > 0.0 );

794template<int xx, int xy, int xz, int yx, int yy, int yz, int zz, typename T,
795 T( *calc )( const T &, const T &, const int & ),
796 bool( *check )( const T &, const T & ),
797 T( *accumulate )( const T &, const T &, const int & )>
798void cast_zlight_segment(
799 const array_of_grids_of<T> &output_caches,
800 const array_of_grids_of<const T> &input_arrays,
801 const array_of_grids_of<const bool> &floor_caches,
802 const array_of_grids_of <const diagonal_blocks> &blocked_caches,
803 const tripoint &offset, int offset_distance,
804 T numerator = 1.0f, int row = 1,
805 float start_major = 0.0f, float end_major = 1.0f,
806 float start_minor = 0.0f, float end_minor = 1.0f,

809template<int xx, int xy, int xz, int yx, int yy, int yz, int zz, typename T,
810 T( *calc )( const T &, const T &, const int & ),
811 bool( *check )( const T &, const T & ),
812 T( *accumulate )( const T &, const T &, const int & )>
813void cast_zlight_segment(
814 const array_of_grids_of<T> &output_caches,
815 const array_of_grids_of<const T> &input_arrays,
816 const array_of_grids_of<const bool> &floor_caches,
817 const array_of_grids_of < const diagonal_blocks > &blocked_caches,
818 const tripoint &offset, const int offset_distance,
819 const T numerator, const int row,
820 float start_major, const float end_major,
821 float start_minor, const float end_minor,
822 T cumulative_transparency )
823{
824 if( start_major >= end_major || start_minor >= end_minor ) {
825 return;
826 }
827
828
829 constexpr quadrant quad = quadrant_from_x_y( xx + xy, yx + yy );
830
831 const auto check_blocked = [ =, &blocked_caches]( const tripoint & p ) -> bool{
832 switch( quad )
833 {
834 case quadrant::NW:
835 return ( *blocked_caches[p.z + OVERMAP_DEPTH] )[p.x][p.y].nw;
836 break;
837 case quadrant::NE:
838 return ( *blocked_caches[p.z + OVERMAP_DEPTH] )[p.x][p.y].ne;
839 break;
840 case quadrant::SE:
841 return ( p.x < MAPSIZE_X - 1 && p.y < MAPSIZE_Y - 1 &&
842 ( *blocked_caches[p.z + OVERMAP_DEPTH] )[p.x + 1][p.y + 1].nw );
843 break;
844 case quadrant::SW:
845 return ( p.x > 1 && p.y < MAPSIZE_Y - 1 &&
846 ( *blocked_caches[p.z + OVERMAP_DEPTH] )[p.x - 1][p.y + 1].ne );
847 break;
848 }
849 };
850
851 float radius = 60.0f - offset_distance;
852
853 constexpr int min_z = -OVERMAP_DEPTH;
854 constexpr int max_z = OVERMAP_HEIGHT;
855
856 float new_start_minor = 1.0f;
857
858 T last_intensity = 0.0;
859 tripoint delta;
860 tripoint current;
861 for( int distance = row; distance <= radius; distance++ ) {
862 delta.y = distance;
863 bool started_block = false;
864 T current_transparency = 0.0f;
865
866 // TODO: Precalculate min/max delta.z based on start/end and distance
867 for( delta.z = 0; delta.z <= std::min( fov_3d_z_range, distance ); delta.z++ ) {
868 float trailing_edge_major = ( delta.z - 0.5f ) / ( delta.y + 0.5f );
869 float leading_edge_major = ( delta.z + 0.5f ) / ( delta.y - 0.5f );
870 current.z = offset.z + delta.x * 00 + delta.y * 00 + delta.z * zz;
871 if( current.z > max_z || current.z < min_z ) {
872 continue;
873 } else if( start_major > leading_edge_major ) {
874 continue;
875 } else if( end_major < trailing_edge_major ) {
876 break;
877 }
878
879 bool started_span = false;
880 bool vehicle_blocked = false;
881 const int z_index = current.z + OVERMAP_DEPTH;
882
883
884 for( delta.x = 0; delta.x <= distance; delta.x++ ) {
885 current.x = offset.x + delta.x * xx + delta.y * xy + delta.z * xz;
886 current.y = offset.y + delta.x * yx + delta.y * yy + delta.z * yz;
887 float trailing_edge_minor = ( delta.x - 0.5f ) / ( delta.y + 0.5f );
888 float leading_edge_minor = ( delta.x + 0.5f ) / ( delta.y - 0.5f );
889
890 if( !( current.x >= 0 && current.y >= 0 &&
891 current.x < MAPSIZE_X &&
892 current.y < MAPSIZE_Y ) || start_minor > leading_edge_minor ) {
893 continue;
894 } else if( end_minor < trailing_edge_minor ) {
895 break;
896 }
897
898 T new_transparency = ( *input_arrays[z_index] )[current.x][current.y];
899 // If we're looking at a tile with floor or roof from the floor/roof side,
900 // that tile is actually invisible to us.
901 bool floor_block = false;
902 if( current.z < offset.z ) {
903 if( z_index < ( OVERMAP_LAYERS - 1 ) &&
904 ( *floor_caches[z_index + 1] )[current.x][current.y] ) {
905 floor_block = true;
906 new_transparency = LIGHT_TRANSPARENCY_SOLID;
907 }
908 } else if( current.z > offset.z ) {
909 if( ( *floor_caches[z_index] )[current.x][current.y] ) {
910 floor_block = true;
911 new_transparency = LIGHT_TRANSPARENCY_SOLID;
912 }
913 }
914
915 if( !started_block ) {
916 started_block = true;
917 current_transparency = new_transparency;
918 }
919
920 const int dist = rl_dist( tripoint_zero, delta ) + offset_distance;
921 last_intensity = calc( numerator, cumulative_transparency, dist );
922
923 if( check_blocked( current ) ) {
924 vehicle_blocked = true;
925 break;
926 }
927
928 if( !floor_block ) {
929 ( *output_caches[z_index] )[current.x][current.y] =
930 std::max( ( *output_caches[z_index] )[current.x][current.y], last_intensity );
931 }
932
933 if( !started_span ) {
934 // Need to reset minor slope, because we're starting a new line
935 new_start_minor = leading_edge_minor;
936 // Need more precision or artifacts happen
937 leading_edge_minor = start_minor;
938 started_span = true;
939 }
940
941 if( new_transparency == current_transparency ) {
942 // All in order, no need to recurse
943 new_start_minor = leading_edge_minor;
944 continue;
945 }
946
947 // We split the block into 4 sub-blocks (sub-frustums actually, this is the view from the origin looking out):
948 // +-------+ <- end major
949 // | D |
950 // +---+---+ <- ???
951 // | B | C |
952 // +---+---+ <- major mid
953 // | A |
954 // +-------+ <- start major
955 // ^ ^
956 // | end minor
957 // start minor
958 // A is previously processed row(s).
959 // B is already-processed tiles from current row.
960 // C is remainder of current row.
961 // D is not yet processed row(s).
962 // One we processed fully in 2D and only need to extend in last D
963 // Only cast recursively horizontally if previous span was not opaque.
964 if( check( current_transparency, last_intensity ) ) {
965 T next_cumulative_transparency = accumulate( cumulative_transparency, current_transparency,
966 distance );
967 // Blocks can be merged if they are actually a single rectangle
968 // rather than rectangle + line shorter than rectangle's width
969 const bool merge_blocks = end_minor <= trailing_edge_minor;
970 // trailing_edge_major can be less than start_major
971 const float trailing_clipped = std::max( trailing_edge_major, start_major );
972 const float major_mid = merge_blocks ? leading_edge_major : trailing_clipped;
973 cast_zlight_segment<xx, xy, xz, yx, yy, yz, zz, T, calc, check, accumulate>(
974 output_caches, input_arrays, floor_caches, blocked_caches,
975 offset, offset_distance, numerator, distance + 1,
976 start_major, major_mid, start_minor, end_minor,
977 next_cumulative_transparency );
978 if( !merge_blocks ) {
979 // One line that is too short to be part of the rectangle above
980 cast_zlight_segment<xx, xy, xz, yx, yy, yz, zz, T, calc, check, accumulate>(
981 output_caches, input_arrays, floor_caches, blocked_caches,
982 offset, offset_distance, numerator, distance + 1,
983 major_mid, leading_edge_major, start_minor, trailing_edge_minor,
984 next_cumulative_transparency );
985 }
986 }
987
988 // One from which we shaved one line ("processed in 1D")
989 const float old_start_minor = start_minor;
990 // The new span starts at the leading edge of the previous square if it is opaque,
991 // and at the trailing edge of the current square if it is transparent.
992 if( !check( current_transparency, last_intensity ) ) {
993 start_minor = new_start_minor;
994 } else {
995 // Note this is the same slope as one of the recursive calls we just made.
996 start_minor = std::max( start_minor, trailing_edge_minor );
997 start_major = std::max( start_major, trailing_edge_major );
998 }
999
1000 // leading_edge_major plus some epsilon
1001 float after_leading_edge_major = ( delta.z + 0.50001f ) / ( delta.y - 0.5f );
1002 cast_zlight_segment<xx, xy, xz, yx, yy, yz, zz, T, calc, check, accumulate>(
1003 output_caches, input_arrays, floor_caches, blocked_caches,
1004 offset, offset_distance, numerator, distance,
1005 after_leading_edge_major, end_major, old_start_minor, start_minor,
1006 cumulative_transparency );
1007
1008 // One we just entered ("processed in 0D" - the first point)
1009 // No need to recurse, we're processing it right now
1010
1011 current_transparency = new_transparency;
1012 new_start_minor = leading_edge_minor;
1013 }
1014
1015 if( !check( current_transparency, last_intensity ) || vehicle_blocked ) {
1016 start_major = leading_edge_major;
1017 }
1018 }
1019
1020 if( !started_block ) {
1021 // If we didn't scan at least 1 z-level, don't iterate further
1022 // Otherwise we may "phase" through tiles without checking them
1023 break;
1024 }
1025
1026 if( !check( current_transparency, last_intensity ) ) {
1027 // If we reach the end of the span with terrain being opaque, we don't iterate further.
1028 break;
1029 }
1030 // Cumulative average of the values encountered.
1031 cumulative_transparency = accumulate( cumulative_transparency, current_transparency, distance );
1032 }
1033}

1130{
1131 constexpr quadrant quad = quadrant_from_x_y( -xx - xy, -yx - yy );
1132
1133 const auto check_blocked = [ =, &blocked_array]( const point & p ) {
1134 switch( quad ) {
1135 case quadrant::NW:
1136 return blocked_array[p.x][p.y].nw;
1137 break;
1138 case quadrant::NE:
1139 return blocked_array[p.x][p.y].ne;
1140 break;
1141 case quadrant::SE:
1142 return ( p.x < MAPSIZE_X - 1 && p.y < MAPSIZE_Y - 1 && blocked_array[p.x + 1][p.y + 1].nw );
1143 break;
1144 case quadrant::SW:
1145 return ( p.x > 1 && p.y < MAPSIZE_Y - 1 && blocked_array[p.x - 1][p.y + 1].ne );
1146 break;
1147 }
1148 };
1149
1150 float newStart = 0.0f;
1151 float radius = 60.0f - offsetDistance;
1152 if( start < end ) {
1153 return;
1154 }
1155 T last_intensity = 0.0;
1156 tripoint delta;
1157 for( int distance = row; distance <= radius; distance++ ) {
1158 delta.y = -distance;
1159 bool started_row = false;
1160 T current_transparency = 0.0;
1161 float away = start - ( -distance + 0.5f ) / ( -distance -
1162 0.5f ); //The distance between our first leadingEdge and start
1163
1164 //We initialize delta.x to -distance adjusted so that the commented start < leadingEdge condition below is never false
1165 delta.x = -distance + std::max( static_cast<int>( std::ceil( away * ( -distance - 0.5f ) ) ), 0 );
1166
1167 for( ; delta.x <= 0; delta.x++ ) {
1168 point current( offset.x + delta.x * xx + delta.y * xy, offset.y + delta.x * yx + delta.y * yy );
1169 float trailingEdge = ( delta.x - 0.5f ) / ( delta.y + 0.5f );
1170 float leadingEdge = ( delta.x + 0.5f ) / ( delta.y - 0.5f );
1171
1172 if( !( current.x >= 0 && current.y >= 0 && current.x < MAPSIZE_X &&
1173 current.y < MAPSIZE_Y ) /* || start < leadingEdge */ ) {
1174 continue;
1175 } else if( end > trailingEdge ) {
1176 break;
1177 }
1178
1179 if( check_blocked( current ) ) {
1180 continue;
1181 }
1182 if( !started_row ) {
1183 started_row = true;
1184 current_transparency = input_array[ current.x ][ current.y ];
1185 }
1186
1187 const int dist = rl_dist( tripoint_zero, delta ) + offsetDistance;
1188 last_intensity = calc( numerator, cumulative_transparency, dist );
1189
1190 T new_transparency = input_array[ current.x ][ current.y ];
1191
1192 if( check( new_transparency, last_intensity ) ) {
1193 update_output( output_cache[current.x][current.y], last_intensity,
1194 quadrant::default_ );
1195 } else {
1196 update_output( output_cache[current.x][current.y], last_intensity, quad );
1197 }
1198
1199 if( new_transparency == current_transparency ) {
1200 newStart = leadingEdge;
1201 continue;
1202 }
1203 // Only cast recursively if previous span was not opaque.
1204 if( check( current_transparency, last_intensity ) ) {
1205 castLight<xx, xy, yx, yy, T, Out, calc, check, update_output, accumulate>(
1206 output_cache, input_array, blocked_array, offset, offsetDistance,
1207 numerator, distance + 1, start, trailingEdge,
1208 accumulate( cumulative_transparency, current_transparency, distance ) );
1209 }
1210 // The new span starts at the leading edge of the previous square if it is opaque,
1211 // and at the trailing edge of the current square if it is transparent.
1212 if( !check( current_transparency, last_intensity ) ) {
1213 start = newStart;
1214 } else {
1215 // Note this is the same slope as the recursive call we just made.
1216 start = trailingEdge;
1217 }
1218 // Trailing edge ahead of leading edge means this span is fully processed.
1219 if( start < end ) {
1220 return;
1221 }
1222 current_transparency = new_transparency;
1223 newStart = leadingEdge;
1224 }
1225 if( !check( current_transparency, last_intensity ) ) {
1226 // If we reach the end of the span with terrain being opaque, we don't iterate further.
1227 break;
1228 }
1229 // Cumulative average of the transparency values encountered.
1230 cumulative_transparency = accumulate( cumulative_transparency, current_transparency, distance );
1231 }
1232}

1311 cur_cache.seen_cache_dirty = false;
1312 }

1315 seen_cache[origin.x][origin.y] = VISIBILITY_FULL;
1316 castLightAll<float, float, sight_calc, sight_check, update_light, accumulate_transparency>(

1318 }
1319 }
1320 } else {

1322 array_of_grids_of<const float> transparency_caches;
1323 array_of_grids_of<float> seen_caches;

1325 array_of_grids_of < const diagonal_blocks > blocked_caches;
1326 for( int z = -OVERMAP_DEPTH; z <= OVERMAP_HEIGHT; z++ ) {
1327 auto &cur_cache = get_cache( z );
1328 transparency_caches[z + OVERMAP_DEPTH] = &cur_cache.vision_transparency_cache;
1329 seen_caches[z + OVERMAP_DEPTH] = &cur_cache.seen_cache;

1340 seen_caches, transparency_caches, floor_caches, blocked_caches, origin, 0, 1.0 );
1341 }

1344 if( !vp ) {
1345 return;
1346 }

1348

1350 std::vector<int> mirrors;
1351 // Do all the sight checks first to prevent fake multiple reflection
1352 // from happening due to mirrors becoming visible due to processing order.
1353 // Cameras are also handled here, so that we only need to get through all vehicle parts once

1355 for( const vpart_reference &vp : veh->get_avail_parts( VPFLAG_EXTENDS_VISION ) ) {
1356 const tripoint mirror_pos = vp.pos();
1357 // We can utilize the current state of the seen cache to determine
1358 // if the player can see the mirror from their position.
1359 if( !vp.info().has_flag( "CAMERA" ) &&
1360 seen_cache[mirror_pos.x][mirror_pos.y] < LIGHT_TRANSPARENCY_SOLID + 0.1 ) {
1361 continue;
1362 } else if( !vp.info().has_flag( "CAMERA_CONTROL" ) ) {
1363 mirrors.emplace_back( static_cast<int>( vp.part_index() ) );
1364 } else {
1365 if( square_dist( origin, mirror_pos ) <= 1 && veh->camera_on ) {
1366 cam_control = static_cast<int>( vp.part_index() );
1367 }
1368 }
1369 }
1370
1371 for( int mirror : mirrors ) {
1372 bool is_camera = veh->part_info( mirror ).has_flag( "CAMERA" );
1373 if( is_camera && cam_control < 0 ) {
1374 continue; // Player not at camera control, so cameras don't work
1375 }
1376
1377 const tripoint mirror_pos = veh->global_part_pos3( mirror );
1378
1379 // Determine how far the light has already traveled so mirrors
1380 // don't cheat the light distance falloff.
1381 int offsetDistance;
1382 if( !is_camera ) {
1383 offsetDistance = rl_dist( origin, mirror_pos );
1384 } else {
1385 offsetDistance = 60 - veh->part_info( mirror ).bonus *
1386 veh->part( mirror ).hp() / veh->part_info( mirror ).durability;
1387 camera_cache[mirror_pos.x][mirror_pos.y] = LIGHT_TRANSPARENCY_OPEN_AIR;
1388 }
1389
1390 // TODO: Factor in the mirror facing and only cast in the
1391 // directions the player's line of sight reflects to.
1392 //
1393 // The naive solution of making the mirrors act like a second player
1394 // at an offset appears to give reasonable results though.
1395 castLightAll<float, float, sight_calc, sight_check, update_light, accumulate_transparency>(
1396 camera_cache, transparency_cache, blocked_cache, mirror_pos.xy(), offsetDistance );
1397 }

1402float fastexp( float x )
1403{
1404 union {
1405 float f;
1406 int i;
1407 } u, v;
1408#pragma GCC diagnostic push
1409#pragma GCC diagnostic ignored "-Wunknown-pragmas"
1410#pragma GCC diagnostic ignored "-Wpragmas"
1411#pragma GCC diagnostic ignored "-Wimplicit-int-float-conversion"
1412 u.i = static_cast<long long>( 6051102 * x + 1056478197 );

1414#pragma GCC diagnostic pop

1416}

1438
1439 const point p2( p.xy() );
1440
1441 if( inbounds( p ) ) {
1442 const float min_light = std::max( static_cast<float>( lit_level::LOW ), luminance );
1443 lm[p2.x][p2.y] = elementwise_max( lm[p2.x][p2.y], min_light );
1444 sm[p2.x][p2.y] = std::max( sm[p2.x][p2.y], luminance );
1445 }

1447 return;
1448 } else if( luminance <= lit_level::BRIGHT_ONLY ) {

1450 }

1453 neighboring fires to the north and west that were applied via light_source_buffer

1455 If there's a 100 luminance magnesium flare south added via apply_light_source instead od

1457
1458 ey
1459 nnnNnnn
1460 w e
1461 w 5 +e
1462 sx W 5*1+E ex
1463 w ++++e
1464 w+++++e
1465 sssSsss
1466 sy
1467 */
1468 const int peer_inbounds = LIGHTMAP_CACHE_X - 1;
1469 bool north = ( p2.y != 0 && light_source_buffer[p2.x][p2.y - 1] < luminance );
1470 bool south = ( p2.y != peer_inbounds && light_source_buffer[p2.x][p2.y + 1] < luminance );
1471 bool east = ( p2.x != peer_inbounds && light_source_buffer[p2.x + 1][p2.y] < luminance );
1472 bool west = ( p2.x != 0 && light_source_buffer[p2.x - 1][p2.y] < luminance );
1473
1474 if( north ) {
1475 castLight < 1, 0, 0, -1, float, four_quadrants, light_calc, light_check,
1476 update_light_quadrants, accumulate_transparency > (
1477 lm, transparency_cache, blocked_cache, p2, 0, luminance );
1478 castLight < -1, 0, 0, -1, float, four_quadrants, light_calc, light_check,
1479 update_light_quadrants, accumulate_transparency > (
1480 lm, transparency_cache, blocked_cache, p2, 0, luminance );
1481 }
1482
1483 if( east ) {
1484 castLight < 0, -1, 1, 0, float, four_quadrants, light_calc, light_check,

1486 lm, transparency_cache, blocked_cache, p2, 0, luminance );

1488 update_light_quadrants, accumulate_transparency > (
1489 lm, transparency_cache, blocked_cache, p2, 0, luminance );
1490 }
1491
1492 if( south ) {
1493 castLight<1, 0, 0, 1, float, four_quadrants, light_calc, light_check,
1494 update_light_quadrants, accumulate_transparency>(
1495 lm, transparency_cache, blocked_cache, p2, 0, luminance );
1496 castLight < -1, 0, 0, 1, float, four_quadrants, light_calc, light_check,
1497 update_light_quadrants, accumulate_transparency > (
1498 lm, transparency_cache, blocked_cache, p2, 0, luminance );
1499 }
1500
1501 if( west ) {
1502 castLight<0, 1, 1, 0, float, four_quadrants, light_calc, light_check,
1503 update_light_quadrants, accumulate_transparency>(
1504 lm, transparency_cache, blocked_cache, p2, 0, luminance );

1551void map::apply_light_arc( const tripoint &p, units::angle angle, float luminance,
1552 units::angle wideangle )
1553{
1554 if( luminance <= LIGHT_SOURCE_LOCAL ) {
1555 return;
1556 }
1557
1558 bool lit[LIGHTMAP_CACHE_X][LIGHTMAP_CACHE_Y] {};
1559
1560 apply_light_source( p, LIGHT_SOURCE_LOCAL );
1561
1562 // Normalize (should work with negative values too)
1563 const units::angle wangle = wideangle / 2.0;
1564
1565 units::angle nangle = fmod( angle, 360_degrees );
1566
1567 tripoint end;
1568 int range = LIGHT_RANGE( luminance );
1569 calc_ray_end( nangle, range, p, end );
1570 apply_light_ray( lit, p, end, luminance );
1571
1572 tripoint test;
1573 calc_ray_end( wangle + nangle, range, p, test );
1574
1575 const float wdist = hypot( end.x - test.x, end.y - test.y );
1576 if( wdist <= 0.5 ) {
1577 return;
1578 }
1579
1580 // attempt to determine beam intensity required to cover all squares
1581 const units::angle wstep = ( wangle / ( wdist * M_SQRT2 ) );
1582
1583 // NOLINTNEXTLINE(clang-analyzer-security.FloatLoopCounter)
1584 for( units::angle ao = wstep; ao <= wangle; ao += wstep ) {
1585 if( trigdist ) {
1586 double fdist = ( ao * M_PI_2 ) / wangle;
1587 end.x = static_cast<int>(
1588 p.x + ( static_cast<double>( range ) - fdist * 2.0 ) * cos( nangle + ao ) );
1589 end.y = static_cast<int>(
1590 p.y + ( static_cast<double>( range ) - fdist * 2.0 ) * sin( nangle + ao ) );
1591 apply_light_ray( lit, p, end, luminance );
1592
1593 end.x = static_cast<int>(
1594 p.x + ( static_cast<double>( range ) - fdist * 2.0 ) * cos( nangle - ao ) );
1595 end.y = static_cast<int>(
1596 p.y + ( static_cast<double>( range ) - fdist * 2.0 ) * sin( nangle - ao ) );
1597 apply_light_ray( lit, p, end, luminance );
1598 } else {
1599 calc_ray_end( nangle + ao, range, p, end );
1600 apply_light_ray( lit, p, end, luminance );
1601 calc_ray_end( nangle - ao, range, p, end );
1602 apply_light_ray( lit, p, end, luminance );
1603 }
1604 }

1607void map::apply_light_ray( bool lit[LIGHTMAP_CACHE_X][LIGHTMAP_CACHE_Y],
1608 const tripoint &s, const tripoint &e, float luminance )
1609{
1610 point a( std::abs( e.x - s.x ) * 2, std::abs( e.y - s.y ) * 2 );
1611 point d( ( s.x < e.x ) ? 1 : -1, ( s.y < e.y ) ? 1 : -1 );
1612 point p( s.xy() );
1613
1614 quadrant quad = quadrant_from_x_y( d.x, d.y );
1615
1616 // TODO: Invert that z comparison when it's sane
1617 if( s.z != e.z || ( s.x == e.x && s.y == e.y ) ) {
1618 return;
1619 }
1620
1621 auto &lm = get_cache( s.z ).lm;
1622 auto &transparency_cache = get_cache( s.z ).transparency_cache;
1623
1624 float distance = 1.0;
1625 float transparency = LIGHT_TRANSPARENCY_OPEN_AIR;
1626 const float scaling_factor = static_cast<float>( rl_dist( s, e ) ) /
1627 static_cast<float>( square_dist( s, e ) );
1628 // TODO: [lightmap] Pull out the common code here rather than duplication
1629 if( a.x > a.y ) {
1630 int t = a.y - ( a.x / 2 );
1631 do {
1632 if( t >= 0 ) {
1633 p.y += d.y;
1634 t -= a.x;
1635 }
1636
1637 p.x += d.x;
