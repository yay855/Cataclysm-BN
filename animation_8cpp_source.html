188void draw_custom_explosion_curses( game &g,
189 const std::list< std::map<tripoint, explosion_tile> > &layers )
190{
191 // calculate screen offset relative to player + view offset position
192 const tripoint center = g.u.pos() + g.u.view_offset;
193 const tripoint topleft( center.x - getmaxx( g.w_terrain ) / 2,
194 center.y - getmaxy( g.w_terrain ) / 2, 0 );
195
196 explosion_animation anim;
197
198 auto last_layer_it = layers.begin();
199 shared_ptr_fast<game::draw_callback_t> explosion_cb =
200 make_shared_fast<game::draw_callback_t>( [&]() {
201 for( auto it = layers.begin(); it != std::next( last_layer_it ); ++it ) {
202 for( const auto &pr : *it ) {
203 // update tripoint in relation to top left corner of curses window
204 // mvwputch already filters out of bounds coordinates
205 const tripoint p = pr.first - topleft;
206 const explosion_neighbors ngh = pr.second.neighborhood;
207 const nc_color col = pr.second.color;
208
209 switch( ngh ) {
210 // '^', 'v', '<', '>'
211 case N_NORTH:
212 mvwputch( g.w_terrain, p.xy(), col, '^' );
213 break;
214 case N_SOUTH:
215 mvwputch( g.w_terrain, p.xy(), col, 'v' );
216 break;
217 case N_WEST:
218 mvwputch( g.w_terrain, p.xy(), col, '<' );
219 break;
220 case N_EAST:
221 mvwputch( g.w_terrain, p.xy(), col, '>' );
222 break;
223 // '|' and '-'
224 case N_NORTH | N_SOUTH:
225 case N_NORTH | N_SOUTH | N_WEST:
226 case N_NORTH | N_SOUTH | N_EAST:
227 mvwputch( g.w_terrain, p.xy(), col, '|' );
228 break;
229 case N_WEST | N_EAST:
230 case N_WEST | N_EAST | N_NORTH:
231 case N_WEST | N_EAST | N_SOUTH:
232 mvwputch( g.w_terrain, p.xy(), col, '-' );
233 break;
234 // '/' and '\'
235 case N_NORTH | N_WEST:
236 case N_SOUTH | N_EAST:
237 mvwputch( g.w_terrain, p.xy(), col, '/' );
238 break;
239 case N_SOUTH | N_WEST:
240 case N_NORTH | N_EAST:
241 mvwputch( g.w_terrain, p.xy(), col, '\\' );
242 break;
243 case N_NO_NEIGHBORS:
244 mvwputch( g.w_terrain, p.xy(), col, '*' );
245 break;
246 case N_WEST | N_EAST | N_NORTH | N_SOUTH:
247 break;
248 }
249 }
250 }
251 } );
252 g.add_draw_callback( explosion_cb );
253
254 for( last_layer_it = layers.begin(); last_layer_it != layers.end(); ++last_layer_it ) {
255 if( is_layer_visible( *last_layer_it ) ) {
256 anim.progress();
257 }
258 }
259}

309void explosion_handler::draw_custom_explosion( const tripoint &,
310 const std::map<tripoint, nc_color> &all_area,
311 const std::string &exp_name )
312{
313#if !defined(TILES)
314 ( void )exp_name;
315#endif
316 if( test_mode ) {
317 // avoid segfault from null tilecontext in tests
318 return;
319 }
320
321 constexpr explosion_neighbors all_neighbors = N_NORTH | N_SOUTH | N_WEST | N_EAST;
322 // We will "shell" the explosion area
323 // Each phase will strip a single layer of points
324 // A layer contains all points that have less than 4 neighbors in cardinal directions
325 // Layers will first be generated, then drawn in inverse order
326
327 // Start by getting rid of everything except current z-level
328 std::map<tripoint, explosion_tile> neighbors;
329#if defined(TILES)
330 if( !use_tiles ) {
331 for( const auto &pr : all_area ) {
332 const tripoint relative_point = relative_view_pos( g->u, pr.first );
333 if( relative_point.z == 0 ) {
334 neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second };
335 }
336 }
337 } else {
338 // In tiles mode, the coordinates have to be absolute
339 const tripoint view_center = relative_view_pos( g->u, g->u.pos() );
340 for( const auto &pr : all_area ) {
341 // Relative point is only used for z level check
342 const tripoint relative_point = relative_view_pos( g->u, pr.first );
343 if( relative_point.z == view_center.z ) {
344 neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second };
345 }
346 }
347 }
348#else
349 for( const auto &pr : all_area ) {
350 const tripoint relative_point = relative_view_pos( g->u, pr.first );
351 if( relative_point.z == 0 ) {
352 neighbors[pr.first] = explosion_tile{ N_NO_NEIGHBORS, pr.second };
353 }
354 }
355#endif
356
357 // Searches for a neighbor, sets the neighborhood flag on current point and on the neighbor
358 const auto set_neighbors = [&]( const tripoint & pos,
359 explosion_neighbors & ngh,
360 explosion_neighbors here,
361 explosion_neighbors there ) {
362 if( ( ngh & here ) == N_NO_NEIGHBORS ) {
363 auto other = neighbors.find( pos );
364 if( other != neighbors.end() ) {
365 ngh = ngh | here;
366 other->second.neighborhood = other->second.neighborhood | there;
367 }
368 }
369 };
370
371 // If the point we are about to remove has a neighbor in a given direction
372 // unset that neighbor's flag that our current point is its neighbor
373 const auto unset_neighbor = [&]( const tripoint & pos,
374 const explosion_neighbors ngh,
375 explosion_neighbors here,
376 explosion_neighbors there ) {
377 if( ( ngh & here ) != N_NO_NEIGHBORS ) {
378 auto other = neighbors.find( pos );
379 if( other != neighbors.end() ) {
380 other->second.neighborhood = ( other->second.neighborhood | there ) ^ there;
381 }
382 }
383 };
384
385 // Find all neighborhoods
386 for( auto &pr : neighbors ) {
387 const tripoint &pt = pr.first;
388 explosion_neighbors &ngh = pr.second.neighborhood;
389
390 set_neighbors( pt + point_west, ngh, N_WEST, N_EAST );
391 set_neighbors( pt + point_east, ngh, N_EAST, N_WEST );
392 set_neighbors( pt + point_north, ngh, N_NORTH, N_SOUTH );
393 set_neighbors( pt + point_south, ngh, N_SOUTH, N_NORTH );
394 }
395
396 // We need to save the layers because we will draw them in reverse order
397 std::list< std::map<tripoint, explosion_tile> > layers;
398 while( !neighbors.empty() ) {
399 std::map<tripoint, explosion_tile> layer;
400 bool changed = false;
401 // Find a layer that can be drawn
402 for( const auto &pr : neighbors ) {
403 if( pr.second.neighborhood != all_neighbors ) {
404 changed = true;
405 layer.insert( pr );
406 }
407 }
408 if( !changed ) {
409 // An error, but a minor one - let it slide
410 return;
411 }
412 // Remove the layer from the area to process
413 for( const auto &pr : layer ) {
414 const tripoint &pt = pr.first;
415 const explosion_neighbors ngh = pr.second.neighborhood;
416
417 unset_neighbor( pt + point_west, ngh, N_WEST, N_EAST );
418 unset_neighbor( pt + point_east, ngh, N_EAST, N_WEST );
419 unset_neighbor( pt + point_north, ngh, N_NORTH, N_SOUTH );
420 unset_neighbor( pt + point_south, ngh, N_SOUTH, N_NORTH );
421 neighbors.erase( pr.first );
422 }
423
424 layers.push_front( std::move( layer ) );
425 }
426
427#if defined(TILES)
428 if( !use_tiles ) {
429 draw_custom_explosion_curses( *g, layers );
430 return;
431 }
432
433 explosion_animation anim;
434 // We need to draw all explosions up to now
435 std::map<tripoint, explosion_tile> combined_layer;
436
437 shared_ptr_fast<game::draw_callback_t> explosion_cb =
438 make_shared_fast<game::draw_callback_t>( [&]() {
439 tilecontext->init_custom_explosion_layer( combined_layer, exp_name );
440 } );
441 g->add_draw_callback( explosion_cb );
442
443 for( const auto &layer : layers ) {
444 combined_layer.insert( layer.begin(), layer.end() );
445 if( is_layer_visible( layer ) ) {
446 anim.progress();
447 }
448 }
449
450 tilecontext->void_custom_explosion();
451#else
452 draw_custom_explosion_curses( *g, layers );
453#endif
454}

983bucketed_points bucket_by_distance( const tripoint &origin,
984 const std::map<tripoint, double> &to_bucket )
985{
986 std::map<int, one_bucket> by_distance;
987 for( const std::pair<const tripoint, double> &pv : to_bucket ) {
988 int dist = trig_dist_squared( origin, pv.first );
989 by_distance[dist].emplace_back( pv.first, pv.second );
990 }
991 bucketed_points buckets;
992 for( const std::pair<const int, one_bucket> &bc : by_distance ) {
993 buckets.emplace_back( bc.second );
994 }
995 return buckets;
996}
997
998bucketed_points optimal_bucketing( const bucketed_points &buckets, size_t max_buckets )
999{
1000 if( buckets.size() <= max_buckets ) {
1001 return buckets;
1002 }
1003 assert( max_buckets > 1 );
1004
1005 std::vector<size_t> sizes = {};
1006 for( const one_bucket &bc : buckets ) {
1007 sizes.emplace_back( bc.size() );
1008 }
1009
1010 bucketed_points optimal = buckets;
1011 // TODO: Good algorithm here, this one is a greedy finder of smallest adjacent size sums
1012 for( size_t i = 0; i < buckets.size() - max_buckets; i++ ) {
1013 auto smallest = sizes.begin();
1014 size_t smallest_sum = *smallest + *( smallest + 1 );
1015 for( auto iter = sizes.begin() + 1; ( iter + 1 ) != sizes.end(); iter++ ) {
1016 size_t sum = *iter + *( iter + 1 );
1017 if( sum < smallest_sum ) {
1018 smallest = iter;
1019 smallest_sum = sum;
1020 }
1021 }
1022
1023 size_t distance = std::distance( sizes.begin(), smallest );
1024 sizes[distance] += sizes[distance + 1];
1025 sizes.erase( smallest + 1 );
1026 auto left_bucket = std::next( optimal.begin(), distance );
1027 auto right_bucket = std::next( left_bucket );
1028 left_bucket->insert( left_bucket->end(), right_bucket->begin(), right_bucket->end() );
1029 optimal.erase( right_bucket );
1030 }
1031
1032 return optimal;
1033}

1035static void draw_cone_aoe_curses( const tripoint &, const bucketed_points &waves )
1036{
1037 // Calculate screen offset relative to player + view offset position
1038 const avatar &u = get_avatar();
1039 const tripoint center = u.pos() + u.view_offset;
1040 const tripoint topleft( center.x - catacurses::getmaxx( g->w_terrain ) / 2,
1041 center.y - catacurses::getmaxy( g->w_terrain ) / 2, 0 );
1042
1043 auto it = waves.begin();
1044 shared_ptr_fast<game::draw_callback_t> wave_cb =
1045 make_shared_fast<game::draw_callback_t>( [&]() {
1046 // All the buckets up until now
1047 for( auto inner_it = waves.begin(); inner_it != std::next( it ); inner_it++ ) {
1048 for( const point_with_value &pr : *inner_it ) {
1049 // update tripoint in relation to top left corner of curses window
1050 // mvwputch already filters out of bounds coordinates
1051 const tripoint p = pr.pt - topleft;
1052 int intensity = ( pr.val >= 1.0 ) + ( pr.val >= 0.5 ) + ( inner_it == it );
1053 nc_color col;
1054 switch( intensity ) {
1055 case 3:
1056 col = c_red;
1057 break;
1058 case 2:
1059 col = c_yellow;
1060 break;
1061 case 1:
1062 col = c_white;
1063 break;
1064 default:
1065 col = c_dark_gray;
1066 break;
1067 }
1068
1069 // TODO: Prettier
1070 mvwputch( g->w_terrain, p.xy(), col, '*' );
1071 }
1072 }
1073 } );
1074 g->add_draw_callback( wave_cb );
1075
1076 wave_animation anim;
1077 for( it = waves.begin(); it != waves.end(); it++ ) {
1078 anim.progress();
1079 }
