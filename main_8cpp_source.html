68static int pfd[2];
69static pthread_t thr;
70static const char *tag = "cdda";
71
72static void *thread_func( void * )
73{
74 ssize_t rdsz;
75 char buf[128];
76 for( ;; ) {
77 if( ( ( rdsz = read( pfd[0], buf, sizeof buf - 1 ) ) > 0 ) ) {
78 if( buf[rdsz - 1] == '\n' ) {
79 --rdsz;
80 }
81 buf[rdsz] = 0; /* add null-terminator */
82 __android_log_write( ANDROID_LOG_DEBUG, tag, buf );
83 }
84 }
85 return 0;
86}

88int start_logger( const char *app_name )
89{
90 tag = app_name;
91
92 /* make stdout line-buffered and stderr unbuffered */
93 setvbuf( stdout, 0, _IOLBF, 0 );
94 setvbuf( stderr, 0, _IONBF, 0 );
95
96 /* create the pipe and redirect stdout and stderr */
97 pipe( pfd );
98 dup2( pfd[1], 1 );
99 dup2( pfd[1], 2 );
100
101 /* spawn the logging thread */
102 if( pthread_create( &thr, 0, thread_func, 0 ) == -1 ) {
103 return -1;
104 }
105 pthread_detach( thr );
106 return 0;
107}

748void printHelpMessage( const arg_handler *first_pass_arguments,
749 size_t num_first_pass_arguments,
750 const arg_handler *second_pass_arguments,
751 size_t num_second_pass_arguments )
752{
753
754 // Group all arguments by help_group.
755 std::multimap<std::string, const arg_handler *> help_map;
756 for( size_t i = 0; i < num_first_pass_arguments; ++i ) {
757 std::string help_group;
758 if( first_pass_arguments[i].help_group ) {
759 help_group = first_pass_arguments[i].help_group;
760 }
761 help_map.insert( std::make_pair( help_group, &first_pass_arguments[i] ) );
762 }
763 for( size_t i = 0; i < num_second_pass_arguments; ++i ) {
764 std::string help_group;
765 if( second_pass_arguments[i].help_group ) {
766 help_group = second_pass_arguments[i].help_group;
767 }
768 help_map.insert( std::make_pair( help_group, &second_pass_arguments[i] ) );
769 }
770
771 cata_printf( "Command line parameters:\n" );
772 std::string current_help_group;
773 auto it = help_map.begin();
774 auto it_end = help_map.end();
775 for( ; it != it_end; ++it ) {
776 if( it->first != current_help_group ) {
777 current_help_group = it->first;
778 cata_printf( "\n%s\n", current_help_group.c_str() );
779 }
780
781 const arg_handler *handler = it->second;
782 cata_printf( "%s", handler->flag );
783 if( handler->param_documentation ) {
784 cata_printf( " %s", handler->param_documentation );
785 }
786 cata_printf( "\n" );
787 if( handler->documentation ) {
788 cata_printf( " %s\n", handler->documentation );
789 }
790 }
791}
