77bool leap_actor::call( monster &z ) const
78{
79 if( !z.can_act() || !z.move_effects( false ) ) {
80 return false;
81 }
82
83 std::vector<tripoint> options;
84 tripoint target = z.move_target();
85 float best_float = trigdist ? trig_dist( z.pos(), target ) : square_dist( z.pos(), target );
86 if( best_float < min_consider_range || best_float > max_consider_range ) {
87 return false;
88 }
89
90 // We wanted the float for range check
91 // int here will make the jumps more random
92 int best = std::numeric_limits<int>::max();
93 if( !allow_no_target && z.attack_target() == nullptr ) {
94 return false;
95 }
96 map &here = get_map();
97 std::multimap<int, tripoint> candidates;
98 for( const tripoint &candidate : here.points_in_radius( z.pos(), max_range ) ) {
99 if( candidate == z.pos() ) {
100 continue;
101 }
102 float leap_dist = trigdist ? trig_dist( z.pos(), candidate ) :
103 square_dist( z.pos(), candidate );
104 if( leap_dist > max_range || leap_dist < min_range ) {
105 continue;
106 }
107 int candidate_dist = rl_dist( candidate, target );
108 if( candidate_dist >= best_float ) {
109 continue;
110 }
111 candidates.emplace( candidate_dist, candidate );
112 }
113 for( const auto &candidate : candidates ) {
114 const int &cur_dist = candidate.first;
115 const tripoint &dest = candidate.second;
116 if( cur_dist > best ) {
117 break;
118 }
119 if( !z.sees( dest ) ) {
120 continue;
121 }
122 if( !g->is_empty( dest ) ) {
123 continue;
124 }
125 bool blocked_path = false;
126 // check if monster has a clear path to the proposed point
127 std::vector<tripoint> line = here.find_clear_path( z.pos(), dest );
128 tripoint prev_point = z.pos();
129 for( auto &i : line ) {
130 if( here.impassable( i ) || here.obstructed_by_vehicle_rotation( prev_point, i ) ) {
131 blocked_path = true;
132 break;
133 }
134 prev_point = i;
135 }
136 // don't leap into water if you could drown (#38038)
137 if( z.is_aquatic_danger( dest ) ) {
138 blocked_path = true;
139 }
140 if( blocked_path ) {
141 continue;
142 }
143
144 options.push_back( dest );
145 best = cur_dist;
146 }
147
148 if( options.empty() ) {
149 return false; // Nowhere to leap!
150 }
151
152 z.moves -= move_cost;
153 const tripoint chosen = random_entry( options );
154 bool seen = g->u.sees( z ); // We can see them jump...
155 z.setpos( chosen );
156 seen |= g->u.sees( z ); // ... or we can see them land
157 if( seen ) {
158 add_msg( _( "The %s leaps!" ), z.name() );
159 }
160
161 return true;
162}

562{

564 if( one_in( 3 ) && g->u.sees( z ) ) {

571 ( require_targeting_npc && target.is_npc() ) ||
572 ( require_targeting_monster && target.is_monster() );
573 const bool not_targeted = require_targeting && !z.has_effect( effect_targeted );
574 const bool not_laser_locked = require_targeting && laser_lock &&
575 !target.has_effect( effect_was_laserlocked );
576

578 if( targeting_volume > 0 && !targeting_sound.empty() ) {

584 }

586 target.add_effect( effect_laserlocked, 5_turns );

598 }
