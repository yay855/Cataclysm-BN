132food_summary stomach_contents::digest( const needs_rates &metabolic_rates, int five_mins )
133{
134 food_summary digested{};
135 stomach_digest_rates rates = get_digest_rates( metabolic_rates );
136
137 // Digest kCal -- use min_kcal by default, but no more than what's in stomach,
138 // and no less than percentage_kcal of what's in stomach.
139 int kcal_fraction = std::lround( nutr.kcal * rates.percent_kcal );
140 digested.nutr.kcal = clamp( five_mins * rates.min_kcal, five_mins * kcal_fraction, nutr.kcal );
141
142 // Digest vitamins just like we did kCal, but we need to do one at a time.
143 for( const std::pair<const vitamin_id, int> &vit : nutr.vitamins ) {
144 int vit_fraction = std::lround( vit.second * rates.percent_vitamin );
145 digested.nutr.vitamins[vit.first] =
146 five_mins * clamp( rates.min_vitamin, vit_fraction, vit.second );
147 }
148
149 nutr -= digested.nutr;
150 return digested;
151}
