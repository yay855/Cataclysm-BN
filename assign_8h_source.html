41inline void report_strict_violation( const JsonObject &jo, const std::string &message,

43{

53template <typename T, typename std::enable_if<std::is_arithmetic<T>::value, int>::type = 0>
54bool assign( const JsonObject &jo, const std::string &name, T &val, bool strict = false,
55 T lo = std::numeric_limits<T>::lowest(), T hi = std::numeric_limits<T>::max() )
56{
57 T out;
58 double scalar;
59
60 // Object via which to report errors which differs for proportional/relative values
61 JsonObject err = jo;
62 err.allow_omitted_members();
63 JsonObject relative = jo.get_object( "relative" );
64 relative.allow_omitted_members();
65 JsonObject proportional = jo.get_object( "proportional" );
66 proportional.allow_omitted_members();
67
68 // Do not require strict parsing for relative and proportional values as rules
69 // such as +10% are well-formed independent of whether they affect base value
70 if( relative.read( name, out ) ) {
71 err = relative;
72 strict = false;
73 out += val;
74
75 } else if( proportional.read( name, scalar ) ) {
76 err = proportional;
77 if( scalar <= 0 || scalar == 1 ) {
78 err.throw_error( "multiplier must be a positive number other than 1", name );
79 }
80 strict = false;
81 out = val * scalar;
82
83 } else if( !jo.read( name, out ) ) {
84 return false;
85 }
86
87 if( out < lo || out > hi ) {
88 err.throw_error( "value outside supported range", name );
89 }
90
91 if( strict && out == val ) {
92 report_strict_violation( err, "cannot assign explicit value the same as default or inherited value",
93 name );
94 }
95
96 val = out;
97
98 return true;
99}

123 bool strict = false, T lo = std::numeric_limits<T>::lowest(), T hi = std::numeric_limits<T>::max() )
124{
125 std::pair<T, T> out;
126
127 if( jo.has_array( name ) ) {
128 auto arr = jo.get_array( name );
129 arr.read( 0, out.first );
130 arr.read( 1, out.second );
131
132 } else if( jo.read( name, out.first ) ) {
133 out.second = out.first;

145 }
146
147 if( strict && out == val ) {
148 report_strict_violation( jo, "cannot assign explicit value the same as default or inherited value",
149 name );

151

178template <typename T>
179typename std::enable_if<std::is_constructible<T, std::string>::value, bool>::type assign(
180 const JsonObject &jo, const std::string &name, std::set<T> &val, bool = false )
181{
182 JsonObject add = jo.get_object( "extend" );
183 add.allow_omitted_members();
184 JsonObject del = jo.get_object( "delete" );
185 del.allow_omitted_members();
186
187 if( jo.has_string( name ) || jo.has_array( name ) ) {
188 val = jo.get_tags<T>( name );
189
190 if( add.has_member( name ) || del.has_member( name ) ) {
191 // ill-formed to (re)define a value and then extend/delete within same definition
192 jo.throw_error( "multiple assignment of value", name );
193 }
194 return true;
195 }
196
197 bool res = false;
198
199 if( add.has_string( name ) || add.has_array( name ) ) {
200 auto tags = add.get_tags<T>( name );
201 val.insert( tags.begin(), tags.end() );
202 res = true;
203 }
204
205 if( del.has_string( name ) || del.has_array( name ) ) {
206 for( const auto &e : del.get_tags<T>( name ) ) {
207 val.erase( e );
208 }
209 res = true;
210 }
211
212 return res;
213}

215/**
216 * Load map from JSON array of [key, value] pairs.
217 * Supports extend by pair and delete by key.
218 */
219template <typename T>
220bool assign_map_from_array( const JsonObject &jo, const std::string &name, T &val, bool = false )
221{
222 using K = typename T::key_type;
223 using V = typename T::mapped_type;
224
225 JsonObject add = jo.get_object( "extend" );
226 add.allow_omitted_members();
227 JsonObject del = jo.get_object( "delete" );
228 del.allow_omitted_members();
229
230 if( jo.has_array( name ) ) {
231 val.clear();
232
233 std::vector<std::pair<K, V>> tmp;
234 jo.get_raw( name )->read( tmp, true );
235 for( const auto &it : tmp ) {
236 val.insert( std::move( it ) );
237 }
238
239 if( add.has_member( name ) || del.has_member( name ) ) {
240 // ill-formed to (re)define a value and then extend/delete within same definition
241 jo.throw_error( "multiple assignment of value", name );
242 }
243
244 return true;
245 }
246
247 bool res = false;
248
249 if( add.has_array( name ) ) {
250 std::vector<std::pair<K, V>> tmp;
251 add.get_raw( name )->read( tmp, true );
252
253 for( const auto &it : tmp ) {
254 val[std::move( it.first )] = std::move( it.second );
255 }
256
257 res = true;
258 }
259
260 if( del.has_array( name ) ) {
261 std::vector<K> tmp;
262 del.get_raw( name )->read( tmp, true );
263
264 for( const auto &it : tmp ) {
265 val.erase( it );
266 }
267
268 res = true;
269 }
270
271 return res;
272}

758 strict = false;
759 out = out + val;
760
761 } else if( proportional.read( name, scalar ) ) {
762 err = proportional;
763 if( scalar <= 0 || scalar == 1 ) {
764 err.throw_error( "multiplier must be a positive number other than 1", name );
765 }
766 strict = false;
767 out = val * scalar;
768
769 } else if( !read_with_factor( jo, name, out, factor ) ) {
770 return false;
771 }
772
773 if( strict && out == val ) {
774 report_strict_violation( err, "cannot assign explicit value the same as default or inherited value",
775 name );
776 }
777
778 val = out;
779
780 return true;
781}

802static void assign_dmg_relative( damage_instance &out, const damage_instance &val,
803 damage_instance relative, bool &strict )
804{
805 for( const damage_unit &val_dmg : val.damage_units ) {
806 for( damage_unit &tmp : relative.damage_units ) {
807 if( tmp.type != val_dmg.type ) {
808 continue;
809 }
810
811 // Do not require strict parsing for relative and proportional values as rules
812 // such as +10% are well-formed independent of whether they affect base value
813 strict = false;
814
815 // res_mult is set to 1 if it's not specified. Set it to zero so we don't accidentally add to it
816 if( tmp.res_mult == 1.0f ) {
817 tmp.res_mult = 0;
818 }
819 // Same for damage_multiplier
820 if( tmp.damage_multiplier == 1.0f ) {
821 tmp.damage_multiplier = 0;
822 }
823
824 damage_unit out_dmg( tmp.type, 0.0f );
825
826 out_dmg.amount = tmp.amount + val_dmg.amount;
827 out_dmg.res_pen = tmp.res_pen + val_dmg.res_pen;
828
829 out_dmg.res_mult = tmp.res_mult + val_dmg.res_mult;
830 out_dmg.damage_multiplier = tmp.damage_multiplier + val_dmg.damage_multiplier;
831
832 out.add( out_dmg );
833 }
834 }

837static void assign_dmg_proportional( const JsonObject &jo, const std::string &name,
838 damage_instance &out,
839 const damage_instance &val,
840 damage_instance proportional, bool &strict )
841{
842 for( const damage_unit &val_dmg : val.damage_units ) {
843 for( damage_unit &scalar : proportional.damage_units ) {
844 if( scalar.type != val_dmg.type ) {
845 continue;
846 }
847
848 // Do not require strict parsing for relative and proportional values as rules
849 // such as +10% are well-formed independent of whether they affect base value
850 strict = false;
851
852 // Can't have negative percent, and 100% is pointless
853 // If it's 0, it wasn't loaded
854 if( scalar.amount == 1 || scalar.amount < 0 ) {
855 jo.throw_error( "Proportional damage multiplier must be a positive number other than 1", name );
856 }
857
858 // If it's 0, it wasn't loaded
859 if( scalar.res_pen < 0 || scalar.res_pen == 1 ) {
860 jo.throw_error( "Proportional armor penetration multiplier must be a positive number other than 1",
861 name );
862 }
863
864 // It wasn't loaded, so set it 100%
865 if( scalar.res_pen == 0 ) {
866 scalar.res_pen = 1.0f;
867 }
868
869 // Ditto
870 if( scalar.amount == 0 ) {
871 scalar.amount = 1.0f;
872 }
873
874 // If it's 1, it wasn't loaded (or was loaded as 1)
875 if( scalar.res_mult <= 0 ) {
876 jo.throw_error( "Proportional armor penetration multiplier must be a positive number", name );
877 }
878
879 // If it's 1, it wasn't loaded (or was loaded as 1)
880 if( scalar.damage_multiplier <= 0 ) {
881 jo.throw_error( "Proportional damage multiplier must be a positive number", name );
882 }
883
884 damage_unit out_dmg( scalar.type, 0.0f );
885
886 out_dmg.amount = val_dmg.amount * scalar.amount;
887 out_dmg.res_pen = val_dmg.res_pen * scalar.res_pen;
888
889 out_dmg.res_mult = val_dmg.res_mult * scalar.res_mult;
890 out_dmg.damage_multiplier = val_dmg.damage_multiplier * scalar.damage_multiplier;
891
892 out.add( out_dmg );
893 }
894 }

897static void check_assigned_dmg( const JsonObject &err, const std::string &name,
898 const damage_instance &out, const damage_instance &lo_inst, const damage_instance &hi_inst )
899{
900 for( const damage_unit &out_dmg : out.damage_units ) {
901 auto lo_iter = std::find_if( lo_inst.damage_units.begin(),
902 lo_inst.damage_units.end(), [&out_dmg]( const damage_unit & du ) {
903 return du.type == out_dmg.type || du.type == DT_NULL;
904 } );
905
906 auto hi_iter = std::find_if( hi_inst.damage_units.begin(),
907 hi_inst.damage_units.end(), [&out_dmg]( const damage_unit & du ) {
908 return du.type == out_dmg.type || du.type == DT_NULL;
909 } );
910
911 if( lo_iter == lo_inst.damage_units.end() ) {
912 err.throw_error( "Min damage type used in assign does not match damage type assigned", name );
913 }
914 if( hi_iter == hi_inst.damage_units.end() ) {
915 err.throw_error( "Max damage type used in assign does not match damage type assigned", name );
916 }
917
918 const damage_unit &hi_dmg = *hi_iter;
919 const damage_unit &lo_dmg = *lo_iter;
920
921 if( out_dmg.amount < lo_dmg.amount || out_dmg.amount > hi_dmg.amount ) {
922 err.throw_error( "value for damage outside supported range", name );
923 }
924 if( out_dmg.res_pen < lo_dmg.res_pen || out_dmg.res_pen > hi_dmg.res_pen ) {
925 err.throw_error( "value for armor penetration outside supported range", name );
926 }
927 if( out_dmg.res_mult < lo_dmg.res_mult || out_dmg.res_mult > hi_dmg.res_mult ) {
928 err.throw_error( "value for armor penetration multiplier outside supported range", name );
929 }
930 if( out_dmg.damage_multiplier < lo_dmg.damage_multiplier ||
931 out_dmg.damage_multiplier > hi_dmg.damage_multiplier ) {
932 err.throw_error( "value for damage multiplier outside supported range", name );
933 }
934 }
935}
