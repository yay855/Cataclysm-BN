49template <typename T, typename std::enable_if<std::is_arithmetic<T>::value, int>::type = 0>
50bool assign( const JsonObject &jo, const std::string &name, T &val, bool strict = false,
51 T lo = std::numeric_limits<T>::lowest(), T hi = std::numeric_limits<T>::max() )
52{
53 T out;
54 double scalar;
55
56 // Object via which to report errors which differs for proportional/relative values
57 JsonObject err = jo;
58 err.allow_omitted_members();
59 JsonObject relative = jo.get_object( "relative" );
60 relative.allow_omitted_members();
61 JsonObject proportional = jo.get_object( "proportional" );
62 proportional.allow_omitted_members();
63
64 // Do not require strict parsing for relative and proportional values as rules
65 // such as +10% are well-formed independent of whether they affect base value
66 if( relative.read( name, out ) ) {
67 err = relative;
68 strict = false;
69 out += val;
70
71 } else if( proportional.read( name, scalar ) ) {
72 err = proportional;
73 if( scalar <= 0 || scalar == 1 ) {
74 err.throw_error( "multiplier must be a positive number other than 1", name );
75 }
76 strict = false;
77 out = val * scalar;
78
79 } else if( !jo.read( name, out ) ) {
80 return false;
81 }
82
83 if( out < lo || out > hi ) {
84 err.throw_error( "value outside supported range", name );
85 }
86
87 if( strict && out == val ) {
88 report_strict_violation( err, "cannot assign explicit value the same as default or inherited value",
89 name );
90 }
91
92 val = out;
93
94 return true;
95}

106

108 auto arr = jo.get_array( name );
109 arr.read( 0, out.first );

158template <typename T>
159typename std::enable_if<std::is_constructible<T, std::string>::value, bool>::type assign(
160 const JsonObject &jo, const std::string &name, std::set<T> &val, bool = false )
161{
162 JsonObject add = jo.get_object( "extend" );
163 add.allow_omitted_members();
164 JsonObject del = jo.get_object( "delete" );
165 del.allow_omitted_members();
166
167 if( jo.has_string( name ) || jo.has_array( name ) ) {
168 val = jo.get_tags<T>( name );
169
170 if( add.has_member( name ) || del.has_member( name ) ) {
171 // ill-formed to (re)define a value and then extend/delete within same definition
172 jo.throw_error( "multiple assignment of value", name );
173 }
174 return true;
175 }
176
177 bool res = false;
178
179 if( add.has_string( name ) || add.has_array( name ) ) {
180 auto tags = add.get_tags<T>( name );
181 val.insert( tags.begin(), tags.end() );
182 res = true;
183 }
184
185 if( del.has_string( name ) || del.has_array( name ) ) {
186 for( const auto &e : del.get_tags<T>( name ) ) {
187 val.erase( e );
188 }
189 res = true;
190 }
191
192 return res;
193}

195/**
196 * Load map from JSON array of [key, value] pairs.
197 * Supports extend by pair and delete by key.
198 */
199template <typename T>
200bool assign_map_from_array( const JsonObject &jo, const std::string &name, T &val, bool = false )
201{
202 using K = typename T::key_type;
203 using V = typename T::mapped_type;
204
205 JsonObject add = jo.get_object( "extend" );
206 add.allow_omitted_members();
207 JsonObject del = jo.get_object( "delete" );
208 del.allow_omitted_members();
209
210 if( jo.has_array( name ) ) {
211 val.clear();
212
213 std::vector<std::pair<K, V>> tmp;
214 jo.get_raw( name )->read( tmp, true );
215 for( const auto &it : tmp ) {
216 val.insert( std::move( it ) );
217 }
218
219 if( add.has_member( name ) || del.has_member( name ) ) {
220 // ill-formed to (re)define a value and then extend/delete within same definition
221 jo.throw_error( "multiple assignment of value", name );
222 }
223
224 return true;
225 }
226
227 bool res = false;
228
229 if( add.has_array( name ) ) {
230 std::vector<std::pair<K, V>> tmp;
231 add.get_raw( name )->read( tmp, true );
232
233 for( const auto &it : tmp ) {
234 val[std::move( it.first )] = std::move( it.second );
235 }
236
237 res = true;
238 }
239
240 if( del.has_array( name ) ) {
241 std::vector<K> tmp;
242 del.get_raw( name )->read( tmp, true );
243
244 for( const auto &it : tmp ) {
245 val.erase( it );
246 }
247
248 res = true;
249 }
250
251 return res;
252}
