40stat_mod get_pain_penalty( const Character &ch )
41{
42 stat_mod ret;
43 int pain = ch.get_perceived_pain();
44 if( pain <= 0 ) {
45 return ret;
46 }
47
48 int stat_penalty = std::floor( std::pow( pain, 0.8f ) / 10.0f );
49
50 bool ceno = ch.has_trait( trait_CENOBITE );
51 if( !ceno ) {
52 ret.strength = stat_penalty;
53 ret.dexterity = stat_penalty;
54 }
55
56 if( !ch.has_trait( trait_INT_SLIME ) ) {
57 ret.intelligence = stat_penalty;
58 } else {
59 ret.intelligence = pain / 5;
60 }
61
62 ret.perception = stat_penalty * 2 / 3;
63
64 ret.speed = std::pow( pain, 0.7f );
65 if( ceno ) {
66 ret.speed /= 2;
67 }
68
69 ret.speed = std::min( ret.speed, 30 );
70 return ret;
71}

177int calc_focus_equilibrium( const Character &who )
178{
179 int focus_equilibrium = 100;
180
181 if( who.activity.id() == ACT_READ ) {
182 item_location loc = who.activity.targets[0];
183 if( loc && loc->is_book() ) {
184 auto &bt = *loc->type->book;
185 // apply a penalty when we're actually learning something
186 const SkillLevel &skill_level = who.get_skill_level_object( bt.skill );
187 if( skill_level.can_train() && skill_level < bt.level ) {
188 focus_equilibrium -= 50;
189 }
190 }
191 }
192
193 int eff_morale = who.get_morale_level();
194 // Factor in perceived pain, since it's harder to rest your mind while your body hurts.
195 // Cenobites don't mind, though
196 if( !who.has_trait( trait_CENOBITE ) ) {
197 eff_morale = eff_morale - who.get_perceived_pain();
198 }
199
200 // as baseline morale is 100, calc_fatigue_cap() has to -100 to apply accurate penalties.
201 if( calc_morale_fatigue_cap( who.get_fatigue() ) != 0 &&
202 eff_morale > calc_morale_fatigue_cap( who.get_fatigue() ) - 100 ) {
203 eff_morale = calc_morale_fatigue_cap( who.get_fatigue() ) - 100;
204 }
205
206 if( eff_morale < -99 ) {
207 // At very low morale, focus is at it's minimum
208 focus_equilibrium = 1;
209 } else if( eff_morale <= 50 ) {
210 // At -99 to +50 morale, each point of morale gives or takes 1 point of focus
211 focus_equilibrium += eff_morale;
212 } else {
213 /* Above 50 morale, we apply strong diminishing returns.
214 * Each block of 50 takes twice as many morale points as the previous one:
215 * 150 focus at 50 morale (as before)
216 * 200 focus at 150 morale (100 more morale)
217 * 250 focus at 350 morale (200 more morale)
218 * ...
219 * Cap out at 400% focus gain with 3,150+ morale, mostly as a sanity check.
220 */
221
222 int block_multiplier = 1;
223 int morale_left = eff_morale;
224 while( focus_equilibrium < 400 ) {
225 if( morale_left > 50 * block_multiplier ) {
226 // We can afford the entire block. Get it and continue.
227 morale_left -= 50 * block_multiplier;
228 focus_equilibrium += 50;
229 block_multiplier *= 2;
230 } else {
231 // We can't afford the entire block. Each block_multiplier morale
232 // points give 1 focus, and then we're done.
233 focus_equilibrium += morale_left / block_multiplier;
234 break;
235 }
236 }
237 }
238
239 // This should be redundant, but just in case...
240 if( focus_equilibrium < 1 ) {
241 focus_equilibrium = 1;
242 } else if( focus_equilibrium > 400 ) {
243 focus_equilibrium = 400;
244 }
245 return focus_equilibrium;
246}

248int calc_focus_change( const Character &who )
249{
250 int focus_gap = calc_focus_equilibrium( who ) - who.focus_pool;
251
252 // handle negative gain rates in a symmetric manner
253 int base_change = 1;
254 if( focus_gap < 0 ) {
255 base_change = -1;
256 focus_gap = -focus_gap;
257 }
258
259 // for every 100 points, we have a flat gain of 1 focus.
260 // for every n points left over, we have an n% chance of 1 focus
261 int gain = focus_gap / 100;
262 if( rng( 1, 100 ) <= focus_gap % 100 ) {
263 gain++;
264 }
265
266 gain *= base_change;
267
268 return gain;
269}
