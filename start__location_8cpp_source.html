109// check if tile at p should be boarded with some kind of furniture.
110static void add_boardable( const map &m, const tripoint &p, std::vector<tripoint> &vec )
111{
112 if( m.has_furn( p ) ) {
113 // Don't need to board this up, is already occupied
114 return;
115 }
116 if( m.ter( p ) != t_floor ) {
117 // Other terrain (door, wall, ...), not boarded either
118 return;
119 }
120 if( m.is_outside( p ) ) {
121 // Don't board up the outside
122 return;
123 }
124 if( std::find( vec.begin(), vec.end(), p ) != vec.end() ) {
125 // Already registered to be boarded
126 return;
127 }
128 vec.push_back( p );
129}

131static void board_up( map &m, const tripoint_range<tripoint> &range )
132{
133 std::vector<tripoint> furnitures1;
134 std::vector<tripoint> furnitures2;
135 std::vector<tripoint> boardables;
136 for( const tripoint &p : range ) {
137 bool must_board_around = false;
138 const ter_id t = m.ter( p );
139 if( t == t_window_domestic || t == t_window || t == t_window_no_curtains ) {
140 // Windows are always to the outside and must be boarded
141 must_board_around = true;
142 m.ter_set( p, t_window_boarded );
143 } else if( t == t_door_c || t == t_door_locked || t == t_door_c_peep ) {
144 // Only board up doors that lead to the outside
145 if( m.is_outside( p + tripoint_north ) || m.is_outside( p + tripoint_south ) ||
146 m.is_outside( p + tripoint_east ) || m.is_outside( p + tripoint_west ) ) {
147 m.ter_set( p, t_door_boarded );
148 must_board_around = true;
149 } else {
150 // internal doors are opened instead
151 m.ter_set( p, t_door_o );
152 }
153 }
154 if( must_board_around ) {
155 // Board up the surroundings of the door/window
156 for( const tripoint &neigh : points_in_radius( p, 1 ) ) {
157 if( neigh == p ) {
158 continue;
159 }
160 add_boardable( m, neigh, boardables );
161 }
162 }
163 }
164 // Find all furniture that can be used to board up some place
165 for( const tripoint &p : range ) {
166 if( std::find( boardables.begin(), boardables.end(), p ) != boardables.end() ) {
167 continue;
168 }
169 if( !m.has_furn( p ) ) {
170 continue;
171 }
172 // If the furniture is movable and the character can move it, use it to barricade
173 // g->u is workable here as NPCs by definition are not starting the game. (Let's hope.)
174 ///\EFFECT_STR determines what furniture might be used as a starting area barricade
175 if( m.furn( p ).obj().is_movable() && m.furn( p ).obj().move_str_req < g->u.get_str() ) {
176 if( m.furn( p ).obj().movecost == 0 ) {
177 // Obstacles are better, prefer them
178 furnitures1.push_back( p );
179 } else {
180 furnitures2.push_back( p );
181 }
182 }
183 }
184 while( ( !furnitures1.empty() || !furnitures2.empty() ) && !boardables.empty() ) {
185 const tripoint fp = random_entry_removed( furnitures1.empty() ? furnitures2 : furnitures1 );
186 const tripoint bp = random_entry_removed( boardables );
187 m.furn_set( bp, m.furn( fp ) );
188 m.furn_set( fp, f_null );
189 auto destination_items = m.i_at( bp );
190 for( const item &moved_item : m.i_at( fp ) ) {
191 destination_items.insert( moved_item );
192 }
193 m.i_clear( fp );
194 }

208tripoint_abs_omt start_location::find_player_initial_location() const
209{
210 // Spiral out from the world origin scanning for a compatible starting location,
211 // creating overmaps as necessary.
212 const int radius = 3;
213 std::vector<point_abs_om> overmaps = closest_points_first( point_abs_om(), radius );
214 // Skip overmap (0,0), that's endgame
215 overmaps.erase( overmaps.begin() );
216 // Shuffle 8 first ones so that we don't always start at (1,0)
217 std::shuffle( overmaps.begin(), overmaps.begin() + 7, rng_get_engine() );
218 for( const point_abs_om &omp : overmaps ) {
219 overmap &omap = overmap_buffer.get( omp );
220 const tripoint_om_omt omtstart = omap.find_random_omt( random_target() );
221 if( omtstart.raw() != tripoint_min ) {
222 return project_combine( omp, omtstart );
223 }
224 }
225 // Should never happen, if it does we messed up.
226 popup( _( "Unable to generate a valid starting location %s [%s] in a radius of %d overmaps, please report this failure." ),
227 name(), id.str(), radius );
228 return overmap::invalid_tripoint;

242/** Helper for place_player
243 * Flood-fills the area from a given point, then returns the area it found.
244 * Considers unpassable tiles passable if they can be bashed down or opened.
245 * Will return INT_MAX if it reaches upstairs or map edge.
246 * We can't really use map::route here, because it expects a destination
247 * Maybe TODO: Allow "picking up" items or parts of bashable furniture
248 * and using them to help with bash attempts.
249 */
250static int rate_location( map &m, const tripoint &p, const bool must_be_inside,
251 const int bash_str, const int attempt,
252 int ( &checked )[MAPSIZE_X][MAPSIZE_Y] )
253{
254 if( ( must_be_inside && m.is_outside( p ) ) ||
255 m.impassable( p ) ||
256 checked[p.x][p.y] > 0 ) {
257 return 0;
258 }
259
260 // Vector that will be used as a stack
261 std::vector<tripoint> st;
262 st.reserve( MAPSIZE_X * MAPSIZE_Y );
263 st.push_back( p );
264
265 // If not checked yet and either can be moved into, can be bashed down or opened,
266 // add it on the top of the stack.
267 const auto maybe_add = [&]( const tripoint & pt, const tripoint & from ) {
268 if( checked[pt.x][pt.y] >= attempt ) {
269 return;
270 }
271
272 if( m.passable( pt ) ||
273 m.bash_resistance( pt ) <= bash_str ||
274 m.open_door( pt, !m.is_outside( from ), true ) ) {
275 st.push_back( pt );
276 }
277 };
278
279 int area = 0;
280 while( !st.empty() ) {
281 area++;
282 const tripoint cur = st.back();
283 st.pop_back();
284
285 checked[cur.x][cur.y] = attempt;
286 if( cur.x == 0 || cur.x == MAPSIZE_X - 1 ||
287 cur.y == 0 || cur.y == MAPSIZE_Y - 1 ||
288 m.has_flag( "GOES_UP", cur ) ) {
289 return INT_MAX;
290 }
291
292 for( const tripoint &delta : eight_horizontal_neighbors ) {
293 tripoint pt = cur + delta;
294 pt.z = p.z;
295 maybe_add( pt, cur );
296 }
297 }
298
299 return area;
300}

302void start_location::place_player( player &u ) const
303{
304 // Need the "real" map with it's inside/outside cache and the like.
305 map &m = g->m;
306 // Start us off somewhere in the center of the map
307 u.setx( HALF_MAPSIZE_X );
308 u.sety( HALF_MAPSIZE_Y );
309 u.setz( g->get_levz() );
310 m.invalidate_map_cache( m.get_abs_sub().z );
311 m.build_map_cache( m.get_abs_sub().z );
312 const bool must_be_inside = flags().count( "ALLOW_OUTSIDE" ) == 0;
313 ///\EFFECT_STR allows player to start behind less-bashable furniture and terrain
314 // TODO: Allow using items here
315 const int bash = u.get_str();
316
317 // Remember biggest found location
318 // Sometimes it may be impossible to automatically found an ideal location
319 // but the player may be more creative than this algorithm and do away with just "good"
320 int best_rate = 0;
321 // In which attempt did this area get checked?
322 // We can overwrite earlier attempts, but not start in them
323 int checked[MAPSIZE_X][MAPSIZE_Y];
324 std::fill_n( &checked[0][0], MAPSIZE_X * MAPSIZE_Y, 0 );
325
326 bool found_good_spot = false;
327 // Try some random points at start
328
329 int tries = 0;
330 const auto check_spot = [&]( const tripoint & pt ) {
331 tries++;
332 const int rate = rate_location( m, pt, must_be_inside, bash, tries, checked );
333 if( best_rate < rate ) {
334 best_rate = rate;
335 u.setpos( pt );
336 if( rate == INT_MAX ) {
337 found_good_spot = true;
338 }
339 }
340 };
341
342 while( !found_good_spot && tries < 100 ) {
343 tripoint rand_point( HALF_MAPSIZE_X + rng( 0, SEEX * 2 - 1 ),
344 HALF_MAPSIZE_Y + rng( 0, SEEY * 2 - 1 ),
345 u.posz() );
346 check_spot( rand_point );
347 }
348 // If we haven't got a good location by now, screw it and brute force it
349 // This only happens in exotic locations (deep of a science lab), but it does happen
350 if( !found_good_spot ) {
351 tripoint tmp = u.pos();
352 int &x = tmp.x;
353 int &y = tmp.y;
354 for( x = 0; x < MAPSIZE_X; x++ ) {
355 for( y = 0; y < MAPSIZE_Y; y++ ) {
356 check_spot( tmp );
357 }
358 }
359 }
360
361 if( !found_good_spot ) {
362 debugmsg( "Could not find a good starting place for character" );
363 }

366void start_location::burn( const tripoint_abs_omt &omtstart, const size_t count,
367 const int rad ) const
368{
369 const tripoint_abs_sm player_location = project_to<coords::sm>( omtstart );
370 tinymap m;
371 m.load( player_location, false );
372 m.build_outside_cache( m.get_abs_sub().z );
373 const point u( g->u.posx() % HALF_MAPSIZE_X, g->u.posy() % HALF_MAPSIZE_Y );
374 std::vector<tripoint> valid;
375 for( const tripoint &p : m.points_on_zlevel() ) {
376 if( !( m.has_flag_ter( "DOOR", p ) ||
377 m.has_flag_ter( "OPENCLOSE_INSIDE", p ) ||
378 m.is_outside( p ) ||
379 ( p.x >= u.x - rad && p.x <= u.x + rad && p.y >= u.y - rad && p.y <= u.y + rad ) ) ) {
380 if( m.has_flag( "FLAMMABLE", p ) || m.has_flag( "FLAMMABLE_ASH", p ) ) {
381 valid.push_back( p );
382 }
383 }
384 }
385 std::shuffle( valid.begin(), valid.end(), rng_get_engine() );
386 for( size_t i = 0; i < std::min( count, valid.size() ); i++ ) {
387 m.add_field( valid[i], fd_fire, 3 );
388 }
389 m.save();
390}

406{
407 for( const bodypart_id &bp : u.get_all_body_parts( true ) ) {
408 if( bp == bodypart_id( "head" ) || bp == bodypart_id( "torso" ) ) {
409 continue;// Skip head + torso for balance reasons.
410 }
411 const int roll = static_cast<int>( rng( 1, 8 ) );
412 switch( roll ) {
413 // Damage + Bleed
414 case 1:
415 case 2:
416 u.add_effect( effect_bleed, 6_minutes, bp->token );
417 /* fallthrough */
418 case 3:
419 case 4:
420 // Just damage
421 case 5: {
422 const int maxHp = u.get_hp_max( bp );
423 // Body part health will range from 33% to 66% with occasional bleed
424 const int dmg = static_cast<int>( rng( maxHp / 3, maxHp * 2 / 3 ) );
425 u.apply_damage( nullptr, bp, dmg );
426 break;
427 }
428 // No damage
