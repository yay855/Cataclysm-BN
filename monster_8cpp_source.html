332void monster::try_upgrade( bool pin_time )
333{
334 if( !can_upgrade() ) {
335 return;
336 }
337
338 const int current_day = to_days<int>( calendar::turn - calendar::turn_zero );
339 //This should only occur when a monster is created or upgraded to a new form
340 if( upgrade_time < 0 ) {
341 upgrade_time = next_upgrade_time();
342 if( upgrade_time < 0 ) {
343 return;
344 }
345 if( pin_time || type->age_grow > 0 ) {
346 // offset by today, always true for growing creatures
347 upgrade_time += current_day;
348 } else {
349 // offset by starting season
350 // TODO: revisit this and make it simpler
351 upgrade_time += to_days<int>( calendar::start_of_cataclysm - calendar::turn_zero );
352 }
353 }
354
355 // Here we iterate until we either are before upgrade_time or can't upgrade any more.
356 // This is so that late into game new monsters can 'catch up' with all that half-life
357 // upgrades they'd get if we were simulating whole world.
358 while( true ) {
359 if( upgrade_time > current_day ) {
360 // not yet
361 return;
362 }
363
364 if( type->upgrade_into ) {
365 //If we upgrade into a blacklisted monster, treat it as though we are non-upgradeable
366 if( MonsterGroupManager::monster_is_blacklisted( type->upgrade_into ) ) {
367 return;
368 }
369 poly( type->upgrade_into );
370 } else {
371 const mtype_id &new_type = MonsterGroupManager::GetRandomMonsterFromGroup( type->upgrade_group );
372 if( new_type ) {
373 poly( new_type );
374 }
375 }
376
377 if( !upgrades ) {
378 // upgraded into a non-upgradeable monster
379 return;
380 }
381
382 const int next_upgrade = next_upgrade_time();
383 if( next_upgrade < 0 ) {
384 // hit never_upgrade
385 return;
386 }
387 upgrade_time += next_upgrade;
388 }

391void monster::try_reproduce()
392{
393 if( !reproduces ) {
394 return;
395 }
396 // This can happen if the monster type has changed (from reproducing to non-reproducing monster)
397 if( !type->baby_timer ) {
398 return;
399 }
400
401 if( !baby_timer ) {
402 // Assume this is a freshly spawned monster (because baby_timer is not set yet), set the point when it reproduce to somewhere in the future.
403 baby_timer.emplace( calendar::turn + *type->baby_timer );
404 }
405
406 bool season_spawn = false;
407 bool season_match = true;
408
409 // only 50% of animals should reproduce
410 bool female = one_in( 2 );
411 for( auto &elem : type->baby_flags ) {
412 if( elem == "SUMMER" || elem == "WINTER" || elem == "SPRING" || elem == "AUTUMN" ) {
413 season_spawn = true;
414 }
415 }
416
417 // add a decreasing chance of additional spawns when "catching up" an existing animal
418 int chance = -1;
419 while( true ) {
420 if( *baby_timer > calendar::turn ) {
421 return;
422 }
423
424 if( season_spawn ) {
425 season_match = false;
426 for( auto &elem : type->baby_flags ) {
427 if( ( season_of_year( *baby_timer ) == SUMMER && elem == "SUMMER" ) ||
428 ( season_of_year( *baby_timer ) == WINTER && elem == "WINTER" ) ||
429 ( season_of_year( *baby_timer ) == SPRING && elem == "SPRING" ) ||
430 ( season_of_year( *baby_timer ) == AUTUMN && elem == "AUTUMN" ) ) {
431 season_match = true;
432 }
433 }
434 }
435
436 chance += 2;
437 if( season_match && female && one_in( chance ) ) {
438 int spawn_cnt = rng( 1, type->baby_count );
439 if( type->baby_monster ) {
440 g->m.add_spawn( type->baby_monster, spawn_cnt, pos() );
441 } else {
442 g->m.add_item_or_charges( pos(), item( type->baby_egg, *baby_timer, spawn_cnt ), true );
443 }
444 }
445
446 *baby_timer += *type->baby_timer;
447 }

451{
452 if( type->starting_ammo.empty() ) {
453 debugmsg( "monster %s has no starting ammo to refill udders", get_name() );
454 return;
455 }

463 if( current_milk != ammo.end() ) {

466 // Erase old key-value from map
467 ammo.erase( current_milk );

469 }
470 // if we got here, we got milk.

1040Creature::Attitude monster::attitude_to( const Creature &other ) const
1041{
1042 const monster *m = other.is_monster() ? static_cast< const monster *>( &other ) : nullptr;
1043 const player *p = other.as_player();
1044 if( m != nullptr ) {
1045 if( m == this ) {
1046 return A_FRIENDLY;
1047 }
1048
1049 auto faction_att = faction.obj().attitude( m->faction );
1050 if( ( friendly != 0 && m->friendly != 0 ) ||
1051 ( friendly == 0 && m->friendly == 0 && faction_att == MFA_FRIENDLY ) ) {
1052 // Friendly (to player) monsters are friendly to each other
1053 // Unfriendly monsters go by faction attitude
1054 return A_FRIENDLY;
1055 } else if( ( friendly == 0 && m->friendly == 0 && faction_att == MFA_HATE ) ) {
1056 // Stuff that hates a specific faction will always attack that faction
1057 return A_HOSTILE;
1058 } else if( ( friendly == 0 && m->friendly == 0 && faction_att == MFA_NEUTRAL ) ||
1059 morale < 0 || anger < 10 ) {
1060 // Stuff that won't attack is neutral to everything
1061 return A_NEUTRAL;
1062 } else {
1063 return A_HOSTILE;
1064 }
1065 } else if( p != nullptr ) {
1066 switch( attitude( const_cast<player *>( p ) ) ) {
1067 case MATT_FRIEND:
1068 case MATT_ZLAVE:
1069 return A_FRIENDLY;
1070 case MATT_FPASSIVE:
1071 case MATT_FLEE:
1072 case MATT_IGNORE:
1073 case MATT_FOLLOW:
1074 return A_NEUTRAL;
1075 case MATT_ATTACK:
1076 return A_HOSTILE;
1077 case MATT_NULL:
1078 case NUM_MONSTER_ATTITUDES:
1079 break;
1080 }
1081 }
1082 // Should not happen!, creature should be either player or monster
1083 return A_NEUTRAL;
1084}

1086monster_attitude monster::attitude( const Character *u ) const
1087{
1088 if( friendly != 0 ) {
1089 if( has_effect( effect_docile ) ) {
1090 return MATT_FPASSIVE;
1091 }
1092 if( u == &g->u ) {
1093 return MATT_FRIEND;
1094 }
1095 // Zombies don't understand not attacking NPCs, but dogs and bots should.
1096 const npc *np = dynamic_cast< const npc * >( u );
1097 if( np != nullptr && np->get_attitude() != NPCATT_KILL && !type->in_species( ZOMBIE ) ) {
1098 return MATT_FRIEND;
1099 }
1100 if( np != nullptr && np->is_hallucination() ) {
1101 return MATT_IGNORE;
1102 }
1103 }
1104 if( effect_cache[FLEEING] ) {
1105 return MATT_FLEE;
1106 }
1107 if( has_effect( effect_pacified ) ) {
1108 return MATT_ZLAVE;
1109 }
1110
1111 int effective_anger = anger;
1112 int effective_morale = morale;
1113
1114 if( u != nullptr ) {
1115 // Those are checked quite often, so avoiding string construction is a good idea
1116 static const string_id<monfaction> faction_bee( "bee" );
1117 if( faction == faction_bee ) {
1118 if( u->has_trait( trait_BEE ) ) {
1119 return MATT_FRIEND;
1120 } else if( u->has_trait( trait_FLOWERS ) ) {
1121 effective_anger -= 10;
1122 }
1123 }
1124
1125 if( type->in_species( FUNGUS ) && ( u->has_trait( trait_THRESH_MYCUS ) ||
1126 u->has_trait( trait_MYCUS_FRIEND ) ) ) {
1127 return MATT_FRIEND;
1128 }
1129
1130 if( effective_anger >= 10 &&
1131 ( ( type->in_species( MAMMAL ) && u->has_trait( trait_PHEROMONE_MAMMAL ) ) ||
1132 ( type->in_species( INSECT ) && u->has_trait( trait_PHEROMONE_INSECT ) ) ) ) {
1133 effective_anger -= 20;
1134 }
1135
1136 if( u->has_trait( trait_TERRIFYING ) ) {
1137 effective_morale -= 10;
1138 }
1139
1140 if( has_flag( MF_ANIMAL ) ) {
1141 if( u->has_trait( trait_ANIMALEMPATH ) ) {
1142 effective_anger -= 10;
1143 if( effective_anger < 10 ) {
1144 effective_morale += 55;
1145 }
1146 } else if( u->has_trait( trait_ANIMALEMPATH2 ) ) {
1147 effective_anger -= 20;
1148 if( effective_anger < 20 ) {
1149 effective_morale += 80;
1150 }
1151 } else if( u->has_trait( trait_ANIMALDISCORD ) ) {
1152 if( effective_anger >= 10 ) {
1153 effective_anger += 10;
1154 }
1155 if( effective_anger < 10 ) {
1156 effective_morale -= 5;
1157 }
1158 } else if( u->has_trait( trait_ANIMALDISCORD2 ) ) {
1159 if( effective_anger >= 20 ) {
1160 effective_anger += 20;
1161 }
1162 if( effective_anger < 20 ) {
1163 effective_morale -= 5;
1164 }
1165 }
1166 }
1167
1168 for( const trait_id &mut : u->get_mutations() ) {
1169 for( const std::pair<const species_id, int> &elem : mut.obj().anger_relations ) {
1170 if( type->in_species( elem.first ) ) {
1171 effective_anger += elem.second;
1172 }
1173 }
1174 }
1175
1176 for( const trait_id &mut : u->get_mutations() ) {
1177 for( const species_id &spe : mut.obj().ignored_by ) {
1178 if( type->in_species( spe ) ) {
1179 return MATT_IGNORE;
1180 }
1181 }
1182 }
1183 }
1184
1185 if( effective_morale < 0 ) {
1186 if( effective_morale + effective_anger > 0 && get_hp() > get_hp_max() / 3 ) {
1187 return MATT_FOLLOW;
1188 }
1189 return MATT_FLEE;
1190 }
1191
1192 if( effective_anger <= 0 ) {
1193 if( get_hp() <= 0.6 * get_hp_max() ) {
1194 return MATT_FLEE;
1195 } else {
1196 return MATT_IGNORE;
1197 }
1198 }
1199
1200 if( effective_anger < 10 ) {
1201 return MATT_FOLLOW;
1202 }
1203
1204 return MATT_ATTACK;
1205}

1213{

1215 return anger > 0 && one_in( 5 ) ? 1 : 0;
1216 } );
1217
1218 process_trigger( mon_trigger::FIRE, [this]() {
1219 int ret = 0;
1220 for( const auto &p : g->m.points_in_radius( pos(), 3 ) ) {
1221 ret += 5 * g->m.get_field_intensity( p, fd_fire );
1222 }
1223 return ret;
1224 } );
1225

1227 // It's hard to ever see it in action
1228 // and even harder to balance it without making it exploitable
1229
1230 if( morale != type->morale && one_in( 10 ) ) {
1231 if( morale < type->morale ) {
1232 morale++;
1233 } else {
1234 morale--;
1235 }
1236 }
1237
1238 if( anger != type->agro && one_in( 10 ) ) {
1239 if( anger < type->agro ) {
1240 anger++;
1241 } else {
1242 anger--;
1243 }
1244 }

1246 // Cap values at [-100, 100] to prevent perma-angry moose etc.
1247 morale = std::min( 100, std::max( -100, morale ) );

1394void monster::melee_attack( Creature &target, float accuracy )
1395{
1396 mod_moves( -type->attack_cost );
1397 if( type->melee_dice == 0 ) {
1398 // We don't attack, so just return
1399 return;
1400 }
1401
1402 if( this == &target ) {
1403 // This happens sometimes
1404 return;
1405 }
1406
1407 if( !can_squeeze_to( target.pos() ) ) {
1408 return;
1409 }
1410
1411 int hitspread = target.deal_melee_attack( this, melee::melee_hit_range( accuracy ) );
1412
1413 if( target.is_player() ||
1414 ( target.is_npc() && g->u.attitude_to( target ) == A_FRIENDLY ) ) {
1415 // Make us a valid target for a few turns
1416 add_effect( effect_hit_by_player, 3_turns );
1417 }
1418
1419 if( has_flag( MF_HIT_AND_RUN ) ) {
1420 add_effect( effect_run, 4_turns );
1421 }
1422
1423 const bool u_see_me = g->u.sees( *this );
1424
1425 damage_instance damage = !is_hallucination() ? type->melee_damage : damage_instance();
1426 if( !is_hallucination() && type->melee_dice > 0 ) {
1427 damage.add_damage( DT_BASH, dice( type->melee_dice, type->melee_sides ) );
1428 }
1429
1430 dealt_damage_instance dealt_dam;
1431
1432 if( hitspread >= 0 ) {
1433 target.deal_melee_hit( this, hitspread, false, damage, dealt_dam );
1434 }
1435 body_part bp_hit = dealt_dam.bp_hit;
1436
1437 const int total_dealt = dealt_dam.total_damage();
1438 if( hitspread < 0 ) {
1439 // Miss

1444 add_msg( _( "%1$s dodges %2$s attack." ),
1445 target.disp_name(), name() );
1446 } else {
1447 add_msg( _( "The %1$s misses %2$s!" ),
1448 name(), target.disp_name() );
1449 }
1450 } else if( target.is_player() ) {
1451 add_msg( _( "You dodge an attack from an unseen source." ) );
1452 }
1453 } else if( is_hallucination() || total_dealt > 0 ) {
1454 // Hallucinations always produce messages but never actually deal damage
1455 if( u_see_me ) {
1456 if( target.is_player() ) {
1457 sfx::play_variant_sound( "melee_attack", "monster_melee_hit",
1458 sfx::get_heard_volume( target.pos() ) );
1459 sfx::do_player_death_hurt( dynamic_cast<player &>( target ), false );
1460 //~ 1$s is attacker name, 2$s is bodypart name in accusative.
1461 add_msg( m_bad, _( "The %1$s hits your %2$s." ), name(),
1462 body_part_name_accusative( bp_hit ) );
1463 } else if( target.is_npc() ) {
1464 if( has_effect( effect_ridden ) && has_flag( MF_RIDEABLE_MECH ) && pos() == g->u.pos() ) {
1465 //~ %1$s: name of your mount, %2$s: target NPC name, %3$d: damage value
1466 add_msg( m_good, _( "Your %1$s hits %2$s for %3$d damage!" ), name(), target.disp_name(),
1467 total_dealt );
1468 } else {
1469 //~ %1$s: attacker name, %2$s: target NPC name, %3$s: bodypart name in accusative
1470 add_msg( _( "The %1$s hits %2$s %3$s." ), name(),
1471 target.disp_name( true ),
1472 body_part_name_accusative( bp_hit ) );
1473 }
1474 } else {
1475 if( has_effect( effect_ridden ) && has_flag( MF_RIDEABLE_MECH ) && pos() == g->u.pos() ) {
1476 //~ %1$s: name of your mount, %2$s: target creature name, %3$d: damage value
1477 add_msg( m_good, _( "Your %1$s hits %2$s for %3$d damage!" ), get_name(), target.disp_name(),
1478 total_dealt );
1479 } else {
1480 //~ %1$s: attacker name, %2$s: target creature name
1481 add_msg( _( "The %1$s hits %2$s!" ), name(), target.disp_name() );
1482 }
1483 }
1484 } else if( target.is_player() ) {
1485 //~ %s is bodypart name in accusative.
1486 add_msg( m_bad, _( "Something hits your %s." ),
1487 body_part_name_accusative( bp_hit ) );
1488 }
1489 } else {
1490 // No damage dealt
1491 if( u_see_me ) {
1492 if( target.is_player() ) {
1493 //~ 1$s is attacker name, 2$s is bodypart name in accusative, 3$s is armor name
1494 add_msg( _( "The %1$s hits your %2$s, but your %3$s protects you." ), name(),
1495 body_part_name_accusative( bp_hit ), target.skin_name() );
1496 } else if( target.is_npc() ) {
1497 //~ $1s is monster name, %2$s is that monster target name,
1498 //~ $3s is target bodypart name in accusative, $4s is the monster target name,
1499 //~ 5$s is target armor name.
1500 add_msg( _( "The %1$s hits %2$s %3$s but is stopped by %4$s %5$s." ), name(),
1501 target.disp_name( true ),
1502 body_part_name_accusative( bp_hit ),
1503 target.disp_name( true ),
1504 target.skin_name() );
1505 } else {
1506 //~ $1s is monster name, %2$s is that monster target name,
1507 //~ $3s is target armor name.
1508 add_msg( _( "The %1$s hits %2$s but is stopped by its %3$s." ),
1509 name(),
1510 target.disp_name(),
1511 target.skin_name() );
1512 }
1513 } else if( target.is_player() ) {
1514 //~ 1$s is bodypart name in accusative, 2$s is armor name.
1515 add_msg( _( "Something hits your %1$s, but your %2$s protects you." ),
1516 body_part_name_accusative( bp_hit ), target.skin_name() );
1517 }
1518 }
1519
1520 target.check_dead_state();
1521
1522 if( is_hallucination() ) {
1523 if( one_in( 7 ) ) {
1524 die( nullptr );
1525 }
1526 return;
1527 }
1528
1529 if( total_dealt <= 0 ) {
1530 return;
1531 }
1532
1533 // Add any on damage effects
1534 for( const auto &eff : type->atk_effs ) {
1535 if( x_in_y( eff.chance, 100 ) ) {
1536 const body_part affected_bp = eff.affect_hit_bp ? bp_hit : eff.bp;
1537 target.add_effect( eff.id, time_duration::from_turns( eff.duration ), affected_bp );
1538 if( eff.permanent ) {
1539 target.get_effect( eff.id, affected_bp ).set_permanent();
1540 }
1541 }
1542 }
1543
1544 const int stab_cut = dealt_dam.type_damage( DT_CUT ) + dealt_dam.type_damage( DT_STAB );
1545
1546 if( stab_cut > 0 && has_flag( MF_VENOM ) ) {
1547 target.add_msg_if_player( m_bad, _( "You're envenomed!" ) );
1548 target.add_effect( effect_poison, 3_minutes );
1549 }
1550
1551 if( stab_cut > 0 && has_flag( MF_BADVENOM ) ) {
1552 target.add_msg_if_player( m_bad,
1553 _( "You feel venom flood your body, wracking you with pain" ) );
1554 target.add_effect( effect_badpoison, 4_minutes );
1555 }
1556
1557 if( stab_cut > 0 && has_flag( MF_PARALYZE ) ) {
1558 target.add_msg_if_player( m_bad, _( "You feel venom enter your body!" ) );
1559 target.add_effect( effect_paralyzepoison, 10_minutes );
1560 }
1561
1562 if( total_dealt > 6 && stab_cut > 0 && has_flag( MF_BLEED ) ) {
1563 // Maybe should only be if DT_CUT > 6... Balance question
1564 target.add_effect( effect_bleed, 6_minutes, bp_hit );
1565 }
