48void mdefense::zapback( monster &m, Creature *const source,
49 dealt_projectile_attack const *proj )
50{
51 if( source == nullptr ) {
52 return;
53 }
54 // If we have a projectile, we're a ranged attack, no zapback.
55 if( proj != nullptr ) {
56 return;
57 }
58
59 if( const player *const foe = dynamic_cast<player *>( source ) ) {
60 // Players/NPCs can avoid the shock if they wear non-conductive gear on their hands
61 for( const item &i : foe->worn ) {
62 if( ( i.covers( bp_hand_l ) || i.covers( bp_hand_r ) ) &&
63 !i.conductive() && i.get_coverage() >= 95 ) {
64 return;
65 }
66 }
67 // Players/NPCs can avoid the shock by using non-conductive weapons
68 if( !foe->weapon.conductive() ) {
69 if( foe->reach_attacking ) {
70 return;
71 }
72 if( !foe->used_weapon().is_null() ) {
73 return;
74 }
75 }
76 }
77
78 if( source->is_elec_immune() ) {
79 return;
80 }
81
82 if( get_avatar().sees( source->pos() ) ) {
83 const auto msg_type = source == &get_avatar() ? m_bad : m_info;
84 add_msg( msg_type, _( "Striking the %1$s shocks %2$s!" ),
85 m.name(), source->disp_name() );
86 }
87
88 const damage_instance shock {
89 DT_ELECTRIC, static_cast<float>( rng( 1, 5 ) )

97void mdefense::acidsplash( monster &m, Creature *const source,
98 dealt_projectile_attack const *const proj )
99{
100 if( source == nullptr ) {
101 return;
102 }
103 size_t num_drops = rng( 4, 6 );
104 // Would be useful to have the attack data here, for cutting vs. bashing etc.
105 if( proj ) {
106 // Projectile didn't penetrate the target, no acid will splash out of it.
107 if( proj->dealt_dam.total_damage() <= 0 ) {
108 return;
109 }
110 // Less likely for a projectile to deliver enough force
111 if( !one_in( 3 ) ) {
112 return;
113 }
114 } else {
115 if( const player *const foe = dynamic_cast<player *>( source ) ) {
116 if( foe->weapon.is_melee( DT_CUT ) || foe->weapon.is_melee( DT_STAB ) ) {
117 num_drops += rng( 3, 4 );
118 }
119 if( foe->unarmed_attack() ) {
120 const damage_instance acid_burn{
121 DT_ACID, static_cast<float>( rng( 1, 5 ) )
122 };
123 source->deal_damage( &m, one_in( 2 ) ? bodypart_id( "hand_l" ) : bodypart_id( "hand_r" ),
124 acid_burn );
125 source->add_msg_if_player( m_bad, _( "Acid covering %s burns your hand!" ), m.disp_name() );
126 }
127 }
128 }
129
130 // Don't splatter directly on the `m`, that doesn't work well
131 std::vector<tripoint> pts = closest_points_first( source->pos(), 1 );
132 pts.erase( std::remove( pts.begin(), pts.end(), m.pos() ), pts.end() );
133
134 projectile prj;
135 prj.speed = 10;
136 prj.range = 4;
137 prj.add_effect( ammo_effect_DRAW_AS_LINE );
138 prj.add_effect( ammo_effect_NO_DAMAGE_SCALING );
139 prj.impact.add_damage( DT_ACID, rng( 1, 3 ) );
140 for( size_t i = 0; i < num_drops; i++ ) {
141 const tripoint &target = random_entry( pts );
142 projectile_attack( prj, m.pos(), target, dispersion_sources{ 1200 }, &m );
143 }
144
145 if( get_avatar().sees( m.pos() ) ) {
146 add_msg( m_warning, _( "Acid sprays out of %s as it is hit!" ), m.disp_name() );
147 }
148}

150void mdefense::return_fire( monster &m, Creature *source, const dealt_projectile_attack *proj )
151{
152 // No return fire for untargeted projectiles, i.e. from explosions.
153 if( source == nullptr ) {
154 return;
155 }
156
157 // No return fire from dead monsters.
158 if( m.is_dead_state() ) {
159 return;
160 }
161
162 // If target actually was not damaged by projectile - then do not bother
163 // Also it covers potential exploit - peek throwing potential can be used to exhaust turret ammo
164 if( proj != nullptr && proj->dealt_dam.total_damage() == 0 ) {
165 return;
166 }
167
168 // No return fire if attacker is seen
169 if( m.sees( *source ) ) {
170 return;
171 }
172
173 const int distance_to_source = rl_dist( m.pos(), source->pos() );
174
175 // TODO: implement different rule, dependent on sound and probably some other things
176 const tripoint fire_point = source->pos();
177 // Add some innacuracy since it is blind fire
178 int dispersion = 150;
179
180 for( const std::pair<const std::string, mtype_special_attack> &attack : m.type->special_attacks ) {
181 if( attack.second->id == "gun" ) {
182
183 const gun_actor *gunactor = dynamic_cast<const gun_actor *>( attack.second.get() );
184 if( gunactor->get_max_range() < distance_to_source ) {
185 continue;
186 }
187 sounds::sound( m.pos(), 50, sounds::sound_t::alert,
188 _( "Detected shots from unseen attacker, return fire mode engaged." ) );
189
190 gunactor->shoot( m, fire_point, gun_mode_id( "DEFAULT" ), dispersion );
191
192 // We only return fire once with one gun.
193 return;
194 }
195 }
