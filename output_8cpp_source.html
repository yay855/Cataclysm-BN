68std::vector<std::string> foldstring( const std::string &str, int width, const char split )
69{
70 std::vector<std::string> lines;
71 if( width < 1 ) {
72 lines.push_back( str );
73 return lines;
74 }
75 std::stringstream sstr( str );
76 std::string strline;
77 std::vector<std::string> tags;
78 while( std::getline( sstr, strline, '\n' ) ) {
79 if( strline.empty() ) {
80 // Special case empty lines as std::getline() sets failbit immediately
81 // if the line is empty.
82 lines.emplace_back();
83 } else {
84 std::string wrapped = word_rewrap( strline, width, split );
85 std::stringstream swrapped( wrapped );
86 std::string wline;
87 while( std::getline( swrapped, wline, '\n' ) ) {
88 // Ensure that each line is independently color-tagged
89 // Re-add tags closed in the previous line
90 const std::string rawwline = wline;
91 if( !tags.empty() ) {
92 std::string swline;
93 for( const std::string &tag : tags ) {
94 swline += tag;
95 }
96 swline += wline;
97 wline = swline;
98 }
99 // Process the additional tags in the current line
100 const std::vector<size_t> tags_pos = get_tag_positions( rawwline );
101 for( const size_t tag_pos : tags_pos ) {
102 if( tag_pos + 1 < rawwline.size() && rawwline[tag_pos + 1] == '/' ) {
103 if( !tags.empty() ) {
104 tags.pop_back();
105 }
106 } else {
107 auto tag_end = rawwline.find( '>', tag_pos );
108 if( tag_end != std::string::npos ) {
109 tags.emplace_back( rawwline.substr( tag_pos, tag_end + 1 - tag_pos ) );
110 }
111 }
112 }
113 // Close any unclosed tags
114 if( !tags.empty() ) {
115 std::string swline;
116 swline += wline;
117 for( auto it = tags.rbegin(); it != tags.rend(); ++it ) {
118 // currently the only closing tag is </color>
119 swline += "</color>";
120 }
121 wline = swline;
122 }
123 // The resulting line can be printed independently and have the correct color
124 lines.emplace_back( wline );
125 }
126 }
127 }
128 return lines;
129}

132{

759 hotkeys.push_back( s.substr( lastsep + 1, end - lastsep - 1 ) );
760 }
761 return hotkeys;

854std::string format_item_info( const std::vector<iteminfo> &vItemDisplay,
855 const std::vector<iteminfo> &vItemCompare )
856{
857 std::string buffer;
858 bool bIsNewLine = true;
859
860 for( const auto &i : vItemDisplay ) {
861 if( i.sType == "DESCRIPTION" ) {
862 // Always start a new line for sType == "DESCRIPTION"
863 if( !bIsNewLine ) {
864 buffer += "\n";
865 }
866 if( i.bDrawName ) {
867 buffer += i.sName;
868 }
869 // Always end with a linebreak for sType == "DESCRIPTION"
870 buffer += "\n";
871 bIsNewLine = true;
872 } else {
873 if( i.bDrawName ) {
874 buffer += i.sName;
875 }
876
877 std::string sFmt = i.sFmt;
878 std::string sPost;
879
880 //A bit tricky, find %d and split the string
881 size_t pos = sFmt.find( "<num>" );
882 if( pos != std::string::npos ) {
883 buffer += sFmt.substr( 0, pos );
884 sPost = sFmt.substr( pos + 5 );
885 } else {
886 buffer += sFmt;
887 }
888
889 if( i.sValue != "-999" ) {
890 nc_color thisColor = c_yellow;
891 for( auto &k : vItemCompare ) {
892 if( k.sValue != "-999" ) {
893 if( i.sName == k.sName && i.sType == k.sType ) {
894 if( i.dValue > k.dValue - .1 &&
895 i.dValue < k.dValue + .1 ) {
896 thisColor = c_light_gray;
897 } else if( i.dValue > k.dValue ) {
898 if( i.bLowerIsBetter ) {
899 thisColor = c_light_red;
900 } else {
901 thisColor = c_light_green;
902 }
903 } else if( i.dValue < k.dValue ) {
904 if( i.bLowerIsBetter ) {
905 thisColor = c_light_green;
906 } else {
907 thisColor = c_light_red;
908 }
909 }
910 break;
911 }
912 }
913 }
914 buffer += colorize( i.sValue, thisColor );
915 }
916 buffer += sPost;
917
918 // Set bIsNewLine in case the next line should always start in a new line
919 if( ( bIsNewLine = i.bNewLine ) ) {
920 buffer += "\n";
921 }
922 }
923 }
924
925 return buffer;
926}

1135// utf-8 version
1136std::string word_rewrap( const std::string &in, int width, const uint32_t split )
1137{
1138 std::string o;
1139
1140 // find non-printing tags
1141 std::vector<size_t> tag_positions = get_tag_positions( in );
1142
1143 int lastwb = 0; //last word break
1144 int lastout = 0;
1145 const char *instr = in.c_str();
1146 bool skipping_tag = false;
1147 bool just_wrapped = false;
1148
1149 for( int j = 0, x = 0; j < static_cast<int>( in.size() ); ) {
1150 const char *ins = instr + j;
1151 int len = ANY_LENGTH;
1152 uint32_t uc = UTF8_getch( &ins, &len );
1153
1154 if( uc == '<' ) { // maybe skip non-printing tag
1155 std::vector<size_t>::iterator it;
1156 for( it = tag_positions.begin(); it != tag_positions.end(); ++it ) {
1157 if( static_cast<int>( *it ) == j ) {
1158 skipping_tag = true;
1159 break;
1160 }
1161 }
1162 }
1163
1164 const int old_j = j;
1165 j += ANY_LENGTH - len;
1166
1167 if( skipping_tag ) {
1168 if( uc == '>' ) {
1169 skipping_tag = false;
1170 }
1171 continue;
1172 }
1173
1174 if( just_wrapped && uc == ' ' ) { // ignore spaces after wrapping
1175 lastwb = lastout = j;
1176 continue;
1177 }
1178
1179 x += mk_wcwidth( uc );
1180
1181 if( uc == split || uc >= 0x2E80 ) { // param split (default ' ') or CJK characters
1182 if( x <= width ) {
1183 lastwb = j; // break after character
1184 } else {
1185 lastwb = old_j; // break before character
1186 }
1187 }
1188
1189 if( x > width ) {
1190 if( lastwb == lastout ) {
1191 lastwb = old_j;
1192 }
1193 // old_j may equal to lastout, this checks it and ensures there's at least one character in the line.
1194 if( lastwb == lastout ) {
1195 lastwb = j;
1196 }
1197 for( int k = lastout; k < lastwb; k++ ) {
1198 o += in[k];
1199 }
1200 o += '\n';
1201 x = 0;
1202 lastout = j = lastwb;
1203 just_wrapped = true;
1204 } else {
1205 just_wrapped = false;
1206 }
1207 }
1208 for( int k = lastout; k < static_cast<int>( in.size() ); k++ ) {
1209 o += in[k];
1210 }
1211
1212 return o;
1213}

1648/**
1649 * Inserts a table into a window, with data right-aligned.
1650 * @param pad Reduce table width by padding left side.
1651 * @param line Line to insert table.
1652 * @param columns Number of columns. Can be 1.
1653 * @param nc_color &FG Default color of table text.
1654 * @param divider To insert a character separating table entries. Can be blank.
1655 * @param r_align true for right aligned, false for left aligned.
1656 * @param data Text data to fill.
1657 * Make sure each entry of the data vector fits into one cell, including divider if any.
1658 */
1659void insert_table( const catacurses::window &w, int pad, int line, int columns,
1660 const nc_color &FG, const std::string &divider, bool r_align,
1661 const std::vector<std::string> &data )

1664 const int rows = getmaxy( w );
1665 const int col_width = ( ( width - pad ) / columns );
1666 int indent = 1; // 1 for right window border

1668 indent = ( col_width * columns ) + 1;
1669 }
1670 int div = columns - 1;
1671 int offset = 0;
1672
1673#if defined(__ANDROID__)
1674 input_context ctxt( "INSERT_TABLE" );
1675#endif
1676 wattron( w, FG );
1677 for( int i = 0; i < rows * columns; i++ ) {
1678 if( i + offset * columns >= static_cast<int>( data.size() ) ) {
1679 break;
1680 }
1681 int y = line + ( i / columns );
1682 if( r_align ) {
1683 indent -= col_width;
1684 }
1685 if( div != 0 ) {
1686 if( r_align ) {
1687 right_print( w, y, indent - utf8_width( divider ), FG, divider );

1692 div--;
1693 } else {
1694 div = columns - 1;

1700 indent = ( col_width * columns ) + 1;
