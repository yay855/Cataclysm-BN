85{

265void dependency_tree::build_connections(
266 std::map<mod_id, std::vector<mod_id > > key_dependency_map,
267 std::map<mod_id, std::vector<mod_id > > key_conflict_map
268)
269{
270 for( auto &elem : key_dependency_map ) {
271 const auto iter = master_node_map.find( elem.first );
272 if( iter != master_node_map.end() ) {
273 dependency_node *knode = &iter->second;
274
275 // apply parents list
276 std::vector<mod_id> vnode_parents = elem.second;
277 for( auto &vnode_parent : vnode_parents ) {
278 const auto iter = master_node_map.find( vnode_parent );
279 if( iter != master_node_map.end() ) {
280 dependency_node *vnode = &iter->second;
281
282 knode->add_parent( vnode );
283 vnode->add_child( knode );
284 } else {
285 // missing dependency!
286 knode->all_errors[node_error_type::missing_dep].push_back( "[" + vnode_parent.str() + "]" );
287 }
288 }
289 }
290 }
291
292 for( const std::pair<const mod_id, std::vector<mod_id>> &elem : key_conflict_map ) {
293 const auto iter = master_node_map.find( elem.first );
294 if( iter != master_node_map.end() ) {
295 dependency_node *knode = &iter->second;
296
297 const std::vector<mod_id> &vnode_conflicts = elem.second;
298 for( const mod_id &vnode_conflict : vnode_conflicts ) {
299 const auto iter = master_node_map.find( vnode_conflict );
300 if( iter != master_node_map.end() ) {
301 dependency_node *vnode = &iter->second;
302
303 knode->add_conflict( vnode );
304 vnode->add_conflict( knode );
305 }
306 }
307 }
308 }
309
310 // finalize and check for circular dependencies
311 check_for_strongly_connected_components();
312
313 for( auto &elem : master_node_map ) {
314 elem.second.inherit_errors();
315 }
316
317 // check for conflicts between dependencies
318 check_for_conflicting_dependencies();

383{
384 strongly_connected_components = std::vector<std::vector<dependency_node * > >();

412void dependency_tree::strong_connect( dependency_node *dnode )
413{
414 dnode->index = open_index;
415 dnode->lowlink = open_index;
416 ++open_index;
417 connection_stack.push( dnode );
418 dnode->on_stack = true;
419
420 for( std::vector<dependency_node *>::iterator it = dnode->parents.begin();
421 it != dnode->parents.end(); ++it ) {
422 if( ( *it )->index < 0 ) {
423 strong_connect( *it );
424 dnode->lowlink = std::min( dnode->lowlink, ( *it )->lowlink );
425 } else if( ( *it )->on_stack ) {
426 dnode->lowlink = std::min( dnode->lowlink, ( *it )->index );
427 }
428 }
429
430 if( dnode->lowlink == dnode->index ) {
431 std::vector<dependency_node *> scc;
432
433 dependency_node *d;
434 do {
435 d = connection_stack.top();
436 scc.push_back( d );
437 connection_stack.pop();
438 d->on_stack = false;
439 } while( dnode->key != d->key );
440
441 strongly_connected_components.push_back( scc );
442 dnode->on_stack = false;
443 }
444}

451void dependency_tree::check_for_conflicting_dependencies()
452{
453 for( auto &node : master_node_map ) {
454 dependency_node *this_node = &node.second;
455 std::vector<dependency_node *> all_deps = get_dependencies_of_X_as_nodes( node.first );
456 std::sort( all_deps.begin(), all_deps.end(), depnode_comparator );
457
458 for( auto dep_it = all_deps.begin(); dep_it != all_deps.end(); dep_it++ ) {
459 const dependency_node *this_dep = *dep_it;
460 std::vector<const dependency_node *> this_dep_conflicts = this_dep->conflicts;
461 std::sort( this_dep_conflicts.begin(), this_dep_conflicts.end(), depnode_comparator );
462
463 std::vector<const dependency_node *> both;
464 std::set_intersection(
465 std::next( dep_it ), all_deps.end(),
466 this_dep_conflicts.begin(), this_dep_conflicts.end(),
467 std::back_inserter( both ), depnode_comparator
468 );

471 //~ Single entry in list of conflicting dependencies, both %s are mod ids.
