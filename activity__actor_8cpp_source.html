158{
159 act.set_to_null();
160 item *weapon = get_weapon();
161 if( !weapon ) {
162 restore_view();

164 }
165 if( aborted ) {
166 if( reload_requested ) {

168 // May assign ACT_RELOAD

170 }
171 restore_view();
172 return;
173 }
174
175 // Fire!
176 gun_mode gun = weapon->gun_current_mode();
177
178 int shots_fired = ranged::fire_gun( who, fin_trajectory.back(), gun.qty, *gun, reload_loc );
179
180 if( shots_fired > 0 ) {
181 // TODO: bionic power cost of firing should be derived from a value of the relevant weapon.
182 if( bp_cost_per_shot > 0_J ) {
183 who.mod_power_level( -bp_cost_per_shot * shots_fired );
184 }
185 if( stamina_cost_per_shot > 0 ) {
186 who.mod_stamina( -stamina_cost_per_shot * shots_fired );
187 }
188 }
189
190 if( !get_option<bool>( "AIM_AFTER_FIRING" ) ) {
191 restore_view();
192 return;
193 }
194
195 // re-enter aiming UI with same parameters
196 aim_activity_actor aim_actor;
197 aim_actor.abort_if_no_targets = true;
198 aim_actor.fake_weapon = this->fake_weapon;
199 aim_actor.bp_cost_per_shot = this->bp_cost_per_shot;
200 aim_actor.initial_view_offset = this->initial_view_offset;
201
202 // if invalid target or it's dead - reset it so a new one is acquired
203 shared_ptr_fast<Creature> last_target = who.last_target.lock();
204 if( !last_target || last_target->is_dead_state() ) {
205 who.last_target.reset();
206 }
207 who.assign_activity( player_activity( aim_actor ), false );

893void move_items_activity_actor::do_turn( player_activity &act, Character &who )
894{
895 const tripoint dest = relative_destination + who.pos();
896
897 while( who.moves > 0 && !target_items.empty() ) {
898 item_location target = std::move( target_items.back() );
899 const int quantity = quantities.back();
900 target_items.pop_back();
901 quantities.pop_back();
902
903 if( !target ) {
904 debugmsg( "Lost target item of ACT_MOVE_ITEMS" );
905 continue;
906 }
907
908 // Don't need to make a copy here since movement can't be canceled
909 item &leftovers = *target;
910 // Make a copy to be put in the destination location
911 item newit = leftovers;
912
913 // Check that we can pick it up.
914 if( !newit.made_of( LIQUID ) ) {
915 // This is for hauling across zlevels, remove when going up and down stairs
916 // is no longer teleportation
917 if( newit.is_owned_by( who, true ) ) {
918 newit.set_owner( who );
919 } else {
920 continue;
921 }
922 // Handle charges, quantity == 0 means move all
923 if( quantity != 0 && newit.count_by_charges() ) {
924 newit.charges = std::min( newit.charges, quantity );
925 leftovers.charges -= quantity;
926 } else {
927 leftovers.charges = 0;
928 }
929 const tripoint src = target.position();
930 const int distance = src.z == dest.z ? std::max( rl_dist( src, dest ), 1 ) : 1;
931 who.mod_moves( -pickup::cost_to_move_item( who, newit ) * distance );
932 if( to_vehicle ) {
933 put_into_vehicle_or_drop( who, item_drop_reason::deliberate, { newit }, dest );
934 } else {
935 drop_on_map( who, item_drop_reason::deliberate, { newit }, dest );
936 }
937 // If we picked up a whole stack, remove the leftover item
938 if( leftovers.charges <= 0 ) {
939 target.remove_item();
940 }
941 }
942 }
943
944 if( target_items.empty() ) {
945 // Nuke the current activity, leaving the backlog alone.
946 act.set_to_null();
