22// event_transformation and event_statistic are both objects defined in json
23// and managed via generic_factory.
24// They both use the pimpl idiom and dynamic polymorphism to support their
25// implementation.
26// Their values can be calculated directly, or watched for updates.
27// To register something that watches for updates, call
28// stats_tracker::add_watcher.
29//
30// The way watching works is as follows:
31// - Each event_transformation and event_statistic has a corresponding state
32// object, which can be stored in the stats_tracker to hold the current
33// state of that transformation or statistic.
34// - Also, the stats_tracker stores all the watchers; any object watching any
35// event type or the value of any transformation or statistic.
36// - Each stat object will also register itself as a watcher for any event
37// type, transformation, or statistic it depends on.
38// - When an event occurs, watchers of that event_type will be informed.
39// - When stats are thus informed, they update their value if appropriate, and
40// inform the stats_tracker of the new value.
41// - The stats_tracker then notifies watchers again in turn.
42//
43// In this way, the updates cascade to all relevant objects and ultimately the
44// end-users (also in the form of watchers) are informed. For example, an
45// achievement might watch in this way so that it can trigger itself when its
46// conditions become true.

167struct value_constraint {
168 cata::optional<cata_variant> equals_;
169 cata::optional<std::string> equals_string_;
170 cata::optional<string_id<event_statistic>> equals_statistic_;
171
172 bool permits( const cata_variant &v, stats_tracker &stats ) const {
173 if( equals_ && *equals_ != v ) {
174 return false;
175 }
176 if( equals_string_ && *equals_string_ != v.get_string() ) {
177 return false;
178 }
179 if( equals_statistic_ && stats.value_of( *equals_statistic_ ) != v ) {
180 return false;
181 }
182 return true;
183 }
184
185 void deserialize( JsonIn &jsin ) {
186 JsonObject jo = jsin.get_object();
187 int equals_int;
188 if( jo.read( "equals", equals_int, false ) ) {
189 equals_ = cata_variant::make<cata_variant_type::int_>( equals_int );
190 }
191
192 bool equals_bool;
193 if( jo.read( "equals", equals_bool, false ) ) {
194 equals_ = cata_variant::make<cata_variant_type::bool_>( equals_bool );
195 }
196
197 std::string equals_string;
198 if( jo.read( "equals", equals_string, false ) ) {
199 equals_string_ = equals_string;
200 }
201
202 string_id<event_statistic> stat;
203 if( jo.read( "equals_statistic", stat ) ) {
204 equals_statistic_ = stat;
205 }
206
207 if( !equals_ && !equals_string_ && !equals_statistic_ ) {
208 jo.throw_error( "No valid value constraint found" );
209 }
210 }
211
212 void check( const std::string &name, const cata_variant_type input_type ) const {
213 if( equals_statistic_ ) {
214 if( !equals_statistic_->is_valid() ) {
215 debugmsg( "constraint for event_transformation %s refers to invalid statistic %s",
216 name, equals_statistic_->str() );
217 return;
218 }
219
220 cata_variant_type stat_type = ( *equals_statistic_ )->type();
221 if( stat_type != input_type ) {
222 debugmsg( "constraint for event_transformation %s matches statistic %s of "
223 "different type. Statistic has type %s but value compared with it has "
224 "type %s",
225 name, equals_statistic_->str(), io::enum_to_string( stat_type ),
226 io::enum_to_string( input_type ) );
227 }
228 }
229
230 if( equals_ ) {
231 if( input_type != equals_->type() ) {
232 debugmsg( "constraint for event_transformation %s matches constant of type %s but "
233 "value compared with it has type %s",
234 name, io::enum_to_string( equals_->type() ),
235 io::enum_to_string( input_type ) );
236 }
237 }
238 }
239
240 bool is_constant() const {
241 return !equals_statistic_ ||
242 ( *equals_statistic_ )->monotonicity() == monotonically::constant;
243 }

246struct new_field {
247 event_field_transformation transformation;
248 std::string input_field;
249
250 void deserialize( JsonIn &jsin ) {
251 JsonObject jo = jsin.get_object();
252 if( jo.size() != 1 ) {
253 jo.throw_error( "new field specifications must have exactly one entry" );
254 }
255 for( const JsonMember &m : jo ) {
256 const std::string &transformation_name = m.name();
257 auto it = event_field_transformations.find( transformation_name );
258 if( it == event_field_transformations.end() ) {
259 m.throw_error(
260 string_format( "unknown event field transformation '%s'", transformation_name )
261 );
262 }
263 transformation = it->second;
264 input_field = m.get_string();
265 }
266 }
267
268 void check( const std::string &context_name,
269 const cata::event::fields_type &input_fields ) const {
270 auto it = input_fields.find( input_field );
271 if( it == input_fields.end() ) {
272 debugmsg( "event_transformation %s specifies transformation on field %s but not such "
273 "field exists on the input", context_name, input_field );
274 return;
275 }
276 if( transformation.argument_types.size() != 1 ) {
277 debugmsg( "event_field_transformations of arity not 1 not supported" );
278 return;
279 }
280 if( it->second != transformation.argument_types[0] ) {
281 debugmsg( "event_transformation %s specifies transformation on field %s of incorrect "
282 "type. Transformation expects %s; field was %s.", context_name, input_field,
283 io::enum_to_string( transformation.argument_types[0] ),
284 io::enum_to_string( it->second ) );
285 }
286 }

288 std::vector<cata::event::data_type> transform( const cata::event::data_type &data,
289 const std::string &new_field_name ) const {
290 std::vector<cata::event::data_type> result;
291 auto it = data.find( input_field );
292 if( it == data.end() ) {
293 // Field missing, probably due to stale data. Reporting an error
294 // would spam, so we rely on the fact that it should already have
295 // been reported at startup time.
296 return result;
297 }
298 for( cata_variant v : transformation.function( it->second ) ) {
299 result.push_back( data );
300 result.back().emplace( new_field_name, v );
301 }
302 return result;
303 }
304
305 cata_variant_type type( const cata::event::fields_type &/*input*/ ) const {
306 return transformation.return_type;

411struct event_transformation_impl : public event_transformation::impl {
412 template<typename NewFields, typename Constraints, typename DropFields>
413 event_transformation_impl( const string_id<event_transformation> &id,
414 std::unique_ptr<event_source> source,
415 const NewFields &new_fields, const Constraints &constraints,
416 const DropFields &drop_fields ) :
417 id_( id ),
418 source_( std::move( source ) ),
419 new_fields_( new_fields.begin(), new_fields.end() ),
420 constraints_( constraints.begin(), constraints.end() ),
421 drop_fields_( drop_fields.begin(), drop_fields.end() )
422 {}
423
424 string_id<event_transformation> id_;
425 cata::clone_ptr<event_source> source_;
426 std::vector<std::pair<std::string, new_field>> new_fields_;
427 std::vector<std::pair<std::string, value_constraint>> constraints_;
428 std::vector<std::string> drop_fields_;
429
430 using EventVector = std::vector<cata::event::data_type>;
431
432 EventVector match_and_transform( const cata::event::data_type &input_data,
433 stats_tracker &stats ) const {
434 EventVector result = { input_data };
435 for( const std::pair<std::string, new_field> &p : new_fields_ ) {
436 EventVector before_this_pass = std::move( result );
437 result.clear();
438 for( const cata::event::data_type &data : before_this_pass ) {
439 EventVector from_this_event = p.second.transform( data, p.first );
440 result.insert( result.end(), from_this_event.begin(), from_this_event.end() );
441 }
442 }
443
444 auto violates_constraints = [&]( const cata::event::data_type & data ) {
445 for( const std::pair<std::string, value_constraint> &p : constraints_ ) {
446 const std::string &field = p.first;
447 const value_constraint &constraint = p.second;
448 const auto it = data.find( field );
449 if( it == data.end() || !constraint.permits( it->second, stats ) ) {
450 return true;
451 }
452 }
453 return false;
454 };
455
456 result.erase( std::remove_if( result.begin(), result.end(), violates_constraints ),
457 result.end() );
458
459 for( cata::event::data_type &data : result ) {
460 for( const std::string &drop_field_name : drop_fields_ ) {
461 data.erase( drop_field_name );
462 }
463 }
464
465 return result;
466 }
467
468 event_multiset initialize( const event_multiset::counts_type &input,
469 stats_tracker &stats ) const {
470 event_multiset result;
471
472 for( const std::pair<const cata::event::data_type, int> &p : input ) {
473 cata::event::data_type event_data = p.first;
474 EventVector transformed = match_and_transform( event_data, stats );
475 for( cata::event::data_type &d : transformed ) {
476 result.add( { d, p.second } );
477 }
478 }
479 return result;
480 }
481
482 event_multiset initialize( stats_tracker &stats ) const override {
483 return initialize( source_->get( stats ).counts(), stats );
484 }
485
486 void check( const std::string &name ) const override {
487 const cata::event::fields_type input_fields = source_->fields();
488
489 for( const std::pair<std::string, new_field> &p : new_fields_ ) {
490 if( input_fields.count( p.first ) ) {
491 debugmsg( "event_transformation %s tries to add field with name %s but a field "
492 "with that name already exists", name, p.first );
493 }
494 p.second.check( name, input_fields );
495 }
496
497 for( const std::pair<std::string, value_constraint> &p : constraints_ ) {
498 auto it = input_fields.find( p.first );
499 if( it == input_fields.end() ) {
500 debugmsg( "event_transformation %s applies constraint to field %s, but no field "
501 "with that name exists in the input", name, p.first );
502 } else {
503 p.second.check( name, it->second );
504 }
505 }
506
507 for( const std::string &drop_field_name : drop_fields_ ) {
508 if( input_fields.find( drop_field_name ) == input_fields.end() ) {
509 debugmsg( "event_transformation %s lists field %s to be dropped, but no field "
510 "with that name exists in the input", name, drop_field_name );
511 }
512 }
513 }

593{

1004 impl_ = std::make_unique<event_statistic_unique_value>(
