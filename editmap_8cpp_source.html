67static std::vector<std::string> fld_string( const std::string &str, int width )
68{
69 std::vector<std::string> lines;
70 if( width < 1 ) {
71 lines.push_back( str );
72 return lines;
73 }
74
75 int linepos = width;
76 int linestart = 0;
77 int crpos = -2;
78 while( linepos < static_cast<int>( str.length() ) || crpos != -1 ) {
79 crpos = str.find( '\n', linestart );
80 if( crpos != -1 && crpos <= linepos ) {
81 lines.push_back( str.substr( linestart, crpos - linestart ) );
82 linepos = crpos + width + 1;
83 linestart = crpos + 1;
84 } else {
85 int spacepos = str.rfind( ',', linepos );
86 if( spacepos == -1 ) {
87 spacepos = str.find( ',', linepos );
88 }
89 if( spacepos < linestart ) {
90 spacepos = linestart + width;
91 if( spacepos < static_cast<int>( str.length() ) ) {
92 lines.push_back( str.substr( linestart, width ) );
93 linepos = spacepos + width;
94 linestart = spacepos;
95 }
96 } else {
97 lines.push_back( str.substr( linestart, spacepos - linestart ) );
98 linepos = spacepos + width;
99 linestart = spacepos;
100 }
101 }
102 }
103 lines.push_back( str.substr( linestart ) );
104 return lines;
105}

841static ter_id get_alt_ter( bool isvert, ter_id sel_ter )
842{
843 std::map<std::string, std::string> alts;
844 alts["_v"] = "_h";
845 alts["_vertical"] = "_horizontal";
846 alts["_v_alarm"] = "_h_alarm";
847 const std::string tersid = sel_ter.obj().id.str();
848 const int sidlen = tersid.size();
849 for( std::map<std::string, std::string>::const_iterator it = alts.begin(); it != alts.end();
850 ++it ) {
851 const std::string suffix = isvert ? it->first : it->second;
852 const int slen = suffix.size();
853 if( sidlen > slen && tersid.substr( sidlen - slen, slen ) == suffix ) {
854 const std::string asuffix = isvert ? it->second : it->first;
855 const std::string terasid = tersid.substr( 0, sidlen - slen ) + asuffix;
856 const ter_str_id tid( terasid );
857
858 if( tid.is_valid() ) {
859 return tid.id();
860 }
861 }
862 }
863 return undefined_ter_id;
864}

1002template<>
1003void apply<ter_t>( const ter_t &t, const shapetype editshape, const tripoint &target,
1004 const tripoint &origin, const std::vector<tripoint> &target_list )
1005{
1006 bool isvert = false;
1007 bool ishori = false;
1008 bool doalt = false;
1009 ter_id teralt = undefined_ter_id;
1010 int alta = -1;
1011 int altb = -1;
1012 const ter_id sel_ter = t.id.id();
1013 if( editshape == editmap_rect ) {
1014 if( t.symbol() == LINE_XOXO || t.symbol() == '|' ) {
1015 isvert = true;
1016 teralt = get_alt_ter( isvert, sel_ter );
1017 } else if( t.symbol() == LINE_OXOX || t.symbol() == '-' ) {
1018 ishori = true;
1019 teralt = get_alt_ter( isvert, sel_ter );
1020 }
1021 if( teralt != undefined_ter_id ) {
1022 if( isvert ) {
1023 alta = target.y;
1024 altb = origin.y;
1025 } else {
1026 alta = target.x;
1027 altb = origin.x;
1028 }
1029 doalt = true;
1030 }
1031 }
1032
1033 map &here = get_map();
1034 for( auto &elem : target_list ) {
1035 ter_id wter = sel_ter;
1036 if( doalt ) {
1037 if( isvert && ( elem.y == alta || elem.y == altb ) ) {
1038 wter = teralt;
1039 } else if( ishori && ( elem.x == alta || elem.x == altb ) ) {
1040 wter = teralt;
1041 }
1042 }
1043 here.ter_set( elem, wter );
1044 const tripoint above = elem + tripoint_above;
1045
1046 if( wter->roof && here.ter( above ) == t_open_air ) {
1047 here.ter_set( above, wter->roof );
1048 }
1049 }

1533 */
1534void editmap::recalc_target( shapetype shape )
1535{
1536 const int z = target.z;
1537 target_list.clear();
1538 switch( shape ) {
1539 case editmap_circle: {
1540 int radius = rl_dist( origin, target );
1541 map &here = get_map();
1542 for( const tripoint &p : here.points_in_radius( origin, radius ) ) {
1543 if( rl_dist( p, origin ) <= radius ) {
1544 if( editmap_boundaries.contains( p ) ) {
1545 target_list.push_back( p );
1546 }
1547 }
1548 }
1549 }
1550 break;
1551 case editmap_rect_filled:
1552 case editmap_rect:
1553 int sx;
1554 int sy;
1555 int ex;
1556 int ey;
1557 if( target.x < origin.x ) {
1558 sx = target.x;
1559 ex = origin.x;
1560 } else {
1561 sx = origin.x;
1562 ex = target.x;
1563 }
1564 if( target.y < origin.y ) {
1565 sy = target.y;
1566 ey = origin.y;
1567 } else {
1568 sy = origin.y;
1569 ey = target.y;
1570 }
1571 for( int x = sx; x <= ex; x++ ) {
1572 for( int y = sy; y <= ey; y++ ) {
1573 if( shape == editmap_rect_filled || x == sx || x == ex || y == sy || y == ey ) {
1574 const tripoint p( x, y, z );
1575 if( editmap_boundaries.contains( p ) ) {
1576 target_list.push_back( p );
1577 }
1578 }
1579 }
1580 }
1581 break;
1582 case editmap_line:
1583 target_list = line_to( origin, target, 0, 0 );
1584 break;
1585 }

1588/*
1589 * Shift 'var' (i.e., part of a coordinate plane) by 'shift'.
1590 * If the result is not >= min and < 'max', constrain the result and adjust 'shift',
1591 * so it can adjust subsequent points of a set consistently.
1592 */
1593static int limited_shift( int var, int &shift, int min, int max )
1594{
1595 if( var + shift < min ) {
1596 shift = min - var;
1597 } else if( var + shift >= max ) {
1598 shift = shift + ( max - 1 - ( var + shift ) );
1599 }
1600 return var += shift;
1601}

1603/*
1604 * Move point 'editmap.target' via keystroke. 'moveorigin' determines if point 'editmap.origin' is moved as well:
1605 * 0: no, 1: yes, -1 (or none): as per bool 'editmap.moveall'.
1606 * if input or ch are not valid movement keys, do nothing and return false
1607 */
1608bool editmap::move_target( const std::string &action, int moveorigin )
1609{
1610 tripoint mp;
1611 bool move_origin = moveorigin == 1 ? true :
1612 moveorigin == 0 ? false : moveall;
1613 if( eget_direction( mp, action ) ) {
1614 target.x = limited_shift( target.x, mp.x, 0, MAPSIZE_X );
1615 target.y = limited_shift( target.y, mp.y, 0, MAPSIZE_Y );
1616 target.z = limited_shift( target.z, mp.z, -OVERMAP_DEPTH, OVERMAP_HEIGHT + 1 );
1617 if( move_origin ) {
1618 origin += mp;
1619 }
1620 return true;
1621 }
1622 return false;
1623}
