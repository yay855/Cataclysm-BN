48int recipe::batch_time( int batch, float multiplier, size_t assistants ) const
49{
50 // 1.0f is full speed
51 // 0.33f is 1/3 speed
52 if( multiplier == 0.0f ) {
53 // If an item isn't craftable in the dark, show the time to complete as if you could craft it
54 multiplier = 1.0f;
55 }
56
57 const float local_time = static_cast<float>( time ) / multiplier;
58
59 // if recipe does not benefit from batching and we have no assistants, don't do unnecessary additional calculations
60 if( batch_rscale == 0.0 && assistants == 0 ) {
61 return static_cast<int>( local_time ) * batch;
62 }
63
64 float total_time = 0.0;
65 // if recipe does not benefit from batching but we do have assistants, skip calculating the batching scale factor
66 if( batch_rscale == 0.0 ) {
67 total_time = local_time * batch;
68 } else {
69 // recipe benefits from batching, so batching scale factor needs to be calculated
70 // At batch_rsize, incremental time increase is 99.5% of batch_rscale
71 const double scale = batch_rsize / 6.0;
72 for( int x = 0; x < batch; x++ ) {
73 // scaled logistic function output
74 const double logf = ( 2.0 / ( 1.0 + std::exp( -( x / scale ) ) ) ) - 1.0;
75 total_time += local_time * ( 1.0 - ( batch_rscale * logf ) );
76 }
77 }
78
79 //Assistants can decrease the time for production but never less than that of one unit
80 if( assistants == 1 ) {
81 total_time = total_time * .75;
82 } else if( assistants >= 2 ) {
83 total_time = total_time * .60;
84 }
85 if( total_time < local_time ) {
86 total_time = local_time;
87 }
88
89 return static_cast<int>( total_time );
90}

350 autolearn_requirements = required_skills;

368 auto it = flags.find( "UNCRAFT_BY_QUANTITY" );

373 ident() );

387std::string recipe::get_consistency_error() const
388{
389 if( category == "CC_BUILDING" ) {
390 if( is_blueprint() || oter_str_id( result_.c_str() ).is_valid() ) {
391 return std::string();
392 }
393 return "defines invalid result";
394 }
395
396 if( !result_.is_valid() ) {
397 return "defines invalid result";
398 }
399
400 if( charges && !item::count_by_charges( result_ ) ) {
401 return "specifies charges but result is not counted by charges";
402 }
403
404 if( charges && result_mult != 1 ) {
405 return "specifies both charges and result_mult";
406 }
407
408 if( result_mult != 1 && reversible ) {
409 return "is reversible, so can't use result_mult";
410 }
411
412 const auto is_invalid_bp = []( const std::pair<itype_id, int> &elem ) {
413 return !elem.first.is_valid();
414 };
415
416 if( std::any_of( byproducts.begin(), byproducts.end(), is_invalid_bp ) ) {
417 return "defines invalid byproducts";
418 }
419
420 if( !contained && !container.is_null() ) {
421 return "defines container but not contained";
422 }
423
424 if( !container.is_valid() ) {
425 return "specifies unknown container";
426 }
427
428 const auto is_invalid_skill = []( const std::pair<skill_id, int> &elem ) {
429 return !elem.first.is_valid();
430 };
431
432 if( ( skill_used && !skill_used.is_valid() ) ||
433 std::any_of( required_skills.begin(), required_skills.end(), is_invalid_skill ) ) {
434 return "uses invalid skill";
435 }
436
437 const auto is_invalid_book = []( const std::pair<itype_id, int> &elem ) {
438 return !elem.first->book;
439 };
440
441 if( std::any_of( booksets.begin(), booksets.end(), is_invalid_book ) ) {
442 return "defines invalid book";
443 }
444
445 // TODO: IDs instead of raw string
446 if( !blueprint.empty() && !mapgen::has_update_id( blueprint ) ) {
447 return "has invalid mapgen id";
448 }
449
450 return std::string();

453item recipe::create_result() const
454{
455 item newit( result_, calendar::turn, item::default_charges_tag{} );
456 if( charges ) {
457 newit.charges = *charges;
458 }
459
460 if( !newit.craft_has_charges() ) {
461 newit.charges = 0;
462 } else if( result_mult != 1 ) {
463 // TODO: Make it work for charge-less items (update makes amount)
464 newit.charges *= result_mult;
465 }
466
467 // Show crafted items as fitting
468 // They might end up not fitting, but it's rare
469 if( newit.has_flag( flag_VARSIZE ) ) {
470 newit.item_tags.insert( flag_FIT );
471 }
472
473 if( contained ) {
474 newit = newit.in_container( container );
475 }
476
477 return newit;
478}

636std::function<bool( const item & )> recipe::get_component_filter(
637 const recipe_filter_flags flags ) const
638{
639 const item result = create_result();
640
641 // Disallow crafting of non-perishables with rotten components
642 // Make an exception for items with the ALLOW_ROTTEN flag such as seeds
643 const bool recipe_forbids_rotten =
644 result.is_food() && !result.goes_bad() && !has_flag( "ALLOW_ROTTEN" );
645 const bool flags_forbid_rotten =
646 static_cast<bool>( flags & recipe_filter_flags::no_rotten ) && result.goes_bad_after_opening();
647 std::function<bool( const item & )> rotten_filter = return_true<item>;
648 if( recipe_forbids_rotten || flags_forbid_rotten ) {
649 rotten_filter = []( const item & component ) {
650 return !component.rotten();
651 };
652 }
653
654 // Disallow usage of non-full magazines as components
655 // This is primarily used to require a fully charged battery, but works for any magazine.
656 std::function<bool( const item & )> magazine_filter = return_true<item>;
657 if( has_flag( "FULL_MAGAZINE" ) ) {
658 magazine_filter = []( const item & component ) {
659 return !component.is_magazine() || ( component.ammo_remaining() >= component.ammo_capacity() );
660 };
661 }
662
663 // Filter out filthy components here instead of with is_crafting_component
664 // Make an exception for recipes with the ALLOW_FILTHY flag

715static std::string dump_requirements(
716 const requirement_data &reqs,
717 int move_cost,
718 const std::map<skill_id, int> &skills )
719{
720 std::ostringstream os;
721 JsonOut jsout( os, /*pretty_print=*/true );
722
723 jsout.start_object();
724
725 jsout.member( "time" );
726 if( move_cost % 100 == 0 ) {
727 dump_to_json_string( time_duration::from_turns( move_cost / 100 ),
728 jsout, time_duration::units );
729 } else {
730 // cannot precisely represent the value using time_duration format,
731 // write integer instead.
732 jsout.write( move_cost );
733 }
734
735 jsout.member( "skills" );
736 jsout.start_array( /*wrap=*/!skills.empty() );
737 for( const std::pair<const skill_id, int> &p : skills ) {
738 jsout.start_array();
739 jsout.write( p.first );
740 jsout.write( p.second );
741 jsout.end_array();
742 }
743 jsout.end_array();
744
745 jsout.member( "inline" );
746 reqs.dump( jsout );
747
748 jsout.end_object();
749 return os.str();
750}

752void recipe::check_blueprint_requirements()
753{
754 if( !blueprint.empty() && !mapgen::has_update_id( blueprint ) ) {
755 debugmsg( "Blueprint %1$s has invalid mapgen id %2$s", ident_.str(), blueprint );
756 return;
757 }
758 build_reqs total_reqs;
759 const std::pair<std::map<ter_id, int>, std::map<furn_id, int>> &changed_ids
760 = get_changed_ids_from_update( blueprint );
761 get_build_reqs_for_furn_ter_ids( ident(), changed_ids, total_reqs );
762
763 requirement_data req_data_blueprint = std::accumulate(
764 reqs_blueprint.begin(), reqs_blueprint.end(), requirement_data(),
765 []( const requirement_data & lhs, const std::pair<requirement_id, int> &rhs ) {
766 return lhs + ( *rhs.first * rhs.second );
767 } );
768 requirement_data req_data_calc = std::accumulate(
769 total_reqs.reqs.begin(), total_reqs.reqs.end(), requirement_data(),
770 []( const requirement_data & lhs, const std::pair<requirement_id, int> &rhs ) {
771 return lhs + ( *rhs.first * rhs.second );
772 } );
773 // do not consolidate req_data_blueprint: it actually changes the meaning of the requirement.
774 // instead we enforce specifying the exact consolidated requirement.
775 req_data_calc.consolidate();
776 if( time_blueprint != total_reqs.time || skills_blueprint != total_reqs.skills
777 || !req_data_blueprint.has_same_requirements_as( req_data_calc ) ) {
778 std::string calc_req_str = dump_requirements( req_data_calc, total_reqs.time, total_reqs.skills );
779 std::string got_req_str = dump_requirements( req_data_blueprint, time_blueprint, skills_blueprint );
780
781 std::stringstream ss;
782 for( auto &id_count : changed_ids.first ) {
783 ss << string_format( "%s: %d\n", id_count.first.id(), id_count.second );
784 }
785 for( auto &id_count : changed_ids.second ) {
786 ss << string_format( "%s: %d\n", id_count.first.id(), id_count.second );
787 }
788
789 debugmsg( "Specified blueprint requirements of %1$s does not match calculated requirements. "
790 "Specify \"check_blueprint_needs\": false to disable the check or "
791 "Update \"blueprint_needs\" to the following value (you can use tools/update_blueprint_needs.py):\n"
792 // mark it for the auto-update python script
793 "~~~ auto-update-blueprint: %1$s\n"
794 "%2$s\n"
795 "~~~ end-auto-update\n"
796 "If the stated requirements match the above, but the error still appears, it is a bug.\n"
797 "Stated requirements are expanded to:\n"
798 "%3$s\n"
799 "---Begin expected tile changes---\n%4$s---End expected tile changes--",
800 ident_.str(), calc_req_str, got_req_str, ss.str() );
801 }
802}

804bool recipe::hot_result() const
805{
806 // Check if the recipe tools make this food item hot upon making it.
807 // We don't actually know which specific tool the player used/will use here, but
808 // we're checking for a class of tools; because of the way requirements
809 // processing works, the "surface_heat" id gets nuked into an actual
810 // list of tools, see data/json/recipes/cooking_tools.json.
811 //
812 // Currently it's only checking for a hotplate because that's a
813 // suitable item in both the "surface_heat" and "water_boiling_heat"
814 // tools, and it's usually the first item in a list of tools so if this
815 // does get heated we'll find it right away.
816 //
817 // TODO: Make this less of a hack
818 if( create_result().is_food() ) {
819 const requirement_data::alter_tool_comp_vector &tool_lists = simple_requirements().get_tools();
820 for( const std::vector<tool_comp> &tools : tool_lists ) {
821 for( const tool_comp &t : tools ) {
822 if( t.type == itype_hotplate ) {
823 return true;
824 }
825 }
826 }
827 }
828 return false;
829}
