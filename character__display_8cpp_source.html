61static bool should_combine_bps( const Character &ch, body_part l, body_part r,
62 const item *selected_clothing )
63{
64 const char_encumbrance_data enc_data = ch.get_encumbrance();
65 return l != r && // are different parts
66 l == other_part( r ) && r == other_part( l ) && // are complementary parts
67 // same encumberance & temperature
68 enc_data.elems[l] == enc_data.elems[r] &&
69 temperature_print_rescaling( ch.temp_conv[l] ) == temperature_print_rescaling( ch.temp_conv[r] ) &&
70 // selected_clothing covers both or neither parts
71 ( !selected_clothing || ( selected_clothing->covers( l ) == selected_clothing->covers( r ) ) );
72}

74static std::vector<std::pair<body_part, bool>> list_and_combine_bps( const Character &ch,
75 const item *selected_clothing )
76{
77 // bool represents whether the part has been combined with its other half
78 std::vector<std::pair<body_part, bool>> bps;
79 for( auto bp : all_body_parts ) {
80 // assuming that a body part has at most one other half
81 if( other_part( other_part( bp ) ) != bp ) {
82 debugmsg( "Bodypart %d has more than one other half!", bp );
83 }
84 if( should_combine_bps( ch, bp, other_part( bp ), selected_clothing ) ) {
85 if( bp < other_part( bp ) ) {
86 // only add the earlier one
87 bps.emplace_back( bp, true );
88 }
89 } else {
90 bps.emplace_back( bp, false );
91 }
92 }
93 return bps;

96void character_display::print_encumbrance( const catacurses::window &win, const Character &ch,
97 const int line, const item *selected_clothing )
98{
99 // bool represents whether the part has been combined with its other half
100 const std::vector<std::pair<body_part, bool>> bps = list_and_combine_bps( ch, selected_clothing );
101
102 // width/height excluding title & scrollbar
103 const int height = getmaxy( win ) - 1;
104 bool draw_scrollbar = height < static_cast<int>( bps.size() );
105 const int width = getmaxx( win ) - ( draw_scrollbar ? 1 : 0 );
106 // index of the first printed bodypart from `bps`
107 const int firstline = clamp( line - height / 2, 0, std::max( 0,
108 static_cast<int>( bps.size() ) - height ) );
109
110 /*** I chose to instead only display X+Y instead of X+Y=Z. More room was needed ***
111 *** for displaying triple digit encumbrance, due to new encumbrance system. ***
112 *** If the player wants to see the total without having to do them maths, the ***
113 *** armor layers ui shows everything they want :-) -Davek ***/
114 const char_encumbrance_data enc_data = ch.get_encumbrance();
115 for( int i = 0; i < height; ++i ) {
116 int thisline = firstline + i;
117 if( thisline < 0 ) {
118 continue;
119 }
120 if( static_cast<size_t>( thisline ) >= bps.size() ) {
121 break;
122 }
123 const body_part bp = bps[thisline].first;
124 const bool combine = bps[thisline].second;
125 const encumbrance_data &e = enc_data.elems[bp];
126 const bool highlighted = selected_clothing ? selected_clothing->covers( bp ) : false;
127 std::string out = body_part_name_as_heading( bp, combine ? 2 : 1 );
128 if( utf8_width( out ) > 7 ) {
129 out = utf8_truncate( out, 7 );
130 }
131
132 // Two different highlighting schemes, highlight if the line is selected as per line being set.
133 // Make the text green if this part is covered by the passed in item.
134 nc_color limb_color = thisline == line ?
135 ( highlighted ? h_green : h_light_gray ) :
136 ( highlighted ? c_green : c_light_gray );
137 mvwprintz( win, point( 1, 1 + i ), limb_color, "%s", out );
138 // accumulated encumbrance from clothing, plus extra encumbrance from layering
139 mvwprintz( win, point( 8, 1 + i ), encumb_color( e.encumbrance ), "%3d",
140 e.encumbrance - e.layer_penalty );
141 // separator in low toned color
142 mvwprintz( win, point( 11, 1 + i ), c_light_gray, "+" );
143 // take into account the new encumbrance system for layers
144 mvwprintz( win, point( 12, 1 + i ), encumb_color( e.encumbrance ), "%-3d", e.layer_penalty );
145 // print warmth, tethered to right hand side of the window
146 mvwprintz( win, point( width - 6, 1 + i ), ch.bodytemp_color( bp ), "(% 3d)",
147 temperature_print_rescaling( ch.temp_conv[bp] ) );
148 }
149
150 if( draw_scrollbar ) {
151 scrollbar().
152 offset_x( width ).
153 offset_y( 1 ).
154 content_size( bps.size() ).
155 viewport_pos( firstline ).
156 viewport_size( height ).
157 scroll_to_last( false ).
158 apply( win );
159 }
160}

925static bool handle_player_display_action( Character &you, unsigned int &line,
926 player_display_tab &curtab, input_context &ctxt,
927 const ui_adaptor &ui_tip, const ui_adaptor &ui_info,
928 const ui_adaptor &ui_stats, const ui_adaptor &ui_encumb,
929 const ui_adaptor &ui_traits, const ui_adaptor &ui_bionics,
930 const ui_adaptor &ui_effects, const ui_adaptor &ui_skills,
931 const std::vector<trait_id> &traitslist,
932 const std::vector<bionic> &bionicslist,
933 const std::vector<std::pair<std::string, std::string>> &effect_name_and_text,
934 const std::vector<HeaderSkill> &skillslist )
935{
936 const auto invalidate_tab = [&]( const player_display_tab tab ) {
937 switch( tab ) {
938 case player_display_tab::stats:
939 ui_stats.invalidate_ui();
940 break;
941 case player_display_tab::encumbrance:
942 ui_encumb.invalidate_ui();
943 break;
944 case player_display_tab::traits:
945 ui_traits.invalidate_ui();
946 break;
947 case player_display_tab::bionics:
948 ui_bionics.invalidate_ui();
949 break;
950 case player_display_tab::effects:
951 ui_effects.invalidate_ui();
952 break;
953 case player_display_tab::skills:
954 ui_skills.invalidate_ui();
955 break;
956 case player_display_tab::num_tabs:
957 abort();
958 }
959 };
960
961 unsigned int line_beg = 0;
962 unsigned int line_end = 0;
963 switch( curtab ) {
964 case player_display_tab::stats:
965 line_end = 6;
966 break;
967 case player_display_tab::encumbrance: {
968 const std::vector<std::pair<body_part, bool>> bps = list_and_combine_bps( you, nullptr );
969 line_end = bps.size();
970 break;
971 }
972 case player_display_tab::traits:
973 line_end = traitslist.size();
974 break;
975 case player_display_tab::bionics:
976 line_end = bionicslist.size();
977 break;
978 case player_display_tab::effects:
979 line_end = effect_name_and_text.size();
980 break;
981 case player_display_tab::skills:
982 line_beg = 1; // skip first header
983 line_end = skillslist.size();
984 break;
985 case player_display_tab::num_tabs:
986 abort();
987 }
988 if( line_beg >= line_end || line < line_beg ) {
989 line = line_beg;
990 } else if( line > line_end - 1 ) {
991 line = line_end - 1;
992 }
993
994 bool done = false;
995 std::string action = ctxt.handle_input();
996
997 if( action == "UP" ) {
998 if( line > line_beg ) {
999 --line;
1000 } else {
1001 line = line_end - 1;
1002 }
1003 if( curtab == player_display_tab::skills && skillslist[line].is_header ) {
1004 --line;
1005 }
1006 invalidate_tab( curtab );
1007 ui_info.invalidate_ui();
1008 } else if( action == "DOWN" ) {
1009 if( line + 1 < line_end ) {
1010 ++line;
1011 } else {
1012 line = line_beg;
1013 }
1014 if( curtab == player_display_tab::skills && skillslist[line].is_header ) {
1015 ++line;
1016 }
1017 invalidate_tab( curtab );
1018 ui_info.invalidate_ui();
1019 } else if( action == "NEXT_TAB" || action == "PREV_TAB" ) {
1020 line = 0;
1021 invalidate_tab( curtab );
1022 curtab = action == "NEXT_TAB" ? next_tab( curtab ) : prev_tab( curtab );
1023 invalidate_tab( curtab );
1024 ui_info.invalidate_ui();
1025 } else if( action == "QUIT" ) {
1026 done = true;
1027 } else if( action == "CONFIRM" ) {
1028 switch( curtab ) {
1029 default:
1030 break;
1031 case player_display_tab::stats:
1032 if( line < 4 && get_option<bool>( "STATS_THROUGH_KILLS" ) && you.is_avatar() ) {
1033 character_display::upgrade_stat_prompt( *you.as_avatar(), static_cast<character_stat>( line ) );
1034 }
1035 invalidate_tab( curtab );
1036 break;
1037 case player_display_tab::skills: {
1038 const Skill *selectedSkill = nullptr;
1039 if( line < skillslist.size() && !skillslist[line].is_header ) {
1040 selectedSkill = skillslist[line].skill;
1041 }
1042 if( selectedSkill ) {
1043 you.get_skill_level_object( selectedSkill->ident() ).toggleTraining();
1044 }
1045 invalidate_tab( curtab );
1046 break;
1047 }
1048 }
1049 } else if( action == "CHANGE_PROFESSION_NAME" ) {
1050 string_input_popup popup;
1051 popup.title( _( "Profession Name: " ) )
1052 .width( 25 )
1053 .text( "" )
1054 .max_length( 25 )
1055 .query();
1056
1057 you.custom_profession = popup.text();
1058 ui_tip.invalidate_ui();
1059 }
1060 return done;
1061}
