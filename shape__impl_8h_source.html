37/**
38 * Cone centered along y axis.
39 * Tip is at zero, base at (0, -h, 0).
40 * A 90 degree half angle will cause zero division.
41 */
42class cone : public shape_impl
43{
44 private:
45 double h;
46 double sin_angle;
47 double cos_angle;
48
49 public:
50 cone( units::angle half_angle, double h )
51 : h( h )
52 , sin_angle( sin( half_angle ) )
53 , cos_angle( cos( half_angle ) )
54 {}
55
56 double signed_distance( const rl_vec3d &p ) const override {
57 // Based on https://www.iquilezles.org/www/articles/distfunctions/distfunctions.htm
58 rl_vec2d q( h * sin_angle / cos_angle, -h );
59 rl_vec2d w( std::sqrt( p.x * p.x + p.z * p.z ), p.y );
60 rl_vec2d a = w - q * clamp( w.dot_product( q ) / q.dot_product( q ), 0.0f, 1.0f );
61 rl_vec2d b = w - rl_vec2d( q.x * clamp( w.x / q.x, 0.0f, 1.0f ), q.y );
62 float k = q.y >= 0.0f ? 1.0f : -1.0f;
63 float d = std::min( a.dot_product( a ), b.dot_product( b ) );
64 float s = std::max( k * ( w.x * q.y - w.y * q.x ), k * ( w.y - q.y ) );
65 return std::sqrt( d ) * sign( s );
66 }
67
68 inclusive_cuboid<rl_vec3d> bounding_box() const override {
69 return inclusive_cuboid<rl_vec3d>( rl_vec3d( -sin_angle * h, -h, 0 ),
70 rl_vec3d( sin_angle * h, 0, 0 ) );
71 }
72};

74class cylinder : public shape_impl
75{
76 private:
77 double half_length;
78 double radius;
79
80 public:
81 cylinder( double length, double radius )
82 : half_length( 0.5 * length )
83 , radius( radius )
84 {}
85
86 double signed_distance( const rl_vec3d &p ) const override {
87 // https://www.iquilezles.org/www/articles/distfunctions/distfunctions.htm
88 // Plus adjustments (and fix!)
89 double dx = std::sqrt( p.x * p.x + p.z * p.z ) - radius;
90 double dy = std::abs( p.y + half_length ) - half_length;
91 double lx = std::max( dx, 0.0 );
92 double ly = std::max( dy, 0.0 );
93
94 return std::min( std::max( dx, dy ), 0.0 ) + std::sqrt( lx * lx + ly * ly );
95 }
96
97 inclusive_cuboid<rl_vec3d> bounding_box() const override {
98 return inclusive_cuboid<rl_vec3d>( rl_vec3d( radius, -2 * half_length, 0 ),
99 rl_vec3d( radius, 0, 0 ) );
100 }

146class rotate_z_shape : public shape_impl
147{
148 private:

150 matrix_3d mat;

152
153 public:
154 rotate_z_shape( const std::shared_ptr<shape_impl> &shape, units::angle angle_z )
155 : shape( shape )
156 , mat( matrices::rotation_z_axis( -angle_z ) )

161 return shape->signed_distance( mat * p );

163
164 inclusive_cuboid<rl_vec3d> bounding_box() const override {
165 const inclusive_cuboid<rl_vec3d> &bb = shape->bounding_box();
166 const std::set<rl_vec3d> pts = {
167 rl_vec3d( bb.p_max.x, bb.p_min.y, bb.p_min.z ),
168 rl_vec3d( bb.p_min.x, bb.p_max.y, bb.p_min.z ),
169 rl_vec3d( bb.p_max.x, bb.p_max.y, bb.p_min.z ),
170 rl_vec3d( bb.p_min.x, bb.p_min.y, bb.p_max.z ),
171 rl_vec3d( bb.p_max.x, bb.p_min.y, bb.p_max.z ),
172 rl_vec3d( bb.p_min.x, bb.p_max.y, bb.p_max.z )
173 };
174 rl_vec3d min = mat_inv * bb.p_min;
175 rl_vec3d max = mat_inv * bb.p_max;
176 for( const rl_vec3d &unrotated : pts ) {
177 rl_vec3d v = mat_inv * unrotated;
178 min.x = std::min( min.x, v.x );
179 min.y = std::min( min.y, v.y );
180 min.z = std::min( min.z, v.z );
181 max.x = std::max( max.x, v.x );
182 max.y = std::max( max.y, v.y );
183 max.z = std::max( max.z, v.z );
184 }
185 return inclusive_cuboid<rl_vec3d>( min, max );
186 }
