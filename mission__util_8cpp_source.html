161static cata::optional<tripoint_abs_omt> find_or_create_om_terrain(
162 const tripoint_abs_omt &origin_pos, const mission_target_params &params )
163{
164 tripoint_abs_omt target_pos = overmap::invalid_tripoint;
165
166 omt_find_params find_params;
167 std::vector<std::pair<std::string, ot_match_type>> temp_types;
168 std::pair<std::string, ot_match_type> temp_pair;
169 temp_pair.first = params.overmap_terrain;
170 temp_pair.second = params.overmap_terrain_match_type;
171 temp_types.push_back( temp_pair );
172 find_params.types = temp_types;
173 find_params.search_range = params.search_range;
174 find_params.min_distance = params.min_distance;
175 find_params.must_see = params.must_see;
176 find_params.cant_see = params.cant_see;
177 find_params.existing_only = true;
178 find_params.popup = make_shared_fast<throbber_popup>( _( "Please wait…" ) );
179
180 auto get_target_position = [&]() {
181 // Either find a random or closest match, based on the criteria.
182 if( params.random ) {
183 target_pos = overmap_buffer.find_random( origin_pos, find_params );
184 } else {
185 target_pos = overmap_buffer.find_closest( origin_pos, find_params );
186 }
187
188 // If we didn't find a match, and we're allowed to create new terrain, and the player didn't
189 // have to see the location beforehand, then we can attempt to create the new terrain.
190 if( target_pos == overmap::invalid_tripoint && params.create_if_necessary &&
191 !params.must_see ) {
192 // If this terrain is part of an overmap special...
193 if( params.overmap_special ) {
194 // ...then attempt to place the whole special.
195 const bool placed = overmap_buffer.place_special( *params.overmap_special, origin_pos,
196 params.search_range );
197 // If we succeeded in placing the special, then try and find the particular location
198 // we're interested in.
199 if( placed ) {
200 find_params.must_see = false;
201 target_pos = overmap_buffer.find_closest( origin_pos, find_params );
202 }
203 } else if( params.replaceable_overmap_terrain ) {
204 // This terrain wasn't part of an overmap special, but we do have a replacement
205 // terrain specified. Find a random location of that replacement type.
206 find_params.must_see = false;
207 find_params.types.front().first = *params.replaceable_overmap_terrain;
208 target_pos = overmap_buffer.find_random( origin_pos, find_params );
209
210 // We didn't find it, so allow this search to create new overmaps and try again.
211 find_params.existing_only = true;
212 if( target_pos == overmap::invalid_tripoint ) {
213 target_pos = overmap_buffer.find_random( origin_pos, find_params );
214 }
215
216 // We found a match, so set this position (which was our replacement terrain)
217 // to our desired mission terrain.
218 if( target_pos != overmap::invalid_tripoint ) {
219 overmap_buffer.ter_set( target_pos, oter_id( params.overmap_terrain ) );
220 }
221 }
222 }
223 };
224
225 // First try to get the position where we only allow existing overmaps.
226 get_target_position();
227
228 if( target_pos == overmap::invalid_tripoint ) {
229 // If it's invalid, then that means we couldn't find it or create it (if allowed) on
230 // our current overmap. We'll now go ahead and try again but allow it to create new overmaps.
231 find_params.existing_only = false;
232 get_target_position();
233 }
234
235 // If we got here and this is still invalid, it means that we couldn't find it nor create it
236 // on any overmap (new or existing) within the allowed search range.
237 if( target_pos == overmap::invalid_tripoint ) {
238 debugmsg( "Unable to find and assign mission target %s.", params.overmap_terrain );
239 return cata::nullopt;
240 }
241 return target_pos;
242}
