137// read an expansion's terrain ID of the form faction_base_$TYPE_$CURLEVEL
138// find the last underbar, strip off the prefix of faction_base_ (which is 13 chars),
139// and the pull out the $TYPE and $CURLEVEL
140// This is legacy support for existing camps; future camps don't use cur_level at all
141expansion_data basecamp::parse_expansion( const std::string &terrain,
142 const tripoint_abs_omt &new_pos )
143{
144 expansion_data e;
145 int last_bar = terrain.find_last_of( '_' );
146 e.type = terrain.substr( base_camps::prefix_len, last_bar - base_camps::prefix_len );
147 e.cur_level = std::stoi( terrain.substr( last_bar + 1 ) );
148 e.pos = new_pos;
149 return e;
150}

299std::vector<basecamp_upgrade> basecamp::available_upgrades( const point &dir )
300{
301 std::vector<basecamp_upgrade> ret_data;
302 auto e = expansions.find( dir );
303 if( e != expansions.end() ) {
304 expansion_data &e_data = e->second;
305 for( const recipe *recp_p : recipe_dict.all_blueprints() ) {
306 const recipe &recp = *recp_p;
307 const std::string &bldg = recp.result().str();
308 // skip buildings that are completed
309 if( e_data.provides.find( bldg ) != e_data.provides.end() ) {
310 continue;
311 }
312 // skip building that have unmet requirements
313 size_t needed_requires = recp.blueprint_requires().size();
314 size_t met_requires = 0;
315 for( const auto &bp_require : recp.blueprint_requires() ) {
316 if( e_data.provides.find( bp_require.first ) == e_data.provides.end() ) {
317 break;
318 }
319 if( e_data.provides[bp_require.first] < bp_require.second ) {
320 break;
321 }
322 met_requires += 1;
323 }
324 if( met_requires < needed_requires ) {
325 continue;
326 }
327 bool should_display = true;
328 bool in_progress = false;
329 for( const auto &bp_exclude : recp.blueprint_excludes() ) {
330 // skip buildings that are excluded by previous builds
331 if( e_data.provides.find( bp_exclude.first ) != e_data.provides.end() ) {
332 if( e_data.provides[bp_exclude.first] >= bp_exclude.second ) {
333 should_display = false;
334 break;
335 }
336 }
337 // track buildings that are currently being built
338 if( e_data.in_progress.find( bp_exclude.first ) != e_data.in_progress.end() ) {
339 if( e_data.in_progress[bp_exclude.first] >= bp_exclude.second ) {
340 in_progress = true;
341 break;
342 }
343 }
344 }
345 if( !should_display ) {
346 continue;
347 }
348 basecamp_upgrade data;
349 data.bldg = bldg;
350 data.name = recp.blueprint_name();
351 const auto &reqs = recp.deduped_requirements();
352 data.avail = reqs.can_make_with_inventory( _inv, recp.get_component_filter(), 1 );
353 data.in_progress = in_progress;
354 ret_data.emplace_back( data );
355 }
356 }
357 return ret_data;
358}

584std::list<item> basecamp::use_charges( const itype_id &fake_id, int &quantity )
585{
586 std::list<item> ret;
587 if( quantity <= 0 ) {
588 return ret;
589 }

591 if( bcp_r.fake_id == fake_id ) {

593 camp_item.charges = std::min( bcp_r.available, quantity );
594 quantity -= camp_item.charges;

596 bcp_r.consumed += camp_item.charges;
597 ret.push_back( camp_item );
598 if( quantity <= 0 ) {

601 }
602 }

606void basecamp::form_crafting_inventory( map &target_map )
607{
608 _inv.clear();

610 const tripoint &origin = target_map.getlocal( dump_spot );
611 auto &mgr = zone_manager::get_manager();
612 if( get_map().check_vehicle_zones( g->get_levz() ) ) {
613 mgr.cache_vzones();
614 }
615 if( mgr.has_near( zone_type_camp_storage, dump_spot, 60 ) ) {
616 std::unordered_set<tripoint> src_set = mgr.get_near( zone_type_camp_storage, dump_spot, 60 );
617 _inv.form_from_zone( target_map, src_set, nullptr, false );
618 }
619 /*
620 * something of a hack: add the resources we know the camp has
621 * the hacky part is that we're adding resources based on the camp's flags, which were
622 * generated based on upgrade missions, instead of having resources added when the
623 * map changes
624 */
625 // make sure the array is empty
626 fuels.clear();
627 for( const itype_id &fuel_id : fuel_types ) {
628 basecamp_fuel bcp_f;
629 bcp_f.available = 0;

631 fuels.emplace_back( bcp_f );
632 }

641 break;

648 bcp_r.consumed = 0;

714{
715 const auto filter = is_crafting_component;
716 const requirement_data *req =
717 making_.deduped_requirements().select_alternative( g->u, base_._inv, filter, batch_size_ );
718 if( !req ) {
719 return false;
720 }
721 if( !item_selections_.empty() || !tool_selections_.empty() ) {
722 debugmsg( "Reused basecamp_action_components" );
723 return false;
724 }
725 for( const auto &it : req->get_components() ) {
726 comp_selection<item_comp> is =
727 g->u.select_item_component( it, batch_size_, base_._inv, true, filter,
728 !base_.by_radio );
729 if( is.use_from == cancel ) {
730 return false;
731 }
732 item_selections_.push_back( is );
733 }
734 // this may consume pseudo-resources from fake items
735 for( const auto &it : req->get_tools() ) {
736 const Character *player_with_inventory = base_.by_radio ? nullptr : &get_avatar();
737 comp_selection<tool_comp> ts =
738 crafting::select_tool_component( it, batch_size_, base_._inv,
739 player_with_inventory, true );
740 if( ts.use_from == cancel ) {
741 return false;
742 }
743 tool_selections_.push_back( ts );
744 }
745 return true;
746}

748void basecamp_action_components::consume_components()
749{
750 map *target_map = &get_map();
751 if( base_.by_radio ) {
752 map_ = std::make_unique<tinymap>();
753 // TODO: fix point types
754 map_->load( project_to<coords::sm>( base_.camp_omt_pos() ).raw(), false );
755 target_map = map_.get();
756 }
757 const tripoint &origin = target_map->getlocal( base_.get_dumping_spot() );
758 for( const comp_selection<item_comp> &sel : item_selections_ ) {
759 g->u.consume_items( *target_map, sel, batch_size_, is_crafting_component, origin,
760 basecamp::inv_range );
761 }
762 // this may consume pseudo-resources from fake items
763 for( const comp_selection<tool_comp> &sel : tool_selections_ ) {
764 g->u.consume_tools( *target_map, sel, batch_size_, origin, basecamp::inv_range, &base_ );
765 }
766 // go back and consume the actual resources
767 for( basecamp_resource &bcp_r : base_.resources ) {
768 if( bcp_r.consumed > 0 ) {
769 target_map->use_charges( origin, basecamp::inv_range, bcp_r.ammo_id, bcp_r.consumed );
770 bcp_r.consumed = 0;
771 }
772 }
773 if( map_ ) {
774 map_->save();
775 map_.reset();
776 }
