132void mission::on_creature_death( Creature &poor_dead_dude )
133{
134 if( poor_dead_dude.is_hallucination() ) {
135 return;
136 }
137 monster *mon = dynamic_cast<monster *>( &poor_dead_dude );
138 if( mon != nullptr ) {
139 if( mon->mission_id == -1 ) {
140 return;
141 }
142 const auto mission = mission::find( mon->mission_id );
143 const auto type = mission->type;
144 if( type->goal == MGOAL_FIND_MONSTER ) {
145 mission->fail();
146 }
147 if( type->goal == MGOAL_KILL_MONSTER ) {
148 mission->step_complete( 1 );
149 }
150 return;
151 }
152 npc *p = dynamic_cast<npc *>( &poor_dead_dude );
153 if( p == nullptr ) {
154 // Must be the player
155 for( auto &miss : g->u.get_active_missions() ) {
156 // mission is free and can be reused
157 miss->player_id = character_id();
158 }
159 // The missions remains assigned to the (dead) character. This should not cause any problems
160 // as the character is dismissed anyway.
161 // Technically, the active missions could be moved to the failed mission section.
162 return;
163 }
164 const auto dead_guys_id = p->getID();
165 for( auto &e : world_missions ) {
166 mission &i = e.second;
167 if( !i.in_progress() ) {
168 continue;
169 }
170 //complete the mission if you needed killing
171 if( i.type->goal == MGOAL_ASSASSINATE && i.target_npc_id == dead_guys_id ) {
172 i.step_complete( 1 );
173 }
174 //fail the mission if the mission giver dies
175 if( i.npc_id == dead_guys_id ) {
176 i.fail();
177 }
178 //fail the mission if recruit target dies
179 if( i.type->goal == MGOAL_RECRUIT_NPC && i.target_npc_id == dead_guys_id ) {
180 i.fail();
181 }
182 }

211{

232 } else {

280void mission::wrap_up()
281{
282 auto &u = g->u;
283 if( u.getID() != player_id ) {
284 // This is called from npctalk.cpp, the npc should only offer the option to wrap up mission
285 // that have been assigned to the current player.

287 player_id.get_value(), u.getID().get_value() );
288 }
289
290 status = mission_status::success;
291 u.on_mission_finished( *this );
292 std::vector<item_comp> comps;
293 switch( type->goal ) {
294 case MGOAL_FIND_ITEM_GROUP: {
295 inventory tmp_inv = u.crafting_inventory();
296 std::vector<item *> items = std::vector<item *>();
297 tmp_inv.dump( items );
298 item_group_id grp_type = type->group_id;
299 itype_id container = type->container_id;

301 bool remove_container = type->remove_container;

303
304 std::map<itype_id, int> matches = std::map<itype_id, int>();

307 container, itype_id( "null" ), specific_container_required );
308

310 comps.push_back( item_comp( cnt.first, cnt.second ) );
311
312 }
313
314 u.consume_items( comps );
315
316 if( remove_container ) {
317 std::vector<item_comp> container_comp = std::vector<item_comp>();
318 if( !empty_container.is_null() ) {
319 container_comp.push_back( item_comp( empty_container, type->item_count ) );
320 u.consume_items( container_comp );

322 container_comp.push_back( item_comp( container, type->item_count ) );
323 u.consume_items( container_comp );
324 }
325 }
326 }
327 break;
328
329 case MGOAL_FIND_ITEM:
330 comps.push_back( item_comp( type->item_id, item_count ) );
331 u.consume_items( comps );
332 break;
333 case MGOAL_FIND_ANY_ITEM:
334 u.remove_mission_items( uid );
335 break;
336 default:
337 //Suppress warnings
338 break;
339 }
340
341 type->end( this );

344bool mission::is_complete( const character_id &_npc_id ) const
345{
346 if( status == mission_status::success ) {
347 return true;
348 }
349
350 auto &u = g->u;
351 switch( type->goal ) {
352 case MGOAL_GO_TO: {
353 const tripoint_abs_omt cur_pos = u.global_omt_location();
354 return ( rl_dist( cur_pos, target ) <= 1 );
355 }
356
357 case MGOAL_GO_TO_TYPE: {
358 const auto cur_ter = overmap_buffer.ter( g->u.global_omt_location() );
359 return is_ot_match( type->target_id.str(), cur_ter, ot_match_type::type );
360 }
361
362 case MGOAL_FIND_ITEM_GROUP: {
363 inventory tmp_inv = u.crafting_inventory();
364 std::vector<item *> items = std::vector<item *>();
365 tmp_inv.dump( items );
366 item_group_id grp_type = type->group_id;
367 itype_id container = type->container_id;
368 bool specific_container_required = !container.is_null();
369
370 std::map<itype_id, int> matches = std::map<itype_id, int>();
371 get_all_item_group_matches(
372 items, grp_type, matches,
373 container, itype_id( "null" ), specific_container_required );
374
375 int total_match = std::accumulate( matches.begin(), matches.end(), 0,
376 []( const std::size_t previous, const std::pair<const itype_id, std::size_t> &p ) {
377 return static_cast<int>( previous + p.second );
378 } );
379
380 if( total_match >= ( type->item_count ) ) {
381 return true;
382
383 }
384 }
385 return false;
386
387 case MGOAL_FIND_ITEM: {
388 if( npc_id.is_valid() && npc_id != _npc_id ) {
389 return false;
390 }
391 const inventory &tmp_inv = u.crafting_inventory();
392 // TODO: check for count_by_charges and use appropriate player::has_* function
393 if( !tmp_inv.has_amount( type->item_id, item_count ) ) {
394 return tmp_inv.has_amount( type->item_id, 1 ) && tmp_inv.has_charges( type->item_id, item_count );
395 }
396 }
397 return true;
398
399 case MGOAL_FIND_ANY_ITEM:
400 return u.has_mission_item( uid ) && ( !npc_id.is_valid() || npc_id == _npc_id );
401
402 case MGOAL_FIND_MONSTER:
403 if( npc_id.is_valid() && npc_id != _npc_id ) {
404 return false;
405 }
406 return g->get_creature_if( [&]( const Creature & critter ) {
407 const monster *const mon_ptr = dynamic_cast<const monster *>( &critter );
408 return mon_ptr && mon_ptr->mission_id == uid;
409 } );
410
411 case MGOAL_RECRUIT_NPC: {
412 npc *p = g->find_npc( target_npc_id );
413 return p != nullptr && p->get_attitude() == NPCATT_FOLLOW;
414 }
415
416 case MGOAL_RECRUIT_NPC_CLASS: {
417 const auto npcs = overmap_buffer.get_npcs_near_player( 100 );
418 for( auto &npc : npcs ) {
419 if( npc->myclass == recruit_class && npc->get_attitude() == NPCATT_FOLLOW ) {
420 return true;
421 }
422 }
423 return false;
424 }
425
426 case MGOAL_FIND_NPC:
427 return npc_id == _npc_id;
428
429 case MGOAL_TALK_TO_NPC:
430 case MGOAL_ASSASSINATE:
431 case MGOAL_KILL_MONSTER:
432 case MGOAL_COMPUTER_TOGGLE:
433 return step >= 1;
434
435 case MGOAL_KILL_MONSTER_TYPE:
436 return g->get_kill_tracker().kill_count( monster_type ) >= kill_count_to_reach;
437
438 case MGOAL_KILL_MONSTER_SPEC:
439 return g->get_kill_tracker().kill_count( monster_species ) >= kill_count_to_reach;
440
441 case MGOAL_CONDITION: {
442 // For now, we only allow completing when talking to the mission originator.
443 if( npc_id != _npc_id ) {
444 return false;
445 }
446
447 npc *n = g->find_npc( _npc_id );
448 if( n == nullptr ) {
449 return false;
450 }
451
452 mission_goal_condition_context cc;
453 cc.alpha = &u;
454 cc.beta = n;
455
456 for( auto &mission : n->chatbin.missions_assigned ) {
457 if( mission->get_assigned_player_id() == g->u.getID() ) {
458 cc.missions_assigned.push_back( mission );
459 }
460 }
461
462 return type->test_goal_condition( cc );
463 }
464
465 default:
466 return false;
467 }

470void mission::get_all_item_group_matches( std::vector<item *> &items,
471 item_group_id &grp_type, std::map<itype_id, int> &matches,
472 const itype_id &required_container, const itype_id &actual_container,
473 bool &specific_container_required )
474{
475 for( item *itm : items ) {
476 bool correct_container = ( required_container == actual_container ) ||
477 !specific_container_required;
478
479 bool item_in_group = item_group::group_contains_item( grp_type, itm->typeId() );
480
481 //check whether item itself is target
482 if( item_in_group && correct_container ) {
483 std::map<itype_id, int>::iterator it = matches.find( itm->typeId() );
484 if( it != matches.end() ) {
485 it->second = ( it->second ) + 1;
486 } else {
487 matches.insert( std::make_pair( itm->typeId(), 1 ) );
488 }
489 }
490
491 //recursivly check item contents for target
492 if( itm->is_container() && !itm->is_container_empty() ) {
493 std::list<item *> content_list = itm->contents.all_items_top();
494 std::vector<item *> content = std::vector<item *>();
495
496 //list of item into list item*
497 std::transform(
498 content_list.begin(), content_list.end(),
499 std::back_inserter( content ),
500 []( item * p ) {
501 return p;
502 } );
503
504 get_all_item_group_matches(
505 content, grp_type, matches,
506 required_container, ( itm->typeId() ), specific_container_required );
507 }
508 }
