230void submap::rotate( int turns )
231{
232 turns = turns % 4;
233
234 if( turns == 0 ) {
235 return;
236 }
237
238 const auto rotate_point = [turns]( const point & p ) {
239 return p.rotate( turns, { SEEX, SEEY } );
240 };
241
242 if( turns == 2 ) {
243 // Swap horizontal stripes.
244 for( int j = 0, je = SEEY / 2; j < je; ++j ) {
245 for( int i = j, ie = SEEX - j; i < ie; ++i ) {
246 swap_soa_tile( { i, j }, rotate_point( { i, j } ) );
247 }
248 }
249 // Swap vertical stripes so that they don't overlap with
250 // the already swapped horizontals.
251 for( int i = 0, ie = SEEX / 2; i < ie; ++i ) {
252 for( int j = i + 1, je = SEEY - i - 1; j < je; ++j ) {
253 swap_soa_tile( { i, j }, rotate_point( { i, j } ) );
254 }
255 }
256 } else {
257 maptile_soa<1, 1> tmp;
258
259 for( int j = 0, je = SEEY / 2; j < je; ++j ) {
260 for( int i = j, ie = SEEX - j - 1; i < ie; ++i ) {
261 auto p = point{ i, j };
262
263 swap_soa_tile( p, tmp );
264
265 for( int k = 0; k < 4; ++k ) {
266 p = rotate_point( p );
267 swap_soa_tile( p, tmp );
268 }
269 }
270 }
271 }
