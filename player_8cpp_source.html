214bool character_martial_arts::pick_style( const avatar &you ) // Style selection menu
215{
216 enum style_selection {
217 KEEP_HANDS_FREE = 0,
218 STYLE_OFFSET
219 };
220
221 // If there are style already, cursor starts there
222 // if no selected styles, cursor starts from no-style
223
224 // Any other keys quit the menu
225 const std::vector<matype_id> &selectable_styles = you.has_active_bionic(
226 bio_cqb ) ? bio_cqb_styles :
227 ma_styles;
228
229 input_context ctxt( "MELEE_STYLE_PICKER" );
230 ctxt.register_action( "SHOW_DESCRIPTION" );
231
232 uilist kmenu;
233 kmenu.text = string_format( _( "Select a style.\n"
234 "\n"
235 "STR: <color_white>%d</color>, DEX: <color_white>%d</color>, "
236 "PER: <color_white>%d</color>, INT: <color_white>%d</color>\n"
237 "Press [<color_yellow>%s</color>] for more info.\n" ),
238 you.get_str(), you.get_dex(), you.get_per(), you.get_int(),
239 ctxt.get_desc( "SHOW_DESCRIPTION" ) );
240 ma_style_callback callback( static_cast<size_t>( STYLE_OFFSET ), selectable_styles );
241 kmenu.callback = &callback;
242 kmenu.input_category = "MELEE_STYLE_PICKER";
243 kmenu.additional_actions.emplace_back( "SHOW_DESCRIPTION", translation() );
244 kmenu.desc_enabled = true;
245 kmenu.addentry_desc( KEEP_HANDS_FREE, true, 'h',
246 keep_hands_free ? _( "Keep hands free (on)" ) : _( "Keep hands free (off)" ),
247 _( "When this is enabled, player won't wield things unless explicitly told to." ) );
248
249 kmenu.selected = STYLE_OFFSET;
250
251 for( size_t i = 0; i < selectable_styles.size(); i++ ) {
252 auto &style = selectable_styles[i].obj();
253 //Check if this style is currently selected
254 const bool selected = selectable_styles[i] == style_selected;
255 std::string entry_text = style.name.translated();
256 if( selected ) {
257 kmenu.selected = i + STYLE_OFFSET;
258 entry_text = colorize( entry_text, c_pink );
259 }
260 kmenu.addentry_desc( i + STYLE_OFFSET, true, -1, entry_text, style.description.translated() );
261 }
262
263 kmenu.query();
264 int selection = kmenu.ret;
265
266 if( selection >= STYLE_OFFSET ) {
267 style_selected = selectable_styles[selection - STYLE_OFFSET];
268 martialart_use_message( you );
269 } else if( selection == KEEP_HANDS_FREE ) {
270 keep_hands_free = !keep_hands_free;
271 } else {
272 return false;
273 }
274
275 return true;
276}

291recipe_subset player::get_recipes_from_books( const inventory &crafting_inv,
292 recipe_filter filter ) const
293{
294 recipe_subset res;
295
296 for( const auto &stack : crafting_inv.const_slice() ) {
297 const item &candidate = stack->front();
298
299 for( std::pair<const recipe *, int> recipe_entry :
300 candidate.get_available_recipes( *this ) ) {
301 if( filter && !filter( *recipe_entry.first ) ) {
302 continue;
303 }
304 res.include( recipe_entry.first, recipe_entry.second );
305 }
306 }
307
308 return res;
309}

311recipe_subset player::get_available_recipes( const inventory &crafting_inv,
312 const std::vector<npc *> *helpers, recipe_filter filter ) const
313{
314 recipe_subset res;
315
316 if( filter ) {
317 res.include_if( get_learned_recipes(), filter );
318 } else {
319 res.include( get_learned_recipes() );
320 }
321
322 res.include( get_recipes_from_books( crafting_inv, filter ) );
323
324 if( helpers != nullptr ) {
325 for( npc *np : *helpers ) {
326 // Directly form the helper's inventory
327 res.include( get_recipes_from_books( np->inv, filter ) );
328 // Being told what to do
329 res.include_if( np->get_learned_recipes(), [ this, &filter ]( const recipe & r ) {
330 if( filter && !filter( r ) ) {
331 return false;
332 }
333 // Skilled enough to understand
334 return get_skill_level( r.skill_used ) >= static_cast<int>( r.difficulty * 0.8f );
335 } );
336 }
337 }
338
339 return res;
340}
